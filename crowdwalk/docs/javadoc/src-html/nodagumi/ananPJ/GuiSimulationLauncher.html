221    public void simulate() {
222        // 既に終わっていたら、警告メッセージ
223        if (!finished) {
224            JOptionPane.showMessageDialog(null,
225                    "Previous simulation not finished?",
226                    "Could not start simulation",
227                    JOptionPane.ERROR_MESSAGE);
228            return;
229        }
230
231        System.gc();
232
233        // スクリーンショットディレクトリのクリア
234        if (recordSimulationScreen && clearScreenshotDir) {
235            File dir = new File(screenshotDir);
236            if (dir.isDirectory()) {
237                for (String filename : dir.list(imageFileFilter)) {
238                    File file = new File(dir, filename);
239                    if (! file.delete()) {
240                        Itk.logError("File can not delete", file.toString());
241                    }
242                }
243            }
244        }
245
246        // シミュレータの実体の初期化
247        initializeSimulatorEntity() ;
248
249        // ウィンドウとGUIの構築
250        setupFrame();
251
252        System.gc();
253
254        // メインループの Runnable 作成。
255        simulationRunnable = new Runnable() {
256            public void run() {
257                simulateMainLoop() ;
258                paused = true ;
259            }
260        };
261    }

263    /**
264     * シミュレータを終了または再起動する
265     */
266    public void quit() {
267        if (quitting) {
268            return;
269        }
270
271        Itk.logInfo("Simulation window closed.") ;
272        // スクリーンショットの保存中ならば完了するのを待つ
273        while (getSaveThreadCount() > 0) {
274            synchronized (this) {
275                try {
276                    wait(100);
277                } catch (InterruptedException e) {}
278            }
279        }
280        if (! finished) {
281            // ログファイルのクローズ処理
282            simulator.finish();
283            Itk.logInfo("Status", getStatusLine());
284            finished = true;
285        }
286        if (resetting) {
287            resetting = false;
288            initPropertyValues();
289            if (editor == null) {
290                init(getPropertiesFile(), settings, commandLineFallbacks);
291            } else {
292                init(editor, settings);
293            }
294            simulate();
295        } else {
296            quitting = true;
297            if (editor != null) {
298                editor.getFrame().setDisableSimulationButtons(false);
299            }
300            if (exitOnClose) {
301                if (editor != null) {
302                    editor.getFrame().exit();
303                } else {
304                    exit(0);
305                }
306            }
307        }
308    }

313                paused = false ;
314                Thread thread = new Thread(simulationRunnable);
315                thread.start();
316            }
317        }

326    public void step() {
327        synchronized (simulationRunnable) {
328            simulateOneStepBare() ;
329            if (exitCount > 0 && counter > exitCount) {
330                finished = true;
331            }
332            if (finished) {
333                finish() ;
334            }
335        }

436    /**
437     * 画面出力用properties設定
438     */
439    protected void setPropertiesForDisplay() {
440        try {
441            deferFactor = properties.getInteger("defer_factor", deferFactor);
442            if (deferFactor < 0 || deferFactor > 299) {
443                throw new Exception("Property error - 設定値が範囲(0～299)外です: defer_factor:" + deferFactor);
444            }
445            verticalScale = properties.getDouble("vertical_scale", verticalScale);
446            if (verticalScale < 0.1 || verticalScale > 10.0) {
447                throw new Exception("Property error - 設定値が範囲(0.1～10.0)外です: vertical_scale:" + verticalScale);
448            }
449            agentSize = properties.getDouble("agent_size", agentSize);
450            zoom = properties.getDouble("zoom", zoom);
451            if (zoom < 0.0 || zoom > 9.9) {
452                throw new Exception("Property error - 設定値が範囲(0.0～9.9)外です: zoom:" + zoom);
453            }
454            cameraFile = properties.getFilePath("camera_file", null);
455            agentAppearanceFile = properties.getFilePath("agent_appearance_file", null);
456            linkAppearanceFile = properties.getFilePath("link_appearance_file", null);
457            nodeAppearanceFile = properties.getFilePath("node_appearance_file", null);
458            showBackgroundImage = properties.getBoolean("show_background_image", false);
459            showBackgroundMap = properties.getBoolean("show_background_map", false);
460            showTheSea = properties.getBoolean("show_the_sea", false);
461            recordSimulationScreen = properties.getBoolean("record_simulation_screen", recordSimulationScreen);
462            screenshotDir = properties.getDirectoryPath("screenshot_dir", screenshotDir).replaceFirst("[/\\\\]+$", "");
463            clearScreenshotDir = properties.getBoolean("clear_screenshot_dir", clearScreenshotDir);
464            if (clearScreenshotDir && ! properties.isDefined("screenshot_dir")) {
465                throw new Exception("Property error - clear_screenshot_dir を有効にするためには screenshot_dir の設定が必要です。");
466            }
467            if (recordSimulationScreen && ! clearScreenshotDir && new File(screenshotDir).list(imageFileFilter).length > 0) {
468                throw new Exception("Property error - スクリーンショットディレクトリに画像ファイルが残っています: screenshot_dir: " + screenshotDir);
469            }
470            screenshotImageType =
471                properties.getStringInPattern("screenshot_image_type",
472                                              screenshotImageType,
473                                              IMAGE_TYPES);
474            hideLinks = properties.getBoolean("hide_links", hideLinks);
475            densityMode = properties.getBoolean("density_mode", densityMode);
476            changeAgentColorDependingOnSpeed =
477                properties.getBoolean("change_agent_color_depending_on_speed", changeAgentColorDependingOnSpeed);
478            drawingAgentByTriageAndSpeedOrder = properties.getBoolean("drawing_agent_by_triage_and_speed_order", drawingAgentByTriageAndSpeedOrder);
479            String show_status =
480                properties.getStringInPattern("show_status", "none",
481                                              SHOW_STATUS_VALUES);
482            if (show_status.equals("none")) {
483                showStatus = false;
484            } else {
485                showStatus = true;
486                showStatusPosition = show_status;
487            }
488            showLogo = properties.getBoolean("show_logo", showLogo);
489            show3dPolygon = properties.getBoolean("show_3D_polygon", show3dPolygon);
490            simulationWindowOpen = properties.getBoolean("simulation_window_open", simulationWindowOpen);
491            autoSimulationStart = properties.getBoolean("auto_simulation_start", autoSimulationStart);
492            exitWithSimulationFinished = properties.getBoolean("exit_with_simulation_finished",
493                    exitWithSimulationFinished);
494        } catch(Exception e) {
495            Itk.logFatal("Property file error", e.getMessage()) ;
496            Itk.quitByError() ;
497        }
498    }

551    /**
552     * 旧書式の link appearance オブジェクトを新書式に変換する.
553     *
554     * ※ width_ratio でリンク幅の倍率を変える機能は廃止した。
555     */
556    public ArrayList<HashMap> convertLinkAppearanceForm(Map<String, Object> object) throws Exception {
557        ArrayList<HashMap> appearances = new ArrayList();
558        JsonicHashMapGetter jsonMap = new JsonicHashMapGetter();
559        for (Map.Entry<String, Object> entry : object.entrySet()) {
560            String tag = entry.getKey();
561            BigDecimal widthRatio = new BigDecimal(1.0);
562            boolean widthFixed = false;
563            String colorName = null;
564            BigDecimal transparency = new BigDecimal(0.75);
565
566            jsonMap.setParameters((HashMap)entry.getValue());
567            widthRatio = jsonMap.getBigDecimalParameter("width", widthRatio);
568            widthRatio = jsonMap.getBigDecimalParameter("width_ratio", widthRatio);
569            widthFixed = jsonMap.getBooleanParameter("width_fixed", widthFixed);
570            colorName = jsonMap.getStringParameter("color", colorName);
571            transparency = jsonMap.getBigDecimalParameter("transparency", transparency);
572
573            HashMap<String, Object> appearance = new HashMap();
574            appearance.put("tag", tag);
575            for (int dimension = 2; dimension <= 3; dimension++) {
576                HashMap<String, Object> view = new HashMap();
577                HashMap<String, Object> parameters = new HashMap();
578                if (widthFixed) {
579                    view.put("className", "FixedWidth" + dimension + "D");
580                    parameters.put("width", widthRatio);
581                } else {
582                    view.put("className", "ActualWidth" + dimension + "D");
583                }
584                if (colorName != null) {
585                    parameters.put("color", colorName);
586                }
587                parameters.put("transparency", transparency);
588                parameters.put("method", "filling");
589                view.put("parameters", parameters);
590                appearance.put("" + dimension + "D_View", view);
591            }
592            appearances.add(appearance);
593        }
594        return appearances;
595    }

677    /**
678     * 海岸線の準備
679     */
680    protected void loadCoastlines() {
681        if (mapTiles == null) {
682            return;
683        }
684        try {
685            ArrayList<String> fileNames = new ArrayList();
686            for (String fileName : properties.getString("coastline_file", "").split(",")) {
687                if (! fileName.isEmpty()) {
688                    fileNames.add(properties.furnishPropertiesDirPath(fileName, true, false));
689                }
690            }
691            if (fileNames.isEmpty()) {
692                return;
693            }
694
695            Rectangle2D boundary = new Rectangle2D.Double();
696            for (GsiTile mapTile : mapTiles) {
697                if (boundary.isEmpty()) {
698                    boundary.setFrameFromDiagonal(mapTile.getPoint(), mapTile.getLowerRightPoint());
699                } else {
700                    boundary.add(mapTile.getPoint());
701                    boundary.add(mapTile.getLowerRightPoint());
702                }
703            }
704            Itk.logInfo("Region of coastline", String.format("(%f, %f)-(%f, %f)", boundary.getX(), boundary.getY(), boundary.getMaxX(), boundary.getMaxY()));
705
706            String cachePath = GsiTile.makeCachePath();
707            GsiTile mapTile = mapTiles.get(0);
708            double elevation = GsiAccessor.getElevation(cachePath, mapTile.getTileNumberX(), mapTile.getTileNumberY(), gsiTileZoom);
709
710            coastline = new Coastline();
711            for (String fileName : fileNames) {
712                coastline.read(fileName, boundary);
713            }
714            coastline.polygonization(boundary, elevation == 0.0);
715        } catch(Exception e) {
716            Itk.logError("loadCoastlines error", e.getMessage());
717        }
718    }

720    /**
721     * 海岸線を取得する
722     */
723    public Coastline getCoastline() {
724        return coastline;
725    }
