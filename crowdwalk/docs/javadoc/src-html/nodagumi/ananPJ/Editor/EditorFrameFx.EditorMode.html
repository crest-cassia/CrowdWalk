478        // シェープファイル読み込みダイアログのプリセット情報を読み込む
479        JSON json = new JSON(JSON.Mode.TRADITIONAL);

482        } catch (IOException e) {
483            Itk.dumpStackTraceOf(e);
484            shapefileSpecs = new HashMap();
485        }
486
487        // ヘルプ画面の準備
488
489        MutableDataSet options = new MutableDataSet();
490        options.setFrom(ParserEmulationProfile.MARKDOWN);
491        options.set(Parser.EXTENSIONS, Arrays.asList(
492            TablesExtension.create()
493        ));
494        parser = Parser.builder(options).build();
495        renderer = HtmlRenderer.builder(options).build();
496
497        webView.setOnKeyPressed(event -> {
498            if (event.isControlDown()) {
499                if (event.getCode() == KeyCode.W) {
500                    helpStage.close();
501                } else if (event.getCode() == KeyCode.PLUS || event.getCode() == KeyCode.UP) {
502                    helpZoom += 0.1;
503                    webView.setZoom(helpZoom);
504                } else if (event.getCode() == KeyCode.MINUS || event.getCode() == KeyCode.DOWN) {
505                    helpZoom -= 0.1;
506                    webView.setZoom(helpZoom);
507                } else if (event.getCode() == KeyCode.DIGIT0) {
508                    helpZoom = 1.0;
509                    webView.setZoom(helpZoom);
510                }
511            }
512        });
513
514        Button okButton = new Button("  OK  ");
515        okButton.setOnAction(event -> helpStage.close());
516        BorderPane buttonPane = new BorderPane();
517        buttonPane.setPadding(new Insets(4, 8, 4, 8));
518        buttonPane.setRight(okButton);
519
520        BorderPane borderPane = new BorderPane();
521        borderPane.setCenter(webView);
522        borderPane.setBottom(buttonPane);
523
524        Scene scene = new Scene(borderPane);
525        scene.getStylesheets().add("stylesheet.css");
526        helpStage.setScene(scene);
527    }

529    /**
530     * 終了処理
531     */
532    public boolean closing(WindowEvent event) {
533        if (editor.isModified()) {
534            Alert alert = new Alert(AlertType.CONFIRMATION, "Warning:\n    Map data has been modified.\n    Do you want to quit anyway?", ButtonType.YES, ButtonType.NO);
535            alert.initOwner(frame);
536            Optional<ButtonType> result = alert.showAndWait();
537            if (! result.isPresent() || result.get() == ButtonType.NO) {
538                if (event != null) {
539                    event.consume();
540                }
541                return false;
542            }
543        }
544        settings.put("_editorPositionX", (int)frame.getX());
545        settings.put("_editorPositionY", (int)frame.getY());
546        settings.put("_editorWidth", (int)frame.getWidth());
547        settings.put("_editorHeight", (int)frame.getHeight());
548        settings.put("dividerPosition", "" + splitPane.getDividerPositions()[0]);
549        return true;
550    }

564                        Itk.dumpStackTraceOf(ex);

951    public void updateRotationMenu() {
952        try {
953            double angle = editor.getProperties().getDouble("rotation_angle", 0.0);
954            if (angle < -180.0 || angle > 180) {
955                Itk.logWarn("Parameter out of range", "rotation_angle", "" + angle);
956                angle %= 360.0;
957                if (angle < -180.0) {
958                    angle = (angle % 180.0) + 180.0;
959                } else if (angle > 180.0) {
960                    angle = (angle % 180.0) - 180.0;
961                }
962            }
963            canvas.rotate(canvas.getWidth() / 2.0, canvas.getHeight() / 2.0, angle);
964
965            boolean angleLocking = editor.getProperties().getBoolean("rotation_angle_locking", false);
966            miResetRotation.setDisable(angleLocking);
967            canvas.setAngleLocking(angleLocking);
968        } catch (Exception e) {
969            Itk.quitWithStackTrace(e);
970        }
971    }

1703    /**
1704     * リンクを表すライン群表示用の Canvas
1705     */
1706    private class LinkViewCanvas extends Canvas {
1707        private HashMap<String, ArrayList<ArrayList<Point2D>>> linesOfMeshCode;
1708        private Rectangle2D boundary = Rectangle2D.EMPTY;
1709
1710        public LinkViewCanvas(HashMap<String, ArrayList<ArrayList<Point2D>>> linesOfMeshCode) {
1711            this.linesOfMeshCode = linesOfMeshCode;
1712            widthProperty().addListener(evt -> draw());
1713            heightProperty().addListener(evt -> draw());
1714        }
1715
1716        public void update() {
1717            boolean passing = false;
1718            double minX = 0.0;
1719            double maxX = 0.0;
1720            double minY = 0.0;
1721            double maxY = 0.0;
1722            for (ArrayList<ArrayList<Point2D>> lines : linesOfMeshCode.values()) {
1723                for (ArrayList<Point2D> line : lines) {
1724                    for (Point2D point : line) {
1725                        if (passing) {
1726                            if (point.getX() < minX) {
1727                                minX = point.getX();
1728                            }
1729                            if (point.getX() > maxX) {
1730                                maxX = point.getX();
1731                            }
1732                            if (point.getY() < minY) {
1733                                minY = point.getY();
1734                            }
1735                            if (point.getY() > maxY) {
1736                                maxY = point.getY();
1737                            }
1738                        } else {
1739                            minX = point.getX();
1740                            maxX = point.getX();
1741                            minY = point.getY();
1742                            maxY = point.getY();
1743                            passing = true;
1744                        }
1745                    }
1746                }
1747            }
1748            boundary = new Rectangle2D(minX, minY, maxX - minX, maxY - minY);
1749            draw();
1750        }
1751
1752        private void draw() {
1753            double width = getWidth();
1754            double height = getHeight();
1755
1756            GraphicsContext gc = getGraphicsContext2D();
1757            gc.clearRect(0, 0, width, height);
1758            if (boundary.getHeight() == 0.0) {
1759                return;
1760            }
1761
1762            double k = Math.min(width / boundary.getWidth(), height / boundary.getHeight());
1763            double offsetX = (width - boundary.getWidth() * k) / 2.0;
1764            double offsetY = (height - boundary.getHeight() * k) / 2.0;
1765
1766            gc.setStroke(FxColor.DEFAULT_LINK_COLOR);
1767            for (ArrayList<ArrayList<Point2D>> lines : linesOfMeshCode.values()) {
1768                for (ArrayList<Point2D> line : lines) {
1769                    Point2D lastPoint = null;
1770                    for (Point2D point : line) {
1771                        Point2D _point = new Point2D((point.getX() - boundary.getMinX()) * k + offsetX, (point.getY() - boundary.getMinY()) * k + offsetY);
1772                        if (lastPoint != null) {
1773                            gc.strokeLine(lastPoint.getX(), lastPoint.getY(), _point.getX(), _point.getY());
1774                        }
1775                        lastPoint = _point;
1776                    }
1777                }
1778            }
1779        }
1780
1781        @Override
1782        public boolean isResizable() {
1783            return true;
1784        }
1785
1786        @Override
1787        public double prefWidth(double height) {
1788            return getWidth();
1789        }
1790
1791        @Override
1792        public double prefHeight(double width) {
1793            return getHeight();
1794        }
1795    }
