036     * データ・モデル
037     */
038    public static class AreaModel {
039        private MapArea area;
040
041        public AreaModel(MapArea area) {
042            this.area = area;
043        }
044
045        public MapArea getArea() { 
046            return area;
047        }
048
049        public String getId() {
050            return area.getID();
051        }
052
053        public String getTags() {
054            return String.join(", ", area.getTags());
055        }
056
057        public Double getWest() {
058            return area.getShape().getBounds2D().getMinX();
059        }
060
061        public Double getNorth() {
062            return area.getShape().getBounds2D().getMinY();
063        }
064
065        public Double getEast() {
066            return area.getShape().getBounds2D().getMaxX();
067        }
068
069        public Double getSouth() {
070            return area.getShape().getBounds2D().getMaxY();
071        }
072
073        public String getGroup() {
074            return area.getParent().toString();
075        }
076    }

078    /**
079     * マップエディタ
080     */
081    private MapEditor editor;
082
083    /**
084     * マップエディタのウィンドウフレーム
085     */
086    private EditorFrameFx frame;
087
088    private TableView<AreaModel> tableView;
089    private ObservableList<AreaModel> areaList = FXCollections.observableArrayList();
090    private FilteredList<AreaModel> filteredData;
091    private TableView.TableViewSelectionModel<AreaModel> selectionModel;
092    private String tag = "";
093    private TextField tagField = new TextField();
094    private Button selectButton = new Button("Select");
095    private ToggleButton filterButton = new ToggleButton("Filter");
096    private CheckBox regexpCheckBox = new CheckBox("regexp");
097    private boolean eventDisabled = false;
098    private HashSet<MapArea> filteredSet = new HashSet();
099
100    private Label obstructerFileLabel;
101    private Button openButton = new Button("Open");

103    /**
104     * コンストラクタ
105     */
106    public AreaPanelFx(MapEditor editor, EditorFrameFx frame) {
107        this.editor = editor;
108        this.frame = frame;
109
110        // Obstructer ファイルペイン
111
112        String fileName = editor.getPollutionFile();
113        obstructerFileLabel = new Label("Obstructer file: " + (fileName == null ? "None" : fileName));
114        obstructerFileLabel.setPadding(new Insets(4, 0, 0, 0));
115        obstructerFileLabel.setTextOverrun(OverrunStyle.CENTER_ELLIPSIS);
116        openButton.setOnAction(e -> {
117            FileChooser fileChooser = new FileChooser();
118            fileChooser.setTitle("Open Obstructer File");
119            frame.setInitialPath(fileChooser, editor.getPollutionFile());
120            fileChooser.getExtensionFilters().addAll(
121                new FileChooser.ExtensionFilter("CSV", "*.csv"),
122                new FileChooser.ExtensionFilter("All", "*.*")
123            );
124            File file = fileChooser.showOpenDialog(frame.getStage());
125            if (file != null) {
126                editor.setPollutionFile(editor.getRelativePath(file));
127                obstructerFileLabel.setText("Obstructer file: " + editor.getRelativePath(file));
128            }
129        });
130        BorderPane obstructerFilePane = new BorderPane();
131        obstructerFilePane.setPadding(new Insets(4, 20, 4, 8));
132        obstructerFilePane.setLeft(obstructerFileLabel);
133        obstructerFilePane.setRight(openButton);
134        this.setTop(obstructerFilePane);
135
136        // テーブル
137
138        tableView = createTableView();
139        this.setCenter(tableView);
140
141        // タグ検索ペイン
142
143        tagField.setMinWidth(250);
144
145        selectButton.setOnAction(e -> {
146            selectionModel.clearSelection();
147            editor.deselectAreas();
148
149            tag = tagField.getText().trim();
150            if (! tag.isEmpty()) {
151                ArrayList<Integer> indices = new ArrayList();
152                int lastIndex = -1;
153                for (int index = 0; index < areaList.size(); index++) {
154                    AreaModel area = areaList.get(index);
155                    if (matching(area.getArea(), tag)) {
156                        indices.add(index);
157                        lastIndex = index;
158                    }
159                }
160                if (! indices.isEmpty()) {
161                    setEventDisabled(true);     // 無駄なイベント処理をスキップさせる
162                    for (Integer index : indices) {
163                        if (index == lastIndex) {
164                            setEventDisabled(false);
165                        }
166                        selectionModel.select(index.intValue());
167                    }
168                }
169            }
170        });
171
172        filterButton.setOnAction(e -> {
173            selectionModel.clearSelection();
174            editor.deselectAreas();
175            frame.getCanvas().repaintLater();
176
177            if (filterButton.isSelected()) {
178                selectButton.setDisable(true);
179                tag = tagField.getText().trim();
180            } else {
181                selectButton.setDisable(false);
182                tag = "";
183            }
184            filteredSet.clear();
185            filteredData.setPredicate(area -> {
186                if (! filterButton.isSelected() || tag.isEmpty() || matching(area.getArea(), tag)) {
187                    filteredSet.add((MapArea)editor.getMap().getObject(area.getId()));
188                    return true;
189                }
190                return false;
191            });
192        });
193
194        FlowPane flowPane = new FlowPane();
195        flowPane.setPadding(new Insets(4));
196        flowPane.setHgap(8);
197        flowPane.setAlignment(Pos.CENTER);
198        flowPane.getChildren().addAll(new Label("Tag"), tagField, selectButton, filterButton, regexpCheckBox);
199        flowPane.setMargin(regexpCheckBox, new Insets(0, 0, 0, 8));
200        this.setBottom(flowPane);
201    }
202
203    /**
204     * テーブルビューの生成
205     */
206    private TableView<AreaModel> createTableView() {
207        TableView<AreaModel> tableView = new TableView<>();
208        selectionModel = tableView.getSelectionModel();
209        selectionModel.setSelectionMode(SelectionMode.MULTIPLE);
210
211        TableColumn<AreaModel, String> idCol = new TableColumn("ID");
212        TableColumn<AreaModel, String> tagsCol = new TableColumn("Tags");
213        TableColumn<AreaModel, Double> westCol = new TableColumn("West");
214        TableColumn<AreaModel, Double> northCol = new TableColumn("North");
215        TableColumn<AreaModel, Double> eastCol = new TableColumn("East");
216        TableColumn<AreaModel, Double> southCol = new TableColumn("South");
217        TableColumn<AreaModel, String> groupCol = new TableColumn("Group");
218
219        idCol.setCellValueFactory(new PropertyValueFactory<>("id"));
220        tagsCol.setCellValueFactory(new PropertyValueFactory<>("tags"));
221        westCol.setCellValueFactory(new PropertyValueFactory<>("west"));
222        northCol.setCellValueFactory(new PropertyValueFactory<>("north"));
223        eastCol.setCellValueFactory(new PropertyValueFactory<>("east"));
224        southCol.setCellValueFactory(new PropertyValueFactory<>("south"));
225        groupCol.setCellValueFactory(new PropertyValueFactory<>("group"));
226
227        reset();
228        filteredData = new FilteredList<>(areaList, p -> true);
229        tableView.setItems(filteredData);
230        tableView.getColumns().addAll(idCol, tagsCol, westCol, northCol, eastCol, southCol, groupCol);
231
232        selectionModel.getSelectedItems().addListener(new ListChangeListener<AreaModel>() {
233            public void onChanged(ListChangeListener.Change<? extends AreaModel> changed) {
234                if (isEventDisabled()) {
235                    return;
236                }
237                editor.deselectAreas();
238                // 選択状態の要素すべて
239                for (AreaModel model : changed.getList()) {
240                    int index = getIndex(model);
241                    MapArea area = (MapArea)editor.getMap().getObject(model.getId());
242                    area.selected = selectionModel.isSelected(index);
243                }
244                frame.getCanvas().repaintLater();
245            }
246        });
247
248        tableView.setContextMenu(frame.getEditAreaMenu());
249
250        return tableView;
251    }
252
253    /**
254     * テーブルの要素を再設定する
255     */
256    public void reset() {
257        String fileName = editor.getPollutionFile();
258        obstructerFileLabel.setText("Obstructer file: " + (fileName == null ? "None" : fileName));
259
260        ArrayList<AreaModel> areaModels = new ArrayList();
261        filteredSet.clear();
262        for (MapArea area : editor.getMap().getAreas()) {
263            areaModels.add(new AreaModel(area));
264            if (! filterButton.isSelected() || tag.isEmpty() || matching(area, tag)) {
265                filteredSet.add(area);
266            }
267        }
268        areaList.setAll(areaModels);
269    }

274    public void widthChanged(double width) {
275        obstructerFileLabel.setPrefWidth(width - 28 - openButton.getWidth());
276    }

278    /**
279     * タグの照合
280     */
281    private boolean matching(MapArea area, String tag) {
282        if (regexpCheckBox.isSelected()) {
283            return area.matchTag(tag) != null;
284        }
285        return area.hasSubTag(tag);
286    }

288    /**
289     * 選択をすべてクリアしてから指定された要素を選択状態にする
290     */
291    public boolean clearAndSelect(MapArea area) {
292        int index = getIndex(area);
293        if (index != -1) {
294            selectionModel.clearAndSelect(index);
295            return true;
296        }
297        return false;
298    }

307    /**
308     * 指定された要素の選択状態を解除する

310    public void clearSelection(MapArea area) {
311        int index = getIndex(area);

313            selectionModel.clearSelection(index);
314        }

317    /**
318     * 指定された要素を選択状態にする
319     */
320    public boolean select(MapArea area) {
321        int index = getIndex(area);
322        if (index != -1) {
323            selectionModel.select(index);
324            return true;
325        }
326        return false;
327    }
328
329    /**
330     * 指定された要素を選択状態にする
331     */
332    public void select(ArrayList<MapArea> areas) {
333        if (areas.isEmpty()) {
334            return;
335        }
336        MapArea lastObject = areas.get(areas.size() - 1);
337        setEventDisabled(true);     // 無駄なイベント処理をスキップさせる
338        for (MapArea area : areas) {
339            if (area == lastObject) {
340                setEventDisabled(false);
341            }
342            selectionModel.select(getIndex(area));
343        }
344    }
345
346    /**
347     * 指定された要素のインデックスを取得する
348     */
349    public int getIndex(MapArea area) {
350        for (int index = 0; index < filteredData.size(); index++) {
351            if (filteredData.get(index).getId().equals(area.getID())) {
352                return index;
353            }
354        }
355        return -1;
356    }
357
358    /**
359     * 指定された要素のインデックスを取得する
360     */
361    public int getIndex(AreaModel area) {
362        for (int index = 0; index < filteredData.size(); index++) {
363            if (filteredData.get(index) == area) {
364                return index;
365            }
366        }
367        return -1;
368    }

384    private synchronized void setEventDisabled(boolean eventDisabled) {
385        this.eventDisabled = eventDisabled;

388    private synchronized boolean isEventDisabled() {
