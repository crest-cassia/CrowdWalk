266        @Override
267        public void actionPerformed(ActionEvent e) {
268                if (e.getActionCommand().equals("Update Frame")) {
269                        settings.updateFrame();
270                        updateButton.setEnabled(false);
271                        refresh();
272                } else if (e.getActionCommand().equals("Remove Frame")) {
273                        ArrayList<EditorFrame> framesToRemove = new ArrayList<EditorFrame>();
274                        ListSelectionModel lsm = frameTable.getSelectionModel();
275                        for (int r = 0; r < frameTable.getRowCount(); ++r) {
276                                if(lsm.isSelectedIndex(r)){
277                                        framesToRemove.add(sortedFrames.get(r));
278                                }
279                        }
280                        //editor.getLayers().removeAll(framesToRemove);
281                        refresh();
282                }

285        private Boolean lockRefresh = false;
286        private Boolean lockValueChanged = false;
287
288        public void refresh() {
289                if (lockRefresh) return;
290
291                //sortedFrames = (ArrayList<EditorFrame>) editor.getLayers().clone();
292                Collections.sort(sortedFrames, new Comparator<EditorFrame>() {
293                        @Override
294                        public int compare(EditorFrame lhs, EditorFrame rhs) {
295                                //if (lhs.layer != rhs.layer) {
296                                        //return lhs.layer - rhs.layer;
297                                //} else 
298                                if (lhs.getDefaultHeight() != rhs.getDefaultHeight()) {
299                                        return (int)(lhs.getDefaultHeight() - rhs.getDefaultHeight());
300                                }
301                                return 0;
302                        }
303                });
304                lockValueChanged = true;
305                dataModel.fireTableDataChanged();
306                updateSelectionFromFrames();
307                SwingUtilities.invokeLater(new Runnable() {
308                        public void run() {
309                                adjustColumnPreferredWidth(frameTable);
310                                frameTable.revalidate();
311                        }
312                });
313                lockValueChanged = false;
314        }
315
316        private void adjustColumnPreferredWidth(JTable table) {
317                TableColumnModel columnModel = table.getColumnModel();
318                for (int col = 0; col < table.getColumnCount(); ++ col) {
319                        int max_width = 0;
320                        
321                        for (int row = 0; row < table.getRowCount(); ++ row) {
322                                TableCellRenderer rend = table.getCellRenderer(row, col);
323                                Object o = table.getValueAt(row,col);
324                                Component c = rend.getTableCellRendererComponent(table, o,
325                                                false, false, row, col);
326                                max_width = Math.max(c.getPreferredSize().width, max_width);
327                        }
328                        TableColumn column = columnModel.getColumn(col);
329                        column.setPreferredWidth(max_width);
330                }
331        }
332
333        public void updateSelectionFromFrames() {
334                ListSelectionModel lsm = frameTable.getSelectionModel();
335                lsm.clearSelection();
336                for (int r = 0; r < frameTable.getRowCount(); ++r) {
337                        if (sortedFrames.get(r).isVisible()) {
338                                lsm.addSelectionInterval(r, r);
339                        }
340                }
341        }
342
343        @Override
344        /* when the selection in the table is changed */
345        public void valueChanged(ListSelectionEvent e) {
346                if (lockValueChanged) return;
347                ListSelectionModel lsm = frameTable.getSelectionModel();
348                int selected_count = 0;
349                EditorFrame selectedFrame = null;
350                for (int r = 0; r < frameTable.getRowCount(); ++r) {
351                        if (lsm.isSelectedIndex(r)) {
352                                sortedFrames.get(r).setVisible(true);
353                                selectedFrame = sortedFrames.get(r);
354                                ++selected_count;
355                        } else {
356                                sortedFrames.get(r).setVisible(false);
357                        }
358                }
359                if (selected_count == 1) {
360                        settings.setFrameInInterest(selectedFrame);
361                        updateButton.setEnabled(true);
362                } else {
363                        settings.setFrameInInterest(null);
364                        updateButton.setEnabled(false);
365                }
366                lockRefresh = true;
367                if (!e.getValueIsAdjusting())editor.updateAll();
368                lockRefresh = false;
369        }
