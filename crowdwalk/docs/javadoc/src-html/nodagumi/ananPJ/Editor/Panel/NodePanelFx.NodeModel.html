029/**
030 * マップエディタの Nodes タブ用のペイン
031 */
032public class NodePanelFx extends BorderPane {
033    /**
034     * データ・モデル
035     */
036    public static class NodeModel {
037        private MapNode node;
038
039        public NodeModel(MapNode node) {
040            this.node = node;
041        }
042
043        public MapNode getNode() { 
044            return node;
045        }
046
047        public String getId() {
048            return node.getID();
049        }
050
051        public String getTags() {
052            return String.join(", ", node.getTags());
053        }
054
055        public Double getX() {
056            return node.getX();
057        }
058
059        public Double getY() {
060            return node.getY();
061        }
062
063        public Double getHeight() {
064            return node.getHeight();
065        }
066
067        public String getGroup() {
068            return node.getParent().toString();
069        }
070    }
071
072    /**
073     * マップエディタ
074     */
075    private MapEditor editor;
076
077    /**
078     * マップエディタのウィンドウフレーム
079     */
080    private EditorFrameFx frame;
081
082    private TableView<NodeModel> tableView;
083    private ObservableList<NodeModel> nodeList = FXCollections.observableArrayList();
084    private FilteredList<NodeModel> filteredData;
085    private TableView.TableViewSelectionModel<NodeModel> selectionModel;
086    private String tag = "";
087    private TextField tagField = new TextField();
088    private Button selectButton = new Button("Select");
089    private ToggleButton filterButton = new ToggleButton("Filter");
090    private CheckBox regexpCheckBox = new CheckBox("regexp");
091    private boolean eventDisabled = false;
092    private HashSet<MapNode> filteredSet = new HashSet();
093
094    /**
095     * コンストラクタ
096     */
097    public NodePanelFx(MapEditor editor, EditorFrameFx frame) {
098        this.editor = editor;
099        this.frame = frame;
100
101        tableView = createTableView();
102        this.setCenter(tableView);
103
104        // タグ検索ペイン
105
106        tagField.setMinWidth(250);
107
108        selectButton.setOnAction(e -> {
109            selectionModel.clearSelection();
110            editor.deselectNodes();
111
112            tag = tagField.getText().trim();
113            if (! tag.isEmpty()) {
114                ArrayList<Integer> indices = new ArrayList();
115                int lastIndex = -1;
116                for (int index = 0; index < nodeList.size(); index++) {
117                    NodeModel node = nodeList.get(index);
118                    if (matching(node.getNode(), tag)) {
119                        indices.add(index);
120                        lastIndex = index;
121                    }
122                }
123                if (! indices.isEmpty()) {
124                    setEventDisabled(true);     // 無駄なイベント処理をスキップさせる
125                    for (Integer index : indices) {
126                        if (index == lastIndex) {
127                            setEventDisabled(false);
128                        }
129                        selectionModel.select(index.intValue());
130                    }
131                }
132            }
133        });
134
135        filterButton.setOnAction(e -> {
136            selectionModel.clearSelection();
137            editor.deselectNodes();
138            frame.getCanvas().repaintLater();
139
140            if (filterButton.isSelected()) {
141                selectButton.setDisable(true);
142                tag = tagField.getText().trim();
143            } else {
144                selectButton.setDisable(false);
145                tag = "";
146            }
147            filteredSet.clear();
148            filteredData.setPredicate(node -> {
149                if (! filterButton.isSelected() || tag.isEmpty() || matching(node.getNode(), tag)) {
150                    filteredSet.add((MapNode)editor.getMap().getObject(node.getId()));
151                    return true;
152                }
153                return false;
154            });
155        });
156
157        FlowPane flowPane = new FlowPane();
158        flowPane.setPadding(new Insets(4));
159        flowPane.setHgap(8);
160        flowPane.setAlignment(Pos.CENTER);
161        flowPane.getChildren().addAll(new Label("Tag"), tagField, selectButton, filterButton, regexpCheckBox);
162        flowPane.setMargin(regexpCheckBox, new Insets(0, 0, 0, 8));
163        this.setBottom(flowPane);
164    }
165
166    /**
167     * テーブルビューの生成
168     */
169    private TableView<NodeModel> createTableView() {
170        TableView<NodeModel> tableView = new TableView<>();
171        selectionModel = tableView.getSelectionModel();
172        selectionModel.setSelectionMode(SelectionMode.MULTIPLE);
173
174        TableColumn<NodeModel, String> idCol = new TableColumn("ID");
175        TableColumn<NodeModel, String> tagsCol = new TableColumn("Tags");
176        TableColumn<NodeModel, Double> xCol = new TableColumn("X");
177        TableColumn<NodeModel, Double> yCol = new TableColumn("Y");
178        TableColumn<NodeModel, Double> heightCol = new TableColumn("Height");
179        TableColumn<NodeModel, String> groupCol = new TableColumn("Group");
180
181        idCol.setCellValueFactory(new PropertyValueFactory<>("id"));
182        tagsCol.setCellValueFactory(new PropertyValueFactory<>("tags"));
183        xCol.setCellValueFactory(new PropertyValueFactory<>("x"));
184        yCol.setCellValueFactory(new PropertyValueFactory<>("y"));
185        heightCol.setCellValueFactory(new PropertyValueFactory<>("height"));
186        groupCol.setCellValueFactory(new PropertyValueFactory<>("group"));
187
188        reset();
189        filteredData = new FilteredList<>(nodeList, p -> true);
190        tableView.setItems(filteredData);
191        tableView.getColumns().addAll(idCol, tagsCol, xCol, yCol, heightCol, groupCol);
192
193        selectionModel.getSelectedItems().addListener(new ListChangeListener<NodeModel>() {
194            public void onChanged(ListChangeListener.Change<? extends NodeModel> changed) {
195                if (isEventDisabled()) {
196                    return;
197                }
198                editor.deselectNodes();
199                // 選択状態の要素すべて
200                for (NodeModel model : changed.getList()) {
201                    int index = getIndex(model);
202                    MapNode node = (MapNode)editor.getMap().getObject(model.getId());
203                    node.selected = selectionModel.isSelected(index);
204                }
205                frame.getCanvas().repaintLater();
206            }
207        });
208
209        tableView.setContextMenu(frame.getEditNodeMenu());
210
211        return tableView;
212    }
213
214    /**
215     * テーブルの要素を再設定する
216     */
217    public void reset() {
218        ArrayList<NodeModel> nodeModels = new ArrayList();
219        filteredSet.clear();
220        for (MapNode node : editor.getMap().getNodes()) {
221            nodeModels.add(new NodeModel(node));
222            if (! filterButton.isSelected() || tag.isEmpty() || matching(node, tag)) {
223                filteredSet.add(node);
224            }
225        }
226        nodeList.setAll(nodeModels);
227    }
228
229    /**
230     * タグの照合
231     */
232    private boolean matching(MapNode node, String tag) {
233        if (regexpCheckBox.isSelected()) {
234            return node.matchTag(tag) != null;
235        }
236        return node.hasSubTag(tag);
237    }
238
239    /**
240     * 選択をすべてクリアしてから指定された要素を選択状態にする
241     */
242    public boolean clearAndSelect(MapNode node) {
243        int index = getIndex(node);
244        if (index != -1) {
245            selectionModel.clearAndSelect(index);
246            return true;
247        }
248        return false;
249    }
250
251    /**
252     * すべての指定された要素の選択状態を解除する
253     */
254    public void clearSelection() {
255        selectionModel.clearSelection();
256    }
257
258    /**
259     * 指定された要素の選択状態を解除する
260     */
261    public void clearSelection(MapNode node) {
262        int index = getIndex(node);
263        if (index != -1) {
264            selectionModel.clearSelection(index);
265        }
266    }
267
268    /**
269     * 指定された要素を選択状態にする
270     */
271    public boolean select(MapNode node) {
272        int index = getIndex(node);
273        if (index != -1) {
274            selectionModel.select(index);
275            return true;
276        }
277        return false;
278    }
279
280    /**
281     * 指定された要素を選択状態にする
282     */
283    public void select(ArrayList<MapNode> nodes) {
284        if (nodes.isEmpty()) {
285            return;
286        }
287        MapNode lastObject = nodes.get(nodes.size() - 1);
288        setEventDisabled(true);     // 無駄なイベント処理をスキップさせる
289        for (MapNode node : nodes) {
290            if (node == lastObject) {
291                setEventDisabled(false);
292            }
293            selectionModel.select(getIndex(node));
294        }
295    }
296
297    /**
298     * 指定された要素のインデックスを取得する
299     */
300    public int getIndex(MapNode node) {
301        for (int index = 0; index < filteredData.size(); index++) {
302            if (filteredData.get(index).getId().equals(node.getID())) {
303                return index;
304            }
305        }
306        return -1;
307    }
308
309    /**
310     * 指定された要素のインデックスを取得する
311     */
312    public int getIndex(NodeModel node) {
313        for (int index = 0; index < filteredData.size(); index++) {
314            if (filteredData.get(index) == node) {
315                return index;
316            }
317        }
318        return -1;
319    }
320
321    /**
322     * 表示を更新する
323     */
324    public void refresh() {
325        tableView.refresh();
326    }
327
328    /**
329     * フィルタリングされたノードのセットを取得する
330     */
331    public HashSet<MapNode> getFilteredSet() {
332        return filteredSet;
333    }
334
335    private synchronized void setEventDisabled(boolean eventDisabled) {
336        this.eventDisabled = eventDisabled;
337    }
338
339    private synchronized boolean isEventDisabled() {
340        return eventDisabled;
341    }
