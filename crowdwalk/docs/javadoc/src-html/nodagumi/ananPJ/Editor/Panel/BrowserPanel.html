214        private boolean recursiveTreeAround(OBNode _OBNode,String tag,OP op){
215                switch (op){ 
216                case ADD:
217                        _OBNode.addTag(Itk.intern(tag)) ;
218                        break;
219                case ALLCLEAR:
220                        _OBNode.allTagsClear();
221                        break;
222                }
223                if(_OBNode.isLeaf()){
224                        return true;
225                }else{
226                        if(_OBNode.getChildCount() != 0){
227                                for(int i = 0; i < _OBNode.getChildCount();i++ ){
228                                        recursiveTreeAround((OBNode)_OBNode.getChildAt(i),tag,op);
229                                }
230                        }
231                        else
232                                return true;
233                }
234                return true;
235        }

298        private void make_group(DefaultMutableTreeNode parentNode){
299                TagNameAndHeightDialog get_tag_and_height =
300                        new TagNameAndHeightDialog(((MapPartGroup)parentNode).getDefaultHeight());
301                get_tag_and_height.setLocation(getMousePosition());
302                get_tag_and_height.setVisible(true);
303                if (get_tag_and_height.getTagName() == null) { return; }
304                
305                MapPartGroup group = editor.getMap().createGroupNode((MapPartGroup) parentNode);
306
307                double default_height = get_tag_and_height.getDefaultHeight();
308                group.setDefaultHeight(default_height);
309                group.setMinHeight(default_height - 5);
310                group.setMaxHeight(default_height + 5);
311                group.addTag(Itk.intern(get_tag_and_height.getTagName())) ;
312
313                groupCount++;
314        tree.repaint();
315        }

348        public void mouseClicked(MouseEvent arg0) {
349                OBNode ob_node = (OBNode)tree.getLastSelectedPathComponent();
350        if (ob_node == null) return;
351
352            if ((arg0.getClickCount() == 2) &&
353                        (javax.swing.SwingUtilities.isLeftMouseButton(arg0))) {
354                /* Generate editor frame */
355                if(!ob_node.isLeaf()){
356                        MapPartGroup group = (MapPartGroup)ob_node;
357                            showEditorForGroup(group);
358                } else {
359                        /* double click on non-group node */
360                }
361            } else {
362                /* single click */
363                MapPartGroup group = (MapPartGroup)ob_node.getParent();
364                if (group != null) {
365                EditorFrame frame = editor.getEditorFrame(group);
366                if (frame != null) {
367                                frame.clearSelection();
368                                ob_node.selected = true;
369                                frame.repaint();
370                        }
371                }
372            }
373        }

442        private EditorFrame showEditorForGroup(MapPartGroup group) {
443        EditorFrame frame = editor.getEditorFrame(group);
444        if (frame == null) {
445            frame = editor.openEditorFrame(group);
446                editor.updateAll();
447                refresh();
448            } else {
449                if (!frame.isVisible()) {
450                        frame.setVisible(true);
451                } else {
452                        /* already shown*/
453                }
454            }
455            return frame;
456        }
