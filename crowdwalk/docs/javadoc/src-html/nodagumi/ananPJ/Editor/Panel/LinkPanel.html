050    final static String[] COLUMN_NAMES = { "Tags", "Length", "Width", "Parent" };
051
052    private class LinksDataModel extends AbstractTableModel {
053
054        public LinksDataModel(final MapLinkTable _links) {
055            shownLinks = _links;
056        }
057        
058        public int getColumnCount() { return 4; }
059        public int getRowCount() { 
060            synchronized(shownLinks){
061                return shownLinks.size();}
062        }
063
064        public String getColumnName(final int col) {
065            return COLUMN_NAMES[col].toString();
066        }
067
068        public Object getValueAt(final int row, final int col) {
069            synchronized(shownLinks) {
070                switch(col) {
071                case 0:
072                    return shownLinks.get(row).getTagString();
073                case 1:
074                    return shownLinks.get(row).getLength();
075                case 2:
076                    return shownLinks.get(row).getWidth();
077                case 3:
078                    return ((OBNode)shownLinks.get(row).getParent()).getTagString();
079                default:
080                    return "ERR(" + row + ", " + col + ")";
081                }
082            }
083        }
084    }

086    public LinkPanel (GuiSimulationEditorLauncher _editor) {
087        editor = _editor;
088        shownLinks = editor.getLinks();
089        synchronized(shownLinks){
090            dataModel = new LinksDataModel(shownLinks);
091        }
092        linkTable = new JTable(dataModel);
093        /* list of links */
094        JPanel link_list_panel = new JPanel(new BorderLayout());
095        setLayout(new BorderLayout());
096        JScrollPane scrollpane = new JScrollPane(linkTable,
097                ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS,
098                ScrollPaneConstants.HORIZONTAL_SCROLLBAR_ALWAYS);
099        scrollpane.setPreferredSize(new Dimension(300, 400));
100
101        linkTable.getSelectionModel().addListSelectionListener(this);
102        link_list_panel.add(scrollpane, BorderLayout.CENTER);
103
104        /* list manipulation */
105        JPanel list_manipulation_panel = new JPanel(new GridBagLayout());
106        GridBagConstraints c;
107        
108        filter_link = new JCheckBox("filter");
109        filter_link.setSelected(false);
110        filter_link.addItemListener(new ItemListener() {
111            @Override
112            public void itemStateChanged(ItemEvent e) {
113                refresh();
114            }
115        });
116        c = new GridBagConstraints();
117        list_manipulation_panel.add(filter_link, c);
118
119        searchText = new JTextArea("");
120        searchText.setPreferredSize(new Dimension(200, 24));
121        c = new GridBagConstraints();
122        c.gridx = 1;
123        list_manipulation_panel.add(searchText, c);
124        searchButton = new JButton("Select");
125        searchButton.addActionListener(new ActionListener() {
126            @Override
127            public void actionPerformed(ActionEvent e) {
128                int count = 0;
129                synchronized(shownLinks) {
130                    for (MapLink link : editor.getLinks()) {
131                        link.selected = (null != link.matchTag(searchText.getText()));
132                        if (link.selected) ++count;
133                    }
134                }
135                repaint();
136                JOptionPane.showMessageDialog(null,
137                        "Found " + count + "links",
138                        "Search finished",
139                        JOptionPane.INFORMATION_MESSAGE);
140                refresh();
141            }
142            
143        });
144        c = new GridBagConstraints();
145        c.gridx = 2;
146        list_manipulation_panel.add(searchButton, c);       
147        
148        placeLink = new JCheckBox("place link");
149        placeLink.addItemListener(new ItemListener() {
150            @Override
151            public void itemStateChanged(ItemEvent e) {
152                if (e.getStateChange() == 2) {
153                    editor.setMode(EditorMode.EDIT_LINK);
154                } else {
155                    editor.setMode(EditorMode.PLACE_LINK);
156                }
157            }
158        });
159        c = new GridBagConstraints();
160        c.gridy = 1;
161        list_manipulation_panel.add(placeLink, c);
162        c = new GridBagConstraints();
163        c.gridx = 1; c.gridy = 1;
164        list_manipulation_panel.add(new JLabel(""), c);
165        
166        JButton deleteLinks = new JButton("Delete selected");
167        deleteLinks.addActionListener(new ActionListener() {
168            @Override
169            public void actionPerformed(ActionEvent e) {
170                if (JOptionPane.showConfirmDialog(null, "Really remove links?", "Proceed",
171                        JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE)
172                        == JOptionPane.NO_OPTION) return;
173                MapLinkTable linksToRemove = new MapLinkTable();
174                synchronized(shownLinks) {
175                    for (MapLink link : shownLinks) {
176                        if (link.selected) linksToRemove.add(link);
177                    }
178                    shownLinks.removeAll(linksToRemove);
179                }
180                for (MapLink link : linksToRemove) {
181                    System.err.println(link.getTagString());
182                    OBNode parent = (OBNode) link.getParent();
183                    editor.getMap().removeOBNode(parent, link, false);
184                }
185                refresh();
186            }
187        });
188        c = new GridBagConstraints();
189        c.gridx = 2; c.gridy = 1;
190        list_manipulation_panel.add(deleteLinks, c);        
191        link_list_panel.add(list_manipulation_panel, BorderLayout.SOUTH);
192        add(link_list_panel, BorderLayout.CENTER);
193        
194        /* attributes of each link */
195        attributePanel = MapLink.getAttributePanel();
196        attributePanel.addListner(this);
197        add(attributePanel, BorderLayout.SOUTH);
198    }

205    private Boolean lockRefresh = false;
206    private Boolean lockValueChanged = false;
207
208    public void refresh() {
209        if (lockRefresh) return;
210
211        if (filter_link.getSelectedObjects() != null) {
212            /* filter mode */
213            shownLinks = new MapLinkTable();
214            synchronized(shownLinks) {
215                for (MapLink link : editor.getLinks()) {
216                    if (null != link.matchTag(searchText.getText())) {
217                        shownLinks.add(link);
218                    }
219                }
220            }
221        } else { 
222            shownLinks = editor.getMap().getLinks();
223        }
224
225        lockValueChanged = true;
226        dataModel.fireTableDataChanged();
227        updateSelectionFromLinks();
228        SwingUtilities.invokeLater(new Runnable() {
229            public void run() {
230                adjustColumnPreferredWidth(linkTable);
231                linkTable.revalidate();
232            }
233        });
234        lockValueChanged = false;
235
236        for (EditorFrame editorFrame : editor.getEditorFrames()) {
237            editorFrame.panel.repaint();
238        }
239    }
240
241    public void updateSelectionFromLinks() {
242        ListSelectionModel lsm = linkTable.getSelectionModel();
243        lsm.clearSelection();
244        for (int r = 0; r < linkTable.getRowCount(); ++r) {
245            synchronized(shownLinks){
246                if (shownLinks.get(r).selected) {
247                    lsm.addSelectionInterval(r, r);
248                }
249            }
250        }
251        updateAttributePanel();
252    }

262    @Override
263    public void valueChanged(ListSelectionEvent e) {
264        if (lockValueChanged) return;
265        
266        updateAttributePanel();
267        lockRefresh = true;
268        if (!e.getValueIsAdjusting())editor.updateAll();
269        lockRefresh = false;
270    }
271    
272    private void updateAttributePanel() {
273        ListSelectionModel lsm = linkTable.getSelectionModel();
274        int count = 0;
275        double length = 0.0, width = 0.0;
276        String label = "";
277        synchronized(shownLinks) {
278            for (int r = 0; r < linkTable.getRowCount(); ++r) {
279                MapLink link = shownLinks.get(r);
280                link.selected = lsm.isSelectedIndex(r);
281                if (link.selected) {
282                    count++;
283                    length += link.getLength();
284                    width += link.getWidth();
285                    for (final String tag : link.getTags()) {
286                        if (!label.contains(tag)){
287                            label += ", " + tag;
288                        }
289                    }
290                }
291            }
292        }
293        if (count == 0) return;
294        length /= count;
295        width /= count;
296
297        boolean single = (count == 1);
298
299        attributePanel.setDetectChange(false);
300        if (single) {
301            attributePanel.setLinkLength(length);
302            attributePanel.setLengthEnabled(true);
303        } else {
304            attributePanel.setLinkLength(length);
305            attributePanel.setLengthEnabled(false);
306        }
307        attributePanel.setLinkWidth(width);
308        attributePanel.setDetectChange(true);
309    }
310
311    @Override
312    public void valueChanged() {
313        ListSelectionModel lsm = linkTable.getSelectionModel();
314        int count = 0;
315        synchronized(shownLinks){
316            for (int r = 0; r < linkTable.getRowCount(); ++r) {
317                MapLink link = shownLinks.get(r);
318                link.selected = lsm.isSelectedIndex(r);
319                if (link.selected) {
320                    count++;
321                }
322            }
323            if (count == 0) return;
324            boolean single = count == 1;
325            for (int r = 0; r < linkTable.getRowCount(); ++r) {
326                MapLink link = shownLinks.get(r);
327                link.selected = lsm.isSelectedIndex(r);
328                if (link.selected) {
329                    link.setWidth(attributePanel.getLinkWidth());
330                    if (single) {
331                        link.setLength(attributePanel.getLinkLength());
332                        return;
333                    }
334                }
335            }
336        }
337        refresh();
338    }
