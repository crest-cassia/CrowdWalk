041public class AreaPanel extends JPanel
042        implements ActionListener, ListSelectionListener {
043        GuiSimulationEditorLauncher editor = null;

050        final static String[] COLUMN_NAMES = { "ID", "Area",
051                "Tags"};
052
053        private class AreaDataModel extends AbstractTableModel {
054                
055                public AreaDataModel() {
056                }
057
058                public int getColumnCount() { return 3; }
059                public int getRowCount() { return editor.getMap().getAreas().size();}
060
061                public String getColumnName(final int col) {
062                        return COLUMN_NAMES[col].toString();
063                }
064
065                public Object getValueAt(final int row, final int col) {
066                        ArrayList<MapArea> areas = editor.getMap().getAreas();
067                        if (areas.isEmpty()) {
068                                return "ERR(" + row + ", " + col + ")";
069                        }
070                        MapArea area = areas.get(row);
071                        switch(col) {
072                        case 0:
073                        {
074                                Matcher match = area.matchTag("(\\d+)");
075                                if (match == null) { return "(not assigned)";}
076                                else { return match.group(0); }
077                        }
078                        case 1:
079                                Rectangle bounds = area.getShape().getBounds();
080                                return "(" + bounds.x + ", " +
081                                bounds.y + ")-(" + (bounds.x + bounds.width) + ", " +
082                                (bounds.y + bounds.height) + ")";
083                        case 2:
084                                return area.getTagString();
085                        }
086                        return "ERR(" + row + ", " + col + ")";
087                }
088        }

090        private JPanel obstructerFilePanel;
091        public AreaPanel(GuiSimulationEditorLauncher _editor) {
092                editor = _editor;
093                setLayout(new BorderLayout());
094                
095                /* obstructer file */
096                obstructerFilePanel = new JPanel(new GridLayout(2, 1));
097                obstructerFileLabel = new JLabel("Obstructer Settings: " +
098                                editor.getSetupFileInfo().getPollutionFile());
099
100                obstructerFilePanel.add(obstructerFileLabel);
101                JButton openButton = new JButton("Open");
102                openButton.addActionListener(this);
103                obstructerFilePanel.add(openButton);
104                add(obstructerFilePanel, BorderLayout.NORTH);
105
106                placeArea = new JCheckBox("place area");
107                placeArea.addItemListener(new ItemListener() {
108                        @Override
109                        public void itemStateChanged(ItemEvent e) {
110                                if (e.getStateChange() == 2) {
111                                        editor.setMode(EditorMode.EDIT_AREA);

113                                        editor.setMode(EditorMode.PLACE_AREA);

115                        }
116                });
117                add(placeArea, BorderLayout.CENTER);

119                dataModel = new AreaDataModel();
120                areaTable = new JTable(dataModel);
121                
122                JScrollPane scrollpane = new JScrollPane(areaTable,
123                                ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS,
124                                ScrollPaneConstants.HORIZONTAL_SCROLLBAR_ALWAYS);

126
127                areaTable.getSelectionModel().addListSelectionListener(this);
128                add(scrollpane, BorderLayout.SOUTH);
129        }

131        private boolean lockRefresh = false;
132        private boolean lockValueChanged = false;
133
134        @Override
135        public String getName() {
136                return "AreaPanel";
137        }
138
139        public void refresh() {
140                if (lockRefresh) return;
141
142                lockValueChanged = true;
143                dataModel.fireTableDataChanged();
144                updateSelectionFromAreas();
145                SwingUtilities.invokeLater(new Runnable() {
146                        public void run() {
147                                adjustColumnPreferredWidth(areaTable);
148                                areaTable.revalidate();
149                        }
150                });
151                lockValueChanged = false;
152                if (editor.getSetupFileInfo().getPollutionFile() != null) {
153                        File obstructerFile =
154                new File(editor.getSetupFileInfo().getPollutionFile());
155                        String message = "";
156                        if (!obstructerFile.exists()) {
157                                message += " (error: does not exist)";
158                        } else if (obstructerFile.isDirectory()) {
159                                message += " (error: is directory)";
160                        } else if (!obstructerFile.canRead()) {
161                                message += " (error: unreadable)";
162                        }
163                        obstructerFileLabel.setText("Obstructer Settings: " +
164                                        obstructerFile.getName() +
165                                        message);
166                } else {                
167                        obstructerFileLabel.setText("Obstructer Settings: file not set");
168                }
169        }
170
171        @Override
172        public void actionPerformed(ActionEvent e) {
173                if (e.getActionCommand().equals("Open")) {
174                        FileDialog fd = new FileDialog(editor.getFrame(), 
175                                        "Set obstructer scenario", FileDialog.LOAD);
176
177            editor.setFileDialogPath(fd, editor.getPollutionFile(), "obstructer");
178                fd.setVisible (true);
179                
180                if (fd.getFile() == null) return;
181                String filename = fd.getDirectory() + fd.getFile();
182                obstructerFileLabel.setText("Obstructer Settings: " + filename);
183                editor.setPollutionFile(filename);
184
185                repaint();
186                }
187        }
188
189        @Override
190        public void valueChanged(ListSelectionEvent e) {
191                if (lockValueChanged) return;
192                ListSelectionModel lsm = areaTable.getSelectionModel();
193                for (int r = 0; r < areaTable.getRowCount(); ++r) {
194                        editor.getMap().getAreas().get(r).selected = lsm.isSelectedIndex(r);
195                }
196                lockRefresh = true;
197                if (!e.getValueIsAdjusting())editor.updateAll();
198                lockRefresh = false;
199        }
200
201        public void updateSelectionFromAreas() {
202                ListSelectionModel lsm = areaTable.getSelectionModel();
203                lsm.clearSelection();
204                for (int r = 0; r < areaTable.getRowCount(); ++r) {
205                        if (editor.getMap().getAreas().get(r).selected) {
206                                lsm.addSelectionInterval(r, r);
207                        }
208                }
209        }
210
211        private void adjustColumnPreferredWidth(JTable table) {
212                TableColumnModel columnModel = table.getColumnModel();
213                for (int col = 0; col < table.getColumnCount(); ++ col) {
214                        int max_width = 0;
215                        
216                        for (int row = 0; row < table.getRowCount(); ++ row) {
217                                TableCellRenderer rend = table.getCellRenderer(row, col);
218                                Object o = table.getValueAt(row,col);
219                                Component c = rend.getTableCellRendererComponent(table, o,
220                                                false, false, row, col);
221                                max_width = Math.max(c.getPreferredSize().width, max_width);
222                        }
223                        TableColumn column = columnModel.getColumn(col);
224                        column.setPreferredWidth(max_width);
225                }
226        }
