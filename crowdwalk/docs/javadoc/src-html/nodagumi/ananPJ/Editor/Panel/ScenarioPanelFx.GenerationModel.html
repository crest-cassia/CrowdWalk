028/**
029 * マップエディタの Scenario タブ用のペイン
030 */
031public class ScenarioPanelFx extends BorderPane {
032    /**
033     * データ・モデル
034     */
035    public static class GenerationModel {
036        private AgentFactory factory;
037
038        public GenerationModel(AgentFactory factory) {
039            this.factory = factory;
040        }
041
042        public AgentFactory getAgentFactory() {
043            return factory;
044        }
045
046        public String getGoal() {
047            if (factory.getGoal() == null) {
048                return "---";
049            }
050            return factory.getGoal().toString();
051        }
052
053        public String getStart() {
054            if (factory.getStartInfo() == null) {
055                return "---";
056            }
057            return factory.getStartInfo();
058        }
059
060        public String getPath() {
061            if (factory.getPlannedRoute() == null) {
062                return "---";
063            }
064            return factory.getPlannedRoute().toString();
065        }
066    }
067
068    /**
069     * マップエディタ
070     */
071    private MapEditor editor;
072
073    /**
074     * マップエディタのウィンドウフレーム
075     */
076    private EditorFrameFx frame;
077
078    private Label generationFileLabel;
079    private Label scenarioFileLabel;
080    private Label fallbackFileLabel;
081    private Button generationFileButton = new Button("Open");
082    private Button scenarioFileButton = new Button("Open");
083    private Button fallbackFileButton = new Button("Open");
084    private TableView<GenerationModel> tableView;
085    private ObservableList<GenerationModel> generationList = FXCollections.observableArrayList();
086    private TableView.TableViewSelectionModel<GenerationModel> selectionModel;
087
088    /**
089     * コンストラクタ
090     */
091    public ScenarioPanelFx(MapEditor editor, EditorFrameFx frame) {
092        this.editor = editor;
093        this.frame = frame;
094
095        // ジェネレーションファイル
096        String fileName = editor.getGenerationFile();
097        generationFileLabel = new Label("Generation file: " + (fileName == null ? "None" : fileName));
098        generationFileLabel.setPadding(new Insets(4, 0, 0, 0));
099        generationFileLabel.setTextOverrun(OverrunStyle.CENTER_ELLIPSIS);
100        generationFileButton.setOnAction(e -> {
101            FileChooser fileChooser = new FileChooser();
102            fileChooser.setTitle("Open Generation File");
103            frame.setInitialPath(fileChooser, editor.getGenerationFile());
104            fileChooser.getExtensionFilters().addAll(
105                new FileChooser.ExtensionFilter("All", "*.*"),
106                new FileChooser.ExtensionFilter("CSV", "*.csv"),
107                new FileChooser.ExtensionFilter("JSON", "*.json")
108            );
109            File file = fileChooser.showOpenDialog(frame.getStage());
110            if (file != null) {
111                editor.setGenerationFile(editor.getRelativePath(file));
112                reset();
113            }
114        });
115        BorderPane generationFilePane = new BorderPane();
116        generationFilePane.setLeft(generationFileLabel);
117        generationFilePane.setRight(generationFileButton);
118
119        // シナリオファイル
120        fileName = editor.getScenarioFile();
121        scenarioFileLabel = new Label("Scenario file: " + (fileName == null ? "None" : fileName));
122        scenarioFileLabel.setPadding(new Insets(4, 0, 0, 0));
123        scenarioFileLabel.setTextOverrun(OverrunStyle.CENTER_ELLIPSIS);
124        scenarioFileButton.setOnAction(e -> {
125            FileChooser fileChooser = new FileChooser();
126            fileChooser.setTitle("Open Scenario File");
127            frame.setInitialPath(fileChooser, editor.getScenarioFile());
128            fileChooser.getExtensionFilters().addAll(
129                new FileChooser.ExtensionFilter("All", "*.*"),
130                new FileChooser.ExtensionFilter("CSV", "*.csv"),
131                new FileChooser.ExtensionFilter("JSON", "*.json")
132            );
133            File file = fileChooser.showOpenDialog(frame.getStage());
134            if (file != null) {
135                editor.setScenarioFile(editor.getRelativePath(file));
136                scenarioFileLabel.setText("Scenario file: " + editor.getRelativePath(file));
137            }
138        });
139        BorderPane scenarioFilePane = new BorderPane();
140        scenarioFilePane.setLeft(scenarioFileLabel);
141        scenarioFilePane.setRight(scenarioFileButton);
142
143        // フォールバックファイル
144        fileName = editor.getFallbackFile();
145        fallbackFileLabel = new Label("Fallback file: " + (fileName == null ? "None" : fileName));
146        fallbackFileLabel.setPadding(new Insets(4, 0, 0, 0));
147        fallbackFileLabel.setTextOverrun(OverrunStyle.CENTER_ELLIPSIS);
148        fallbackFileButton.setOnAction(e -> {
149            FileChooser fileChooser = new FileChooser();
150            fileChooser.setTitle("Open Fallback File");
151            frame.setInitialPath(fileChooser, editor.getFallbackFile());
152            fileChooser.getExtensionFilters().addAll(
153                new FileChooser.ExtensionFilter("All", "*.*"),
154                new FileChooser.ExtensionFilter("JSON", "*.json")
155            );
156            File file = fileChooser.showOpenDialog(frame.getStage());
157            if (file != null) {
158                editor.setFallbackFile(editor.getRelativePath(file), null);
159                fallbackFileLabel.setText("Fallback file: " + editor.getRelativePath(file));
160            }
161        });
162        BorderPane fallbackFilePane = new BorderPane();
163        fallbackFilePane.setLeft(fallbackFileLabel);
164        fallbackFilePane.setRight(fallbackFileButton);
165
166        VBox configPane = new VBox(2, generationFilePane, scenarioFilePane, fallbackFilePane);
167        configPane.setPadding(new Insets(4, 20, 4, 8));
168
169        tableView = createTableView();
170
171        this.setTop(configPane);
172        this.setCenter(tableView);
173    }
174
175    /**
176     * テーブルビューの生成
177     */
178    private TableView<GenerationModel> createTableView() {
179        TableView<GenerationModel> tableView = new TableView<>();
180        selectionModel = tableView.getSelectionModel();
181        selectionModel.setSelectionMode(SelectionMode.MULTIPLE);
182
183        TableColumn<GenerationModel, String> goalCol = new TableColumn("Goal");
184        TableColumn<GenerationModel, String> startCol = new TableColumn("Start");
185        TableColumn<GenerationModel, String> pathCol = new TableColumn("Path");
186
187        goalCol.setCellValueFactory(new PropertyValueFactory<>("goal"));
188        startCol.setCellValueFactory(new PropertyValueFactory<>("start"));
189        pathCol.setCellValueFactory(new PropertyValueFactory<>("path"));
190
191        reset();
192        tableView.setItems(generationList);
193        tableView.getColumns().addAll(goalCol, startCol, pathCol);
194 
195        selectionModel.getSelectedItems().addListener(new ListChangeListener<GenerationModel>() {
196            public void onChanged(ListChangeListener.Change<? extends GenerationModel> changed) {
197                editor.deselectNodes();
198                editor.deselectLinks();
199                // 選択状態の要素すべて
200                for (GenerationModel model : changed.getList()) {
201                    if (model.getAgentFactory().getStartObject() != null) {
202                        int index = getIndex(model);
203                        model.getAgentFactory().getStartObject().selected = selectionModel.isSelected(index);
204                    }
205                }
206                frame.getCanvas().repaintLater();
207            }
208        });
209
210        return tableView;
211    }
212
213    /**
214     * テーブルの要素を再設定する
215     */
216    public void reset() {
217        String fileName = editor.getGenerationFile();
218        generationFileLabel.setText("Generation file: " + (fileName == null ? "None" : fileName));
219
220        fileName = editor.getScenarioFile();
221        scenarioFileLabel.setText("Scenario file: " + (fileName == null ? "None" : fileName));
222
223        fileName = editor.getFallbackFile();
224        fallbackFileLabel.setText("Fallback file: " + (fileName == null ? "None" : fileName));
225
226        generationList.clear();
227        ArrayList<GenerationModel> generationModels = new ArrayList();
228        fileName = editor.getGenerationFile();
229        if (fileName != null && ! fileName.isEmpty()) {
230            try {
231                Term fallbackParameters = editor.getSetupFileInfo().fallbackParameters;
232                AgentFactoryList factoryList =
233                    new AgentFactoryList(fileName, editor.getMap(),
234                                         fallbackParameters, true, 1.0,
235                                         new Random());
236                for (AgentFactory factory : factoryList) {
237                    generationModels.add(new GenerationModel(factory));
238                }
239            } catch(Exception ex) {
240                editor.displayMessage("Error", ex.getMessage(), "Illegal AgentGenerationFile: " + fileName);
241                return;
242            }
243        }
244        generationList.addAll(generationModels);
245    }
246
247    /**
248     * スプリットペインのディバイダが移動した時に呼び出されるコールバック
249     */
250    public void widthChanged(double width) {
251        generationFileLabel.setPrefWidth(width - 28 - generationFileButton.getWidth());
252        scenarioFileLabel.setPrefWidth(width - 28 - scenarioFileButton.getWidth());
253        fallbackFileLabel.setPrefWidth(width - 28 - fallbackFileButton.getWidth());
254    }
255
256    /**
257     * 指定された要素のインデックスを取得する
258     */
259    public int getIndex(GenerationModel model) {
260        for (int index = 0; index < generationList.size(); index++) {
261            if (generationList.get(index) == model) {
262                return index;
263            }
264        }
265        return -1;
266    }
267
268    /**
269     * 表示を更新する
270     */
271    public void refresh() {
272        tableView.refresh();
273    }
