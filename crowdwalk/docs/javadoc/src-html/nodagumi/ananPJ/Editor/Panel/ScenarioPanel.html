045    private class AgentFactoryListModel extends AbstractTableModel {
046        AgentFactoryList factoryList = null;
047        public AgentFactoryListModel() {
048        }
049
050        public void setFile(File _file) {
051            if (_file != null) {
052                try {
053                                        Term fallbackParameters =
054                                                editor.getSetupFileInfo().fallbackParameters ;
055                    factoryList = new AgentFactoryList(_file.getPath(),
056                                                       editor.getMap(),
057                                                       fallbackParameters,
058                                                       true, 1.0, random);
059                } catch(Exception ex) {
060                                        ex.printStackTrace();
061                    System.err.printf("Illegal AgentGenerationFile: %s\n%s", _file.getPath(), ex.getMessage());
062                    System.exit(1);
063                }
064            } else {
065                factoryList = null;
066            }
067        }
068        
069        public int getColumnCount() { return 3; }
070        public int getRowCount() { 
071            if (factoryList == null) return 0;
072            else return factoryList.size();
073        }
074
075        public String getColumnName(final int col) {
076            return COLUMN_NAMES[col].toString();
077        }
078
079        public Object getValueAt(final int row, final int col) {
080            switch(col) {
081            case 0:
082                return factoryList.get(row).getGoal();
083            case 1:
084                return factoryList.get(row).getStartInfo();
085            case 2:
086                return factoryList.get(row).getPlannedRoute();
087            }
088            return "ERR(" + row + ", " + col + ")";
089        }
090        
091        public OBNode getStartObject(final int row) {
092            return factoryList.get(row).getStartObject();
093        }
094    }

150    private void open_generation_file() {
151        FileDialog fd = new FileDialog(editor.getFrame(), 
152                "Set agent generation scenario", FileDialog.LOAD);
153
154        editor.setFileDialogPath(fd, editor.getGenerationFile(), "generation");
155        fd.setVisible (true);
156        if (fd.getFile() == null) return;
157
158        String filename = fd.getDirectory() + fd.getFile();
159        editor.setGenerationFile(filename);
160
161        refresh();
162    }

170        if (fd.getFile() == null) return;

201    public void refresh() {
202        if (editor.getSetupFileInfo().getGenerationFile() != null) {
203            File generation_file =
204                new File(editor.getSetupFileInfo().getGenerationFile());
205            String message = "";
206            if (!generation_file.exists()) {
207                message += " (error: does not exist)";
208            } else if (generation_file.isDirectory()) {
209                message += " (error: is directory)";
210            } else if (!generation_file.canRead()) {
211                message += " (error: unreadable)";
212            }
213            generateFileLabel.setText("Generation: " +
214                    generation_file.getName() +
215                    message);
216            generateFileButton.setToolTipText(generation_file.getPath());
217            factoryListModel.setFile(generation_file);
218        } else {
219            generateFileLabel.setText("Generation: not given");
220            generateFileButton.setToolTipText(null);
221            factoryListModel.setFile(null);
222        }
223
224        if (editor.getSetupFileInfo().getScenarioFile() != null) {
225            File scenario_file =
226                new File(editor.getSetupFileInfo().getScenarioFile());
227            SwingUtilities.invokeLater(new Runnable() {
228                public void run() {
229                    adjustColumnPreferredWidth(factoryTable);
230                    factoryTable.revalidate();
231                }
232            });
233            String message = "";
234            if (!scenario_file.exists()) {
235                message += " (error: does not exist)";
236            } else if (scenario_file.isDirectory()) {
237                message += " (error: is directory)";
238            } else if (!scenario_file.canRead()) {
239                message += " (error: unreadable)";
240            }
241            scenarioFileLabel.setText("Scenario  : " +
242                    scenario_file.getName() +
243                    message);
244            scenarioFileButton.setToolTipText(scenario_file.getPath());
245        } else {
246            scenarioFileLabel.setText("Scenario  : not given");
247            scenarioFileButton.setToolTipText(null);
248        }
249        repaint();
250    }

252    @Override
253    public void valueChanged(ListSelectionEvent e) {
254        ListSelectionModel lsm = factoryTable.getSelectionModel();
255        for (int r = 0; r < factoryTable.getRowCount(); ++r) {
256            if (lsm.isSelectedIndex(r)) {
257                factoryListModel.getStartObject(r).selected = true;
258            }
259        }
260    }
