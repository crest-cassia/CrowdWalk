028/**
029 * マップエディタの Polygon タブ用のペイン
030 */
031public class PolygonPanel extends BorderPane {
032    /**
033     * データ・モデル
034     */
035    public static class PolygonModel {
036        private MapPolygon polygon;
037
038        public PolygonModel(MapPolygon polygon) {
039            this.polygon = polygon;
040        }
041
042        public MapPolygon getPolygon() { 
043            return polygon;
044        }
045
046        public String getId() {
047            return polygon.getID();
048        }
049
050        public String getTags() {
051            return String.join(", ", polygon.getTags());
052        }
053
054        public Integer getZindex() {
055            return polygon.getZIndex();
056        }
057
058        public String getType() {
059            if (polygon.isTriangleMeshes()) {
060                return "Triangle Mesh";
061            }
062            if (polygon.isPlanePolygon()) {
063                return String.format("Plane Polygon[height=%s, %d innerBoundaries]", polygon.getOuterBoundary().getHeight(), polygon.getInnerBoundaries().size());
064            }
065            return "None";
066        }
067
068        public String getGroup() {
069            return polygon.getParent().toString();
070        }
071    }
072
073    /**
074     * マップエディタ
075     */
076    private MapEditor editor;
077
078    /**
079     * マップエディタのウィンドウフレーム
080     */
081    private EditorFrameFx frame;
082
083    private TableView<PolygonModel> tableView;
084    private ObservableList<PolygonModel> polygonList = FXCollections.observableArrayList();
085    private FilteredList<PolygonModel> filteredData;
086    private TableView.TableViewSelectionModel<PolygonModel> selectionModel;
087    private String tag = "";
088    private TextField tagField = new TextField();
089    private Button selectButton = new Button("Select");
090    private ToggleButton filterButton = new ToggleButton("Filter");
091    private CheckBox regexpCheckBox = new CheckBox("regexp");
092    private boolean eventDisabled = false;
093    private HashSet<MapPolygon> filteredSet = new HashSet();
094
095    /**
096     * コンストラクタ
097     */
098    public PolygonPanel(MapEditor editor, EditorFrameFx frame) {
099        this.editor = editor;
100        this.frame = frame;
101
102        tableView = createTableView();
103        this.setCenter(tableView);
104
105        // タグ検索ペイン
106
107        tagField.setMinWidth(250);
108
109        selectButton.setOnAction(e -> {
110            selectionModel.clearSelection();
111            editor.deselectPolygons();
112
113            tag = tagField.getText().trim();
114            if (! tag.isEmpty()) {
115                ArrayList<Integer> indices = new ArrayList();
116                int lastIndex = -1;
117                for (int index = 0; index < polygonList.size(); index++) {
118                    PolygonModel polygon = polygonList.get(index);
119                    if (matching(polygon.getPolygon(), tag)) {
120                        indices.add(index);
121                        lastIndex = index;
122                    }
123                }
124                if (! indices.isEmpty()) {
125                    setEventDisabled(true);     // 無駄なイベント処理をスキップさせる
126                    for (Integer index : indices) {
127                        if (index == lastIndex) {
128                            setEventDisabled(false);
129                        }
130                        selectionModel.select(index.intValue());
131                    }
132                }
133            }
134        });
135
136        filterButton.setOnAction(e -> {
137            selectionModel.clearSelection();
138            editor.deselectPolygons();
139            frame.getCanvas().repaintLater();
140
141            if (filterButton.isSelected()) {
142                selectButton.setDisable(true);
143                tag = tagField.getText().trim();
144            } else {
145                selectButton.setDisable(false);
146                tag = "";
147            }
148            filteredSet.clear();
149            filteredData.setPredicate(polygon -> {
150                if (! filterButton.isSelected() || tag.isEmpty() || matching(polygon.getPolygon(), tag)) {
151                    filteredSet.add((MapPolygon)editor.getMap().getObject(polygon.getId()));
152                    return true;
153                }
154                return false;
155            });
156        });
157
158        FlowPane flowPane = new FlowPane();
159        flowPane.setPadding(new Insets(4));
160        flowPane.setHgap(8);
161        flowPane.setAlignment(Pos.CENTER);
162        flowPane.getChildren().addAll(new Label("Tag"), tagField, selectButton, filterButton, regexpCheckBox);
163        flowPane.setMargin(regexpCheckBox, new Insets(0, 0, 0, 8));
164        this.setBottom(flowPane);
165    }
166
167    /**
168     * テーブルビューの生成
169     */
170    private TableView<PolygonModel> createTableView() {
171        TableView<PolygonModel> tableView = new TableView<>();
172        selectionModel = tableView.getSelectionModel();
173        selectionModel.setSelectionMode(SelectionMode.MULTIPLE);
174
175        TableColumn<PolygonModel, String> idCol = new TableColumn("ID");
176        TableColumn<PolygonModel, String> tagsCol = new TableColumn("Tags");
177        TableColumn<PolygonModel, Integer> zindexCol = new TableColumn("Z-index");
178        TableColumn<PolygonModel, String> typeoCol = new TableColumn("Type");
179        TableColumn<PolygonModel, String> groupCol = new TableColumn("Group");
180
181        idCol.setCellValueFactory(new PropertyValueFactory<>("id"));
182        tagsCol.setCellValueFactory(new PropertyValueFactory<>("tags"));
183        zindexCol.setCellValueFactory(new PropertyValueFactory<>("zindex"));
184        typeoCol.setCellValueFactory(new PropertyValueFactory<>("type"));
185        groupCol.setCellValueFactory(new PropertyValueFactory<>("group"));
186
187        reset();
188        filteredData = new FilteredList<>(polygonList, p -> true);
189        tableView.setItems(filteredData);
190        tableView.getColumns().addAll(idCol, tagsCol, zindexCol, typeoCol, groupCol);
191
192        selectionModel.getSelectedItems().addListener(new ListChangeListener<PolygonModel>() {
193            public void onChanged(ListChangeListener.Change<? extends PolygonModel> changed) {
194                if (isEventDisabled()) {
195                    return;
196                }
197                editor.deselectPolygons();
198                // 選択状態の要素すべて
199                for (PolygonModel model : changed.getList()) {
200                    int index = getIndex(model);
201                    MapPolygon polygon = (MapPolygon)editor.getMap().getObject(model.getId());
202                    polygon.selected = selectionModel.isSelected(index);
203                }
204                frame.getCanvas().repaintLater();
205            }
206        });
207
208        tableView.setContextMenu(frame.getEditPolygonMenu());
209
210        return tableView;
211    }
212
213    /**
214     * テーブルの要素を再設定する
215     */
216    public void reset() {
217        ArrayList<PolygonModel> polygonModels = new ArrayList();
218        filteredSet.clear();
219        for (MapPolygon polygon : editor.getMap().getPolygons()) {
220            polygonModels.add(new PolygonModel(polygon));
221            if (! filterButton.isSelected() || tag.isEmpty() || matching(polygon, tag)) {
222                filteredSet.add(polygon);
223            }
224        }
225        polygonList.setAll(polygonModels);
226    }
227
228    /**
229     * タグの照合
230     */
231    private boolean matching(MapPolygon polygon, String tag) {
232        if (regexpCheckBox.isSelected()) {
233            return polygon.matchTag(tag) != null;
234        }
235        return polygon.hasSubTag(tag);
236    }
237
238    /**
239     * 選択をすべてクリアしてから指定された要素を選択状態にする
240     */
241    public boolean clearAndSelect(MapPolygon polygon) {
242        int index = getIndex(polygon);
243        if (index != -1) {
244            selectionModel.clearAndSelect(index);
245            return true;
246        }
247        return false;
248    }
249
250    /**
251     * すべての指定された要素の選択状態を解除する
252     */
253    public void clearSelection() {
254        selectionModel.clearSelection();
255    }
256
257    /**
258     * 指定された要素の選択状態を解除する
259     */
260    public void clearSelection(MapPolygon polygon) {
261        int index = getIndex(polygon);
262        if (index != -1) {
263            selectionModel.clearSelection(index);
264        }
265    }
266
267    /**
268     * 指定された要素を選択状態にする
269     */
270    public boolean select(MapPolygon polygon) {
271        int index = getIndex(polygon);
272        if (index != -1) {
273            selectionModel.select(index);
274            return true;
275        }
276        return false;
277    }
278
279    /**
280     * 指定された要素を選択状態にする
281     */
282    public void select(ArrayList<MapPolygon> polygons) {
283        if (polygons.isEmpty()) {
284            return;
285        }
286        MapPolygon lastObject = polygons.get(polygons.size() - 1);
287        setEventDisabled(true);     // 無駄なイベント処理をスキップさせる
288        for (MapPolygon polygon : polygons) {
289            if (polygon == lastObject) {
290                setEventDisabled(false);
291            }
292            selectionModel.select(getIndex(polygon));
293        }
294    }
295
296    /**
297     * 指定された要素のインデックスを取得する
298     */
299    public int getIndex(MapPolygon polygon) {
300        for (int index = 0; index < filteredData.size(); index++) {
301            if (filteredData.get(index).getId().equals(polygon.getID())) {
302                return index;
303            }
304        }
305        return -1;
306    }
307
308    /**
309     * 指定された要素のインデックスを取得する
310     */
311    public int getIndex(PolygonModel polygon) {
312        for (int index = 0; index < filteredData.size(); index++) {
313            if (filteredData.get(index) == polygon) {
314                return index;
315            }
316        }
317        return -1;
318    }
319
320    /**
321     * 表示を更新する
322     */
323    public void refresh() {
324        tableView.refresh();
325    }
326
327    /**
328     * フィルタリングされたリンクのセットを取得する
329     */
330    public HashSet<MapPolygon> getFilteredSet() {
331        return filteredSet;
332    }
333
334    private synchronized void setEventDisabled(boolean eventDisabled) {
335        this.eventDisabled = eventDisabled;
336    }
337
338    private synchronized boolean isEventDisabled() {
339        return eventDisabled;
340    }
