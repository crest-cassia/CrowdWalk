061    /**
062     * コンストラクタ
063     */
064    public GroupPanel(MapEditor editor) {
065        this.editor = editor;
066
067        // コンテキストメニューの生成
068        createContextMenu();
069
070        this.setOnMouseClicked(event -> {
071            if (event.getButton() == MouseButton.PRIMARY || event.getButton() == MouseButton.SECONDARY) {
072                TreeItem<OBNode> item = getSelectedItem();
073                if (item != null) {
074                    selectedGroup = (MapPartGroup)item.getValue();
075                    if (selectedGroup != editor.getCurrentGroup() && selectedGroup != editor.getMap().getRoot()) {
076                        editor.getFrame().clickGroupButton(selectedGroup);
077                    }
078                }
079            }
080            if (event.getButton() == MouseButton.SECONDARY) {
081                showContextMenu(event.getScreenX(), event.getScreenY());
082            }
083        });
084
085        construct();
086    }

088    /**
089     * コンテキスト・メニューを生成する
090     */
091    private void createContextMenu() {
092        // ルートグループメニュー
093
094        MenuItem miSetRootGroupAttributes = new MenuItem("Set group attributes");
095        miSetRootGroupAttributes.setOnAction(e -> openGroupAttributesDialog(selectedGroup));
096
097        MenuItem miAddGroup = new MenuItem("Add group");
098        miAddGroup.setOnAction(e -> {
099            MapPartGroup root = (MapPartGroup)editor.getMap().getRoot();
100            if (editor.getMap().getGroups().size() == 1 && ! root.getChildNodes().isEmpty()) {
101                // ルートグループにマップが存在する場合にはサブグループを作ることは出来ない
102                Alert alert = new Alert(AlertType.WARNING, "Can not create subgroup.", ButtonType.OK);
103                alert.initOwner(getStage());
104                alert.showAndWait();
105                return;
106            }
107            openGroupAdditionDialog();
108        });
109
110        rootGroupMenu = new ContextMenu();
111        rootGroupMenu.getItems().addAll(miSetRootGroupAttributes, miAddGroup);
112
113        // グループメニュー
114
115        MenuItem miSetGroupAttributes = new MenuItem("Set group attributes");
116        miSetGroupAttributes.setOnAction(e -> openGroupAttributesDialog(selectedGroup));
117
118        MenuItem miDuplicateFloor = new MenuItem("Duplicate floor");
119        miDuplicateFloor.setOnAction(e -> editor.getFrame().openDuplicateFloorDialog());
120
121        MenuItem miAddTagToAllNodes = new MenuItem("Add tag to all nodes");
122        miAddTagToAllNodes.setOnAction(e -> addTagTo(selectedGroup.getChildNodes(), "Add tag to all nodes", "node"));
123
124        MenuItem miAddTagToAllLinks = new MenuItem("Add tag to all links");
125        miAddTagToAllLinks.setOnAction(e -> addTagTo(selectedGroup.getChildLinks(), "Add tag to all links", "link"));
126
127        MenuItem miAddTagToAllAreas = new MenuItem("Add tag to all areas");
128        miAddTagToAllAreas.setOnAction(e -> addTagTo(selectedGroup.getChildMapAreas(), "Add tag to all areas", "area"));
129
130        MenuItem miRemoveTagFromAllNodes = new MenuItem("Remove tag from all nodes");
131        miRemoveTagFromAllNodes.setOnAction(e -> removeTagFrom(selectedGroup.getChildNodes(), "Remove tag from all nodes", "node"));
132
133        MenuItem miRemoveTagFromAllLinks = new MenuItem("Remove tag from all links");
134        miRemoveTagFromAllLinks.setOnAction(e -> removeTagFrom(selectedGroup.getChildLinks(), "Remove tag from all links", "link"));
135
136        MenuItem miRemoveTagFromAllAreas = new MenuItem("Remove tag from all areas");
137        miRemoveTagFromAllAreas.setOnAction(e -> removeTagFrom(selectedGroup.getChildMapAreas(), "Remove tag from all areas", "area"));
138
139        MenuItem miRemoveGroup = new MenuItem("Remove group");
140        miRemoveGroup.setOnAction(e -> {
141            if (selectedGroup.children().hasMoreElements()) {
142                Alert alert = new Alert(AlertType.WARNING, "It can not be removed until the contents are empty.", ButtonType.OK);
143                alert.initOwner(getStage());
144                alert.showAndWait();
145                return;
146            }
147            editor.invokeSingleCommand(new RemoveGroup(selectedGroup));
148        });
149
150        groupMenu = new ContextMenu();
151        groupMenu.getItems().addAll(miSetGroupAttributes, miDuplicateFloor, miAddTagToAllNodes, miAddTagToAllLinks, miAddTagToAllAreas, miRemoveTagFromAllNodes, miRemoveTagFromAllLinks, miRemoveTagFromAllAreas, miRemoveGroup);
152    }

154    /**
155     * コンテキスト・メニューを表示する
156     */
157    private void showContextMenu(double x, double y) {
158        TreeItem<OBNode> item = getSelectedItem();
159        if (item == null) {
160            return;
161        }
162        selectedGroup = (MapPartGroup)item.getValue();
163        if (selectedGroup == editor.getMap().getRoot()) {
164            setContextMenu(rootGroupMenu);
165        } else {
166            setContextMenu(groupMenu);
167        }
168    }

170    /**
171     * マウスがクリックしているアイテムを返す
172     */
173    private TreeItem<OBNode> getSelectedItem() {
174        if (this.getSelectionModel().getSelectedItems().size() != 1) {
175            return null;
176        }
177        return this.getSelectionModel().getSelectedItems().get(0);
178    }

180    /**
181     * 複数の OBNode に指定したタグを付加する
182     */
183    private void addTagTo(ArrayList<? extends OBNode> obNodes, String title, String objectName) {
184        if (obNodes.isEmpty()) {
185            Alert alert = new Alert(AlertType.WARNING, "There is no " + objectName + ".", ButtonType.OK);
186            alert.initOwner(getStage());
187            alert.showAndWait();
188            return;
189        }
190        TextInputDialog dialog = new TextInputDialog();
191        dialog.initOwner(getStage());
192        dialog.getDialogPane().setPrefWidth(250);
193        dialog.setTitle(title);
194        dialog.setHeaderText("Enter the tag");
195        String tag = dialog.showAndWait().orElse("").trim();
196        if (! tag.isEmpty()) {
197            editor.addTag(obNodes, tag);
198        }
199    }
200
201    /**
202     * 複数の OBNode から指定したタグを削除する
203     */
204    private void removeTagFrom(ArrayList<? extends OBNode> obNodes, String title, String objectName) {
205        if (obNodes.isEmpty()) {
206            Alert alert = new Alert(AlertType.WARNING, "There is no " + objectName + ".", ButtonType.OK);
207            alert.initOwner(getStage());
208            alert.showAndWait();
209            return;
210        }
211        TextInputDialog dialog = new TextInputDialog();
212        dialog.initOwner(getStage());
213        dialog.getDialogPane().setPrefWidth(250);
214        dialog.setTitle(title);
215        dialog.setHeaderText("Enter the tag");
216        String tag = dialog.showAndWait().orElse("").trim();
217        if (! tag.isEmpty()) {
218            editor.removeTag(obNodes, tag);
219        }
220    }

222    /**
223     * グループ設定ダイアログを開く
224     */
225    private void openGroupAttributesDialog(MapPartGroup group) {
226        // TODO: この処理はマップを読み込んだ時点でおこなってもよい
227        if (group.getTags().size() != 1) {
228            Alert alert = new Alert(AlertType.ERROR, "Tags: " + group.getTagString() + "\n\nOnly one group tag is allowed.", ButtonType.OK);
229            alert.initOwner(getStage());
230            alert.showAndWait();
231            return;
232        }
233
234        Dialog dialog = new Dialog();
235        dialog.initOwner(getStage());
236        dialog.setTitle("Set group attributes");
237
238        Label paramLabel = new Label("Parameters");
239        paramLabel.setFont(Font.font("Arial", FontWeight.BOLD, paramLabel.getFont().getSize()));
240
241        // defaultHeight field
242        Label defaultHeightLabel = new Label("Default height");
243        defaultHeightLabel.setPadding(new Insets(0, 0, 0, 4));
244        TextField defaultHeightField = new TextField("" + group.getDefaultHeight());
245        defaultHeightField.setMinWidth(100);
246        Button defaultHeightButton = new Button("Set");
247        EventHandler defaultHeightHandler = new EventHandler<ActionEvent>() {
248            public void handle(ActionEvent event) {
249                Double value = editor.getFrame().convertToDouble(defaultHeightField.getText());
250                if (value != null) {
251                    double defaultHeight = value;
252                    if (defaultHeight != group.getDefaultHeight()) {
253                        editor.invokeSingleCommand(new SetDefaultHeight(group, defaultHeight));
254                    }
255                    dialog.close();
256                }
257            }
258        };
259        defaultHeightField.setOnAction(defaultHeightHandler);
260        defaultHeightButton.setOnAction(defaultHeightHandler);
261
262        // zone ChoiceBox
263        Label zoneLabel = new Label("Zone");
264        zoneLabel.setPadding(new Insets(0, 0, 0, 4));
265        ArrayList<String> zones = new ArrayList();
266        for (int zone = 0; zone <= GsiTile.JGD2000_JPR_EPSG_NAMES.length - 1; zone++) {
267            zones.add("" + zone);
268        }
269        ChoiceBox<String> zoneChoiceBox = new ChoiceBox(FXCollections.observableArrayList(zones));
270        zoneChoiceBox.setValue("" + group.getZone());
271        Button zoneButton = new Button("Set");
272        zoneButton.setOnAction(e -> {
273            int zone = Integer.parseInt((String)zoneChoiceBox.getValue());
274            if (zone != group.getZone()) {
275                editor.invokeSingleCommand(new SetZone(group, zone));
276            }
277            dialog.close();
278        });
279        if (group != editor.getMap().getRoot()) {
280            zoneLabel.setDisable(true);
281            zoneChoiceBox.setDisable(true);
282            zoneButton.setDisable(true);
283        }
284
285        // scale field
286        Label scaleLabel = new Label("Scale");
287        scaleLabel.setPadding(new Insets(0, 0, 0, 4));
288        TextField scaleField = new TextField("" + group.getScale());
289        scaleField.setMinWidth(100);
290        Button scaleButton = new Button("Set");
291        EventHandler scaleHandler = new EventHandler<ActionEvent>() {
292            public void handle(ActionEvent event) {
293                Double value = editor.getFrame().convertToDouble(scaleField.getText());
294                if (value != null) {
295                    double scale = value;
296                    Alert alert = new Alert(AlertType.CONFIRMATION, "Recalculate link lengths of this group?", ButtonType.YES, ButtonType.NO, ButtonType.CANCEL);
297                    alert.initOwner(getStage());
298                    Optional<ButtonType> result = alert.showAndWait();
299                    if (result.isPresent()) {
300                        if (result.get() == ButtonType.YES) {
301                            editor.setScaleAndRecalculateLinkLength(group, scale, true, false);
302                        } else if (result.get() == ButtonType.NO) {
303                            editor.setScaleAndRecalculateLinkLength(group, scale, false, false);
304                        }
305                    }
306                    dialog.close();
307                }
308            }
309        };
310        scaleField.setOnAction(scaleHandler);
311        scaleButton.setOnAction(scaleHandler);
312
313        GridPane grid = new GridPane();
314        grid.setHgap(8);
315        grid.setVgap(10);
316        grid.add(defaultHeightLabel, 1, 1);
317        grid.add(defaultHeightField, 2, 1);
318        grid.add(defaultHeightButton, 3, 1);
319        grid.add(zoneLabel, 1, 2);
320        grid.add(zoneChoiceBox, 2, 2);
321        grid.add(zoneButton, 3, 2);
322        grid.add(scaleLabel, 1, 3);
323        grid.add(scaleField, 2, 3);
324        grid.add(scaleButton, 3, 3);
325
326        Separator separator = new Separator();
327        separator.setPadding(new Insets(10, 0, 8, 0));
328
329        // Tag field
330        Label tagLabel = new Label("Tag");
331        tagLabel.setFont(Font.font("Arial", FontWeight.BOLD, tagLabel.getFont().getSize()));
332        tagLabel.setPadding(new Insets(0, 0, 8, 0));
333        TextField tagField = new TextField(group.getTagString());
334        Button tagButton = new Button("Set");
335        tagField.setMinWidth(250);
336        tagField.setOnKeyReleased(e -> {
337            String text = tagField.getText();
338            tagButton.setDisable(text.isEmpty() || text.indexOf(" ") != -1 || sameTagExists(null, text));
339        });
340        tagField.setOnAction(e -> {
341            String text = tagField.getText();
342            if (! text.equals(group.getTagString())) {
343                if (text.isEmpty() || text.indexOf(" ") != -1 || sameTagExists(group, text)) {
344                    Alert alert = new Alert(AlertType.WARNING, "\"" + text + "\" is an invalid tag.", ButtonType.OK);
345                    alert.initOwner(getStage());
346                    alert.showAndWait();
347                    return;
348                }
349                editor.invokeSingleCommand(new SetGroupTag(group, text));
350            }
351            dialog.close();
352        });
353        tagButton.setDisable(true);
354        tagButton.setOnAction(e -> {
355            editor.invokeSingleCommand(new SetGroupTag(group, tagField.getText()));
356            dialog.close();
357        });
358        FlowPane tagFlowPane = new FlowPane();
359        tagFlowPane.setHgap(8);
360        tagFlowPane.getChildren().addAll(tagField, tagButton);
361
362        VBox vBox = new VBox();
363        vBox.getChildren().addAll(paramLabel, grid, separator, tagLabel, tagFlowPane);
364
365        dialog.getDialogPane().setContent(vBox);
366        dialog.getDialogPane().getButtonTypes().add(ButtonType.CANCEL);
367        dialog.showAndWait();
368    }

370    /**
371     * グループ追加ダイアログを開く
372     */
373    private void openGroupAdditionDialog() {
374        // 初期値として利用するグループ(root 以外の方が参考になる)
375        MapPartGroup root = (MapPartGroup)editor.getMap().getRoot();
376        MapPartGroup group = root;
377        for (MapPartGroup _group : editor.getMap().getGroups()) {
378            if (_group != group) {
379                group = _group;
380                break;
381            }
382        }
383
384        Dialog dialog = new Dialog();
385        dialog.initOwner(getStage());
386        dialog.setTitle("Add group");
387        DialogPane pane = dialog.getDialogPane();
388
389        Label paramLabel = new Label("Parameters");
390        paramLabel.setFont(Font.font("Arial", FontWeight.BOLD, paramLabel.getFont().getSize()));
391
392        // defaultHeight field
393        Label defaultHeightLabel = new Label("Default height");
394        defaultHeightLabel.setPadding(new Insets(0, 0, 0, 4));
395        TextField defaultHeightField = new TextField("" + group.getDefaultHeight());
396        defaultHeightField.setMinWidth(100);
397
398        GridPane grid = new GridPane();
399        grid.setHgap(8);
400        grid.setVgap(10);
401        grid.add(defaultHeightLabel, 1, 1);
402        grid.add(defaultHeightField, 2, 1);
403
404        Separator separator = new Separator();
405        separator.setPadding(new Insets(10, 0, 8, 0));
406
407        // Tag field
408        Label tagLabel = new Label("Tag");
409        tagLabel.setFont(Font.font("Arial", FontWeight.BOLD, tagLabel.getFont().getSize()));
410        tagLabel.setPadding(new Insets(0, 0, 8, 0));
411        TextField tagField = new TextField();
412        tagField.setMinWidth(250);
413        tagField.setOnKeyReleased(e -> {
414            String text = tagField.getText();
415            pane.lookupButton(ButtonType.OK).setDisable(text.isEmpty() || text.indexOf(" ") != -1 || sameTagExists(null, text));
416        });
417        FlowPane tagFlowPane = new FlowPane();
418        tagFlowPane.getChildren().add(tagField);
419
420        VBox vBox = new VBox();
421        vBox.getChildren().addAll(paramLabel, grid, separator, tagLabel, tagFlowPane);
422        pane.setContent(vBox);
423
424        pane.getButtonTypes().addAll(ButtonType.OK, ButtonType.CANCEL);
425        pane.lookupButton(ButtonType.OK).setDisable(true);
426        pane.lookupButton(ButtonType.OK).addEventFilter(ActionEvent.ACTION, e -> {
427            String tag = tagField.getText();
428            Double defaultHeight = editor.getFrame().convertToDouble(defaultHeightField.getText());
429            if (defaultHeight == null) {
430                e.consume();
431                return;
432            }
433            editor.invokeSingleCommand(new AddGroup(tag, defaultHeight, 0));
434        });
435
436        dialog.showAndWait();
437    }

439    /**
440     * グループタグの重複チェック
441     */
442    public boolean sameTagExists(MapPartGroup group, String tag) {
443        for (MapPartGroup _group : editor.getMap().getGroups()) {
444            if (_group != group && _group.getTagString().equals(tag)) {
445                return true;
446            }
447        }
448        return false;
449    }

451    /**
452     * ツリー・ビューを構築する
453     */
454    public void construct() {
455        // 最初に root を登録する
456        MapPartGroup root = (MapPartGroup)editor.getMap().getRoot();
457        TreeItem<OBNode> rootItem = new TreeItem<OBNode>(root, new ImageView(groupImage));
458        rootItem.setExpanded(true);
459        setRoot(rootItem);
460        treeItemMap.put(root, rootItem);
461
462        // 残りのグループを登録する
463        for (MapPartGroup group : editor.getMap().getGroups()) {
464            if (group == root) {
465                continue;
466            }
467            String tag = String.join(", ", group.getTags());
468            TreeItem<OBNode> groupItem = new TreeItem<OBNode>(group, new ImageView(groupImage));
469            rootItem.getChildren().add(groupItem);
470            treeItemMap.put(group, groupItem);
471        }
472    }

474    /**
475     * ツリー・ビューを削除する
476     */
477    public void clear() {
478        TreeItem<OBNode> rootItem = this.getRoot();
479        if (rootItem != null) {
480            for (TreeItem<OBNode> groupItem : rootItem.getChildren()) {
481                groupItem.getChildren().clear();
482            }
483            rootItem.getChildren().clear();
484        }
485        treeItemMap.clear();
486    }
