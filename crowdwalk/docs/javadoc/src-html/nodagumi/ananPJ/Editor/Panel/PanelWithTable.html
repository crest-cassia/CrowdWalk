002package nodagumi.ananPJ.Editor.Panel;
003
004import java.awt.Component;
005
006import java.io.IOException;
007import java.io.ObjectInputStream;
008import java.io.ObjectOutputStream;
009
010import javax.swing.JPanel;
011import javax.swing.JTable;
012import javax.swing.event.ListSelectionListener;
013import javax.swing.table.TableCellRenderer;
014import javax.swing.table.TableColumn;
015import javax.swing.table.TableColumnModel;
016
017public abstract class PanelWithTable extends JPanel
018        implements ListSelectionListener {
019
020        protected void adjustColumnPreferredWidth(JTable table) {
021                TableColumnModel columnModel = table.getColumnModel();
022                for (int col = 0; col < table.getColumnCount(); ++ col) {
023                        int max_width = 0;
024                        
025                        for (int row = 0; row < table.getRowCount(); ++ row) {
026                                TableCellRenderer rend = table.getCellRenderer(row, col);
027                                Object o = table.getValueAt(row,col);
028                                Component c = rend.getTableCellRendererComponent(table, o,
029                                                false, false, row, col);
030                                max_width = Math.max(c.getPreferredSize().width, max_width);
031                        }
032                        TableColumn column = columnModel.getColumn(col);
033                        column.setPreferredWidth(max_width);
034                }
035        }
036}
