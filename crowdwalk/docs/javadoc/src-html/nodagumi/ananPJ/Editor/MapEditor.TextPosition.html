235    public boolean loadNetworkMap() {
236        String fileName = getNetworkMapFile();
237        if (fileName == null || fileName.isEmpty()) {
238            Itk.logError("Map file name is empty.");
239            return false;
240        }
241        initNetworkMap();
242
243        // シミュレーション結果の同一性を確保するため
244        // TODO: これは既に不要かもしれない
245        random.setSeed(properties.getRandseed());
246
247        Document doc = null;
248        try {
249            doc = ItkXmlUtility.singleton.streamToDoc(new FileInputStream(fileName));
250            if (doc == null) {
251                Itk.logError("Could not read map.", fileName);
252                return false;
253            }
254        } catch (IOException e) {
255            Itk.logError("Can't Open Map File", fileName);
256            return false;
257        }
258        if (doc.getChildNodes() == null) {
259            Itk.logError("Invalid inputted DOM object.", fileName);
260            return false;
261        }
262
263        if (! networkMap.fromDOM(doc)) {
264            return false;
265        }
266        // tag を持たない Group があれば適当な名前で tag を付加しておく
267        appendGroupTag();
268        // Group の zone を正規化する(networkMap.getGroups() を使用しているため appendGroupTag() の後に呼ぶこと)
269        normalizeZone();
270
271        if (! loadBackgroundImage()) {
272            return false;
273        }
274        if (! loadGsiTiles()) {
275            return false;
276        }
277
278        Itk.logInfo("Load Map File", fileName);
279
280        return true;
281    }

283    /**
284     * tag を持たない Group があれば適当な名前で tag を付加しておく
285     */
286    private void appendGroupTag() {
287        ArrayList<String> names = new ArrayList();
288        for (MapPartGroup group : networkMap.getGroups()) {
289            names.add(group.getTagString());
290        }
291
292        int n = 1;
293        MapPartGroup root = (MapPartGroup)networkMap.getRoot();
294        for (OBNode node : networkMap.getOBCollection()) {
295            if (node.getNodeType() != OBNode.NType.GROUP) {
296                continue;
297            }
298            MapPartGroup group = (MapPartGroup)node;
299            if (group.getTagString().isEmpty()) {
300                group.allTagsClear();
301                if (group == root) {
302                    group.addTag("root");
303                    normalized = true;
304                } else {
305                    String name = "GROUP" + n;
306                    while (names.contains(name)) {
307                        n++;
308                        name = "GROUP" + n;
309                    }
310                    group.addTag(name);
311                    names.add(name);
312                    normalized = true;
313                }
314            }
315        }
316    }

318    /**
319     * Group の zone を正規化する
320     */
321    private void normalizeZone() {
322        MapPartGroup root = (MapPartGroup)networkMap.getRoot();
323
324        // 異常な値があれば 0 にする
325        for (MapPartGroup group : networkMap.getGroups()) {
326            if (group.getZone() != 0 && ! GsiTile.isCorrectZone(group.getZone())) {
327                Itk.logWarn("Group " + group.getTagString(), "delete zone value: " + group.getZone());
328                group.setZone(0);
329                normalized = true;
330            }
331        }
332
333        // サブグループ側で設定されている場合は root にも設定する
334        if (root.getZone() == 0) {
335            HashSet<Integer> zones = new HashSet();
336            for (MapPartGroup group : networkMap.getGroups()) {
337                if (group.getZone() != 0) {
338                    zones.add(new Integer(group.getZone()));
339                }
340            }
341            switch (zones.size()) {
342            case 0:     // 全グループの zone が 0
343                return;
344            case 1:     // サブグループで zone が設定
345                zones.forEach(zone -> {
346                    Itk.logWarn("Group " + root.getTagString(), "set zone value: " + zone);
347                    root.setZone(zone.intValue());
348                    normalized = true;
349                });
350                break;
351            default:    // サブグループで異なる複数の zone が設定
352                break;
353            }
354        }
355
356        // サブグループの zone はすべて 0 にする
357        for (MapPartGroup group : networkMap.getGroups()) {
358            if (group != root && group.getZone() != 0) {
359                group.setZone(0);
360                normalized = true;
361            }
362        }
363    }

398        backgroundImages.clear();
399        for (MapPartGroup group : networkMap.getGroups()) {
400            String imageFileName = group.getImageFileName();
401            if (imageFileName == null || imageFileName.isEmpty()) {
402                continue;
403            }
404            File file = new File(getDir(), imageFileName);
405            String filePath = file.toURI().toString();
406            if (! file.exists()) {
407                Itk.logWarn("Background image file not exists", filePath);
408                continue;
409            }
410            Image image = new Image(filePath);
411            if (image.isError()) {
412                Itk.logError("Illegal image file", filePath);
413                return false;
414            }
415            setBackgroundImage(group, image);
416        }
417        return true;
418    }

497    /**
498     * 編集コマンドを実行する
499     */
500    public synchronized boolean invoke(EditCommandBase command) {
501        command.setTime();
502        if (isFirstCommand) {
503            command.setFirst(true);
504            initStatusOfEdit();
505            isFirstCommand = false;
506        } else {
507            command.setFirst(false);
508        }
509
510        boolean result = command.invoke(this);
511        if (command.isInvalid()) {
512            initCommandHistory();
513            displayMessage("Warn", "System information", "The map data may have been broken.");
514            lastCommand = null;
515        } else {
516            updateStatusOfEdit(command);
517            frame.setStatusText(command.toString());
518
519            // undo した分のコマンド履歴は削除する
520            if (historyIndex + 1 < commandHistory.size()) {
521                commandHistory.subList(historyIndex + 1, commandHistory.size()).clear();
522            }
523
524            EditCommandBase previousCommand = commandHistory.get(historyIndex);
525            if (previousCommand.isMergeable(command)) {
526                previousCommand.mergeTo(command);
527            } else {
528                commandHistory.add(command);
529                historyIndex++;
530            }
531            lastCommand = command;
532        }
533
534        return result;
535    }

537    /**
538     * 最後に実行した編集コマンドを取り消して実行前の状態に戻す
539     */
540    public synchronized void undo() {
541        if (historyIndex == 0) {
542            beep();
543            return;
544        }
545
546        initStatusOfEdit();
547        boolean heightChangeable = false;
548        while (true) {
549            EditCommandBase command = commandHistory.get(historyIndex);
550            command.undo(this);
551            if (command.isInvalid()) {
552                initCommandHistory();
553                displayMessage("Warn", "System information", "The map data may have been broken.");
554                break;
555            }
556            heightChangeable = heightChangeable || command.isHeightChangeable();
557            updateStatusOfEdit(command);
558            frame.setStatusText("Undo: " + command.toString());
559            historyIndex--;
560
561            if (command.isFirst()) {
562                break;
563            }
564        }
565        if (heightChangeable) {
566            updateHeight();
567        }
568
569        frame.getCanvas().clearEditingStates();
570        refresh();
571    }

573    /**
574     * 最後に undo した編集コマンドを再び実行する
575     */
576    public synchronized void redo() {
577        if (historyIndex + 1 == commandHistory.size()) {
578            beep();
579            return;
580        }
581
582        initStatusOfEdit();
583        boolean heightChangeable = false;
584        while (true) {
585            historyIndex++;
586            EditCommandBase command = commandHistory.get(historyIndex);
587            command.invoke(this);
588            if (command.isInvalid()) {
589                initCommandHistory();
590                displayMessage("Warn", "System information", "The map data may have been broken.");
591                break;
592            }
593            heightChangeable = heightChangeable || command.isHeightChangeable();
594            updateStatusOfEdit(command);
595            frame.setStatusText("Redo: " + command.toString());
596
597            if (command.isLast()) {
598                break;
599            }
600        }
601        if (heightChangeable) {
602            updateHeight();
603        }
604
605        frame.getCanvas().clearEditingStates();
606        refresh();
607    }

685    /**
686     * 表示を更新する
687     */
688    public void refresh() {
689        boolean scenarioPanelUpdated = false;
690
691        frame.getCanvas().repaintLater();
692
693        if (groupVolumeChanged) {
694            // root 以外にグループがあればそのグループに移る
695            if (networkMap.getGroups().size() == 2) {
696                initCurrentGroup();
697            }
698
699            // グループボタンを再構築する
700            frame.updateShowBackgroundGroupMenu();
701            frame.updateGroupSelectionPane();
702
703            // シンボリックリンク追加サブメニューを再構築する
704            frame.updateAddSymbolicLinkMenu();
705
706            // マップツリーを再構築する
707            frame.getGroupPanel().clear();
708            frame.getGroupPanel().construct();
709        } else if (groupTagChanged) {
710            frame.getGroupPanel().refresh();
711
712            // グループタグ表示を更新する
713            for (MapPartGroup group : networkMap.getGroups()) {
714                frame.updateGroupTag(group);
715            }
716
717            // シンボリックリンク追加サブメニューを再構築する
718            frame.updateAddSymbolicLinkMenu();
719        }
720
721        if (nodeTagChanged || nodeVolumeChanged) {
722            frame.getCanvas().updateNodePoints();
723            // フィルタ処理があるため nodeTagChanged でも reset() が必要
724            frame.getNodePanel().reset();
725            frame.getScenarioPanel().reset();
726            scenarioPanelUpdated = true;
727        } else if (nodeParamChanged) {
728            frame.getCanvas().updateNodePoints();
729            frame.getNodePanel().refresh();
730        }
731
732        if (linkTagChanged || linkVolumeChanged) {
733            // フィルタ処理があるため linkTagChanged でも reset() が必要
734            frame.getLinkPanel().reset();
735            if (! scenarioPanelUpdated) {
736                frame.getScenarioPanel().reset();
737            }
738        } else if (linkParamChanged) {
739            frame.getLinkPanel().refresh();
740        }
741
742        if (areaTagChanged || areaVolumeChanged) {
743            // フィルタ処理があるため areaTagChanged でも reset() が必要
744            frame.getAreaPanel().reset();
745        } else if (areaParamChanged) {
746            frame.getAreaPanel().refresh();
747        }
748
749        if (polygonTagChanged || polygonVolumeChanged) {
750            // フィルタ処理があるため polygonTagChanged でも reset() が必要
751            frame.getPolygonPanel().reset();
752        } else if (polygonParamChanged) {
753            frame.getPolygonPanel().refresh();
754        }
755    }

1266    /**
1267     * GeoJSON ファイルを読み込んでマップデータに追加する
1268     */
1269    public void readGeoJSON(String filePath, String srcEpsg, int scaleOfRoundOff, String lengthName, boolean useGeodeticLength, String widthName, Double correctionFactor, ArrayList<Double> referenceTable, MapPartGroup group, int zone) throws Exception {
1270        CoordinateTransform transform = GsiTile.createCoordinateTransform(srcEpsg, GsiTile.JGD2000_JPR_EPSG_NAMES[zone]);
1271        JsonicHashMapGetter jsonMap = new JsonicHashMapGetter();
1272
1273        FileInputStream is = new FileInputStream(filePath);
1274        jsonMap.setParameters(JSON.decode(is));
1275        is.close();
1276
1277        ArrayList<HashMap> features = (ArrayList<HashMap>)jsonMap.getArrayListParameter("features", null);
1278        if (features == null) {
1279            throw new Exception("File parsing error - \"" + filePath + "\" : \"features\" elements are missing.");
1280        }
1281
1282        startOfCommandBlock();
1283        HashMap<String, MapNode> nodes = new HashMap<>();
1284        HashMap<String, MapLink> links = new HashMap<>();
1285        for (HashMap feature : features) {
1286            jsonMap.setParameters(feature);
1287            HashMap _properties = jsonMap.getHashMapParameter("properties", null);
1288            if (_properties == null) {
1289                Itk.logWarn_("File parsing error", "\"" + filePath + "\" : \"properties\" elements are missing.");
1290                continue;
1291            }
1292            HashMap geometry = jsonMap.getHashMapParameter("geometry", null);
1293            if (geometry == null) {
1294                Itk.logWarn_("File parsing error", "\"" + filePath + "\" : \"geometry\" elements are missing.");
1295                continue;
1296            }
1297
1298            jsonMap.setParameters(_properties);
1299            double lineLength = jsonMap.getDoubleParameter(lengthName, 0.0);
1300            double width = jsonMap.getDoubleParameter(widthName, 0.0);
1301            if (referenceTable != null && ! referenceTable.isEmpty()) {
1302                width = referenceTable.get((int)width);
1303            } else if (correctionFactor != null) {
1304                width *= correctionFactor.doubleValue();
1305            }
1306
1307            jsonMap.setParameters(geometry);
1308            ArrayList<ArrayList<BigDecimal>> coordinates = (ArrayList<ArrayList<BigDecimal>>)jsonMap.getArrayListParameter("coordinates", null);
1309            if (coordinates == null) {
1310                Itk.logWarn_("File parsing error", "\"" + filePath + "\" : \"coordinates\" elements are missing.");
1311                continue;
1312            }
1313
1314            double[] ratio = new double[coordinates.size() - 1];
1315            int ratioIndex = 0;
1316            if (useGeodeticLength) {
1317                java.awt.geom.Point2D lastPoint = null;
1318                double totalDistance = 0.0;
1319                for (ArrayList<BigDecimal> coordinate : coordinates) {
1320                    if (coordinate.size() < 2) {
1321                        endOfCommandBlock();
1322                        updateHeight();
1323                        throw new Exception("File parsing error - \"" + filePath + "\" : invalid coordinates: " + coordinate);
1324                    }
1325                    double x = coordinate.get(0).doubleValue();
1326                    double y = coordinate.get(1).doubleValue();
1327                    java.awt.geom.Point2D point = new java.awt.geom.Point2D.Double(x, y);
1328
1329                    if (lastPoint != null) {
1330                        double d = point.distance(lastPoint);
1331                        ratio[ratioIndex] = d;
1332                        totalDistance += d;
1333                        ratioIndex++;
1334                    }
1335                    lastPoint = point;
1336                }
1337                for (int index = 0; index < ratio.length; index++) {
1338                    ratio[index] /= totalDistance;
1339                }
1340            }
1341
1342            ratioIndex = 0;
1343            MapNode from = null;
1344            for (ArrayList<BigDecimal> coordinate : coordinates) {
1345                if (coordinate.size() < 2) {
1346                    endOfCommandBlock();
1347                    updateHeight();
1348                    throw new Exception("File parsing error - \"" + filePath + "\" : invalid coordinates: " + coordinate);
1349                }
1350                double x = roundValue(coordinate.get(0).doubleValue(), scaleOfRoundOff);
1351                double y = roundValue(coordinate.get(1).doubleValue(), scaleOfRoundOff);
1352                double height = 0.0;
1353                // 経緯度を CrowdWalk 座標に変換する
1354                java.awt.geom.Point2D jpr = GsiTile.transformCoordinate(transform, x, y);
1355                Point2D point = new Point2D(roundValue(jpr.getX(), 4), roundValue(-jpr.getY(), 4));
1356                String pointStr = point.toString().replaceAll(" ", "");
1357
1358                MapNode node = null;
1359                if (nodes.containsKey(pointStr)) {
1360                    node = nodes.get(pointStr);
1361                } else {
1362                    AddNode command = new AddNode(group, point, height);
1363                    if (! invoke(command)) {
1364                        endOfCommandBlock();
1365                        updateHeight();
1366                        throw new Exception("Edit command error: " + command.toString());
1367                    }
1368                    node = (MapNode)networkMap.getObject(command.getId());
1369                    nodes.put(pointStr, node);
1370                }
1371
1372                if (from == node) {
1373                    Itk.logWarn_("readGeoJSON", "from === node");
1374                    ratioIndex++;
1375                } else if (from != null) {
1376                    String nodeIdPair = makeNodeIdPair(from, node);
1377                    if (links.containsKey(nodeIdPair)) {
1378                        Itk.logWarn_("readGeoJSON", "Duplicate link found at node[" + nodeIdPair + "]. Ignore this.");
1379                    } else {
1380                        double length = 0.0;
1381                        if (useGeodeticLength) {
1382                            length = roundValue(lineLength * ratio[ratioIndex], 4);
1383                        } else {
1384                            length = from.getPosition().distance(node.getPosition());
1385                        }
1386                        AddLink command = new AddLink(from, node, length, width);
1387                        if (! invoke(command)) {
1388                            endOfCommandBlock();
1389                            updateHeight();
1390                            throw new Exception("Edit command error: " + command.toString());
1391                        }
1392                        MapLink link = (MapLink)networkMap.getObject(command.getId());
1393                        links.put(nodeIdPair, link);
1394                    }
1395                    ratioIndex++;
1396                }
1397                from = node;
1398            }
1399        }
1400
1401        // root グループの zone にセットする
1402        MapPartGroup root = (MapPartGroup)networkMap.getRoot();
1403        if (root.getZone() != zone) {
1404            if (! invoke(new SetZone(root, zone))) {
1405                ;   // do nothing
1406            }
1407        }
1408
1409        endOfCommandBlock();
1410        updateHeight();
1411    }

1538     * フロアを複製する
1539     */
1540    public void duplicateFloor(MapPartGroup orgGroup, int direction, int numberOfFloors, double heightDiff) throws Exception {
1541        if (orgGroup == networkMap.getRoot()) {
1542            throw new Exception("Root group can not be duplicated.");
1543        }
1544
1545        Matcher match = orgGroup.matchTag("^(B?)(\\d+)F$");
1546        if (match == null) {
1547            throw new Exception("No floor number given for this group.");
1548        }
1549        String floorName = match.group(0);
1550        int floor = Integer.parseInt(match.group(2)) * (match.group(1).isEmpty() ? 1 : -1);
1551
1552        int nextFloor = floor + direction;
1553        if (nextFloor == 0) {
1554            nextFloor += direction;
1555        }
1556        String nextFloorName = ((nextFloor < 0) ? "B" : "") + Math.abs(nextFloor) + "F";
1557
1558        // next floor が衝突しないかのチェック
1559        for (MapPartGroup group : networkMap.getGroups()) {
1560            if (group.getTagString().equals(nextFloorName)) {
1561                throw new Exception("The generated floor names conflict: " + nextFloorName);
1562            }
1563        }
1564
1565        startOfCommandBlock();
1566        for (int n = 1; n <= numberOfFloors; n++) {
1567            // グループの複製
1568            double defaultHeight = orgGroup.getDefaultHeight() + heightDiff * direction * n;
1569            AddGroup _command = new AddGroup(nextFloorName, defaultHeight, orgGroup.getZone());
1570            if (! invoke(_command)) {
1571                break;
1572            }
1573            MapPartGroup newGroup = (MapPartGroup)networkMap.getObject(_command.getId());
1574            if (! invoke(new SetScale(newGroup, orgGroup.getScale()))) {
1575                break;
1576            }
1577
1578            // ノードの複製
1579            HashMap<MapNode, MapNode> nodeToNode = new HashMap<MapNode, MapNode>();
1580            boolean failed = false;
1581            for (MapNode node : orgGroup.getChildNodes()) {
1582                double height = node.getHeight() - orgGroup.getDefaultHeight() + defaultHeight;
1583                AddNode command = new AddNode(newGroup, new Point2D(node.getX(), node.getY()), height);
1584                if (! invoke(command)) {
1585                    failed = true;
1586                    break;
1587                }
1588                MapNode newNode = (MapNode)networkMap.getObject(command.getId());
1589                for (String tag : node.getTags()) {
1590                    if (! invoke(new AddTag(newNode, tag.replaceFirst(floorName, nextFloorName)))) {
1591                        failed = true;
1592                        break;
1593                    }
1594                }
1595                if (failed) {
1596                    break;
1597                }
1598                nodeToNode.put(node, newNode);
1599            }
1600            updateHeight();
1601            if (failed) {
1602                break;
1603            }
1604
1605            // リンクの複製
1606            for (MapLink link : orgGroup.getChildLinks()) {
1607                MapNode fromNode = nodeToNode.get(link.getFrom());
1608                MapNode toNode = nodeToNode.get(link.getTo());
1609                if (fromNode != null && toNode != null) {
1610                    AddLink command = new AddLink(fromNode, toNode, link.getLength(), link.getWidth());
1611                    if (! invoke(command)) {
1612                        endOfCommandBlock();
1613                        return;
1614                    }
1615                    MapLink newLink = (MapLink)networkMap.getObject(command.getId());
1616                    for (String tag : link.getTags()) {
1617                        if (! invoke(new AddTag(newLink, tag.replaceFirst(floorName, nextFloorName)))) {
1618                            endOfCommandBlock();
1619                            return;
1620                        }
1621                    }
1622                }
1623            }
1624
1625            // TODO: 階段の複製
1626
1627            nextFloor += direction;
1628            if (nextFloor == 0) {
1629                nextFloor += direction;
1630            }
1631            nextFloorName = ((nextFloor < 0) ? "B" : "") + Math.abs(nextFloor) + "F";
1632            for (MapPartGroup group : networkMap.getGroups()) {
1633                if (group.getTagString().equals(nextFloorName)) {
1634                    endOfCommandBlock();
1635                    return;
1636                }
1637            }
1638        }
1639        endOfCommandBlock();

1848    /**
1849     * マップ座標の正規化
1850     */
1851    public void normalizeCoordinates(MapNode node1, MapNode node2, int zone, double node1Longitude, double node1Latitude, double node2Longitude, double node2Latitude, String scaleName, boolean rotationEnabled, boolean lengthCalculating) {
1852        // 経緯度を CrowdWalk 座標に変換する
1853        CoordinateTransform transform = GsiTile.createCoordinateTransform("EPSG:4326", GsiTile.JGD2000_JPR_EPSG_NAMES[zone]);
1854        java.awt.geom.Point2D jpr1 = GsiTile.transformCoordinate(transform, node1Longitude, node1Latitude);
1855        java.awt.geom.Point2D jpr2 = GsiTile.transformCoordinate(transform, node2Longitude, node2Latitude);
1856        Point2D point1 = toFxPoint2D(GsiTile.convertJPR2CW(jpr1.getY(), jpr1.getX()));
1857        Point2D point2 = toFxPoint2D(GsiTile.convertJPR2CW(jpr2.getY(), jpr2.getX()));
1858
1859        Point2D nodePoint1 = toFxPoint2D(node1.getPosition());
1860        Point2D nodePoint2 = toFxPoint2D(node2.getPosition());
1861
1862        double angularDifference = 0.0;
1863        if (rotationEnabled) {
1864            double angle = getAngle(nodePoint1, nodePoint2);
1865            double correctAngle = getAngle(point1, point2);
1866            angularDifference = correctAngle - angle;
1867            Affine affine = new Affine();
1868            affine.prependRotation(angularDifference, nodePoint1);
1869            nodePoint2 = affine.transform(nodePoint2);
1870            Itk.logInfo("Normalize coordinates", "angular_difference = " + angularDifference);
1871        }
1872
1873        Affine affine = new Affine();
1874        affine.prependTranslation(-nodePoint1.getX(), -nodePoint1.getY());
1875        if (rotationEnabled) {
1876            affine.prependRotation(angularDifference);
1877        }
1878
1879        double scaleX = (point2.getX() - point1.getX()) / (nodePoint2.getX() - nodePoint1.getX());
1880        double scaleY = (point2.getY() - point1.getY()) / (nodePoint2.getY() - nodePoint1.getY());
1881        Itk.logInfo("Normalize coordinates", "scaleX = " + scaleX + ", scaleY = " + scaleY);
1882        switch (scaleName) {
1883        case "separately":
1884            affine.prependScale(scaleX, scaleY);
1885            Itk.logInfo("Normalize coordinates", "applied scale = " + scaleName + ", value = " + scaleX + ", " + scaleY);
1886            break;
1887        case "average":
1888            double scale = (scaleX + scaleY) / 2.0;
1889            affine.prependScale(scale, scale);
1890            Itk.logInfo("Normalize coordinates", "applied scale = " + scaleName + ", value = " + scale);
1891            break;
1892        case "X":
1893            affine.prependScale(scaleX, scaleX);
1894            Itk.logInfo("Normalize coordinates", "applied scale = " + scaleName + ", value = " + scaleX);
1895            break;
1896        case "Y":
1897            affine.prependScale(scaleY, scaleY);
1898            Itk.logInfo("Normalize coordinates", "applied scale = " + scaleName + ", value = " + scaleY);
1899            break;
1900        }
1901
1902        affine.prependTranslation(point1.getX(), point1.getY());
1903
1904        startOfCommandBlock();
1905        for (MapNode node : networkMap.getNodes()) {
1906            Point2D point = affine.transform(node.getX(), node.getY());
1907            if (! invoke(new MoveNode(node, point.getX(), point.getY()))) {
1908                endOfCommandBlock();
1909                return;
1910            }
1911        }
1912        if (lengthCalculating) {
1913            for (MapLink link : networkMap.getLinks()) {
1914                if (! invoke(new SetLength(link, calcLinkLength(link)))) {
1915                    endOfCommandBlock();
1916                    return;
1917                }
1918            }
1919        }
1920        for (MapPartGroup group : networkMap.getGroups()) {
1921            if (group.getZone() != zone) {
1922                if (! invoke(new SetZone(group, zone))) {
1923                    break;
1924                }
1925            }
1926            if (group.getScale() != 1.0) {
1927                if (! invoke(new SetScale(group, 1.0))) {
1928                    break;
1929                }
1930            }
1931        }
1932        endOfCommandBlock();
1933    }

2216        }
2217
2218        // links の中からリング状に連なったリンクを探す(一部でもあれば対象)
2219        ArrayList<MapLink> _links = (ArrayList<MapLink>)links.clone();
2220        ArrayList<ArrayList<MapLink>> boundaries = new ArrayList();
2221        while (! _links.isEmpty()) {
2222            ArrayList<MapLink> boundary = getRingedLinks(_links.remove(0));
2223            if (boundary == null) {
2224                throw new Exception("Unclosed links found.");
2225            }

2227            for (MapLink link : boundary) {
2228                _links.remove(link);
2229            }
2230        }
2231
2232        // 標高値とタグに不整合がないかチェックする
2233        double height = boundaries.get(0).get(0).getFrom().getHeight();

2235        String tagString = null;
2236        HashMap<ArrayList, Path2D> paths = new HashMap();
2237        for (ArrayList<MapLink> boundary : boundaries) {
2238            for (MapLink link : boundary) {
2239                if (link.getFrom().getHeight() != height || link.getTo().getHeight() != height) {
2240                    throw new Exception("Height is not constant.");
2241                }
2242                if (! link.getTags().isEmpty()) {
2243                    if (tagString == null) {
2244                        tags = (ArrayList<String>)link.getTags().clone();
2245                        Collections.sort(tags);
2246                        tagString = String.join(",", tags);
2247                    } else {
2248                        ArrayList<String> linkTags = (ArrayList<String>)link.getTags().clone();
2249                        Collections.sort(linkTags);
2250                        if (! String.join(",", linkTags).equals(tagString)) {
2251                            throw new Exception("Link tag is not constant.");
2252                        }

2255            }

2260        ArrayList<MapLink> _outerBoundary = boundaries.get(0);

2272                }
2273            }

2283                }
2284            }
2285        }

2289            for (MapNode node : getBoundaryNodes(boundary)) {
2290                for (ArrayList<MapLink> _boundary : boundaries) {
2291                    if (_boundary == boundary) {
2292                        continue;
2293                    }
2294                    Path2D path = paths.get(_boundary);
2295                    if (path.contains(node.getX(), node.getY())) {
2296                        throw new Exception("Inner boundaries overlap.");
2297                    }
2298                }
2299            }
2300        }
2301
2302        ArrayList<java.awt.geom.Point2D> coordinates = new ArrayList();
2303        for (MapNode node : outerNodes) {

2308        ArrayList<InnerBoundary> innerBoundaries = new ArrayList();

2313            }
2314            innerBoundaries.add(new InnerBoundary(coordinates));
2315        }
2316

2318        if (tags != null) {
2319            for (String tag : (List<String>)tags.clone()) {
2320                if (tag.startsWith("Z_INDEX_")) {
2321                    zIndex = Integer.parseInt(tag.substring(8));
2322                    tags.remove(tag);
2323                    break;
2324                }
2325            }
2326        }
2327
2328        startOfCommandBlock();
2329        AddPolygon command = new AddPolygon(currentGroup, zIndex, outerBoundary, innerBoundaries);
2330        if (invoke(command)) {
2331            if (tags != null) {
2332                MapPolygon polygon = (MapPolygon)networkMap.getObject(command.getId());
2333                for (String tag : tags) {
2334                    if (! invoke(new AddTag(polygon, tag))) {
2335                        endOfCommandBlock();
2336                        return;
2337                    }
2338                }
2339            }
2340
2341            // ポリゴンの元になったノードを削除する(連動してリンクも削除される)
2342            frame.getNodePanel().clearSelection();
2343            frame.getLinkPanel().clearSelection();
2344            for (MapNode node : getBoundaryNodes(_outerBoundary)) {
2345                node.selected = true;

2347            for (ArrayList<MapLink> boundary : boundaries) {
2348                for (MapNode node : getBoundaryNodes(boundary)) {

2355    }
