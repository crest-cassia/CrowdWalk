101    public void zoom(int z) {
102        double scaleOld = scale;
103
104        double r;
105        if (z > 0) {
106            r = 0.8;
107        } else {
108            r = 1.25;
109        }
110
111        z = Math.abs(z);
112        for (int i = 0; i < z; i++) {
113            scale *= r;
114        }
115        
116        if (scale > 100.0) scale = 100.0;
117        if (scale < 0.01) scale = 0.01;
118
119        //TODO: needs calculation for rotation
120        tx += frame.mousePoint.getX() * (scaleOld - scale);
121        ty += frame.mousePoint.getY() * (scaleOld - scale);
122    }

172    public void centering(boolean withScaling) {
173        double minX = Double.MAX_VALUE, minY = Double.MAX_VALUE;
174        // ※Double.MIN_VALUE が何故か正の値と判定されることがあるため Long.MIN_VALUE で代用した
175        double maxX = (double)Long.MIN_VALUE, maxY = (double)Long.MIN_VALUE;
176        for (MapNode node : frame.getChildNodes()) {
177            minX = Math.min(minX, node.getX());
178            maxX = Math.max(maxX, node.getX());
179            minY = Math.min(minY, node.getY());
180            maxY = Math.max(maxY, node.getY());
181        }
182        if (minX == Double.MAX_VALUE) {
183            tx = 0.0;
184            ty = 0.0;
185            scale = 1.0;
186            return;
187        }
188
189        final double width = maxX - minX;
190        final double height = maxY - minY;
191        final double scaleX = (getWidth() - 20) / width;
192        final double scaleY = (getHeight() - 60) / height;
193        
194        if (withScaling) {
195            scale = Math.min(scaleX, scaleY);
196        }
197        tx = -(minX + maxX) / (2) * scale + (getWidth() - 20) / 2;
198        ty = -(minY + maxY) / (2) * scale + (getHeight() - 40) / 2;
199    }

201    /* theoretical value to display */
202    public Point2D calcPos(Point2D p) {
203        /* translate local position  < reference parent position */ 
204        double xx = p.getX() * scale + tx;
205        double yy = p.getY() * scale + ty;
206        return new Point2D.Double(xx, yy);
207    }

209    /* display to theoretical value */
210    /* this method is called by zoom relate function */  
211    public Point2D revCalcPos(int x, int y) {
212        point_on_panel = SwingUtilities.convertPoint(null, x, y, this);
213        
214        AffineTransform trans = new AffineTransform();
215        trans.translate(tx, ty);
216        trans.scale(scale, scale);
217
218        try {
219            trans.invert();
220        } catch (NoninvertibleTransformException e) {
221            e.printStackTrace();
222        }
223        return trans.transform(new Point2D.Double(point_on_panel.getX(), point_on_panel.getY()), null);
224    }

226    @Override
227    public void paintComponent (Graphics g0) {
228        super.paintComponent(g0);
229        
230        // viewArea を使って表示範囲外の無駄な描画処理を省く
231        Dimension frameSize = new Dimension();
232        frame.getSize(frameSize);
233        Point2D position = revCalcPos(0, 0);
234        Point2D size = revCalcPos(frameSize.width, frameSize.height);
235        Rectangle2D viewArea = new Rectangle2D.Double(position.getX(), position.getY(), (size.getX() - position.getX()) + 1, (size.getY() - position.getY()) + 1);
236
237        Graphics2D g = (Graphics2D)g0;
238
239        if (setScaleMode) {
240            //g.drawString("Scale Mode", 0, 20);
241            frame.setStatus("SET_SCALE");
242        } else {
243            if (updateStatusEnabled) {
244                frame.setStatus();
245            }
246        }
247
248        g.translate(tx, ty);
249        g.scale(scale, scale);
250
251        AffineTransform t = g.getTransform();
252
253        if (backgroundImage != null) {
254            g.translate(ob_node.tx, ob_node.ty);
255            g.scale(ob_node.sx, ob_node.sy);
256            g.rotate(ob_node.r);
257
258            g.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, (float)imageStrength));
259            g.drawImage(backgroundImage, 0, 0,
260                    (int)(backgroundImage.getWidth(null)),
261                    (int)(backgroundImage.getHeight(null)),
262                    this);
263            g.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 1.0f));
264            //System.out.println("tx:" + tx + ", ty:" + ty + ", obnode tx:" + ob_node.tx + ", ty:" + ob_node.ty + ", sx:" + ob_node.sx + ", sy:" + ob_node.sy + ", r:" + ob_node.r + ", scale:" + scale);
265
266            g.setTransform(t);
267        }
268
269        /* show frame */
270        if (frame.getRegion() != null){
271            g.setStroke(MapLink.narrow);
272            g.setColor(Color.ORANGE);
273            //g.draw(frame.getRegion());
274            g.draw(new Rectangle2D.Double(
275                    frame.getRegion().getX(),
276                    frame.getRegion().getY(),
277                    frame.getBounds().getWidth()-40,
278                    frame.getBounds().getHeight()-190));
279        }
280        /* background group */
281        if (backgroundGroup != null) {
282            if (showLinks) {
283                for (final MapLink link : backgroundGroup.getChildLinks()) {
284                    if (viewArea.intersectsLine(link.getLine2D())) {
285                        g.setColor(Color.WHITE);
286                        link.drawInEditor(g, showLinkLabels, false, showScaling);
287                    }
288                }
289            }
290            if (showNodes) {
291                g.setStroke(new BasicStroke(1.0f));
292                if (backgroundGroup.getChildNodes() != null) {
293                    for (MapNode node : backgroundGroup.getChildNodes()) {
294                        if (viewArea.contains(node.getX(), node.getY())) {
295                            node.drawInEditor(g, false, false, Color.LIGHT_GRAY);
296                        }
297                    }
298                }
299            }
300        }
301        
302        /* actual objects */
303        if (showArea) {
304            for (final MapArea area : frame.getChildMapAreas()) {
305                // tkokada
306                // apply a rotation rectangle, here.
307                double angle = area.getAngle();
308                if (!Double.isNaN(angle)) {
309                    AffineTransform af = g.getTransform();
310                    af.rotate(angle, ((Rectangle2D) area.getShape()).getCenterX(), ((Rectangle2D) area.getShape()).getCenterY());
311                    // draw center point of map area
312                    /*
313                    g.setColor(Color.GREEN);
314                    g.fill(new Rectangle2D.Double(
315                            ((Rectangle2D) area.getShape()).getCenterX() - 6 /g.getTransform().getScaleX(),
316                            ((Rectangle2D) area.getShape()).getCenterY() - 6 / g.getTransform().getScaleX(),
317                            6 * 2 / g.getTransform().getScaleX(),
318                            6 * 2 / g.getTransform().getScaleX()));
319                    */
320                    //System.out.println("area cx:" + ((Rectangle2D) area.getShape()).getCenterX() + "cy:" + ((Rectangle2D) area.getShape()).getCenterY());
321                    g.setTransform(af);
322                    //System.out.println("area cx:" + ((Rectangle2D) area.getShape()).getCenterX() + "cy:" + ((Rectangle2D) area.getShape()).getCenterY());
323                    //System.out.println("tx:" + tx + ", ty:" + ty + ", obnode tx:" + ob_node.tx + ", ty:" + ob_node.ty + ", sx:" + ob_node.sx + ", sy:" + ob_node.ty
324                    //      + ", r:" + ob_node.r + ", scaleX:" + g.getTransform().getScaleX());
325                    area.drawInEditor(g) ;
326                    g.setTransform(t);
327                } else {
328                    area.drawInEditor(g) ;
329                }
330            }
331        }
332
333        if (showLinks) {
334            for (final MapLink link : frame.getChildLinks()) {
335                if (viewArea.intersectsLine(link.getLine2D())) {
336                    g.setColor(linkColor);
337                    link.drawInEditor(g, showLinkLabels, false, showScaling);
338                }
339            }
340        }
341        
342        if (showNodes) {
343            g.setStroke(new BasicStroke(1.0f));
344            if (frame.getChildNodes() != null) {
345                for (MapNode node : frame.getChildNodes()) {
346                    if (viewArea.contains(node.getX(), node.getY())) {
347                        node.drawInEditor(g, showNodeLabels, false);
348                    }
349                }
350            }
351        }
352        
353        if (showGroups) {
354            for (MapPartGroup group : frame.getChildGroups()) {
355                group.drawInEditor(g, showSubGroups);
356            }
357        }
358        
359        /* symbolic links */
360        /* first try links */
361        if (showLinks) {
362            for (OBNodeSymbolicLink symlink : frame.getSymbolicLinks()) {
363                OBNode orig = symlink.getOriginal();
364                if (orig == null) {
365                    System.err.println("Dangling symlink? " + symlink.ID);
366                }
367                OBNode.NType ntype = orig.getNodeType();
368            
369                if (ntype == OBNode.NType.LINK) {
370                    MapLink link = (MapLink)orig;
371                    link.drawInEditor(g, showLinkLabels, true, showScaling);
372                }
373            }
374        }
375        /* then nodes links */
376        if (showNodes) {
377            for (OBNodeSymbolicLink symlink : frame.getSymbolicLinks()) {
378                OBNode orig = symlink.getOriginal();
379                OBNode.NType ntype = orig.getNodeType();
380            
381                if (ntype == OBNode.NType.NODE) {
382                    MapNode node = (MapNode)orig;
383                    node.drawInEditor(g, showNodeLabels, true);
384                }
385            }
386        }
387
388        /* temporary objects */
389        if (hoverNode != null) {
390            if (hoverNode.orig_node != null) {
391                hoverNode.orig_node.drawInEditor(g, true, true);
392            }
393            g.setColor(Color.MAGENTA);
394            final double scale = g.getTransform().getScaleX();
395            Rectangle2D rect = new Rectangle2D.Double(hoverNode.getX() - 6 / scale,
396                    hoverNode.getY() - 6 / scale,
397                    12 / scale,
398                    12 / scale);
399            g.fill(rect);
400            g.setColor(Color.CYAN);
401            rect = new Rectangle2D.Double(hoverNode.getX() - 5 / scale,
402                    hoverNode.getY() - 5 / scale,
403                    10 / scale,
404                    10 / scale);
405            g.fill(rect);
406            if (frame.getChildNodes().contains(hoverNode)) {
407                showHoverNodeInfo(g);
408            }
409        }
410        
411        if (hoverLink != null) {
412            //g.setStroke(MapLink.narrow);
413            g.setColor(Color.BLUE);
414            //g.draw(hoverLink.getRect(g));
415            g.fill(hoverLink.getRect(g, showScaling));
416            // ラベルの描画
417            if (hoverLink.orig_link != null) {
418                hoverLink.orig_link.drawLabel(g, showScaling);
419            }
420        }
421        
422        if (hoverArea != null) {
423            g.setColor(Color.BLUE);
424            // tkokada
425            // apply the rotated rectangles to show.
426            boolean drawSelectedArea = false;
427            for (MapArea area : frame.getChildMapAreas()) {
428                if (((Rectangle2D) area.getShape()).contains((Rectangle2D) hoverArea.getShape())) {
429                    double angle = area.getAngle();
430                    if (!Double.isNaN(angle)) {
431                        AffineTransform af = g.getTransform();
432                        af.rotate(angle, ((Rectangle2D) area.getShape()).getCenterX(), ((Rectangle2D) area.getShape()).getCenterY());
433                        g.setTransform(af);
434                        g.draw(hoverArea.getShape());
435                        g.setTransform(t); 
436                        drawSelectedArea = true;
437                        break;
438                    }
439                }
440            }
441            if (!drawSelectedArea) {
442                g.draw(hoverArea.getShape());
443            }
444        }
445        // draw selected MapArea by using EDIT_AREA multi selection
446        for (MapArea area: frame.getChildMapAreas()) {
447            if (area.selected) {
448                double angle = area.getAngle();
449                if (!Double.isNaN(angle)) {
450                    AffineTransform af = g.getTransform();
451                    af.rotate(angle, ((Rectangle2D) area.getShape()).getCenterX(), ((Rectangle2D) area.getShape()).getCenterY());
452                    g.setTransform(af);
453                    g.setColor(Color.BLUE);
454                    g.draw(area.getShape());
455                    g.setTransform(t); 
456                }
457            }
458        }
459
460        if (selectedArea != null) {
461            g.setColor(Color.BLUE);
462            g.draw(selectedArea);
463            
464        }
465        
466        if (scaleLine != null) {
467            g.setStroke(MapLink.broad);
468            g.setColor(Color.WHITE);
469            g.draw(scaleLine);
470            g.setStroke(MapLink.narrow);
471            g.setColor(Color.BLACK);
472            g.draw(scaleLine);
473        }
474        
475        if (tempLine != null) {
476            g.setStroke(MapLink.broad);
477            g.setColor(Color.YELLOW);
478            g.draw(tempLine);
479            g.setStroke(MapLink.narrow);
480            g.setColor(Color.BLACK);
481            g.draw(tempLine);
482        }       
483
484        if (showOneWayIndicator) {
485            g.setColor(Color.RED);
486            drawScaleFixedText(g, oneWayfirstNode.getX(), oneWayfirstNode.getY(),
487                    "A", oneWayLabelPositionA, Font.BOLD, 18);
488            drawScaleFixedText(g, oneWaylastNode.getX(), oneWaylastNode.getY(),
489                    "B", oneWayLabelPositionB, Font.BOLD, 18);
490        }
491    }

504    private void showHoverNodeInfo(Graphics2D g) {
505        g.setColor(Color.MAGENTA);
506        g.setFont(new Font("Roman", Font.ITALIC, (int) (24 / scale)));
507        g.drawString(hoverNode.getTagString(),
508                (int)hoverNode.getX(), (int)hoverNode.getY());
509
510        for (MapLink path : hoverNode.getUsableLinkTable()) {
511            MapNode otherNode = path.getOther(hoverNode.getDummyHoverNode());
512            HashMap<String, NavigationHint> hints
513                = otherNode.getHints(NavigationHint.DefaultMentalMode);
514            float dy = 0;
515            for (String key : hints.keySet()) {
516                dy += 10 / scale;
517                NavigationHint hint = hints.get(key);
518                g.drawString(key + " " + hint.distance,
519                             (float)otherNode.getX(),
520                             (float)otherNode.getY() + dy);
521            }
522        }
523    }

525    /**
526     * 指定された位置、フォントスタイル、フォントサイズで text を描画する。
527     * フォントサイズは描画スケールの影響を受けない。
528     */
529    public void drawScaleFixedText(Graphics2D g, double x, double y,
530            String text, TextPosition position, int fontStyle, int fontSize) {
531        double scale = g.getTransform().getScaleX();
532        Font font = new Font("SansSerif", fontStyle, (int)(fontSize / scale));
533        g.setFont(font);
534        FontMetrics fm = g.getFontMetrics();
535        int textWidth = fm.stringWidth(text);
536        int textHeight = fm.getHeight();
537
538        double dx = 0.0;
539        double dy = 0.0;
540        switch (position) {
541        case CENTER:
542            dx = textWidth / -2.0;
543            dy = textHeight / 5.0;
544            break;
545        case UPPER:
546            dx = textWidth / -2.0;
547            dy = textHeight / -3.0;
548            break;
549        case LOWER:
550            dx = textWidth / -2.0;
551            dy = textHeight / 1.3;
552            break;
553        case LEFT:
554            dx = -textWidth - textHeight / 3.0;
555            dy = textHeight / 5.0;
556            break;
557        case RIGHT:
558            dx = textHeight / 3.0;
559            dy = textHeight / 5.0;
560            break;
561        }
562
563        g.drawString(text, (float)(x + dx), (float)(y + dy));
564    }
