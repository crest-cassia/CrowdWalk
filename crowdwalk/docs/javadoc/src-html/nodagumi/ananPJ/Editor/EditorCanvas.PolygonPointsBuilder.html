057        private ArrayList<Point2D> points = new ArrayList();
058        private double[] xPoints = null;
059        private double[] yPoints = null;
060
061        public PolygonPointsBuilder() {}
062
063        public void append(double x, double y) {
064            points.add(new Point2D(x, y));
065            xPoints = null;
066            yPoints = null;
067        }
068
069        public double[] getXPoints() {
070            if (xPoints == null) {
071                xPoints = new double[points.size()];
072                int index = 0;
073                for (Point2D point : points) {
074                    xPoints[index] = point.getX();
075                    index++;
076                }
077            }
078            return xPoints;
079        }
080
081        public double[] getYPoints() {
082            if (yPoints == null) {
083                yPoints = new double[points.size()];
084                int index = 0;
085                for (Point2D point : points) {
086                    yPoints[index] = point.getY();
087                    index++;
088                }
089            }
090            return yPoints;
091        }
092
093        public int size() {
094            return points.size();
095        }
096
097        public void clear() {
098            points.clear();
099            xPoints = null;
100            yPoints = null;
101        }
102    }

333    /**
334     * キー操作の設定
335     */
336    private void setKeyOperation() {
337        setFocusTraversable(true);
338        addEventFilter(MouseEvent.ANY, e -> requestFocus());
339
340        final ArrayList<KeyCode> functionKeys = new ArrayList();
341        functionKeys.add(KeyCode.F1);
342        functionKeys.add(KeyCode.F2);
343        functionKeys.add(KeyCode.F3);
344        functionKeys.add(KeyCode.F4);
345        functionKeys.add(KeyCode.F5);
346        functionKeys.add(KeyCode.F6);
347        functionKeys.add(KeyCode.F7);
348        functionKeys.add(KeyCode.F8);
349        functionKeys.add(KeyCode.F9);
350
351        // 何かのキーが押された
352        setOnKeyPressed(event -> {
353            // ESC
354            if (event.getCode() == KeyCode.ESCAPE) {
355                switch (mode) {
356                case ADD_LINK:
357                case ADD_NODE_LINK:
358                    // ESC: ニュートラル
359                    clearEditingStates();
360                    repaintLater();
361                    break;
362                case EDIT_NODE:
363                    // ESC: 選択を解除
364                    frame.getNodePanel().clearSelection();
365                    break;
366                case EDIT_LINK:
367                    // ESC: 選択を解除
368                    frame.getLinkPanel().clearSelection();
369                    break;
370                case EDIT_AREA:
371                    // ESC: 選択を解除
372                    frame.getAreaPanel().clearSelection();
373                    break;
374                case EDIT_POLYGON:
375                    // ESC: 選択を解除
376                    frame.getPolygonPanel().clearSelection();
377                    break;
378                }
379                event.consume();
380            }
381
382            // Ctrl + A
383            if (event.isControlDown() && event.getCode() == KeyCode.A) {
384                switch (mode) {
385                case EDIT_NODE:
386                    // Ctrl + A (編集画面): カレントグループの全ノードを選択
387                    ArrayList<MapNode> nodes = new ArrayList();
388                    for (MapNode node : editor.getCurrentGroup().getChildNodes()) {
389                        if (frame.getNodePanel().getFilteredSet().contains(node)) {
390                            nodes.add(node);
391                        }
392                    }
393                    frame.getNodePanel().select(nodes);
394                    break;
395                case EDIT_LINK:
396                    // Ctrl + A (編集画面): カレントグループの全リンクを選択
397                    ArrayList<MapLink> links = new ArrayList();
398                    for (MapLink link : editor.getCurrentGroup().getChildLinks()) {
399                        if (frame.getLinkPanel().getFilteredSet().contains(link)) {
400                            links.add(link);
401                        }
402                    }
403                    frame.getLinkPanel().select(links);
404                    break;
405                case EDIT_AREA:
406                    // Ctrl + A (編集画面): カレントグループの全エリアを選択
407                    ArrayList<MapArea> areas = new ArrayList();
408                    for (MapArea area : editor.getCurrentGroup().getChildMapAreas()) {
409                        if (frame.getAreaPanel().getFilteredSet().contains(area)) {
410                            areas.add(area);
411                        }
412                    }
413                    frame.getAreaPanel().select(areas);
414                    break;
415                case EDIT_POLYGON:
416                    // Ctrl + A (編集画面): カレントグループの全ポリゴンを選択
417                    ArrayList<MapPolygon> polygons = new ArrayList();
418                    for (MapPolygon polygon : editor.getCurrentGroup().getChildMapPolygons()) {
419                        if (frame.getPolygonPanel().getFilteredSet().contains(polygon)) {
420                            polygons.add(polygon);
421                        }
422                    }
423                    frame.getPolygonPanel().select(polygons);
424                    break;
425                }
426                event.consume();
427            }
428
429            // アプリケーションキー
430            if (event.getCode() == KeyCode.CONTEXT_MENU) {
431                if (! frame.isContextMenuShowing()) {
432                    popupMenu(lastMouseX, lastMouseY, event.isControlDown());
433                    event.consume();
434                }
435            }
436
437            // ファンクションキー
438            int fKeyIndex = functionKeys.indexOf(event.getCode());
439            if (fKeyIndex != -1) {
440                // Edit mode shortcut
441                if (! event.isAltDown() && ! event.isControlDown() && ! event.isMetaDown() && ! event.isShiftDown()) {
442                    frame.selectEditMode(fKeyIndex);
443                }
444                // Group selection shortcut
445                else if (! event.isAltDown() && ! event.isControlDown() && ! event.isMetaDown() && event.isShiftDown()) {
446                    frame.selectGroup(fKeyIndex);
447                }
448                event.consume();
449            }
450
451            // 何かキーを押したら矩形選択はキャンセルする
452            if (selectionRangeStart != null) {
453                selectionRangeStart = null;
454                selectionRange = null;
455                repaintLater();
456            }
457        });
458
459        // 押されているキーが放された
460        setOnKeyReleased(event -> {
461            // ポリゴン選択の完了
462            if (! event.isAltDown() && ! selectionRangePolygon.isEmpty()) {
463                selection();
464                selectionRangePolygon.clear();
465                repaintLater();
466            }
467        });
468    }

478    /**
479     * マウス操作の設定
480     */
481    private void setMouseOperation() {
482        // カーソルを移動した時
483        setOnMouseMoved(event -> {
484            double x = event.getX();
485            double y = event.getY();
486
487            switch (mode) {
488            case ADD_NODE:
489            case ADD_NODE_LINK:
490                if (lastMouseX != x || lastMouseY != y) {
491                    Point2D snapPoint = null;
492                    if (gridShowing && gridSnapping) {
493                        snapPoint = getGridSnapPoint(x, y);
494                        if (snapPoint == null) {
495                            nodeCursorX = x;
496                            nodeCursorY = y;
497                            indicateMapCoordinates(x, y);
498                        } else {
499                            Point2D mapPoint = convertToOriginal(snapPoint);
500                            frame.setStatusText("Map coordinates: " + mapPoint.getX() + ", " + mapPoint.getY());
501                            Point2D point = pointConvertMapToCanvas(snapPoint);
502                            nodeCursorX = point.getX();
503                            nodeCursorY = point.getY();
504                        }
505                    } else {
506                        nodeCursorX = x;
507                        nodeCursorY = y;
508                        indicateMapCoordinates(x, y);
509                    }
510                    if (fromNode != null) {
511                        MapPartGroup group = editor.getCurrentGroup();
512                        Point2D point = (snapPoint == null ? pointConvertCanvasToMap(x, y) : snapPoint);
513                        double length = editor.calcLinkLength(getRotatedPoint(fromNode), fromNode.getHeight(), point, frame.getCurrentNodeHeight(), group);
514                        frame.setCurrentLinkLength(length);
515                    }
516                    repaintLater();
517                }
518                break;
519            case ADD_LINK:
520                if (fromNode == null) {
521                    // 始点ノード(候補)の更新
522                    if (updateTargetNode(x, y)) {
523                        repaintLater();
524                    }
525                } else {
526                    MapNode lastNode = toNode;
527                    updateTentativeLink(x, y);
528                    if (toNode != null && toNode != lastNode) {
529                        repaintLater();
530                        MapPartGroup group = editor.getCurrentGroup();
531                        double length = editor.calcLinkLength(fromNode, toNode, group);
532                        frame.setCurrentLinkLength(length);
533                    }
534                }
535                if (mapCoordinatesShowing) {
536                    indicateMapCoordinates(x, y);
537                }
538                break;
539            case ADD_AREA:
540                break;
541            case EDIT_NODE:
542                if (event.isAltDown() && ! selectionRangePolygon.isEmpty()) {
543                    repaintLater();
544                } else if (updateTargetNode(x, y)) {
545                    repaintLater();
546                }
547                if (mapCoordinatesShowing) {
548                    indicateMapCoordinates(x, y);
549                }
550                break;
551            case EDIT_LINK:
552                if (event.isAltDown() && ! selectionRangePolygon.isEmpty()) {
553                    repaintLater();
554                } else if (updateTargetLink(x, y)) {
555                    repaintLater();
556                }
557                if (mapCoordinatesShowing) {
558                    indicateMapCoordinates(x, y);
559                }
560                break;
561            case EDIT_AREA:
562                if (event.isAltDown() && ! selectionRangePolygon.isEmpty()) {
563                    repaintLater();
564                } else if (updateTargetArea(x, y)) {
565                    repaintLater();
566                }
567                if (mapCoordinatesShowing) {
568                    indicateMapCoordinates(x, y);
569                }
570                break;
571            case EDIT_POLYGON:
572                if (event.isAltDown() && ! selectionRangePolygon.isEmpty()) {
573                    repaintLater();
574                } else if (updateTargetPolygon(x, y)) {
575                    repaintLater();
576                }
577                if (mapCoordinatesShowing) {
578                    indicateMapCoordinates(x, y);
579                }
580                break;
581            }
582            lastMouseX = x;
583            lastMouseY = y;
584        });
585
586        // ボタンを押した時
587        setOnMousePressed(event -> {
588            lastMouseX = event.getX();
589            lastMouseY = event.getY();
590            mousePressedPoint = pointConvertCanvasToMap(lastMouseX, lastMouseY);
591            dragging = false;
592        });
593
594        // ボタンを放した時
595        setOnMouseReleased(event -> {
596            if (selectionRange != null) {
597                selection();
598                selectionRangeStart = null;
599                selectionRange = null;
600                repaintLater();
601            }
602            if (nodeMoving) {
603                nodeMoving = false;
604                Point2D point = pointConvertCanvasToMap(event.getX(), event.getY());
605                nodeMovingAmount = point.subtract(mousePressedPoint);
606                if (nodeMovingAmount.getX() == 0.0 && nodeMovingAmount.getY() == 0.0) {
607                    repaintLater();
608                    return;
609                }
610                // 選択中の複数ノードを移動またはコピーする
611                Point2D movingAmount = point.subtract(mousePressedPoint);
612                frame.copyOrMoveNodes(movingAmount.getX(), movingAmount.getY(), 0.0);
613            }
614        });
615
616        // ドラッグした時
617        setOnMouseDragged(event -> {
618            dragOperation(event);
619            dragging = true;
620        });
621
622        // ボタンをクリックした時
623        setOnMouseClicked(event -> {
624            if (dragging) {
625                return;
626            }
627
628            MapPartGroup group = editor.getCurrentGroup();
629            Point2D mapPoint = pointConvertCanvasToMap(event.getX(), event.getY());
630            switch (event.getButton()) {
631            case PRIMARY:
632                if (mode == EditorMode.ADD_NODE || mode == EditorMode.ADD_NODE_LINK) {
633                    NetworkMap networkMap = editor.getMap();
634                    if (networkMap.getGroups().size() == 1 && networkMap.getNodes().isEmpty()) {
635                        Alert alert = new Alert(AlertType.CONFIRMATION, "Would you like to add the node?", ButtonType.YES, ButtonType.NO);
636                        alert.initOwner(frame.getStage());
637                        alert.getDialogPane().setHeaderText("If you create a node in the root group,\nyou will not be able to add groups.");
638                        Optional<ButtonType> result = alert.showAndWait();
639                        if (result.isPresent() && result.get() == ButtonType.NO) {
640                            repaintLater();
641                            return;
642                        }
643                    }
644                }
645                switch (mode) {
646                case ADD_NODE:
647                    if (gridShowing && gridSnapping) {
648                        Point2D point = getGridSnapPoint(event.getX(), event.getY());
649                        if (point != null) {
650                            mapPoint = point;
651                        }
652                    }
653                    editor.invokeSingleCommand(new AddNode(group, convertToOriginal(mapPoint), frame.getCurrentNodeHeight()));
654                    editor.updateHeight();
655                    break;
656                case ADD_LINK:
657                    if (fromNode == null) {
658                        // 始点ノードの設定
659                        if (pointedNode == null) {
660                            frame.setStatusText("Point to the start node");
661                            editor.beep();
662                        } else {
663                            fromNode = pointedNode;
664                            toNode = null;
665                        }
666                    } else {
667                        // 新規リンク追加
668                        if (toNode == null) {
669                            frame.setStatusText("Point to the end node");
670                            editor.beep();
671                        } else {
672                            double length = editor.calcLinkLength(fromNode, toNode, group);
673                            frame.setCurrentLinkLength(length);
674                            double width = frame.getCurrentLinkWidth();
675                            if (width <= 0.0) {
676                                return;
677                            }
678                            AddLink command = new AddLink(fromNode, toNode, length, width);
679                            fromNode = toNode;
680                            toNode = null;
681                            editor.invokeSingleCommand(command);
682                        }
683                    }

1305        } else if (deltaY < 0.0) {
1306            scale /= (controlDown ? 1.01 : 1.1);
1307        }
1308        scale = Math.min(scale, 100.0 / SCALE_FACTOR);
1309        scale = Math.max(scale, 0.01 / SCALE_FACTOR);
1310        frame.setStatusText("Scale: " + scale);
1311
1312        originTranslate = new Point2D(ox * scale, oy * scale);
1313        adjustmentTranslate = new Point2D(x, y);
1314        repaintLater();
1315    }

1477    /**
1478     * マップのセンタリングとスケーリング
1479     */
1480    public void centering(boolean withScaling) {
1481        Rectangle2D bounds = calcRotatedMapRectangle();
1482        double cx = bounds.getMinX() + bounds.getWidth() / 2.0;
1483        double cy = bounds.getMinY() + bounds.getHeight() / 2.0;
1484        cx *= SCALE_FACTOR;
1485        cy *= SCALE_FACTOR;
1486
1487        // map is empty
1488        if (cx == 0.0 && cy == 0.0) {
1489            if (withScaling) {
1490                scale = 1.0 / SCALE_FACTOR;
1491            }
1492            originTranslate = new Point2D(0.0, 0.0);
1493            adjustmentTranslate = new Point2D(getWidth() / 2, getHeight() / 2);
1494            return;
1495        }
1496
1497        if (withScaling) {
1498            int centeringMargin = 0;
1499            // リアルマップにはマージンを付加しない
1500            if (editor.getMap().getLinks().size() < MINIMUM_REAL_MAP_LINKS) {
1501                centeringMargin = CENTERING_MARGIN;
1502            }
1503            double scaleX = (getWidth() - centeringMargin * 2) / (bounds.getWidth() * SCALE_FACTOR);
1504            double scaleY = (getHeight() - centeringMargin * 2) / (bounds.getHeight() * SCALE_FACTOR);
1505            scale = Math.min(scaleX, scaleY);
1506        }
1507
1508        // -(マップの中心座標に相当するCanvas座標)
1509        originTranslate = new Point2D(-cx * scale, -cy * scale);
1510        // 表示位置をCanvasの中央に合わせる
1511        adjustmentTranslate = new Point2D(getWidth() / 2, getHeight() / 2);
1512    }

1514    /**
1515     * マップに外接する矩形を算出する
1516     */
1517    public Rectangle2D calcRotatedMapRectangle() {
1518        double north = 0.0;
1519        double south = 0.0;
1520        double west = 0.0;
1521        double east = 0.0;
1522        for (MapNode node : editor.getMap().getNodes()) {
1523            Point2D point = calcRotatedPoint(node.getX(), node.getY());
1524            if (north == 0.0 && south == 0.0) {
1525                north = point.getY();
1526                south = point.getY();
1527                west = point.getX();
1528                east = point.getX();
1529            }
1530            if (point.getY() < north) {
1531                north = point.getY();
1532            }
1533            if (point.getY() > south) {
1534                south = point.getY();
1535            }
1536            if (point.getX() < west) {
1537                west = point.getX();
1538            }
1539            if (point.getX() > east) {
1540                east = point.getX();
1541            }
1542        }
1543        return new Rectangle2D.Double(west, north, east - west, south - north);
1544    }

1660    /**
1661     * 仮リンク(を構成する toNode)を決める
1662     */
1663    private void updateTentativeLink(double x, double y) {
1664        java.awt.geom.Point2D point = toAwtPoint2D(convertToOriginal(pointConvertCanvasToMap(x, y)));
1665        MapPartGroup group = editor.getCurrentGroup();
1666
1667        MapNodeTable nodes = group.getChildNodesAndSymlinks(fromNode, fromNode.getQuadrant(point));
1668        MapLinkTable links = group.getChildLinks();
1669        double minDistance = -1.0;
1670        for (MapNode toNode : nodes) {
1671            if (toNode == fromNode) {
1672                continue;
1673            }
1674            Line2D line = new Line2D.Double(fromNode.getPosition(), toNode.getPosition());
1675            if (line.ptSegDist(point) < 5.0 / (scale * SCALE_FACTOR)) {
1676                double distance = line.getP1().distance(line.getP2());
1677                if (minDistance == -1.0 || distance < minDistance) {
1678                    /* check for existing links */
1679                    boolean exists = false;
1680                    for (MapLink link : links) {
1681                        if ((link.getFrom() == fromNode && link.getTo() == toNode) ||
1682                                (link.getTo() == fromNode && link.getFrom() == toNode)) {
1683                            exists = true;
1684                            break;
1685                        }
1686                    }
1687                    if (exists) {
1688                        continue;
1689                    }
1690                    this.toNode = toNode;
1691                    minDistance = distance;
1692                }
1693            }
1694        }
1695    }

2634     * 座標がスナップエリア内だったらグリッドの交点座標を返す
2635     */
2636    public Point2D getGridSnapPoint(double x, double y) {
2637        double offsetX = Math.abs(gridOffsetX);
2638        while (offsetX > gridWidth) {
2639            offsetX -= gridWidth;
2640        }
2641        offsetX *= Math.signum(gridOffsetX);
2642
2643        double offsetY = Math.abs(gridOffsetY);
2644        while (offsetY > gridHeight) {
2645            offsetY -= gridHeight;
2646        }
2647        offsetY *= Math.signum(gridOffsetY);
2648
2649        double x1 = ((int)((viewArea.getMinX() + offsetX) / gridWidth) - 1) * gridWidth;
2650        double y1 = ((int)((viewArea.getMinY() + offsetY) / gridHeight) - 1) * gridHeight;
2651        double x2 = ((int)((viewArea.getMaxX() + offsetX) / gridWidth) + 1) * gridWidth;
2652        double y2 = ((int)((viewArea.getMaxY() + offsetY) / gridHeight) + 1) * gridHeight;
2653
2654        Point2D point = pointConvertCanvasToMap(x, y);
2655        double mindist = Double.POSITIVE_INFINITY;
2656        double radius = gridSnapSize / (scale * SCALE_FACTOR);
2657        double snappedX = point.getX();
2658        double snappedY = point.getY();
2659        for (double _y = y1; _y < y2; _y += gridHeight) {
2660            for (double _x = x1; _x < x2; _x += gridWidth) {
2661                double dist = point.distance(_x + offsetX, _y + offsetY);
2662                if (dist < mindist && dist < radius) {
2663                    mindist = dist;
2664                    snappedX = _x + offsetX;
2665                    snappedY = _y + offsetY;
2666                }
2667            }
2668        }
2669
2670        return snappedX == point.getX() && snappedY == point.getY() ? null : new Point2D(snappedX, snappedY);
2671    }
