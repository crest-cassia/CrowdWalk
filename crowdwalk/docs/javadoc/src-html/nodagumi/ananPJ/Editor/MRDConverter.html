025 * MAPPLE 道路ネットワークデータをマップデータに変換する.
026 *
027 * 基本道路ノードの接続リンクに付いての仕様の補足:
028 * ・通行可能かどうかがリンクの交通規制種別コードで判断できる場合には、通行可/不可にかかわらず交差点通行コードは 1 となる。
029 * ・リンクの交通規制種別コードだけでは「通行不可」と判断できない時のみ交差点通行コードは 2 となる。
030 * ・カーナビではUターン指示など出さないため、Uターン禁止かどうかの情報はない。Uターンの交差点通行コードは必ず 1 となる。
031 * ・隣接メッシュとの接合ノードは必ず末端ノードとなり、接続リンクは一本だけである。
032 */

081            }
082            properties.put("信号機有無フラグ", reader.getInteger("信号機有無フラグ"));
083        }

086            return String.format("MRD_%s_%s", meshCode, nodeNumber);

118        public void readInterpolationPoints(MRDReader reader) throws IOException {

125            }

129                interpolationPoints.add(new Point2D(x, y));
130            }
131        }

311    /**
312     * コンストラクタ
313     */
314    public MRDConverter(File file, int zone) throws IOException {
315        // 全レコードを読んでデータの種類ごとのハッシュマップに登録する
316        MRDReader reader = new MRDReader(file);
317        for (int position = 0; position < reader.getRecordSize(); position++) {
318            reader.setRecordPosition(position);
319            int recordID = reader.getInt("レコードID");
320            switch (recordID) {
321            case 11: // 管理データ（その１）
322                meshCode = reader.getString("２次メッシュコード");
323                meshPoint = meshCodeToCoordinate(meshCode);
324                break;
325            case 21: // 基本道路ノードデータ
326                BasicRoadNode basicRoadNode = new BasicRoadNode(reader);
327                basicRoadNodes.put(basicRoadNode.nodeNumber, basicRoadNode);
328                break;
329            case 22: // 基本道路リンクデータ
330                BasicRoadLink basicRoadLink = basicRoadLinks.get(reader.getString("リンク番号"));
331                if (basicRoadLink == null) {
332                    basicRoadLink = new BasicRoadLink(reader);
333                    basicRoadLinks.put(basicRoadLink.linkNumber, basicRoadLink);
334                }
335                basicRoadLink.readInterpolationPoints(reader);
336                break;
337            case 24: // 基本道路リンク、全道路リンク対応データ
338                BasicAndAllRoadLinkMapping linkMapping = basicAndAllRoadLinkMappings.get(reader.getString("基本道路リンク番号"));
339                if (linkMapping == null) {
340                    linkMapping = new BasicAndAllRoadLinkMapping(reader);
341                    basicAndAllRoadLinkMappings.put(linkMapping.linkNumber, linkMapping);
342                }
343                linkMapping.readAllRoadLinkNumbers(reader);
344                break;
345            case 31: // 全道路ノードデータ
346                AllRoadNode allRoadNode = new AllRoadNode(reader);
347                allRoadNodes.put(String.format("%d,%d", allRoadNode.x, allRoadNode.y), allRoadNode);
348                break;
349            case 32: // 全道路リンクデータ
350                AllRoadLink allRoadLink = allRoadLinks.get(reader.getString("リンク番号"));
351                if (allRoadLink == null) {
352                    allRoadLink = new AllRoadLink(reader);
353                    allRoadLinks.put(allRoadLink.linkNumber, allRoadLink);
354                }
355                allRoadLink.readInterpolationPoints(reader);
356                break;
357            }
358        }
359        reader.close();
360
361        for (BasicAndAllRoadLinkMapping linkMapping : basicAndAllRoadLinkMappings.values()) {
362            for (String allRoadLinkNumber : linkMapping.allRoadLinkNumbers) {
363                basicAndAllRoadLinkMap.put(allRoadLinkNumber, linkMapping.linkNumber);
364            }
365        }
366
367        // 経緯度(旧日本測地系)を平面直角座標に換算する
368        transform = GsiTile.createCoordinateTransform(EPSG, GsiTile.JGD2000_JPR_EPSG_NAMES[zone]);
369    }

371    /**
372     * リンクを座標リストに変換する(簡易描画用).
373     *
374     * 基本道路リンクと全道路リンクの重複が多少発生する
375     */
376    public ArrayList<ArrayList<Point2D>> convertToLines(boolean allRoadIncluding) {
377        ArrayList<ArrayList<Point2D>> lines = new ArrayList();
378
379        for (BasicRoadLink basicRoadLink : basicRoadLinks.values()) {
380            ArrayList<Point2D> line = new ArrayList();
381            for (Point2D point : basicRoadLink.interpolationPoints) {
382                line.add(toCrowdWalkPoint(point));
383            }
384            if (! line.isEmpty()) {
385                lines.add(line);
386            }
387        }
388
389        if (allRoadIncluding) {
390            for (AllRoadLink allRoadLink : allRoadLinks.values()) {
391                if (basicAndAllRoadLinkMap.containsKey(allRoadLink.linkNumber)) {
392                    continue;
393                }
394                ArrayList<Point2D> line = new ArrayList();
395                BasicRoadLink basicRoadLink = basicRoadLinks.get(allRoadLink.basicLinkNumber);
396                if (basicRoadLink == null) {
397                    for (Point2D point : allRoadLink.interpolationPoints) {
398                        line.add(toCrowdWalkPoint(point));
399                    }
400                    if (! line.isEmpty()) {
401                        lines.add(line);
402                    }
403                }
404            }
405        }
406
407        return lines;
408    }

410    /**
411     * ノードとリンクをマップデータに変換してネットワークマップに追加する
412     */
413    public void addToNetworkMap(MapEditor editor, MapPartGroup group, int zone, boolean allRoadIncluding) throws Exception {
414        NetworkMap networkMap = editor.getMap();
415        HashMap<String, MapNode> nodes = new HashMap<>();
416        HashMap<BasicRoadNode, MapNode> basicNodes = new HashMap<>();
417
418        // 基本道路ノードを MapNode として NetworkMap に追加する
419        for (BasicRoadNode basicRoadNode : basicRoadNodes.values()) {
420            String nextID = String.format("MRD_%s_%s", basicRoadNode.nextMeshCode, basicRoadNode.nextNodeNumber);
421            MapNode node = (MapNode)networkMap.getObject(nextID);
422            if (node == null) {
423                node = addNode(editor, basicRoadNode.getID(), group, toCrowdWalkPoint(basicRoadNode.x, basicRoadNode.y), 0.0);
424                setProperties(editor, node, basicRoadNode.properties);
425                if (! basicRoadNode.nextNodeNumber.equals("0000")) {
426                    addJointNodeTag(editor, node);
427                }
428            }
429            nodes.put(basicRoadNode.getID(), node);
430            basicNodes.put(basicRoadNode, node);
431        }
432
433        // 基本道路リンクを MapLink として NetworkMap に追加する
434        // 補間点は MapNode として NetworkMap に追加する
435        for (BasicRoadLink basicRoadLink : basicRoadLinks.values()) {
436            double width = WIDTH_TABLE[basicRoadLink.widthCode];
437
438            // 通行止・一方通行タグの準備
439            int trafficRegulation = (Integer)basicRoadLink.properties.get("交通規制種別コード");
440            String regulationTag = null;
441            if (trafficRegulation == 2 || trafficRegulation == 3) {
442                regulationTag = MapLink.Tag_RoadClosed;
443            } else if (trafficRegulation == 4 || trafficRegulation == 6) {
444                regulationTag = MapLink.Tag_OneWayForward;
445            } else if (trafficRegulation == 5 || trafficRegulation == 7) {
446                regulationTag = MapLink.Tag_OneWayBackward;
447            }
448
449            MapNode from = null;
450            for (int index = 0; index < basicRoadLink.nInterpolationPoints; index++) {
451                MapNode node = null;
452                if (index == 0) {
453                    node = nodes.get(basicRoadLink.getNode1ID());
454                    if (node == null) {
455                        throw new Exception("Node 1 of BasicRoadLink not found: " + basicRoadLink.getNode1ID());
456                    }
457                } else if (index == basicRoadLink.nInterpolationPoints - 1) {
458                    node = nodes.get(basicRoadLink.getNode2ID());
459                    if (node == null) {
460                        throw new Exception("Node 2 of BasicRoadLink not found: " + basicRoadLink.getNode2ID());
461                    }
462                } else {
463                    Point2D iPoint = basicRoadLink.interpolationPoints.get(index);
464                    AllRoadNode allRoadNode = null;
465                    boolean hasIntersection = false;
466                    if (allRoadIncluding) {
467                        allRoadNode = allRoadNodes.get(String.format("%d,%d", (int)iPoint.getX(), (int)iPoint.getY()));
468                        if (allRoadNode != null) {
469                            hasIntersection = ((Integer)allRoadNode.properties.get("ノード種別コード") == 1);   // 交差点ノード
470                        }
471                    }
472                    // 該当する全道路ノードがあれば補間点ノードに流用する
473                    if (allRoadIncluding && hasIntersection) {
474                        node = addNode(editor, allRoadNode.getID(), group, toCrowdWalkPoint(iPoint), 0.0);
475                        setProperties(editor, node, allRoadNode.properties);
476                        if (! allRoadNode.nextNodeNumber.equals("00000")) {
477                            addJointNodeTag(editor, node);
478                        }
479                        nodes.put(allRoadNode.getID(), node);
480                    }
481                    // 補間点ノードを生成する
482                    else {
483                        node = addNode(editor, null, group, toCrowdWalkPoint(iPoint), 0.0);
484                        // この node が交差点になることはないので nodes には追加しない
485                    }
486                }
487
488                if (from != null) {
489                    String id = String.format("%s_%03d", basicRoadLink.getID(), index);
490                    double length = from.getPosition().distance(node.getPosition());
491                    MapLink link = addLink(editor, id, from, node, length, width);
492                    if (regulationTag != null) {
493                        AddTag _command = new AddTag(link, regulationTag);
494                        if (! editor.invoke(_command)) {
495                            throw new Exception("Edit command error: " + _command.toString());
496                        }
497                    }
498                    setProperties(editor, link, basicRoadLink.properties);
499                }
500                from = node;
501            }
502        }
503
504        // 接続リンクごとの属性情報をノードにセットする
505        String[] names = {"接続リンクID", "交差点通行コード", "接続角度"};
506        for (Map.Entry<BasicRoadNode, MapNode> entry : basicNodes.entrySet()) {
507            HashMap<String, Object> properties = new LinkedHashMap();
508            BasicRoadNode basicRoadNode = entry.getKey();
509            MapNode node = entry.getValue();
510            // 今回初めて登録されたノードの場合
511            if (getPropertyAsString(node, "接続リンクID\\[1\\]") == null) {
512                for (int index = 0; index < basicRoadNode.connectionNodeNumbers.size(); index++) {
513                    String id = getMapLink(networkMap, basicRoadNode, index).getID();
514                    String passCode = basicRoadNode.passCodes.get(index);
515                    Integer connectionAngle = basicRoadNode.connectionAngles.get(index);
516                    properties.put(String.format("接続リンクID[%d]", index + 1), id);
517                    properties.put(String.format("交差点通行コード[%d]", index + 1), passCode);
518                    properties.put(String.format("接続角度[%d]", index + 1), connectionAngle);
519                }
520            }
521            // 隣接メッシュの既存ノードと接合する場合は接続リンク情報をマージする
522            else {
523                // 接合するノード側の接続リンク本数のチェック
524                if (basicRoadNode.connectionNodeNumbers.size() != 1) {
525                    throw new Exception("Incorrect number of connected links: " + basicRoadNode.nodeNumber);
526                }
527                // 既存ノードの接続リンク本数のチェック
528                String str = getPropertyAsString(node, "接続リンク本数");
529                if (str == null || Integer.parseInt(str) != 1) {
530                    throw new Exception("Incorrect number of connected links: " + node.toShortInfo());
531                }
532                removeProperty(editor, node, "接続リンク本数");
533                properties.put("接続リンク本数", new Integer(2));
534
535                // 既存ノードに登録されている属性情報を取得後に削除する
536                String linkIdOfExistingNode = null;
537                Integer connectionAngleOfExistingNode = null;
538                for (int n = 0; n < names.length; n++) {
539                    String name = String.format("%s\\[1\\]", names[n]);
540                    String value = getPropertyAsString(node, name);
541                    if (value != null) {
542                        if (n == 0) {           // 接続リンクID
543                            linkIdOfExistingNode = value;
544                        } else if (n == 2) {    // 接続角度
545                            connectionAngleOfExistingNode = Integer.parseInt(value);
546                        }
547                        removeProperty(editor, node, name);
548                    }
549                }
550
551                String linkID = getMapLink(networkMap, basicRoadNode, 0).getID();
552                String passCode = "11000000";
553                Integer connectionAngle = basicRoadNode.connectionAngles.get(0);
554                if (connectionAngle < connectionAngleOfExistingNode) {
555                    properties.put("接続リンクID[1]", linkID);
556                    properties.put("交差点通行コード[1]", passCode);
557                    properties.put("接続角度[1]", connectionAngle);
558                    properties.put("接続リンクID[2]", linkIdOfExistingNode);
559                    properties.put("交差点通行コード[2]", passCode);
560                    properties.put("接続角度[2]", connectionAngleOfExistingNode);
561                } else {
562                    properties.put("接続リンクID[1]", linkIdOfExistingNode);
563                    properties.put("交差点通行コード[1]", passCode);
564                    properties.put("接続角度[1]", connectionAngleOfExistingNode);
565                    properties.put("接続リンクID[2]", linkID);
566                    properties.put("交差点通行コード[2]", passCode);
567                    properties.put("接続角度[2]", connectionAngle);
568                }
569            }
570            setProperties(editor, node, properties);
571        }
572
573        // root グループの zone を更新する
574        MapPartGroup root = (MapPartGroup)networkMap.getRoot();
575        if (root.getZone() != zone) {
576            SetZone command = new SetZone(root, zone);
577            if (! editor.invoke(command)) {
578                throw new Exception("Edit command error: " + command.toString());
579            }
580        }
581
582        if (! allRoadIncluding) {
583            return;
584        }
585
586        // 全道路ノードを MapNode として NetworkMap に追加する
587        for (AllRoadNode allRoadNode : allRoadNodes.values()) {
588            if (nodes.containsKey(allRoadNode.getID())) {
589                continue;
590            }
591            String nextID = String.format("MRD_%s_%s", allRoadNode.nextMeshCode, allRoadNode.nextNodeNumber);
592            MapNode node = (MapNode)networkMap.getObject(nextID);
593            if (node == null && allRoadNode.getBasicNodeID() != null) {
594                node = (MapNode)networkMap.getObject(allRoadNode.getBasicNodeID());
595            }
596            if (node == null) {
597                node = addNode(editor, allRoadNode.getID(), group, toCrowdWalkPoint(allRoadNode.x, allRoadNode.y), 0.0);
598                setProperties(editor, node, allRoadNode.properties);
599                if (! allRoadNode.nextNodeNumber.equals("00000")) {
600                    addJointNodeTag(editor, node);
601                }
602            }
603            nodes.put(allRoadNode.getID(), node);
604        }
605
606        // 全道路リンクを MapLink として NetworkMap に追加する
607        // 補間点は MapNode として NetworkMap に追加する
608        for (AllRoadLink allRoadLink : allRoadLinks.values()) {
609            if (basicAndAllRoadLinkMap.containsKey(allRoadLink.linkNumber)) {
610                continue;
611            }
612            String basicLinkID = allRoadLink.getBasicLinkID();
613            if (basicLinkID != null) {
614                MapLink link = (MapLink)networkMap.getObject(basicLinkID + "_001");
615                if (link != null) {
616                    continue;
617                }
618            }
619
620            double width = WIDTH_TABLE[allRoadLink.widthCode];
621            MapNode from = null;
622            for (int index = 0; index < allRoadLink.nInterpolationPoints; index++) {
623                MapNode node = null;
624                if (index == 0) {
625                    node = nodes.get(allRoadLink.getNode1ID());
626                    if (node == null) {
627                        throw new Exception("Node 1 of AllRoadLink not found: " + allRoadLink.getNode1ID());
628                    }
629                } else if (index == allRoadLink.nInterpolationPoints - 1) {
630                    node = nodes.get(allRoadLink.getNode2ID());
631                    if (node == null) {
632                        throw new Exception("Node 2 of AllRoadLink not found: " + allRoadLink.getNode2ID());
633                    }
634                } else {
635                    Point2D point = toCrowdWalkPoint(allRoadLink.interpolationPoints.get(index));
636                    String pointStr = String.format("%d,%d", (int)point.getX(), (int)point.getY());
637                    node = nodes.get(pointStr);
638                    if (node == null) {
639                        // 補間点ノードを生成する
640                        node = addNode(editor, null, group, point, 0.0);
641                        nodes.put(pointStr, node);
642                    }
643                }
644
645                if (from != null) {
646                    String id = String.format("%s_%03d", allRoadLink.getID(), index);
647                    double length = from.getPosition().distance(node.getPosition());
648                    MapLink link = addLink(editor, id, from, node, length, width);
649                    setProperties(editor, link, allRoadLink.properties);
650                }
651                from = node;
652            }
653        }
654
655        // 孤立しているノードがあれば削除する
656        for (MapNode node : nodes.values()) {
657            if (node.getLinks().isEmpty()) {
658                RemoveNode command = new RemoveNode(node);
659                if (! editor.invoke(command)) {
660                    throw new Exception("Edit command error: " + command.toString());
661                }
662            }
663        }
664    }

666    /**
667     * 基本道路ノードの接続リンクに対応する MapLink を返す
668     */
669    public MapLink getMapLink(NetworkMap networkMap, BasicRoadNode basicRoadNode, int index) throws Exception {
670        String connectionNodeNumber = basicRoadNode.connectionNodeNumbers.get(index);
671        String linkNumber = makeLinkNumber(basicRoadNode.nodeNumber, connectionNodeNumber);
672        BasicRoadLink basicRoadLink = basicRoadLinks.get(linkNumber);
673        if (basicRoadLink == null) {
674            throw new Exception("BasicRoadLink not found: " + linkNumber);
675        }
676        int n = basicRoadLink.getNodeNumber1().equals(basicRoadNode.nodeNumber) ? 1 : basicRoadLink.nInterpolationPoints - 1;
677        String id = String.format("%s_%03d", basicRoadLink.getID(), n);
678        MapLink link = (MapLink)networkMap.getObject(id);
679        if (link == null) {
680            throw new Exception("MapLink not found: " + id);
681        }
682        return link;
683    }

686     * メッシュコードを経度緯度に変換する.
687     *
688     * ※二次メッシュコードまで対応
689     */
690    public static Point2D meshCodeToCoordinate(String meshCode) {
691        double lat = Integer.parseInt(meshCode.substring(0, 2)) / 1.5;
692        double lon = Integer.parseInt(meshCode.substring(2, 4)) + 100.0;
693        if (meshCode.length() > 4) {
694            lat += ((Integer.parseInt(meshCode.substring(4, 5)) * 300) / 3600.0);
695            lon += ((Integer.parseInt(meshCode.substring(5, 6)) * 450) / 3600.0);
696        }
697        return new Point2D(lon, lat);
698    }

718    /**
719     * 正規化座標を CrowdWalk 座標に変換する.
720     *
721     * 座標は小数点以下第4位で丸める。
722     */
723    public Point2D toCrowdWalkPoint(int x, int y) {
724        Point2D point = toOriginalCoordinate(meshPoint, x, y);
725        java.awt.geom.Point2D jpr = GsiTile.transformCoordinate(transform, point.getX(), point.getY());
726        java.awt.geom.Point2D cwPoint = GsiTile.convertJPR2CW(jpr.getY(), jpr.getX());
727        return new Point2D(MapEditor.roundValue(cwPoint.getX(), 4), MapEditor.roundValue(cwPoint.getY(), 4));
728    }

795     * ノード・リンクの属性情報を文字列化して取得する.
796     *
797     * ※この実装では先頭に "property:" を付けたタグと照合している
798     */
799    public static String getPropertyAsString(OBNode obNode, String name) {
800        Matcher matcher = obNode.matchTag("property:" + name + "=(.*)");
801        return matcher == null ?  null : matcher.group(1);
802    }

804    /**
805     * ノード・リンクの属性情報を削除する.
806     *
807     * ※この実装では先頭に "property:" を付けたタグと照合している
808     */
809    public static boolean removeProperty(MapEditor editor, OBNode obNode, String name) throws Exception {
810        String value = getPropertyAsString(obNode, name);
811        if (value == null) {
812            return false;
813        }
814        String tag = String.format("property:%s=%s", name.replaceAll("\\\\", ""), value);
815        RemoveTag command = new RemoveTag(obNode, tag);
816        if (! editor.invoke(command)) {
817            throw new Exception("Edit command error: " + command.toString());
818        }
819        return true;
820    }
