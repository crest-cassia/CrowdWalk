424    private void init() {
425        // ヘルプ画面の準備
426
427        webView.setOnKeyPressed(event -> {
428            if (event.isControlDown()) {
429                if (event.getCode() == KeyCode.W) {
430                    helpStage.close();
431                } else if (event.getCode() == KeyCode.PLUS || event.getCode() == KeyCode.UP) {
432                    helpZoom += 0.1;
433                    webView.setZoom(helpZoom);
434                } else if (event.getCode() == KeyCode.MINUS || event.getCode() == KeyCode.DOWN) {
435                    helpZoom -= 0.1;
436                    webView.setZoom(helpZoom);
437                } else if (event.getCode() == KeyCode.DIGIT0) {
438                    helpZoom = 1.0;
439                    webView.setZoom(helpZoom);
440                }
441            }
442        });
443
444        Button okButton = new Button("  OK  ");
445        okButton.setOnAction(event -> helpStage.close());
446        BorderPane buttonPane = new BorderPane();
447        buttonPane.setPadding(new Insets(4, 8, 4, 8));
448        buttonPane.setRight(okButton);
449
450        BorderPane borderPane = new BorderPane();
451        borderPane.setCenter(webView);
452        borderPane.setBottom(buttonPane);
453
454        helpStage.setScene(new Scene(borderPane));
455    }

457    /**
458     * 終了処理
459     */
460    public boolean closing(WindowEvent event) {
461        if (editor.isModified()) {
462            Alert alert = new Alert(AlertType.CONFIRMATION, "Warning:\n    Map data has been modified.\n    Do you want to quit anyway?", ButtonType.YES, ButtonType.NO);
463            Optional<ButtonType> result = alert.showAndWait();
464            if (! result.isPresent() || result.get() == ButtonType.NO) {
465                if (event != null) {
466                    event.consume();
467                }
468                return false;
469            }
470        }
471        settings.put("_editorPositionX", (int)frame.getX());
472        settings.put("_editorPositionY", (int)frame.getY());
473        settings.put("_editorWidth", (int)frame.getWidth());
474        settings.put("_editorHeight", (int)frame.getHeight());
475        settings.put("dividerPosition", "" + splitPane.getDividerPositions()[0]);
476        return true;
477    }

784            e.printStackTrace();
