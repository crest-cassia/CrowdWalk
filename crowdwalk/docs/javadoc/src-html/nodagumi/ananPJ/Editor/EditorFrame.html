582    private boolean setBackgroundFile() {
583        FileDialog fd = new FileDialog(this, "Set background image", FileDialog.LOAD);
584        String dirName = null, fileName = null;
585        if (current_group.getImageFileName() != null) {
586            File file = new File(current_group.getImageFileName());
587            dirName = file.getPath();
588            fileName = file.getName();
589        } else {
590            dirName = editor.getDirName();
591            fileName = "";
592        }
593        fd.setFile(fileName);
594        fd.setDirectory(dirName);
595        fd.setVisible (true);
596
597        if (fd.getFile() == null) return false;
598
599        current_group.setImageFileName(fd.getFile());
600        return readBackgroundWithName();
601    }

604     * カレントグループの背景画像をマップデータから削除する。
605     *
606     * sx, sy 値は利便性のため残しておく。
607     */
608    private boolean removeBackgroundFile() {
609        String imageFileName = current_group.getImageFileName();
610        if (imageFileName == null || imageFileName.isEmpty()) {
611            return false;
612        }
613
614        int res = JOptionPane.showConfirmDialog(this, "Remove background file: " + imageFileName,
615                "", JOptionPane.OK_CANCEL_OPTION);
616        if (res == JOptionPane.CANCEL_OPTION) {
617            return false;
618        }
619
620        current_group.setImageFileName("");
621        panel.addBackground(null);
622        background_read = false;
623        backgroundImageScaleMenu.setEnabled(false);
624
625        repaint();
626        return true;
627    }

633    /*
634     * カレントグループの背景画像を読み込んで画面に表示する。
635     *
636     * 画像ファイルはマップファイルと同じディレクトリに存在しなければならない。
637     */
638    private boolean readBackgroundWithName() {
639        String imageFileName = current_group.getImageFileName();
640        if (imageFileName == null || imageFileName.isEmpty()) {
641            return false;
642        }
643
644        String imageFilePath = editor.getDirName() + imageFileName;
645        File imageFile = new File(imageFilePath);
646        if (!imageFile.exists()) {
647            status.setText("Failed to load " + imageFile + ".");
648            return false;
649        }
650        
651        Image image = getToolkit().getImage(imageFilePath);
652        panel.addBackground(image);
653        background_read = true;
654        backgroundImageScaleMenu.setEnabled(true);
655
656        repaint();
657        return true;
658    }

699    private boolean updateHoverNode(Point2D p) {
700        double mindist = Double.POSITIVE_INFINITY;
701        MapNode hoverNodeCandidate = null;
702        for (final MapNode node : getChildNodes()) {
703            double dist = p.distance(node.getPosition());
704            if (dist < mindist &&
705                    dist < (10.0 / panel.getDrawingScale())) {
706                hoverNodeCandidate = node;
707                mindist = dist;
708            }
709        }
710
711        final boolean updated = (selectedNode != hoverNodeCandidate);
712        selectedNode = hoverNodeCandidate;
713        if (selectedNode != null) hoverNode = new Hover(selectedNode,
714                selectedNode.getPosition());
715        else hoverNode = null;
716
717        return updated;
718    }

1323    /**
1324     * candidateLinks をリンクが連なる順に並べ替えたリストを両端のノードと共に返す。
1325     * (candidateLinks に含まれないリンクは接続されていないものと見なす)
1326     * 条件:
1327     * ・閉じたリンクを含んでいないこと。
1328     * ・リンクが分岐している箇所がないこと。
1329     * ・すべてのリンクが途切れなく接続されていること。
1330     */
1331    public HashMap getSeriesLinks(MapLinkTable candidateLinks) throws Exception {
1332        MapLinkTable seriesLinks = new MapLinkTable();
1333        HashSet<MapLink> candidateLinkSet = new HashSet<MapLink>(candidateLinks);
1334        HashMap result = new HashMap();
1335        result.put("linkTable", seriesLinks);
1336
1337        // 端のリンクを見つける
1338        MapLink terminalLink = null;
1339        MapNode terminalNode = null;
1340        for (MapLink link : candidateLinks) {
1341            if (isTerminalNode(link.getFrom(), link, candidateLinkSet)) {
1342                terminalNode = link.getFrom();
1343                terminalLink = link;
1344                break;
1345            }
1346            if (isTerminalNode(link.getTo(), link, candidateLinkSet)) {
1347                terminalNode = link.getTo();
1348                terminalLink = link;
1349                break;
1350            }
1351        }
1352        // 見つからなかった(閉じたリンクのみだった)
1353        if (terminalLink == null) {
1354            throw new Exception("選択したリンクはループしているため、一方通行の設定ができません。");
1355        }
1356        result.put("firstNode", terminalNode);
1357
1358        // リンクをたどりながら、もう一方の端を探す
1359        MapLink link = terminalLink;
1360        MapNode node = link.getOther(terminalNode);
1361        while (true) {
1362            seriesLinks.add(link);
1363            MapLink nextLink = null;
1364            for (MapLink _link : node.getLinks()) {
1365                if (_link != link && candidateLinkSet.contains(_link)) {
1366                    if (nextLink != null) {
1367                        throw new Exception("選択したリンクは分岐を含んでいるため、一方通行の設定ができません。");
1368                    }
1369                    nextLink = _link;
1370                }
1371            }
1372            if (nextLink == null) {
1373                if (seriesLinks.size() < candidateLinks.size()) {

1883    private MapPartGroup duplicateFloor(MapPartGroup group,
1884            int direction,
1885            double stair_width) {
1886        /* see which floor we are in */
1887        Matcher match = group.matchTag("(B?)(\\d+)F");
1888        if (match == null) {
1889            JOptionPane.showMessageDialog(this,
1890                    "No floor number given for this group",
1891                    "Do nothing", JOptionPane.ERROR_MESSAGE);
1892            return null;
1893        }
1894        String floor_string = match.group(0);
1895        int floor = Integer.parseInt(match.group(2));
1896        if (!match.group(1).isEmpty()) { floor = -floor;}
1897        
1898        /* get stairs */
1899        AllStairSets stairs = get_stairset(group, floor_string);
1900        if (stairs == null) return null;
1901
1902        /* next floor */
1903        int next_floor = floor + direction;
1904        if (next_floor == 0) {
1905            if (direction > 0) {
1906                next_floor = 1;
1907            } else {
1908                next_floor = -1;
1909            }
1910        }
1911        String next_floor_string =
1912            ((next_floor < 0) ? "B" : "") + (int)Math.abs(next_floor) + "F";
1913        
1914        double dz = stairs.height_diff * direction;
1915        MapPartGroup next_group = editor.getMap().createGroupNode((MapPartGroup)group.getParent());
1916        next_group.addTag(Itk.intern(next_floor_string)) ;
1917        next_group.setDefaultHeight(group.getDefaultHeight() + dz);
1918        next_group.setScale(group.getScale());
1919        next_group.setMinHeight(group.getMinHeight() + dz);
1920        next_group.setMaxHeight(group.getMaxHeight() + dz);
1921        next_group.setWest(group.getWest());
1922        next_group.setSouth(group.getSouth());
1923        
1924        /* copy nodes */
1925        MapNodeTable nodesToMove = new MapNodeTable();
1926        HashMap<MapNode, MapNode> nodeToNode = new HashMap<MapNode, MapNode>();
1927        for (MapNode node : group.getChildNodes()) {
1928            double height_diff = node.getHeight() - group.getDefaultHeight();
1929            MapNode newNode = editor.getMap().createMapNode(next_group, 
1930                    node.getPosition(),
1931                    next_group.getDefaultHeight() + height_diff);
1932            for (String tag : node.getTags()) {
1933                tag = tag.replaceFirst(floor_string, next_floor_string);
1934                newNode.addTag(Itk.intern(tag)) ;
1935            }
1936            nodesToMove.add(newNode);
1937            nodeToNode.put(node, newNode);
1938        }
1939        
1940        /* copy links */
1941        for (MapLink link : group.getChildLinks()) {
1942            if (group.getChildNodes().contains(link.getFrom()) &&
1943                    group.getChildNodes().contains(link.getTo())) {
1944                MapNode fromNode = nodeToNode.get(link.getFrom());

1952                    tag = tag.replaceFirst(floor_string, next_floor_string);
1953                    newLink.addTag(Itk.intern(tag)) ;
1954                }
1955            }
1956        }
1957        
1958        /* make stairs */
1959        int orig_index = 1; /* make a symlink of UP to next group */
1960        int next_index = 0; /* draw a link from the symlink to DOWN of next group */
1961        if (direction < 0) {
1962            orig_index = 0;
1963            next_index = 1;
1964        }
1965

1967            MapNode fromNode = stair_set.nodes[orig_index];
1968            editor.getMap().createSymLink(next_group, fromNode);
1969            MapNode toNode = nodeToNode.get(stair_set.nodes[next_index]);
1970            MapLink newLink = editor.getMap().createMapLink(next_group,
1971                    fromNode, toNode,
1972                    fromNode.calcDistance(toNode), stair_width);
1973            newLink.addTag(Itk.intern("STAIR_" + stair_set.name + "_"
1974                                      + floor_string + "_"
1975                                      + next_floor_string)) ;
1976        }
1977        return next_group;
