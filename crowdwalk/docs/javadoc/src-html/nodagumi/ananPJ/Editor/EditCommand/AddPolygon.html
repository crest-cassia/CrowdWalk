001package nodagumi.ananPJ.Editor.EditCommand;
002
003import java.util.ArrayList;
004
005import nodagumi.ananPJ.Editor.MapEditor;
006import nodagumi.ananPJ.NetworkMap.MapPartGroup;
007import nodagumi.ananPJ.NetworkMap.NetworkMap;
008import nodagumi.ananPJ.NetworkMap.Polygon.MapPolygon;
009import nodagumi.ananPJ.NetworkMap.Polygon.OuterBoundary;
010import nodagumi.ananPJ.NetworkMap.Polygon.InnerBoundary;
011import nodagumi.ananPJ.NetworkMap.Polygon.TriangleMeshes;
012
013/**
014 * ポリゴンを生成してマップに追加する編集コマンド
015 */
016public class AddPolygon extends EditCommandBase {
017    private String groupId;
018    private String id;
019    private int z_index;
020    private TriangleMeshes triangleMeshes;
021    private OuterBoundary outerBoundary;
022    private ArrayList<InnerBoundary> innerBoundaries;
023
024    /**
025     * コンストラクタ
026     */
027    public AddPolygon(MapPartGroup group, int z_index, TriangleMeshes triangleMeshes) {
028        changeType = ChangeType.POLYGON_VOLUME;
029        groupId = group.getID();
030        this.z_index = z_index;
031        this.triangleMeshes = triangleMeshes;
032    }
033
034    /**
035     * コンストラクタ
036     */
037    public AddPolygon(MapPartGroup group, int z_index, OuterBoundary outerBoundary, ArrayList<InnerBoundary> innerBoundaries) {
038        changeType = ChangeType.POLYGON_VOLUME;
039        groupId = group.getID();
040        this.z_index = z_index;
041        this.outerBoundary = outerBoundary.clone();
042        this.innerBoundaries = innerBoundaries;
043    }
044
045    /**
046     * ID を取得する
047     */
048    public String getId() {
049        return id;
050    }
051
052    /**
053     * 編集を実行する
054     */
055    public boolean invoke(MapEditor editor) {
056        MapPartGroup group = getGroup(editor, groupId);
057        if (group == null) {
058            return false;
059        }
060
061        NetworkMap networkMap = editor.getMap();
062        if (id == null) {
063            id = networkMap.assignNewId();
064        } else {
065            // redo の時は生成済み ID を再利用するので重複チェック
066            if (networkMap.checkObjectId(id)) {
067                editor.displayMessage("Error", getName(), "MapPolygon ID conflicted: ID=" + id);
068                invalid = true;
069                return false;
070            }
071        }
072        MapPolygon polygon = null;
073        if (triangleMeshes != null) {
074            polygon = new MapPolygon(id, z_index, triangleMeshes);
075        } else {
076            polygon = new MapPolygon(id, z_index, outerBoundary, innerBoundaries);
077        }
078        networkMap.insertOBNode(group, polygon);
079
080        invoked = true;
081        return true;
082    }
083
084    /**
085     * 編集を取り消す
086     */
087    public boolean undo(MapEditor editor) {
088        MapPartGroup group = getGroup(editor, groupId);
089        if (group == null) {
090            return false;
091        }
092
093        MapPolygon polygon = getPolygon(editor, id);
094        if (polygon == null) {
095            return false;
096        }
097        editor.getMap().removeOBNode(group, polygon);
098
099        invoked = false;
100        return true;
101    }
102
103    /**
104     * このコマンドの文字列表現を取得する
105     */
106    public String toString() {
107        if (triangleMeshes != null) {
108            return getName() + "[type=TriangleMeshes, group=" + groupId + ", z_index=" + z_index + "]";
109        } else {
110            return getName() + "[type=PlanePolygon, group=" + groupId + ", z_index=" + z_index + ", innerBoundaries=" + innerBoundaries.size() + "]";
111        }
112    }
113}
