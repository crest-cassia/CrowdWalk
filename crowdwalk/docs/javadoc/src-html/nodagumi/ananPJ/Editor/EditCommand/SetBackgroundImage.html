001package nodagumi.ananPJ.Editor.EditCommand;
002
003import java.io.File;
004
005import javafx.geometry.Point2D;
006import javafx.scene.image.Image;
007
008import nodagumi.ananPJ.Editor.MapEditor;
009import nodagumi.ananPJ.Editor.EditorCanvas;
010import nodagumi.ananPJ.NetworkMap.MapPartGroup;
011
012/**
013 * 背景画像をセットする編集コマンド
014 */
015public class SetBackgroundImage extends EditCommandBase {
016    private String groupId;
017    private String fileName;
018    private double tx;
019    private double ty;
020    private double sx;
021    private double sy;
022    private double r;
023    private String originalFileName;
024    private double originalTx;
025    private double originalTy;
026    private double originalSx;
027    private double originalSy;
028    private double originalR;
029    private Image originalImage;
030
031    /**
032     * コンストラクタ
033     */
034    public SetBackgroundImage(MapPartGroup group, String fileName, double tx, double ty) {
035        groupId = group.getID();
036        this.fileName = fileName;
037        this.tx = tx;
038        this.ty = ty;
039    }
040
041    /**
042     * 編集を実行する
043     */
044    public boolean invoke(MapEditor editor) {
045        MapPartGroup group = getGroup(editor, groupId);
046        if (group == null) {
047            return false;
048        }
049
050        String filePath = new File(editor.getDir(), fileName).toURI().toString();
051        Image image = new Image(filePath);
052        if (image.isError()) {
053            editor.displayMessage("Error", getName(), "Illegal image file: " + fileName);
054            invalid = true;
055            return false;
056        }
057
058        // 画像の幅が画面の幅の半分になるスケール
059        EditorCanvas canvas = editor.getFrame().getCanvas();
060        Point2D upperLeft = canvas.pointConvertCanvasToMap(0.0, 0.0);
061        Point2D upperRight = canvas.pointConvertCanvasToMap(canvas.getWidth(), 0.0);
062        sx = Math.abs(upperRight.getX() - upperLeft.getX()) / 2.0 / image.getWidth();
063        sy = sx;
064        r = 0.0;
065
066        originalFileName = group.getImageFileName();
067        originalTx = group.tx;
068        originalTy = group.ty;
069        originalSx = group.sx;
070        originalSy = group.sy;
071        originalR = group.r;
072        originalImage = editor.getBackgroundImage(group);   // これは常に null になるはず
073
074        group.setImageFileName(fileName);
075        group.tx = tx;
076        group.ty = ty;
077        group.sx = sx;
078        group.sy = sy;
079        group.r = r;
080
081        editor.setBackgroundImage(group, image);
082
083        invoked = true;
084        return true;
085    }
086
087    /**
088     * 編集を取り消す
089     */
090    public boolean undo(MapEditor editor) {
091        MapPartGroup group = getGroup(editor, groupId);
092        if (group == null) {
093            return false;
094        }
095
096        group.setImageFileName(originalFileName);
097        group.tx = originalTx;
098        group.ty = originalTy;
099        group.sx = originalSx;
100        group.sy = originalSy;
101        group.r = originalR;
102        editor.setBackgroundImage(group, originalImage);
103
104        invoked = false;
105        return true;
106    }
107
108    /**
109     * このコマンドの文字列表現を取得する
110     */
111    public String toString() {
112        return getName() + "[group=" + groupId + ", fileName=\"" + fileName + "\", tx=" + tx + ", ty=" + ty + ", sx=" + sx + ", sy=" + sy + ", r=" + r + "]";
113    }
114}
