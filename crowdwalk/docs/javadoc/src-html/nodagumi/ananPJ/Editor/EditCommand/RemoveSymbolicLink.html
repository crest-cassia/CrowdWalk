001package nodagumi.ananPJ.Editor.EditCommand;
002
003import nodagumi.ananPJ.Editor.MapEditor;
004import nodagumi.ananPJ.NetworkMap.MapPartGroup;
005import nodagumi.ananPJ.NetworkMap.NetworkMap;
006import nodagumi.ananPJ.NetworkMap.OBNode;
007import nodagumi.ananPJ.NetworkMap.OBNodeSymbolicLink;
008
009/**
010 * シンボリックリンクを削除する編集コマンド
011 */
012public class RemoveSymbolicLink extends EditCommandBase {
013    private String groupId;
014    private String id;
015    private String originalId;
016
017    /**
018     * コンストラクタ
019     */
020    public RemoveSymbolicLink(MapPartGroup group, OBNodeSymbolicLink symlink) {
021        changeType = ChangeType.SYMBOLIC_LINK_VOLUME;
022        groupId = group.getID();
023        id = symlink.getID();
024        originalId = symlink.getOriginal().getID();
025    }
026
027    /**
028     * 編集を実行する
029     */
030    public boolean invoke(MapEditor editor) {
031        MapPartGroup group = getGroup(editor, groupId);
032        if (group == null) {
033            return false;
034        }
035
036        OBNodeSymbolicLink symlink = getSymbolicLink(editor, id);
037        if (symlink == null) {
038            return false;
039        }
040        editor.getMap().removeOBNode(group, symlink);
041
042        invoked = true;
043        return true;
044    }
045
046    /**
047     * 編集を取り消す
048     */
049    public boolean undo(MapEditor editor) {
050        MapPartGroup group = getGroup(editor, groupId);
051        if (group == null) {
052            return false;
053        }
054
055        NetworkMap networkMap = editor.getMap();
056        if (networkMap.checkObjectId(id)) {
057            editor.displayMessage("Error", getName(), "Symbolic link ID conflicted: ID=" + id);
058            invalid = true;
059            return false;
060        }
061        OBNode original = getOBNode(editor, originalId);
062        if (original == null) {
063            return false;
064        }
065        OBNodeSymbolicLink symlink = new OBNodeSymbolicLink(id, original);
066        networkMap.insertOBNode(group, symlink);
067
068        invoked = false;
069        return true;
070    }
071
072    /**
073     * このコマンドの文字列表現を取得する
074     */
075    public String toString() {
076        return getName() + "[group=" + groupId + ", original=" + originalId + "]";
077    }
078}
