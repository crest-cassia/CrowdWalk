001package nodagumi.ananPJ.Editor.EditCommand;
002
003import javafx.scene.image.Image;
004
005import nodagumi.ananPJ.Editor.MapEditor;
006import nodagumi.ananPJ.NetworkMap.MapPartGroup;
007
008/**
009 * 背景画像を削除する編集コマンド
010 */
011public class RemoveBackgroundImage extends EditCommandBase {
012    private String groupId;
013    private String fileName;
014    private double tx;
015    private double ty;
016    private String originalFileName;
017    private double originalTx;
018    private double originalTy;
019    private double originalSx;
020    private double originalSy;
021    private double originalR;
022    private Image originalImage;
023
024    /**
025     * コンストラクタ
026     */
027    public RemoveBackgroundImage(MapPartGroup group) {
028        groupId = group.getID();
029    }
030
031    /**
032     * 編集を実行する
033     */
034    public boolean invoke(MapEditor editor) {
035        MapPartGroup group = getGroup(editor, groupId);
036        if (group == null) {
037            return false;
038        }
039
040        originalFileName = group.getImageFileName();
041        originalTx = group.tx;
042        originalTy = group.ty;
043        originalSx = group.sx;
044        originalSy = group.sy;
045        originalR = group.r;
046        // TODO: 画像オブジェクトをそのまま保存しているのでメモリを消費しすぎる様ならば別の方法を取る
047        originalImage = editor.getBackgroundImage(group);
048
049        group.setImageFileName("");
050        group.tx = 0.0;
051        group.ty = 0.0;
052        group.sx = 1.0;
053        group.sy = 1.0;
054        group.r = 0.0;
055        editor.setBackgroundImage(group, null);
056
057        invoked = true;
058        return true;
059    }
060
061    /**
062     * 編集を取り消す
063     */
064    public boolean undo(MapEditor editor) {
065        MapPartGroup group = getGroup(editor, groupId);
066        if (group == null) {
067            return false;
068        }
069
070        group.setImageFileName(originalFileName);
071        group.tx = originalTx;
072        group.ty = originalTy;
073        group.sx = originalSx;
074        group.sy = originalSy;
075        group.r = originalR;
076        editor.setBackgroundImage(group, originalImage);
077
078        invoked = false;
079        return true;
080    }
081
082    /**
083     * このコマンドの文字列表現を取得する
084     */
085    public String toString() {
086        return getName() + "[group=" + groupId + ", fileName=\"" + fileName + "\", tx=" + tx + ", ty=" + ty + "]";
087    }
088}
