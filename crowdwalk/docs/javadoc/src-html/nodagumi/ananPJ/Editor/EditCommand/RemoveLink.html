001package nodagumi.ananPJ.Editor.EditCommand;
002
003import java.util.ArrayList;
004
005import nodagumi.ananPJ.Editor.MapEditor;
006import nodagumi.ananPJ.NetworkMap.Link.MapLink;
007import nodagumi.ananPJ.NetworkMap.MapPartGroup;
008import nodagumi.ananPJ.NetworkMap.NetworkMap;
009import nodagumi.ananPJ.NetworkMap.Node.MapNode;
010import nodagumi.Itk.Itk;
011
012/**
013 * リンクを削除する編集コマンド
014 */
015public class RemoveLink extends EditCommandBase {
016    private String groupId;
017    private String id;
018    private String fromNodeId;
019    private String toNodeId;
020    private double length;
021    private double width;
022    private ArrayList<String> tags;
023
024    /**
025     * コンストラクタ
026     */
027    public RemoveLink(MapLink link) {
028        changeType = ChangeType.LINK_VOLUME;
029        groupId = ((MapPartGroup)link.getParent()).getID();
030        id = link.getID();
031        fromNodeId = link.getFrom().getID();
032        toNodeId = link.getTo().getID();
033        length = link.getLength();
034        width = link.getWidth();
035        tags = new ArrayList<String>(link.getTags());
036    }
037
038    /**
039     * 編集を実行する
040     */
041    public boolean invoke(MapEditor editor) {
042        MapPartGroup group = getGroup(editor, groupId);
043        if (group == null) {
044            return false;
045        }
046
047        MapLink link = getLink(editor, id);
048        if (link == null) {
049            return false;
050        }
051        link.prepareRemove();
052        editor.getMap().removeOBNode(group, link);
053
054        invoked = true;
055        return true;
056    }
057
058    /**
059     * 編集を取り消す
060     */
061    public boolean undo(MapEditor editor) {
062        MapPartGroup group = getGroup(editor, groupId);
063        if (group == null) {
064            return false;
065        }
066        MapNode fromNode = getNode(editor, fromNodeId);
067        MapNode toNode = getNode(editor, toNodeId);
068        if (fromNode == null || toNode == null) {
069            editor.displayMessage("Error", getName(), "\"from node\" or \"to node\" not found: ID=" + fromNodeId + ", " + toNodeId);
070            invalid = true;
071            return false;
072        }
073
074        NetworkMap networkMap = editor.getMap();
075        if (networkMap.checkObjectId(id)) {
076            editor.displayMessage("Error", getName(), "Link ID conflicted: ID=" + id);
077            invalid = true;
078            return false;
079        }
080        MapLink link = new MapLink(id, fromNode, toNode, length, width);
081        for (String tag : tags) {
082            link.addTag(Itk.intern(tag));
083        }
084        link.prepareAdd();
085        networkMap.insertOBNode(group, link);
086
087        invoked = false;
088        return true;
089    }
090
091    /**
092     * このコマンドの文字列表現を取得する
093     */
094    public String toString() {
095        return getName() + "[group=" + groupId + ", fromNode=" + fromNodeId + ", toNode=" + toNodeId + ", length=" + length + ", width=" + width + "]";
096    }
097}
