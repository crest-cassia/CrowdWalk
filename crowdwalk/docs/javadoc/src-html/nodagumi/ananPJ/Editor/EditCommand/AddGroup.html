001package nodagumi.ananPJ.Editor.EditCommand;
002
003import nodagumi.ananPJ.Editor.MapEditor;
004import nodagumi.ananPJ.NetworkMap.MapPartGroup;
005import nodagumi.ananPJ.NetworkMap.NetworkMap;
006import nodagumi.Itk.Itk;
007
008/**
009 * 新しいグループを生成してルートグループ下に追加する編集コマンド
010 */
011public class AddGroup extends EditCommandBase {
012    private String id = null;
013    private String tag;
014    private double defaultHeight;
015    private int zone;
016
017    /**
018     * コンストラクタ
019     */
020    public AddGroup(String tag, double defaultHeight, int zone) {
021        changeType = ChangeType.GROUP_VOLUME;
022        this.tag = Itk.intern(tag);
023        this.defaultHeight = defaultHeight;
024        this.zone = zone;
025    }
026
027    /**
028     * ID を取得する
029     */
030    public String getId() {
031        return id;
032    }
033
034    /**
035     * 編集を実行する
036     */
037    public boolean invoke(MapEditor editor) {
038        NetworkMap networkMap = editor.getMap();
039        if (id == null) {
040            id = networkMap.assignNewId();
041        } else {
042            // redo の時は生成済み ID を再利用するので重複チェック
043            if (networkMap.checkObjectId(id)) {
044                editor.displayMessage("Error", getName(), "Group ID conflicted: ID=" + id);
045                invalid = true;
046                return false;
047            }
048        }
049        MapPartGroup group = new MapPartGroup(id);
050        group.setDefaultHeight(defaultHeight);
051        group.setZone(zone);
052        networkMap.insertOBNode((MapPartGroup)networkMap.getRoot(), group);
053        group.addTag(tag);
054
055        invoked = true;
056        return true;
057    }
058
059    /**
060     * 編集を取り消す
061     */
062    public boolean undo(MapEditor editor) {
063        MapPartGroup group = getGroup(editor, id);
064        if (group == null) {
065            return false;
066        }
067        editor.getMap().removeOBNode((MapPartGroup)editor.getMap().getRoot(), group);
068
069        invoked = false;
070        return true;
071    }
072
073    /**
074     * このコマンドの文字列表現を取得する
075     */
076    public String toString() {
077        return getName() + "[tag=" + tag + ", defaultHeight=" + defaultHeight + ", zone=" + zone + "]";
078    }
079}
