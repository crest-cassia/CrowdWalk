001package nodagumi.ananPJ.Editor.EditCommand;
002
003import nodagumi.ananPJ.Editor.MapEditor;
004import nodagumi.ananPJ.NetworkMap.Link.MapLink;
005import nodagumi.ananPJ.NetworkMap.MapPartGroup;
006import nodagumi.ananPJ.NetworkMap.NetworkMap;
007import nodagumi.ananPJ.NetworkMap.Node.MapNode;
008
009/**
010 * リンクを生成してマップに追加する編集コマンド
011 */
012public class AddLink extends EditCommandBase {
013    private String groupId;
014    private String id = null;
015    private String fromNodeId;
016    private String toNodeId;
017    private double length;
018    private double width;
019
020    /**
021     * コンストラクタ
022     */
023    public AddLink(String id, MapNode fromNode, MapNode toNode, double length, double width) {
024        this.id = id;
025        changeType = ChangeType.LINK_VOLUME;
026        groupId = ((MapPartGroup)fromNode.getParent()).getID();
027        fromNodeId = fromNode.getID();
028        toNodeId = toNode.getID();
029        this.length = length;
030        this.width = width;
031    }
032
033    /**
034     * コンストラクタ
035     */
036    public AddLink(MapNode fromNode, MapNode toNode, double length, double width) {
037        changeType = ChangeType.LINK_VOLUME;
038        groupId = ((MapPartGroup)fromNode.getParent()).getID();
039        fromNodeId = fromNode.getID();
040        toNodeId = toNode.getID();
041        this.length = length;
042        this.width = width;
043    }
044
045    /**
046     * ID を取得する
047     */
048    public String getId() {
049        return id;
050    }
051
052    /**
053     * 編集を実行する
054     */
055    public boolean invoke(MapEditor editor) {
056        MapPartGroup group = getGroup(editor, groupId);
057        if (group == null) {
058            return false;
059        }
060        MapNode fromNode = getNode(editor, fromNodeId);
061        MapNode toNode = getNode(editor, toNodeId);
062        if (fromNode == null || toNode == null) {
063            editor.displayMessage("Error", getName(), "\"from node\" or \"to node\" not found: ID=" + fromNodeId + ", " + toNodeId);
064            invalid = true;
065            return false;
066        }
067
068        NetworkMap networkMap = editor.getMap();
069        if (id == null) {
070            id = networkMap.assignNewId();
071        } else {
072            // redo の時は生成済み ID を再利用するので重複チェック
073            if (networkMap.checkObjectId(id)) {
074                editor.displayMessage("Error", getName(), "Link ID conflicted: ID=" + id);
075                invalid = true;
076                return false;
077            }
078        }
079        MapLink link = new MapLink(id, fromNode, toNode, length, width);
080        link.prepareAdd();
081        networkMap.insertOBNode(group, link);
082
083        invoked = true;
084        return true;
085    }
086
087    /**
088     * 編集を取り消す
089     */
090    public boolean undo(MapEditor editor) {
091        MapPartGroup group = getGroup(editor, groupId);
092        if (group == null) {
093            return false;
094        }
095
096        MapLink link = getLink(editor, id);
097        if (link == null) {
098            return false;
099        }
100        link.prepareRemove();
101        editor.getMap().removeOBNode(group, link);
102
103        invoked = false;
104        return true;
105    }
106
107    /**
108     * このコマンドの文字列表現を取得する
109     */
110    public String toString() {
111        return getName() + "[group=" + groupId + ", fromNode=" + fromNodeId + ", toNode=" + toNodeId + ", length=" + length + ", width=" + width + "]";
112    }
113}
