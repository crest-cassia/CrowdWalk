001package nodagumi.ananPJ.Editor.EditCommand;
002
003import nodagumi.ananPJ.Editor.MapEditor;
004import nodagumi.ananPJ.NetworkMap.Link.MapLink;
005
006/**
007 * リンクの長さをセットする編集コマンド
008 */
009public class SetLength extends EditCommandBase {
010    private String id;
011    private double length;
012    private double originalLength;
013
014    /**
015     * コンストラクタ
016     */
017    public SetLength(MapLink link, double length) {
018        changeType = ChangeType.LINK_PARAM;
019        id = link.getID();
020        this.length = length;
021    }
022
023    /**
024     * 編集を実行する
025     */
026    public boolean invoke(MapEditor editor) {
027        MapLink link = getLink(editor, id);
028        if (link == null) {
029            return false;
030        }
031
032        originalLength = link.getLength();
033        link.setLength(length);
034
035        invoked = true;
036        return true;
037    }
038
039    /**
040     * 編集を取り消す
041     */
042    public boolean undo(MapEditor editor) {
043        MapLink link = getLink(editor, id);
044        if (link == null) {
045            return false;
046        }
047
048        link.setLength(originalLength);
049
050        invoked = false;
051        return true;
052    }
053
054    /**
055     * このコマンドの文字列表現を取得する
056     */
057    public String toString() {
058        return getName() + "[id=" + id + ", length=" + length + "]";
059    }
060}
