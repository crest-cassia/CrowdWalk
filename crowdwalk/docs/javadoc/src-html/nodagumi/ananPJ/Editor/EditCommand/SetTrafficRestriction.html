001package nodagumi.ananPJ.Editor.EditCommand;
002
003import nodagumi.ananPJ.Editor.MapEditor;
004import nodagumi.ananPJ.NetworkMap.Link.MapLink;
005
006/**
007 * 通行制限をセットする編集コマンド
008 */
009public class SetTrafficRestriction extends EditCommandBase {
010    private String id;
011    private boolean oneWayForward;
012    private boolean oneWayBackward;
013    private boolean roadClosed;
014    private boolean originalOneWayForward;
015    private boolean originalOneWayBackward;
016    private boolean originalRoadClosed;
017
018    /**
019     * コンストラクタ
020     */
021    public SetTrafficRestriction(MapLink link, boolean oneWayForward, boolean oneWayBackward, boolean roadClosed) {
022        changeType = ChangeType.LINK_PARAM;
023        id = link.getID();
024        this.oneWayForward = oneWayForward;
025        this.oneWayBackward = oneWayBackward;
026        this.roadClosed = roadClosed;
027    }
028
029    /**
030     * 編集を実行する
031     */
032    public boolean invoke(MapEditor editor) {
033        MapLink link = getLink(editor, id);
034        if (link == null) {
035            return false;
036        }
037
038        originalOneWayForward = link.isOneWayForward();
039        originalOneWayBackward = link.isOneWayBackward();
040        originalRoadClosed = link.isRoadClosed();
041        link.setOneWayForward(oneWayForward);
042        link.setOneWayBackward(oneWayBackward);
043        link.setRoadClosed(roadClosed);
044
045        invoked = true;
046        return true;
047    }
048
049    /**
050     * 編集を取り消す
051     */
052    public boolean undo(MapEditor editor) {
053        MapLink link = getLink(editor, id);
054        if (link == null) {
055            return false;
056        }
057
058        link.setOneWayForward(originalOneWayForward);
059        link.setOneWayBackward(originalOneWayBackward);
060        link.setRoadClosed(originalRoadClosed);
061
062        invoked = false;
063        return true;
064    }
065
066    /**
067     * このコマンドの文字列表現を取得する
068     */
069    public String toString() {
070        return getName() + "[id=" + id + ", oneWayForward=" + oneWayForward + ", oneWayBackward=" + oneWayBackward + ", roadClosed=" + roadClosed + "]";
071    }
072}
