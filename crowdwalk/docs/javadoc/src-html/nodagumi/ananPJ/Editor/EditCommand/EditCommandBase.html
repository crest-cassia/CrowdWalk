001package nodagumi.ananPJ.Editor.EditCommand;
002
003import nodagumi.ananPJ.Editor.MapEditor;
004import nodagumi.ananPJ.NetworkMap.Area.MapArea;
005import nodagumi.ananPJ.NetworkMap.Link.MapLink;
006import nodagumi.ananPJ.NetworkMap.MapPartGroup;
007import nodagumi.ananPJ.NetworkMap.Node.MapNode;
008import nodagumi.ananPJ.NetworkMap.OBNode;
009import nodagumi.ananPJ.NetworkMap.OBNodeSymbolicLink;
010import nodagumi.ananPJ.NetworkMap.Polygon.MapPolygon;
011
012/**
013 * Undo が可能な編集コマンドの抽象クラス
014 */
015public abstract class EditCommandBase {
016    /**
017     * 変更点の種類
018     */
019    public static enum ChangeType {
020        NONE,
021        GROUP_PARAM,
022        GROUP_TAG,
023        GROUP_VOLUME,
024        NODE_PARAM,
025        NODE_TAG,
026        NODE_VOLUME,
027        LINK_PARAM,
028        LINK_TAG,
029        LINK_VOLUME,
030        AREA_PARAM,
031        AREA_TAG,
032        AREA_VOLUME,
033        POLYGON_PARAM,
034        POLYGON_TAG,
035        POLYGON_VOLUME,
036        SYMBOLIC_LINK_VOLUME
037    }
038
039    protected long time = 0;
040    protected boolean invoked = false;
041    protected boolean invalid = false;
042    protected boolean first = false;
043    protected boolean last = false;
044    protected boolean heightChangeable = false;
045    protected ChangeType changeType = ChangeType.NONE;
046
047    /**
048     * 編集を実行する
049     */
050    public abstract boolean invoke(MapEditor editor);
051
052    /**
053     * 編集を取り消す
054     */
055    public abstract boolean undo(MapEditor editor);
056
057    /**
058     * このコマンドのクラス名を取得する
059     */
060    public String getName() {
061        return this.getClass().getSimpleName();
062    }
063
064    /**
065     * このコマンドのミリ秒で表される実行時刻をセットする
066     */
067    public void setTime() {
068        time = System.currentTimeMillis();
069    }
070
071    /**
072     * このコマンドのミリ秒で表される実行時刻を取得する
073     */
074    public long getTime() {
075        return time;
076    }
077
078    /**
079     * このコマンドは実行済みか?
080     */
081    public boolean isInvoked() {
082        return invoked;
083    }
084
085    /**
086     * このコマンドは無効か?
087     */
088    public boolean isInvalid() {
089        return invalid;
090    }
091
092    /**
093     * 編集コマンドブロックの最初のコマンドかどうかの設定
094     */
095    public void setFirst(boolean first) {
096        this.first = first;
097    }
098
099    /**
100     * 編集コマンドブロックの最初のコマンドか?
101     */
102    public boolean isFirst() {
103        return first;
104    }
105
106    /**
107     * 編集コマンドブロックの最後のコマンドかどうかの設定
108     */
109    public void setLast(boolean last) {
110        this.last = last;
111    }
112
113    /**
114     * 編集コマンドブロックの最後のコマンドか?
115     */
116    public boolean isLast() {
117        return last;
118    }
119
120    /**
121     * 標高可変か?
122     */
123    public boolean isHeightChangeable() {
124        return heightChangeable;
125    }
126
127    /**
128     * 変更点を取得する
129     */
130    public ChangeType getChangeType() {
131        return changeType;
132    }
133
134    /**
135     * 続けて実行したコマンドとマージすることは可能か?
136     *
137     * @param command このコマンドの直後に実行されたコマンド
138     */
139    public boolean isMergeable(EditCommandBase command) {
140        return false;
141    }
142
143    /**
144     * 続けて実行したコマンドとマージする
145     *
146     * @param command このコマンドの直後に実行されたコマンド
147     */
148    public void mergeTo(EditCommandBase command) {}
149
150    /**
151     * OBNode を取得する
152     */
153    public OBNode getOBNode(MapEditor editor, String id) {
154        OBNode obNode = editor.getMap().getObject(id);
155        if (obNode == null) {
156            editor.displayMessage("Error", getName(), "OBNode not found: ID=" + id);
157            invalid = true;
158            return null;
159        }
160        return obNode;
161    }
162
163    /**
164     * グループを取得する
165     */
166    public MapPartGroup getGroup(MapEditor editor, String id) {
167        OBNode obNode = getOBNode(editor, id);
168        if (obNode == null) {
169            return null;
170        }
171        if (obNode.getNodeType() != OBNode.NType.GROUP) {
172            editor.displayMessage("Error", getName(), "Group was lost: ID=" + id);
173            invalid = true;
174            return null;
175        }
176        return (MapPartGroup)obNode;
177    }
178
179    /**
180     * ノードを取得する
181     */
182    public MapNode getNode(MapEditor editor, String id) {
183        OBNode obNode = getOBNode(editor, id);
184        if (obNode == null) {
185            return null;
186        }
187        if (obNode.getNodeType() != OBNode.NType.NODE) {
188            editor.displayMessage("Error", getName(), "Node was lost: ID=" + id);
189            invalid = true;
190            return null;
191        }
192        return (MapNode)obNode;
193    }
194
195    /**
196     * リンクを取得する
197     */
198    public MapLink getLink(MapEditor editor, String id) {
199        OBNode obNode = getOBNode(editor, id);
200        if (obNode == null) {
201            return null;
202        }
203        if (obNode.getNodeType() != OBNode.NType.LINK) {
204            editor.displayMessage("Error", getName(), "Link was lost: ID=" + id);
205            invalid = true;
206            return null;
207        }
208        return (MapLink)obNode;
209    }
210
211    /**
212     * エリアを取得する
213     */
214    public MapArea getArea(MapEditor editor, String id) {
215        OBNode obNode = getOBNode(editor, id);
216        if (obNode == null) {
217            return null;
218        }
219        if (obNode.getNodeType() != OBNode.NType.AREA) {
220            editor.displayMessage("Error", getName(), "Area was lost: ID=" + id);
221            invalid = true;
222            return null;
223        }
224        return (MapArea)obNode;
225    }
226
227    /**
228     * ポリゴンを取得する
229     */
230    public MapPolygon getPolygon(MapEditor editor, String id) {
231        OBNode obNode = getOBNode(editor, id);
232        if (obNode == null) {
233            return null;
234        }
235        if (obNode.getNodeType() != OBNode.NType.POLYGON) {
236            editor.displayMessage("Error", getName(), "Polygon was lost: ID=" + id);
237            invalid = true;
238            return null;
239        }
240        return (MapPolygon)obNode;
241    }
242
243    /**
244     * シンボリックリンクを取得する
245     */
246    public OBNodeSymbolicLink getSymbolicLink(MapEditor editor, String id) {
247        OBNode obNode = getOBNode(editor, id);
248        if (obNode == null) {
249            return null;
250        }
251        if (obNode.getNodeType() != OBNode.NType.SYMLINK) {
252            editor.displayMessage("Error", getName(), "SymbolicLink was lost: ID=" + id);
253            invalid = true;
254            return null;
255        }
256        return (OBNodeSymbolicLink)obNode;
257    }
258
259    /**
260     * このコマンドの文字列表現を取得する
261     */
262    public String toString() {
263        return getName();
264    }
265}
