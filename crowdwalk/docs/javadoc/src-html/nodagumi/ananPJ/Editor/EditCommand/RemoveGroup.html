001package nodagumi.ananPJ.Editor.EditCommand;
002
003import java.awt.geom.Point2D;
004
005import nodagumi.ananPJ.Editor.MapEditor;
006import nodagumi.ananPJ.NetworkMap.MapPartGroup;
007import nodagumi.ananPJ.NetworkMap.NetworkMap;
008import nodagumi.Itk.Itk;
009
010/**
011 * グループを削除する編集コマンド
012 */
013public class RemoveGroup extends EditCommandBase {
014    private String id;
015    private String tag;
016    private double defaultHeight;
017    private double scale;
018    private int zone;
019    private String imageFileName;
020    private double tx;
021    private double ty;
022    private double sx;
023    private double sy;
024    private double r;
025    private Point2D pNorthWest = new Point2D.Double();
026    private Point2D pSouthEast = new Point2D.Double();
027    private double minHeight;
028    private double maxHeight;
029
030    /**
031     * コンストラクタ
032     */
033    public RemoveGroup(MapPartGroup group) {
034        changeType = ChangeType.GROUP_VOLUME;
035        id = group.getID();
036        tag = Itk.intern(group.getTagString());
037        defaultHeight = group.getDefaultHeight();
038        scale = group.getScale();
039        zone = group.getZone();
040        imageFileName = group.getImageFileName();
041        tx = group.tx;
042        ty = group.ty;
043        sx = group.sx;
044        sy = group.sy;
045        r = group.r;
046        pNorthWest.setLocation(group.getWest(), group.getNorth());
047        pSouthEast.setLocation(group.getEast(), group.getSouth());
048        minHeight = group.getMinHeight();
049        maxHeight = group.getMaxHeight();
050    }
051
052    /**
053     * 編集を実行する
054     */
055    public boolean invoke(MapEditor editor) {
056        MapPartGroup group = getGroup(editor, id);
057        if (group == null) {
058            return false;
059        }
060
061        NetworkMap networkMap = editor.getMap();
062        networkMap.removeOBNode((MapPartGroup)networkMap.getRoot(), group);
063
064        invoked = true;
065        return true;
066    }
067
068    /**
069     * 編集を取り消す
070     */
071    public boolean undo(MapEditor editor) {
072        NetworkMap networkMap = editor.getMap();
073        if (networkMap.checkObjectId(id)) {
074            editor.displayMessage("Error", getName(), "Group ID conflicted: ID=" + id);
075            invalid = true;
076            return false;
077        }
078        MapPartGroup group = new MapPartGroup(id, pNorthWest, pSouthEast, 0.0, imageFileName);
079        group.setDefaultHeight(defaultHeight);
080        group.setScale(scale);
081        group.setZone(zone);
082        group.tx = tx;
083        group.ty = ty;
084        group.sx = sx;
085        group.sy = sy;
086        group.r = r;
087        group.setMinHeight(minHeight);
088        group.setMaxHeight(maxHeight);
089        networkMap.insertOBNode((MapPartGroup)networkMap.getRoot(), group);
090        group.addTag(tag);
091
092        invoked = false;
093        return true;
094    }
095
096    /**
097     * このコマンドの文字列表現を取得する
098     */
099    public String toString() {
100        return getName() + "[id=" + id + ", tag=" + tag + ", defaultHeight=" + defaultHeight + ", scale=" + scale + ", zone=" + zone + ", imageFileName=" + imageFileName + ", tx=" + tx + ", ty=" + ty + ", sx=" + sx + ", sy=" + sy + ", r=" + r + "]";
101    }
102}
