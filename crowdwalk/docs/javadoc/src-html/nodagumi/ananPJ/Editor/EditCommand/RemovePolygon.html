001package nodagumi.ananPJ.Editor.EditCommand;
002
003import java.util.ArrayList;
004
005import nodagumi.ananPJ.Editor.MapEditor;
006import nodagumi.ananPJ.NetworkMap.MapPartGroup;
007import nodagumi.ananPJ.NetworkMap.NetworkMap;
008import nodagumi.ananPJ.NetworkMap.Polygon.MapPolygon;
009import nodagumi.ananPJ.NetworkMap.Polygon.OuterBoundary;
010import nodagumi.ananPJ.NetworkMap.Polygon.InnerBoundary;
011import nodagumi.ananPJ.NetworkMap.Polygon.TriangleMeshes;
012import nodagumi.Itk.Itk;
013
014/**
015 * ポリゴンを削除する編集コマンド
016 */
017public class RemovePolygon extends EditCommandBase {
018    private String groupId;
019    private String id;
020    private int z_index;
021    private TriangleMeshes triangleMeshes;
022    private OuterBoundary outerBoundary;
023    private ArrayList<InnerBoundary> innerBoundaries;
024    private ArrayList<String> tags;
025
026    /**
027     * コンストラクタ
028     */
029    public RemovePolygon(MapPolygon polygon) {
030        changeType = ChangeType.POLYGON_VOLUME;
031        groupId = ((MapPartGroup)polygon.getParent()).getID();
032        id = polygon.getID();
033        z_index = polygon.getZIndex();
034        if (polygon.isTriangleMeshes()) {
035            triangleMeshes = polygon.getTriangleMeshes();
036        } else {
037            outerBoundary = polygon.getOuterBoundary();
038            innerBoundaries = polygon.getInnerBoundaries();
039        }
040        tags = new ArrayList<String>(polygon.getTags());
041    }
042
043    /**
044     * 編集を実行する
045     */
046    public boolean invoke(MapEditor editor) {
047        MapPartGroup group = getGroup(editor, groupId);
048        if (group == null) {
049            return false;
050        }
051
052        MapPolygon polygon = getPolygon(editor, id);
053        if (polygon == null) {
054            return false;
055        }
056        editor.getMap().removeOBNode(group, polygon);
057
058        invoked = true;
059        return true;
060    }
061
062    /**
063     * 編集を取り消す
064     */
065    public boolean undo(MapEditor editor) {
066        MapPartGroup group = getGroup(editor, groupId);
067        if (group == null) {
068            return false;
069        }
070
071        NetworkMap networkMap = editor.getMap();
072        if (networkMap.checkObjectId(id)) {
073            editor.displayMessage("Error", getName(), "Polygon ID conflicted: ID=" + id);
074            invalid = true;
075            return false;
076        }
077        MapPolygon polygon = null;
078        if (triangleMeshes != null) {
079            polygon = new MapPolygon(id, z_index, triangleMeshes);
080        } else {
081            polygon = new MapPolygon(id, z_index, outerBoundary, innerBoundaries);
082        }
083        for (String tag : tags) {
084            polygon.addTag(Itk.intern(tag));
085        }
086        networkMap.insertOBNode(group, polygon);
087
088        invoked = false;
089        return true;
090    }
091
092    /**
093     * このコマンドの文字列表現を取得する
094     */
095    public String toString() {
096        if (triangleMeshes != null) {
097            return getName() + "[type=TriangleMeshes, group=" + groupId + ", z_index=" + z_index + "]";
098        } else {
099            return getName() + "[type=PlanePolygon, group=" + groupId + ", z_index=" + z_index + ", innerBoundaries=" + innerBoundaries.size() + "]";
100        }
101    }
102}
