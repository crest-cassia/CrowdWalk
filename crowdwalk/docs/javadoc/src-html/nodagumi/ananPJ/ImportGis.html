054     * Shapefile 形式の地図を読み込んでモデルを作成する． 
055     * 以下の2種類のデータ形式に対応
056     *
057     * 1. ナビゲーション道路地図2004 （shape版） アルプス社
058     *
059     * 読み込みの対象ファイル：
060     *   RNxxxxxxA.shp 道路リンク（高速・有料道路・国道）
061     *   RNxxxxxxB.shp 主要地方道、一般県道、幅員13.0m以上の道路のリンクデータ（道路区間データ）
062     *   RNxxxxxxC.shp 全道路のノードデータ（道路点データ）
063     *
064     * 読み込みデータ：
065     *   ノード１番号 nd1 文字 5 あり 半角数字
066     *   ノード２番号 nd2 文字 5 あり 半角数字
067     *   リンク長 lk_length 整数
068     *   道路幅員区分コード width_tpcd 文字
069     *     1 幅員13．0ｍ以上
070     *     2 幅員 5．5ｍ以上～13．0ｍ未満
071     *     3 幅員 3．0ｍ以上～ 5．5ｍ未満
072     *     4 幅員 3．0ｍ未満
073     *     0 未調査
074     *
075     * 2. 拡張版全国デジタル道路地図データベース （shape版） 2013 住友電気工業株式会社・住友電工システムソリューション株式会社
076     *
077     * 読み込みの対象ファイル：
078     *   全道路リンクデータ(32)
079     *   or
080     *   全道路リンク拡張属性データ(39)
081     *
082     * 読み込みデータ：
083     *   ノード１番号リンク番号nd1 C 5
084     *   ノード２番号リンク番号nd2 C 5
085     *   リンク長(計算値) length C
086     *   道路幅員区分コードrdwdcd C 1
087     *     1 幅員 13.0m以上
088     *     2 幅員 5.5m以上~13.0m未満
089     *     3 幅員 3.0m以上～ 5.5ｍ未満
090     *     4 幅員 3.0ｍ未満
091     *     0 未調査
092     */

210    /**
211     * シェープファイル仕様情報を読み込む
212     */
213    private void readShapefileSpecs() {
214        Map<String, Object> shapefileSpec = (Map<String, Object>)shapefileSpecs.get(shapefileSpecName);
215        JSON json = new JSON();
216        Itk.logInfo("shapeFileSpec", json.encode(shapefileSpec, true));
217
218        coordinateSystem = (String)shapefileSpec.get("coordinate_system");
219        if (! coordinateSystem.equals("Geographic")) {
220            Itk.logError("readShapeFileSpecs",
221                         "coordinate_system は現在 \"Geographic\" のみ有効です。");
222            Itk.quitByError() ;
223        }
224
225        geodeticDatum = (String)shapefileSpec.get("geodetic_datum");
226
227        // 経緯度座標を四捨五入して小数第 scaleOfRoundOff 位に丸める
228        Object object = shapefileSpec.get("round_off_coordinate_to");
229        if (object != null) {
230            scaleOfRoundOff = Integer.parseInt(object.toString());
231        }
232
233        // length
234        Map<String, Object> lengthObjects = (Map<String, Object>)shapefileSpec.get("length");
235        lengthAttributeName = (String)lengthObjects.get("attribute_name");
236        object = lengthObjects.get("correction_factor");
237        if (object != null) {
238            lengthCorrectionFactor = Double.parseDouble(object.toString());
239        }
240
241        // width
242        Map<String, Object> widthObjects = (Map<String, Object>)shapefileSpec.get("width");
243        widthAttributeName = (String)widthObjects.get("attribute_name");
244        object = widthObjects.get("correction_factor");
245        if (object != null) {
246            widthCorrectionFactor = Double.parseDouble(object.toString());
247        }
248        object = widthObjects.get("reference_table");
249        if (object != null) {
250            widthTable = ((ArrayList<Object>)object).toArray();
251        }
252    }

297    /**
298     * シェープファイルを読み込む
299     */
300    private void readShapefile(File source_file) {
301        /* set up map */
302        try {
303            FileDataStore store;
304            store = FileDataStoreFinder.getDataStore(source_file);
305            Itk.logInfo("source file name", source_file.getName()) ;
306            if (store == null) {
307                Itk.logError("readShapeFile", "store is null!!");
308                Itk.quitByError() ;
309            }
310            String fileName = source_file.getName();
311            if (! fileName.toLowerCase().endsWith(".shp")) {
312                Itk.logWarn("Irregular file name", fileName);
313            }
314            SimpleFeatureSource featureSource;
315            featureSource = store.getFeatureSource();
316
317            map.addLayer(featureSource, null);
318        } catch (IOException e) {
319            Itk.dumpStackTraceOf(e) ;
320        }
321    }

464        public double objectToDouble(Object object) {
465            double value = 0.0;
466            if (object instanceof Double) {
467                value = (Double)object;
468            } else if (object instanceof Integer) {
469                value = (Integer)object;
470            } else if (object instanceof Long) {
471                value = (Long)object;
472            } else if (object instanceof String) {
473                value = Double.valueOf((String)object);
474            } else if (object instanceof BigDecimal) {
475                value = ((BigDecimal)object).doubleValue();
476            } else {
477                Itk.logError("Illegal object", object);
478                Itk.quitByError() ;
479            }
480            return value;
481        }

483        @Override
484        public void action(ActionEvent arg0) throws Throwable {
485            if (zone == -1) {
486                zone = chooseSystemNumber();
487                if (zone == -1)
488                    return;
489            }
490            JOptionPane.showMessageDialog(map_frame, "GisToCrowdWalk Version 1.10 uses Proj4J (Java library) to transform point coordinates.", "", JOptionPane.INFORMATION_MESSAGE);
491
492            // シェープファイルの座標系から平面直角座標系に変換する座標変換オブジェクトを生成する
493            String srcEpsg = null;
494            if (geodeticDatum.equals("Tokyo")) {
495                srcEpsg = "EPSG:4301";  // 旧日本測地系
496            } else {
497                srcEpsg = "EPSG:4326";  // WGS84
498            }
499            String targetEpsg = null;
500            if (outputByTokyoDatum) {
501                targetEpsg = TOKYO_JPR_EPSG_CODES[zone];
502            } else {
503                targetEpsg = JGD2000_JPR_EPSG_CODES[zone];
504            }
505            CoordinateTransform transform = createCoordinateTransform(srcEpsg, targetEpsg);
506
507            networkMap = new NetworkMap();
508            ReferencedEnvelope ref = map.getAreaOfInterest();
509            MapPartGroup root = (MapPartGroup)networkMap.getRoot();
510            root.setZone(zone);
511            MapPartGroup group = networkMap.createGroupNode(root);
512            group.addTag("(" + roundCoordinate(ref.getMinX()) + "_"
513                    + roundCoordinate(ref.getMinY()) + ")-("
514                    + roundCoordinate(ref.getMaxX()) + "_"
515                    + roundCoordinate(ref.getMaxY()) + ")");
516            group.setWest(roundCoordinate(ref.getMinX()));
517            group.setSouth(roundCoordinate(ref.getMaxY()));
518            group.setEast(roundCoordinate(ref.getMaxX()));
519            group.setNorth(roundCoordinate(ref.getMinY()));
520            group.setZone(zone);
521
522            double base_x = roundCoordinate(ref.getMinX());
523            double scale_x =15.0* Math.min(800 / roundCoordinate(ref.getWidth()),
524                    800 / roundCoordinate(ref.getHeight()));
525            double base_y = roundCoordinate(ref.getMinY());
526            double scale_y = scale_x;
527
528            Itk.logInfo("","basex: " + base_x + ", basey: " + base_y +
529                        ", scalex: " + scale_x + ", scaley: " + scale_y);
530            FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2(GeoTools
531                    .getDefaultHints());
532            Filter filter = ff.bbox(ff.property("the_geom"), ref);
533            HashMap<String, MapNode> nodes = new HashMap<String, MapNode>();
534            HashMap<String, MapLink> links = new HashMap<String, MapLink>();
535            MapLayer layers[] = map.getLayers();
536            for (int i = 0; i < layers.length; ++i) {
537                MapLayer layer = layers[i];
538                SimpleFeatureSource source = (SimpleFeatureSource) layer
539                    .getFeatureSource();
540                SimpleFeatureCollection features = source.getFeatures(filter);
541                SimpleFeatureType schema = features.getSchema();
542                // for (AttributeType type : schema.getTypes()) {
543                    // Itk.dbgVal("type",type);
544                // }
545                FeatureIterator<SimpleFeature> it = features.features();
546                while (it.hasNext()) {
547                    SimpleFeature feature = it.next();
548                    Object the_geom = feature.getAttribute("the_geom");
549
550                    double length = objectToDouble(feature.getAttribute(lengthAttributeName));
551                    if (lengthCorrectionFactor != 1.0) {
552                        length *= lengthCorrectionFactor;
553                    }
554
555                    double width = objectToDouble(feature.getAttribute(widthAttributeName));
556                    if (widthCorrectionFactor != 1.0) {
557                        width *= widthCorrectionFactor;
558                    }
559                    if (widthTable != null) {
560                        width = objectToDouble(widthTable[(int)width]);
561                    }
562                    width = roundValue(width, 4);
563
564                    if (makeSimpleModel) {
565                        make_nodes_simple(the_geom, feature, group, nodes, links,
566                            length, width, base_x, base_y, scale_x, scale_y,
567                            ! outputByPlaneRectangular, transform);
568                    } else {
569                        make_nodes_precise(the_geom, feature, group, nodes, links,
570                            length, width, base_x, base_y, scale_x, scale_y,
571                            ! outputByPlaneRectangular, transform);
572                    }
573                }
574            }
575            // check the nodes that are placed same coordinate.
576            ArrayList<MapNode> mapNodes = networkMap.getNodes();
577            Itk.logInfo("MapNode size", mapNodes.size());
578            for (int i = 0; i < mapNodes.size(); i++) {
579                for (int j = i + 1; j < mapNodes.size(); j++) {
580                    if (mapNodes.get(i).ID == mapNodes.get(j).ID)
581                        Itk.logWarn("",
582                                    "MapNode " + mapNodes.get(i).ID +
583                                    " is used by two nodes!");
584                }
585            }
586            ArrayList<MapLink> mapLinks = networkMap.getLinks();
587            Itk.logInfo("MapLink size", mapLinks.size());
588            for (int i = 0; i < mapLinks.size(); i++) {
589                for (int j = i + 1; j < mapLinks.size(); j++) {
590                    if (mapLinks.get(i).ID == mapLinks.get(j).ID)
591                        Itk.logWarn("",
592                                    "MapLink " + mapLinks.get(i).ID +
593                                    " is used by two links!");
594                }
595            }
596            if (save_map()) {
597                map.dispose();
598                map_frame.dispose();
599                quit();
600            }
601        }
602    }

611    /**
612     * 実数値 value を小数点以下第 scale 位で四捨五入する
613     */
614    public double roundValue(double value, int scale) {
615        if (scale >= 0 && ! Double.isNaN(value)) {
616            BigDecimal bd = new BigDecimal(String.valueOf(value));
617            return bd.setScale(scale, BigDecimal.ROUND_HALF_UP).doubleValue();
618        }
619        return value;
620    }

654    private void make_nodes_precise(Object the_geom, SimpleFeature feature,
655            MapPartGroup parent_group, HashMap<String, MapNode> nodes, HashMap<String, MapLink> links,
656            double length, double width, double base_x, double base_y,
657            double scale_x, double scale_y, boolean crowdwalk_coordinate, CoordinateTransform transform) {
658        if (!(the_geom instanceof MultiLineString))
659            return;
660        MultiLineString line = (MultiLineString) the_geom;
661        MapNode from = null;
662        Coordinate[] points = line.getCoordinates();
663
664        if (scaleOfRoundOff >= 0) {
665            for (int index = 0; index < points.length; index++) {
666                Coordinate c = points[index];
667                points[index] = new Coordinate(roundCoordinate(c.x), roundCoordinate(c.y), roundCoordinate(c.z));
668            }
669        }
670
671        /* two-pass, first get ratio of each segments */
672        double[] ratio = new double[points.length - 1];
673
674        Coordinate last_c = null;
675        double total_d = 0.0;
676        for (int j = 0; j < points.length; j++) {
677            Coordinate c = points[j];
678            if (last_c != null) {
679                double d = c.distance(last_c);
680                ratio[j - 1] = d;
681                total_d += d;
682            }
683            last_c = c;
684        }
685        for (int j = 0; j < points.length - 1; j++) {
686            ratio[j] /= total_d;
687        }
688
689        /* two-pass, second actually make links */
690        for (int j = 0; j < points.length; j++) {
691            Coordinate c = points[j];
692            ProjCoordinate coordinate = transformCoordinate(transform, c);
693            double x = roundValue(coordinate.y, 4);
694            double y = roundValue(coordinate.x, 4);
695            double z = roundValue(c.z, 4);
696            if (! heightEnable || Double.isNaN(z)) {
697                z = 0.0;
698            }
699
700            Point2D point = null;
701            if (crowdwalk_coordinate) {
702                point = new Point2D.Double(y, -x);
703            } else {
704                point = new Point2D.Double(x, y);
705            }
706
707            String point_str = point.toString().replaceAll(" ", "");
708            // if (j == 0) {
709                // point_str = feature.getAttribute("ND1").toString();
710                // Itk.dbgVal("ND1", point_str);
711            // } else if (j == points.length - 1) {
712                // point_str = feature.getAttribute("ND2").toString();
713                // Itk.dbgVal("ND2", point_str);
714            // } else{
715                // Itk.dbgVal("   ", point_str);
716            // }
717
718            MapNode node = null;
719            if (nodes.containsKey(point_str)) {
720                node = nodes.get(point_str);
721            } else {
722                node = networkMap.createMapNode(parent_group, point, z);
723                //node.addTag(point_str);
724                if (addCoordinateValueTag) {
725                    node.addTag("" + c.x + "_" + c.y);
726                }
727                nodes.put(point_str, node);
728            }
729
730            if (from == node) {
731                Itk.logInfo("","from  === node");
732            } else if (from != null) {
733                if (j == 0)
734                    Itk.logWarn("","\tj 0 but from is not null!");
735                String nodeIdPair = makeNodeIdPair(from, node);
736                if (links.containsKey(nodeIdPair)) {
737                    Itk.logWarn("","Duplicate link! node:" + nodeIdPair);
738                } else {
739                    MapLink link = networkMap.createMapLink(parent_group, from,
740                            node, roundValue(length * ratio[j - 1], 4), width);
741                    links.put(nodeIdPair, link);
742                }
743                /*
744                double dnodes = Math.sqrt(
745                        Math.pow(from.getX() - node.getX(), 2.0) +
746                        Math.pow(from.getY() - node.getY(), 2.0));
747                Itk.dbgVal("","from " + from.ID + " to " + node.ID +
748                    " length " + length + " ratio " + length * ratio[j - 1]
749                    + " distance " + dnodes + " j " + j);
750                */
751            }
752            from = node;
753        }
754    }

835
836        try {
837            FileOutputStream fos = new FileOutputStream(filename);
838
839            Document doc = ItkXmlUtility.singleton.newDocument() ;

841            boolean result = ItkXmlUtility.singleton.docToStream(doc, fos);

843                JOptionPane.showMessageDialog(map_frame,
844                        "Could no save to:\n" + filename
845                        + "\nAn error occured while actually writing to file.",
846                        "Save failed",
847                        JOptionPane.ERROR_MESSAGE);

850        } catch (IOException e) {
851            Itk.dumpStackTraceOf(e) ;
852            JOptionPane.showMessageDialog(map_frame,
853                    "Could no save to:\n" + filename,
854                    "Save failed",
855                    JOptionPane.ERROR_MESSAGE);
856            return false;
857        }
858
859        return true;
860    }

896    /**
897     * コマンドラインオプションを解析して指定された処理を実行する
898     * @param args : main メソッドの args 引数
899     */
900    private void parseCommandLine(String[] args, Options options) {
901        try {
902            CommandLine commandLine = new PosixParser().parse(options, args);
903            // ヘルプ表示オプションもしくはコマンドライン引数エラー
904            if (commandLine.hasOption("help") || commandLine.getArgs().length > 1) {
905                printHelp(options);
906                Itk.quitSafely() ;
907            }
908
909            // シェープファイルの指定あり
910            if (commandLine.getArgs().length == 1) {
911                shapefilePath = commandLine.getArgs()[0];
912            }
913
914            // 読み込むシェープファイルの形式
915            if (commandLine.hasOption("shapefile-spec")) {
916                shapefileSpecName = commandLine.getOptionValue("shapefile-spec");
917                boolean found = false;
918                for (String name : shapefileSpecNames) {
919                    if (name.equals(shapefileSpecName)) {
920                        found = true;
921                    }
922                }
923                if (! found) {
924                    printHelp(options);
925                    Itk.quitByError() ;
926                }
927            }
928
929            // シェープファイルが標高情報を持っていたら、その値を height 属性に反映する
930            // heightEnable = commandLine.hasOption("height-enable");
931            // ※標高値に関しては不明な点があるため当面サポートは見送る
932
933            // 経緯度タグをノードに付加する
934            addCoordinateValueTag = commandLine.hasOption("add-coordinate-value-tag");
935
936            // 旧日本測地系で出力する
937            outputByTokyoDatum = commandLine.hasOption("output-by-tokyo-datum");
938
939            // 平面直角座標系のまま出力する
940            outputByPlaneRectangular = commandLine.hasOption("output-by-plane-rectangular");
941
942            // 簡易モデルのマップを出力する
943            makeSimpleModel = commandLine.hasOption("make-simple-model");
944
945            // 平面直角座標系の系番号(1～19)を指定する
946            if (commandLine.hasOption("zone")) {
947                String _zone = commandLine.getOptionValue("zone");
948                for (Integer num : SYSTEM_NUMBERS) {
949                    if (_zone.equals(num.toString())) {
950                        zone = num;
951                        break;
952                    }
953                }
954                if (zone == -1) {
955                    printHelp(options);
956                    Itk.quitByError() ;
957                }
958            }
959        } catch (ParseException e) {
960            Itk.logError("", e.getMessage());
961            printHelp(options);
962            Itk.quitByError() ;
963        } catch (Exception e) {
964            Itk.quitWithStackTrace(e) ;
965        }
966    }
