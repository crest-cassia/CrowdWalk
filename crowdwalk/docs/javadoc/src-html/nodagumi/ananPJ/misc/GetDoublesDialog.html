004import java.awt.Container;
005import java.awt.GridLayout;
006import java.awt.event.ActionEvent;
007import java.awt.event.ActionListener;
008
009import java.io.IOException;
010import java.io.ObjectInputStream;
011import java.io.ObjectOutputStream;
012
013import javax.swing.JButton;
014import javax.swing.JDialog;
015import javax.swing.JLabel;
016import javax.swing.JSpinner;
017import javax.swing.SpinnerNumberModel;
018
019public class GetDoublesDialog extends JDialog 
020        implements ActionListener {
021        private JSpinner[] values;
022        public boolean proceed = false;
023        public GetDoublesDialog(String title,
024                        String[] labels,
025                        double[][] range) {
026                this.setModal(true);
027                Container panel = getContentPane();
028                panel.setLayout(new GridLayout(3, labels.length));
029                
030                values = new JSpinner[labels.length];
031
032                for (int i = 0; i < labels.length; ++i) {
033                        panel.add(new JLabel(labels[i]));
034                }
035
036                for (int i = 0; i < labels.length; ++i) {
037                        values[i] = new JSpinner(new SpinnerNumberModel(range[i][2],
038                                        range[i][0], range[i][1], 0.1));
039                        panel.add(values[i]);
040                }
041
042                for (int i = 0; i < labels.length - 2; ++i) {
043                        panel.add(new JLabel());
044                }
045
046                JButton ok = new JButton("OK");
047                ok.addActionListener(this);
048                panel.add(ok);
049                JButton cancel = new JButton("Cancel");
050                cancel.addActionListener(this);
051                panel.add(cancel);
052                
053                this.pack();
054        }
055        @Override
056        public void actionPerformed(ActionEvent e) {
057                if (e.getActionCommand() == "OK") {
058                        proceed = true;
059                        dispose();
060                } else if (e.getActionCommand() == "Cancel") {
061                        proceed = false;
062                        dispose();
063                }
064        }
065        
066        public static double[] showDialog(String title, String[] labels) {
067        final double[][] range = {{-1000, 1000, -1000},
068                        {-1000, 1000, -1000},
069                        {-1000, 1000, 1000}}; 
070
071                return showDialog(title, labels, range);
072        }
073        public static double[] showDialog(String title,
074                        String[] labels,
075                        double[][] range) {
076                GetDoublesDialog gdd = new GetDoublesDialog("", labels, range);
077        gdd.setVisible(true);
078
079        if (!gdd.proceed) return null;
080
081        double[] ret = new double[labels.length];
082        for (int i = 0; i < labels.length; ++i) {
083                ret[i] = (Double)(gdd.values[i].getValue());
084        }
085        return ret;
086        }
087}
