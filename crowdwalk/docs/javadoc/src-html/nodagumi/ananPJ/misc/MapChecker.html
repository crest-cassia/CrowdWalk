027class ProgressBarDialog extends JDialog
028    implements ActionListener {
029    int count;
030
031    JProgressBar bar;
032    JButton cancel;
033
034    boolean canceled = false;
035
036    public ProgressBarDialog(String message, int start, int end) {
037        count = start;
038
039        setLayout(new BorderLayout());
040
041        add(new JLabel(message), BorderLayout.WEST);
042
043        bar = new JProgressBar(start, end);
044        bar.setValue(start);
045        bar.setStringPainted(true);
046        add(bar, BorderLayout.CENTER);
047
048        cancel = new JButton("Cancel");
049        cancel.addActionListener(this);
050        add(cancel, BorderLayout.EAST);
051
052        pack();
053    }
054
055    public boolean step() {
056        ++count;
057        bar.setValue(count);
058        repaint();
059
060        return canceled;
061    }
062
063    @Override
064    public void actionPerformed(ActionEvent e) {
065        if (e.getActionCommand() == "Cancel") canceled = true;
066    }
067}

069public class MapChecker {
070    public static boolean checkForPiledNodes(MapNodeTable nodes) {
071        ProgressMonitor bar = new ProgressMonitor(null,
072                                                  "Checking node collisions",
073                                                  "", 0, nodes.size() - 1);
074
075        HashMap<Point2D, MapNodeTable > collisions =
076            new HashMap<Point2D, MapNodeTable >();
077        for (int i = 0; i < nodes.size() - 1; ++i) {
078            bar.setProgress(i);
079            for (int j = i + 1; j < nodes.size(); ++j){
080                MapNode lhs = nodes.get(i);
081                MapNode rhs = nodes.get(j);
082
083                if (lhs.getPosition().equals(rhs.getPosition())) {
084                    Point2D pos = lhs.getPosition();
085                    MapNodeTable nodesHere = collisions.get(pos);
086                    if (nodesHere == null) {
087                        nodesHere = new MapNodeTable();
088                        collisions.put(pos, nodesHere);
089                    }
090                    if (!nodesHere.contains(lhs)) nodesHere.add(lhs);
091                    if (!nodesHere.contains(rhs)) nodesHere.add(rhs);
092                }
093            }
094        }
095        bar.close();
096        if (collisions.size() == 0)     return true;
097
098        int i = JOptionPane.showConfirmDialog(null,
099                                              "" + collisions.size() + " collisions were found.\n"
100                                              + "Resolve them?", "Collisions found", JOptionPane.YES_NO_OPTION);
101        if (i == JOptionPane.NO_OPTION) return false;
102
103        for (MapNodeTable cnodes : collisions.values()) {
104            for (MapNode node : cnodes) {
105                node.selected = true;
106            }
107        }
108
109        return false;
110    }

115    public static MapLinkTable getReachableLinks(MapNodeTable nodes,
116                                                 String targetTag) {
117        final MapNodeTable exits = new MapNodeTable();
118        for (final MapNode node : nodes) {
119            if (node.hasTag(targetTag)) exits.add(node);
120        }
121
122        Stack<MapNode> path = new Stack<MapNode>();
123        Stack<Integer> selection = new Stack<Integer>();
124        MapLinkTable reachableLinks = new MapLinkTable();
125
126        for (MapNode exit : exits) {
127            path.push(exit);
128            selection.push(0);
129
130            while(!path.empty()) {
131                MapNode node = path.peek();
132                Integer index = selection.pop();
133
134                final MapLinkTable linkList  = node.getUsableLinkTable();
135                if (index == linkList.size()) {
136                    path.pop();
137                    continue;
138                }
139
140                final MapLink link = linkList.get(index);
141                ++index;
142                selection.push(index);
143
144                if (!reachableLinks.contains(link)) {
145                    reachableLinks.add(link);
146                    path.push(link.getOther(node));
147                    selection.push(0);
148                }
149            }
150        }
151        return reachableLinks;
152    }
