202    /**
203     * fallback （デフォルトセッティング）の読み込み
204     * @param scanResourceP : resource の fallback も読み込むかどうか
205     */
206    public void scanFallbackFile(ArrayList<String> commandLineFallbacks,
207                                 boolean scanResourceP) {
208        if(fallbackFile != null) {
209            try {
210                BufferedReader buffer =
211                    new BufferedReader(new FileReader(fallbackFile)) ;
212                fallbackParameters = Term.newByScannedJson(JSON.decode(buffer),
213                                                           true) ;
214                Itk.logInfo("Load Fallback File", fallbackFile) ;
215            } catch (Exception ex) {
216                Itk.dumpStackTraceOf(ex) ;
217                Itk.logError("Can not scan a fallback parameter file:",
218                             fallbackFile) ;
219                Itk.logError_("Exception",ex) ;
220            }
221        } else {
222            fallbackParameters = Term.newObjectTerm() ;
223        }
224
225        if(scanResourceP) {
226            try {
227                InputStream istrm =
228                    getClass().getResourceAsStream(FallbackResource) ;
229                Term finalFallback =
230                    Term.newByScannedJson(JSON.decode(istrm),true) ;
231                attachFallback(fallbackParameters, finalFallback) ;
232            } catch (Exception ex) {
233                Itk.dumpStackTraceOf(ex) ;
234                Itk.logError("Can not scan a fallback resource file.") ;
235                Itk.logError_("Exception",ex) ;
236            }
237        }
238
239        // コマンドラインで指定した fallback を先頭に追加していく。
240        // コマンドラインでは、後から追加したものほど優先。
241        if(commandLineFallbacks != null) {
242            for(String fallbackString : commandLineFallbacks) {
243                Term newFallback =
244                    Term.newByScannedJson(JSON.decode(fallbackString), true) ;
245                attachFallback(newFallback, fallbackParameters) ;
246                fallbackParameters = newFallback ;
247            }
248        }
249
250        Itk.logInfo("fallbackParameters",
251                    unifiedFallbackParameters().toJson(true)) ;
252    }

258    /**
259     * パラメータ設定に fallback を追加する。
260     * 内部的には、fallback は、以下のように格納される。
261     * {@code params} の内容が {@literal { "foo" : { "bar" : 1 }, "baz" : 2}}
262     * {@code fallback} の内容が {@literal { "foo" : { "bar2" : 2}}}
263     * の場合、{@code attachFallback} された結果は、
264     * {@literal { "foo" : { "bar" : 1 }, "baz" : 2, "_fallback" : { "foo" : { "bar2" : 2}}}} となる。
265     * <br>
266     * 最終的に、fallback 情報は、以下の形式となる。
267     * <pre>{@literal
268     * { ...<コマンドラインの--fallbackで指定したJSON>... ,
269     *   "_fallback" : { ...<propertiesの"fallback_file"で指定したファイルのJSON>...,
270     *                   "_fallback" : { ..."src/main/resources/fallbackParameters.json"のJSON>...} }
271     * }
272     * }</pre>
273     * @param params : もとになる parameter 設定用 Term
274     * @param fallback : params の後に追加する fallback
275     * @return fallbackを含んだ 設定Term。
276     * @see nodagumi.ananPJ Overview
277     */
278    static public Term attachFallback(Term params, Term fallback) {
279        params.setArg(FallbackSlot, fallback) ;
280        return params ;
281    }

372    /**
373     * fallback の fetch (Lexiconを使ってObjectを取得)。
374     * @param fallbacks : filter する fallback
375     * @param lexicon : 値を変換するための Lexicon。
376     * @param tag : fallbacks の中から、tag をたどって filter する。
377     * @param finalFallbackValue : 最後の値
378     */
379    static public Object fetchFallbackObjectViaLexicon(Term fallbacks,
380                                                       Lexicon lexicon,
381                                                       String tag,
382                                                       Object finalFallbackValue) {
383        String word = fetchFallbackString(fallbacks, tag, null) ;
384        if(word == null) {
385            return finalFallbackValue ;
386        } else {
387            Object value = lexicon.lookUp(word) ;
388            if(value != null) {
389                return value ;
390            } else {
391                Itk.dumpStackTrace() ;
392                Itk.logError("Unknown word in lexicon",
393                             " lookup tag=", tag, ";",
394                             " word=", word) ;
395                Itk.quitByError() ;
396                return null ; // never reach here.
397            }
398        }
399    }
