013package nodagumi.ananPJ.misc ;
014
015import java.util.ArrayList;
016import java.util.List;
017
018import nodagumi.Itk.Itk;
019import nodagumi.Itk.Term;
020
021//======================================================================
022/**
023 * Route の プランを、先頭から順番に使っていく時のためのクラス。
024 */
025public class RoutePlan {
026    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
027    /**
028     * 現在位置
029     */
030    private int index = 0 ;
031
032    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
033    /**
034     * タグの列
035     */
036    private List<Term> route = null ;
037
038    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
039    /**
040     * route が share されているかどうか？
041     */
042    private boolean shared = true ;
043
044    //------------------------------------------------------------
045    /**
046     * コンストラクタ
047     */
048    public RoutePlan() {
049        route = new ArrayList<Term>() ;
050        index = 0 ;
051    } ;
052
053    //------------------------------------------------------------
054    /**
055     * コンストラクタ
056     */
057    public RoutePlan(RoutePlan origin) {
058        copyFrom(origin);
059    } ;
060
061    //------------------------------------------------------------
062    /**
063     * route を取得
064     */
065    public List<Term> getRoute() {
066        return route ;
067    }
068
069    //------------------------------------------------------------
070    /**
071     * route をセット
072     */
073    public RoutePlan setRoute(List<Term> _route) {
074        return setRoute(_route, false) ;
075    }
076
077    //------------------------------------------------------------
078    /**
079     * route をセット
080     */
081    public RoutePlan setRoute(List<Term> _route, boolean copy) {
082        if(copy) {
083            route = new ArrayList<Term>(_route) ;
084            shared = false ;
085        } else {
086            route = _route ;
087            shared = true ;
088        }
089        return this ;
090    }
091
092    //------------------------------------------------------------
093    /**
094     * index を取得
095     */
096    public int getIndex() {
097        return index ;
098    }
099
100    //------------------------------------------------------------
101    /**
102     * index をセット
103     */
104    public RoutePlan setIndex(int _index) {
105        index = _index ;
106        return this ;
107    }
108
109    //------------------------------------------------------------
110    /**
111     * index をリセッt
112     */
113    public RoutePlan resetIndex() {
114        setIndex(0) ;
115        return this ;
116    }
117
118    //------------------------------------------------------------
119    /**
120     * index を最後に持っていく
121     */
122    public RoutePlan makeCompleted() {
123        setIndex(totalLength()) ;
124        return this ;
125    }
126
127    //------------------------------------------------------------
128    /**
129     * shift: index をひとつずらす
130     */
131    public int shift() {
132        return shift(1) ;
133    }
134
135    //------------------------------------------------------------
136    /**
137     * shift: index を n ずらす
138     */
139    public int shift(int n) {
140        index += n ;
141        return index ;
142    }
143
144    //------------------------------------------------------------
145    /**
146     * 残りの長さ
147     */
148    public int length() {
149        return totalLength() - index ;
150    }
151
152    //------------------------------------------------------------
153    /**
154     * もとの長さ
155     */
156    public int totalLength() {
157        return route.size() ;
158    }
159
160    //------------------------------------------------------------
161    /**
162     * 空かどうかのチェック
163     */
164    public boolean isEmpty() {
165        return length() <= 0 ;
166    }
167
168    //------------------------------------------------------------
169    /**
170     * 現在の先頭を見る。（取り除かない）
171     */
172    public Term top() {
173        return top(0) ;
174    }
175
176    //------------------------------------------------------------
177    /**
178     * 現在のindexよりn番目を取り出す。
179     */
180    public Term top(int n) {
181        return route.get(index + n) ;
182    }
183
184    //------------------------------------------------------------
185    /**
186     * 現在の先頭を見る。（取り除く）
187     */
188    public Term get() {
189        Term string = top() ;
190        shift() ;
191        return string ;
192    }
193
194    //------------------------------------------------------------
195    /**
196     * プランの追加(末尾)
197     */
198    public RoutePlan add(Term tag) {
199        route.add(tag) ;
200        return this ;
201    }
202
203    //------------------------------------------------------------
204    /**
205     * プランの追加(末尾)
206     */
207    public RoutePlan addSafely(Term tag) {
208        if(shared) {
209            route = new ArrayList<Term>(route) ;
210            shared = false ;
211        }
212        add(tag) ;
213        return this ;
214    }
215
216    //------------------------------------------------------------
217    /**
218     * プランの追加
219     */
220    public RoutePlan insertSafely(Term tag) {
221        if(shared) {
222            route = new ArrayList<Term>(route) ;
223            shared = false ;
224        }
225        insert(tag) ;
226        return this ;
227    }
228
229    //------------------------------------------------------------
230    /**
231     * プランの追加
232     */
233    public RoutePlan insert(Term tag) {
234        route.add(null) ;
235        for(int i = route.size() -1 ; i > index ; i--) {
236            route.set(i, route.get(i-1)) ;
237        }
238        route.set(index, tag) ;
239        return this ;
240    }
241
242    //------------------------------------------------------------
243    /**
244     * プランの追加
245     */
246    public RoutePlan duplicate() {
247        return new RoutePlan(this) ;
248    }
249
250    //------------------------------------------------------------
251    /**
252     * プランの追加
253     */
254    public RoutePlan copyFrom(RoutePlan origin) {
255        setRoute(origin.getRoute()) ;
256        setIndex(origin.getIndex()) ;
257        shared = origin.shared ;
258        return this ;
259    }
260
261    //------------------------------------------------------------
262    /**
263     * Term へ変更
264     */
265    public Term toTerm() {
266        Term term = Term.newObjectTerm() ;
267        term.setArg("index",index) ;
268        Term taglist = Term.newArrayTerm() ;
269        for(Term subgoal : getRoute()) {
270            taglist.addNth(subgoal) ;
271        }
272        term.setArg("route",taglist) ;
273
274        return term ;
275    }
276
277    //------------------------------------------------------------
278    /**
279     * Term から設定
280     */
281    public Term setByTerm(Term term) {
282        resetIndex() ;
283        route.clear() ;
284        for(Object _subgoal : term.getArgTerm("route").getArray()){
285            Term subgoal = (Term)_subgoal ;
286            route.add(subgoal) ;
287        }
288        setIndex(term.getArgInt("index")) ;
289
290        return term ;
291    }
292    
293
294} // class RoutePlan.java
