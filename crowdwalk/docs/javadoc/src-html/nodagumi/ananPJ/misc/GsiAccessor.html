019/**
020 * 国土地理院の Web サービスを利用するためのインターフェイスを提供する
021 */
022public class GsiAccessor {
023    /**
024     * 緯度経度から Slippy map のタイル番号を求める
025     */
026    public static Point getTileNumber(double lat, double lon, int zoom) {
027        int xtile = (int)Math.floor((lon + 180) / 360 * (1 << zoom));
028        int ytile = (int)Math.floor((1 - Math.log(Math.tan(Math.toRadians(lat)) + 1 / Math.cos(Math.toRadians(lat))) / Math.PI) / 2 * (1 << zoom));
029        if (xtile < 0)
030            xtile = 0;
031        if (xtile >= (1 << zoom))
032            xtile = ((1 << zoom) - 1);
033        if (ytile < 0)
034            ytile = 0;
035        if (ytile >= (1 << zoom))
036            ytile = ((1 << zoom) - 1);
037        return new Point(xtile, ytile);
038    }
039
040    /**
041     * Slippy map のタイル番号から緯度経度とタイルサイズを求める
042     */
043    public static Rectangle2D tile2boundingBox(int x, int y, int zoom) {
044        double north = tile2lat(y, zoom);
045        double south = tile2lat(y + 1, zoom);
046        double west = tile2lon(x, zoom);
047        double east = tile2lon(x + 1, zoom);
048        return new java.awt.geom.Rectangle2D.Double(west, north, east - west, north - south);
049    }
050
051    private static double tile2lon(int x, int z) {
052        return x / Math.pow(2.0, z) * 360.0 - 180;
053    }
054
055    private static double tile2lat(int y, int z) {
056        double n = Math.PI - (2.0 * Math.PI * y) / Math.pow(2.0, z);
057        return Math.toDegrees(Math.atan(Math.sinh(n)));
058    }
059
060    /**
061     * キャッシュされた地理院タイル画像のファイル名を取得する
062     */
063    public static String getFileNameOfGsiTileImage(String cachePath, String tileName, int x, int y, int zoom, String ext) {
064        String _cachePath = cachePath.replaceFirst("[/\\\\]$", "");
065        return String.format("%s/%s/%d/%d/%d.%s", _cachePath, tileName, zoom, x, y, ext);
066    }
067
068    /**
069     * 地理院タイル画像を cachePath ディレクトリにダウンロードする
070     */
071    public static boolean downloadGsiTileImage(String cachePath, String tileName, int x, int y, int zoom, String ext) {
072        String _cachePath = cachePath.replaceFirst("[/\\\\]$", "");
073        String dirPath = String.format("%s/%s/%d/%d", _cachePath, tileName, zoom, x);
074        File dir = new File(dirPath);
075        if (! dir.exists()) {
076            Itk.logInfo("Make directory", dirPath);
077            dir.mkdirs();
078        }
079
080        String filePath = String.format("%s/%d.%s", dirPath, y, ext);
081        File file = new File(filePath);
082        if (file.exists()) {
083            return true;
084        }
085        if (CrowdWalkLauncher.offline) {
086            return false;
087        }
088        // インターネット接続を確認して以降の動作に反映
089        if (! CrowdWalkLauncher.internetEnabled) {
090            if (CrowdWalkLauncher.isInternetEnabled()) {
091                CrowdWalkLauncher.internetEnabled = true;
092            } else {
093                CrowdWalkLauncher.offline = true;
094                return false;
095            }
096        }
097
098        String url = String.format("http://cyberjapandata.gsi.go.jp/xyz/%s/%d/%d/%d.%s", tileName, zoom, x, y, ext);
099        Itk.logInfo("Download", url);
100        try {
101            BufferedImage image = ImageIO.read(new URL(url));
102            Itk.logInfo("Save", filePath);
103            ImageIO.write(image, ext, file);
104        } catch (IOException e) {
105            Itk.dumpStackTraceOf(e) ;
106            return false;
107        }
108        try {
109            // サーバーに負荷を掛けすぎてアクセス拒否されないために毎秒2回とする
110            Thread.sleep(500);
111        } catch(InterruptedException e) {}
112        return true;
113    }
114
115    /**
116     * 地理院タイルの左上点の標高値を取得する
117     */
118    public static double getElevation(String cachePath, int x, int y, int zoom) {
119        String _cachePath = cachePath.replaceFirst("[/\\\\]$", "");
120        String filePath = String.format("%s/elevation_%d_%d_%d.json", _cachePath, zoom, x, y);
121        File file = new File(filePath);
122        if (! file.exists()) {
123            if (CrowdWalkLauncher.offline) {
124                return 0.0;
125            }
126            // インターネット接続を確認して以降の動作に反映
127            if (! CrowdWalkLauncher.internetEnabled) {
128                if (CrowdWalkLauncher.isInternetEnabled()) {
129                    CrowdWalkLauncher.internetEnabled = true;
130                } else {
131                    CrowdWalkLauncher.offline = true;
132                    return 0.0;
133                }
134            }
135            // 国土地理院Webの標高APIにアクセスして標高値を取得し、キャッシュファイルに保存する
136            Rectangle2D tileRect = tile2boundingBox(x, y, zoom);
137            String url = String.format("http://cyberjapandata2.gsi.go.jp/general/dem/scripts/getelevation.php?lon=%s&lat=%s&outtype=JSON", tileRect.getX(), tileRect.getY());
138            try {
139                Itk.logInfo("Access to", url);
140                Files.copy(new URL(url).openStream(), file.toPath());
141            } catch (IOException e) {
142                Itk.logError("IOException", e.getMessage());
143                return 0.0;
144            }
145        }
146        String elevation = "";
147        try {
148            Itk.logInfo("Read elevation file", filePath);
149            HashMap<String, Object> json = JSON.decode(new FileInputStream(filePath));
150            elevation = json.get("elevation").toString();
151            return new Double(elevation).doubleValue();
152        } catch (Exception e) {
153            Itk.logError(e.getMessage(), elevation);
154        }
155        return 0.0;
156    }
