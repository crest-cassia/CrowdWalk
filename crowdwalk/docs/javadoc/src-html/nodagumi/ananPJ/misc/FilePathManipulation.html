011public class FilePathManipulation {
012        public static String getRelativePath(String base_str,
013                        String file_str) {
014                if (base_str.equals(file_str)) return file_str;
015                if (!(new File(file_str)).isFile()) return file_str;
016
017                File base = new File(base_str);
018                /* we want to compare directory names, so we remove the filename */
019                if (!base.isDirectory()) base = base.getParentFile();
020
021                int depth_to_common = 0;
022                while (base != null) {
023                        LinkedList<String> relpath = new LinkedList<String>();
024                        File file = new File(file_str);
025
026                        /* we want to compare directory names, so we remove the filename */
027                        String filename = file.getName();
028                        file = file.getParentFile();
029                        while (file != null) {
030                                if (base.equals(file)) {
031                                        StringBuffer relpathbuf = new StringBuffer();
032                                        
033                                        if (depth_to_common != 0) {
034                                                for (int i = 0; i < depth_to_common; ++i) {
035                                                        relpathbuf.append(".." + File.separator);
036                                                }
037                                        }
038                                        while (!relpath.isEmpty()) {
039                                                relpathbuf.append(relpath.pop() + File.separator);
040                                        }
041                                        relpathbuf.append(filename);
042                                        return relpathbuf.toString();
043                                }
044
045                                relpath.push(file.getName());
046                                file = file.getParentFile();
047                        }
048                        base = base.getParentFile();
049                        depth_to_common++;
050                }
051                /* No common path; returning absolute path */
052                return file_str;                
053        }

071    public static boolean deleteFiles(File file, boolean skip) {
072        if (file.isDirectory()) {
073            for (String filename : file.list()) {
074                File _file = new File(file, filename);
075                if (! deleteFiles(_file, false)) {
076                    System.err.println("File can not delete: " + _file.toString());
077                }
078            }
079        }
080        return skip ? true : file.delete();
081        }
082
083    public static boolean deleteFiles(String filePath, boolean skip) {
084        return deleteFiles(new File(filePath), skip);
085    }
086
087    // ※filter はディレクトリを通さないこと
088    public static void deleteFiles(File dir, FilenameFilter filter) {
089        if (dir.isDirectory()) {
090            for (String filename : dir.list(filter)) {
091                File _file = new File(dir, filename);
092                if (! _file.delete()) {
093                    System.err.println("File can not delete: " + _file.toString());
094                }
095            }
096        }
097        }
098
099    public static void deleteFiles(String dirPath, FilenameFilter filter) {
100        deleteFiles(new File(dirPath), filter);
101    }
