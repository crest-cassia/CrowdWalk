076    public GeneralPath getRect(Graphics2D g, boolean showScaling) {
077        double scale = 1.0;
078        if (!showScaling)
079            scale = g.getTransform().getScaleX();
080        double x1 = from.getX();
081        double y1 = from.getY();
082        double x2 = to.getX();
083        double y2 = to.getY();
084        double fwidth = 4.0 / scale;
085
086        double dx = (x2 - x1);
087        double dy = (y2 - y1);
088        double a = Math.sqrt(dx*dx + dy*dy);
089
090        double edx = fwidth * dx / a / 2;
091        double edy = fwidth * dy / a / 2;
092
093        GeneralPath p = new GeneralPath();
094        p.moveTo(x1 - edy, y1 + edx);
095        p.lineTo(x1 + edy, y1 - edx);
096        p.lineTo(x2 + edy, y2 - edx);
097        p.lineTo(x2 - edy, y2 + edx);
098        p.lineTo(x1 - edy, y1 + edx);
099        p.closePath();
100
101        return p;
102    }
