753    public String furnishPropertiesDirPath(String path,
754                                           boolean forceP,
755                                           boolean absP) {
756        try {
757            boolean useRelFromProp =
758                getBoolean("use_relative_path_from_prop", true) ;
759            if(!useRelFromProp) return path ;
760        } catch(Exception ex) {
761            Itk.logError("property file error." + ex.getMessage()) ;
762        }
763            
764        String propDir = (absP ?
765                          getPropertiesDirAbs() :
766                          getPropertiesDirRel()) ;
767        File file = new File(path) ;
768            
769        if(file.getParent() == null || (forceP && !file.isAbsolute())) {
770            return (propDir.replaceAll("\\\\", "/") + "/" + path) ;
771        } else {
772            return path ;
773        }
774    }

893     * コンストラクタ
894     */
895    public CrowdWalkPropertiesHandler(String _propertiesFile) {
896        // load properties
897        prop = new Term();
898        propertiesFile = _propertiesFile;
899        try {
900            String path = _propertiesFile.toLowerCase();
901            if (path.endsWith(".xml")) {
902                scanXmlPropertiesIntoTerm(new FileInputStream(_propertiesFile),
903                                          prop) ;
904                Itk.logInfo("Load Properties File (XML)",
905                            _propertiesFile);
906            } else if (path.endsWith(".json")) {
907                HashMap<String, Object> map =
908                    (HashMap<String, Object>)
909                    JSON.decode(new FileInputStream(_propertiesFile));
910                prop = new Term().setScannedJson(map, true) ;
911                
912                Itk.logInfo("Load Properties File (JSON)",
913                            _propertiesFile);
914            } else {
915                throw new Exception("Property file error - unknown extention: "
916                                    + _propertiesFile);
917            }
918
919            // パス指定がファイル名のみならば
920            // プロパティファイルのディレクトリパスを付加する
921            /* [2017.06.22 I.Noda] furnishPropertiesDirPath などを使うように、
922             * 全体を改変。
923            file propertyfile = new file(_propertiesfile);
924            string propertydirpath = propertyfile.getparent();
925            if (propertydirpath == null) {
926                propertydirpath = ".";
927            }
928            for (string property_item : definition_file_items) {
929                string filepath = getstring(property_item, null);
930                if (filepath != null) {
931                    file file = new file(filepath);
932                    if (file.getparent() == null) {
933                        prop.setarg(property_item,
934                                    propertydirpath.replaceall("\\\\", "/") +
935                                    "/" + filepath);
936                    }
937                }
938            }
939            */
940
941            // input files
942            networkMapFile = getFurnishedPath("map_file", null);
943            pollutionFile = getFurnishedPath("pollution_file", null);
944            generationFile = getFurnishedPath("generation_file", null);
945            scenarioFile = getFurnishedPath("scenario_file", null);
946            fallbackFile = getFurnishedPath("fallback_file", null) ;
947
948            // create random with seed
949            randseed = getInteger("randseed", 0);
950
951            // exit count
952            exitCount = getInteger("exit_count", 0);
953            isAllAgentSpeedZeroBreak = getBoolean("all_agent_speed_zero_break",
954                                                  true);
955
956            // 早い内に設定ミスをユーザーに知らせるための検査
957            String obstructerType = getString("pollution_type", null);
958            if (obstructerType != null) {
959                AgentBase.setObstructerType(obstructerType);
960                ObstructerBase.createInstance(obstructerType) ;
961            }
962            getDouble("pollution_color_saturation", 0.0);
963
964            /* [2015.01.07 I.Noda] to switch agent queue in the link directions.*/
965            String queueOrderStr = getString("queue_order", null) ;
966            if(queueOrderStr != null) {
967                if(queueOrderStr.equals("front_first")) {
968                    AgentHandler.useFrontFirstOrderQueue(true) ;
969                    Itk.logInfo("use front_first order to sort agent queue.") ;
970                } else if (queueOrderStr.equals("rear_first")) {
971                    AgentHandler.useFrontFirstOrderQueue(false) ;
972                    Itk.logInfo("use rear_first order to sort agent queue.") ;
973                } else {
974                    Itk.logError("unknown queue_order:" + queueOrderStr) ;
975                    Itk.logError_("use default order (rear_first)") ;
976                }
977            }
978
979            // [2016.01.30 I.Noda] mentalRule check
980            mentalMapRules = getTerm("mental_map_rules",null) ;
981
982            // "No hint for goal" ログを出力しない
983            disableNoHintForGoalLog |= getBoolean("disable_no_hint_for_goal_log", false);
984
985            // legacy モード
986            legacy = getBoolean("legacy", legacy);
987
988            // validate モード
989            validate |= getBoolean("validate", false);
990        } catch (IOException ioe) {
991            Itk.logError("IO exception") ;
992            Itk.quitWithStackTrace(ioe) ;
993        } catch(Exception e) {
994            Itk.logError("unknown exception", e.getMessage()) ;
995            Itk.quitWithStackTrace(e) ;
996        }
997    }

1022    /**
1023     * scan XML file as Properties into Term
1024     */
1025    private Term scanXmlPropertiesIntoTerm(InputStream stream, Term term) {
1026        try {
1027            Properties _prop = new Properties() ;
1028            _prop.loadFromXML(stream) ;
1029
1030            for(String key : _prop.stringPropertyNames()) {
1031                term.setArg(Itk.intern(key),
1032                            Itk.intern(_prop.getProperty(key))) ;
1033            }
1034        } catch (IOException ioe) {
1035            Itk.logError("IO exception") ;
1036            Itk.quitWithStackTrace(ioe) ;
1037        } catch(Exception e) {
1038            Itk.logError("unknown exception", e.getMessage());
1039            Itk.quitWithStackTrace(e) ;
1040        }
1041            
1042        return term ;
1043    }

1085    /** */
1086    public String getStringInPattern(String key, String defaultValue,
1087                                     String pattern[]) throws Exception
1088    {
1089        String value = prop.getArgString(key);
1090        if (value == null || value.trim().isEmpty()) {
1091            return defaultValue;
1092        }
1093        value = value.toLowerCase();
1094        for (String str : pattern) {
1095            if (str.toLowerCase().equals(value)) {
1096                return value;
1097            }
1098        }
1099        throw new Exception("Property error - value is not in the pattern:" +
1100                            key + ":" + value + " not in " + pattern);
1101    }

1334     * パスは区切り記号で区切られているとする。
1335     * [2019-02-05: I.Noda]
1336     * もしくは、文字列のリスト（配列）も受け付けるとする。
1337     */
1338    public ArrayList<String> getPathList(String key,
1339                                         ArrayList<String> defaultValue,
1340                                         String separator) {
1341        Term valueTerm = prop.getArgTerm(key) ;
1342        if(Term.isNullTerm(valueTerm)) { return defaultValue ; }
1343
1344        ArrayList<String> pathList = new ArrayList<String>();
1345        if(valueTerm.isArray()) {
1346            for(Object valueObj : valueTerm.getArray()) {
1347                String path = ((valueObj instanceof Term) ?
1348                               ((Term)valueObj).getString() :
1349                               (String)valueObj) ;
1350                pathList.add(path) ;
1351            }
1352        } else {
1353            String value = valueTerm.getString() ;
1354            String[] arrayedPath = value.split(separator);
1355            for(String path : arrayedPath) {
1356                pathList.add(path.trim()) ;
1357            }
1358        }
1359
1360        return pathList ;
1361    }
