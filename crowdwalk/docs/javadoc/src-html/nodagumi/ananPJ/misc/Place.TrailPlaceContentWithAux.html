126    /**
127     * 各値をセット
128     * @param _link : エージェントのいるリンク
129     * @param _node : どちらかのノード
130     * @param _entering : _node が入った側かどうか？
131     * @param _advancingDistance : 進行距離
132     * @param safely : enteringNode の設定で、整合性チェックするかどうか
133     */
134    public Place set(MapLink _link, MapNode _node, boolean _entering,
135                     double _advancingDistance, boolean safely) {
136        link = _link ;
137        if(_entering) {
138            setEnteringNode(_node, safely) ;
139        } else {
140            setEnteringNode(link.getOther(_node), safely) ;
141        }
142
143        advancingDistance = _advancingDistance ;
144
145        return this ;
146    }

504    /**
505     * 前進（制約付き）
506     * @param distance : 進める距離
507     * @param limitOnLink : リンク上にとどまるかどうかのチェック
508     * @return Placeそのものを返す。
509     */
510    public Place makeAdvance(double distance, boolean limitOnLink) {
511        makeAdvance(distance) ;
512        if(limitOnLink) {
513            if(isBeforeLink()) {
514                setAdvancingDistance(0.0) ;
515            }
516            if(isBeyondLink()) {
517                setAdvancingDistance(getLinkLength() * AlmostOne) ;
518            }
519        }
520        return this ;
521    }
