019/* tkokada
020 * This class create a dialog to ask how many grids create with selected area.
021 * The dialog is quite similar with GetDoublesDialog class, but it includes
022 * a division method which separate a selected rectangle area with vertical 
023 * division and horizontal division number. Then the rectangle is divided with
024 * division number.
025 */
026
027public class GridMapAreaDialog extends JDialog 
028    implements ActionListener {
029        
030        private JSpinner[] values;
031        
032        public boolean proceed = false;
033        
034        public GridMapAreaDialog(String title,
035                        String[] labels,
036                        double min_height[],
037                        double max_height[],
038                        int vertical_division[],
039                        int horizontal_division[],
040                        double rotation[]) {
041                this.setModal(true);
042                Container panel = getContentPane();
043                panel.setLayout(new GridLayout(3, labels.length));
044                
045                values = new JSpinner[labels.length];
046
047                for (int i = 0; i < labels.length; ++i) {
048                        panel.add(new JLabel(labels[i]));
049                }
050
051                values[0] = new JSpinner(new SpinnerNumberModel(min_height[2],
052                                min_height[0], min_height[1], 0.1));
053                values[1] = new JSpinner(new SpinnerNumberModel(max_height[2],
054                                max_height[0], max_height[1], 0.1));
055                values[2] = new JSpinner(new SpinnerNumberModel(vertical_division[2],
056                                vertical_division[0], vertical_division[1], 1));
057                values[3] = new JSpinner(new SpinnerNumberModel(horizontal_division[2],
058                                horizontal_division[0], horizontal_division[1], 1));
059                values[4] = new JSpinner(new SpinnerNumberModel(rotation[2],
060                                rotation[0], rotation[1], 1.0));
061                /*values[4].addChangeListener(new ChangeListner() {
062                        public void stateChanged(ChangeEvent ce) {
063                                panel.
064                        }
065                });*/
066                
067                for (int i = 0; i < labels.length; ++i) {
068                        panel.add(values[i]);
069                }
070                
071                for (int i = 0; i < labels.length - 2; ++i) {
072                        panel.add(new JLabel());
073                }
074
075                JButton ok = new JButton("OK");
076                ok.addActionListener(this);
077                panel.add(ok);
078                JButton cancel = new JButton("Cancel");
079                cancel.addActionListener(this);
080                panel.add(cancel);
081                
082                this.pack();
083        }
084        
085        @Override
086        public void actionPerformed(ActionEvent e) {
087                if (e.getActionCommand() == "OK") {
088                        proceed = true;
089                        dispose();
090                } else if (e.getActionCommand() == "Cancel") {
091                        proceed = false;
092                        dispose();
093                }
094        }
095        
096        public static double[] showDialog(String title, String[] labels) {
097        final double[] min_height = {-1000, 1000, -1000};
098        final double[] max_height = {-1000, 1000, -1000};
099        final int[] vertical_division = {0, 1000, 0};
100        final int[] horizontal_division = {0, 1000, 0};
101        final double[] rotation = {0, 360, 0};
102
103                return showDialog(title, labels, min_height, max_height,
104                                vertical_division, horizontal_division, rotation);
105        }
106        
107        public static double[] showDialog(String title,
108                        String[] labels,
109                        double[] min_height, 
110                        double[] max_height,
111                        int[] vertical_division,
112                        int[] horizontal_division,
113                        double[] rotation) {
114                GridMapAreaDialog gdd = new GridMapAreaDialog("", labels,
115                                min_height, max_height, vertical_division, horizontal_division,
116                                rotation);
117        gdd.setVisible(true);
118
119        if (!gdd.proceed) return null;
120
121        double[] ret = new double[labels.length];
122        ret[0] = (Double)(gdd.values[0].getValue());
123        ret[1] = (Double)(gdd.values[1].getValue());
124        ret[2] = (Integer) (gdd.values[2].getValue());
125        ret[3] = (Integer) (gdd.values[3].getValue());
126        ret[4] = (Double) (gdd.values[4].getValue());
127        return ret;
128        }
129}
