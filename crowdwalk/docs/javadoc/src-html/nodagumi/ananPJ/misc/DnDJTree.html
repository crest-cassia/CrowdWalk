054    public void dragGestureRecognized(DragGestureEvent dge) {
055        Point pt = dge.getDragOrigin();
056        TreePath path = getPathForLocation(pt.x, pt.y);
057        if(path==null || path.getParentPath()==null) {
058            return;
059        }
060
061        draggedNode = (TreeNode) path.getLastPathComponent();
062        Transferable trans = new RJLTransferable(draggedNode);
063        new DragSource().startDrag(dge, Cursor.getDefaultCursor(), trans, this);
064    }

096    public void dragOver(DropTargetDragEvent dtde) {
097        DataFlavor[] f = dtde.getCurrentDataFlavors();
098        boolean isDataFlavorSupported = f[0].getHumanPresentableName().equals(NAME);
099        if(!isDataFlavorSupported) {
100            //
101            rejectDrag(dtde);
102            return;
103        }
104        /* figure out which cell it's over, no drag to self */
105        Point pt = dtde.getLocation();
106        TreePath path = getPathForLocation(pt.x, pt.y);
107        if(path==null) {
108            //
109            rejectDrag(dtde);
110            return;
111        }
112        Object droppedObject;
113        try {
114            droppedObject = dtde.getTransferable().getTransferData(localObjectFlavor);
115        }catch(Exception ex) {
116            rejectDrag(dtde);
117            return;
118        }
119        MutableTreeNode droppedNode       = (MutableTreeNode) droppedObject;
120        DefaultMutableTreeNode targetNode = (DefaultMutableTreeNode) path.getLastPathComponent();
121        DefaultMutableTreeNode parentNode = (DefaultMutableTreeNode) targetNode.getParent();
122        while(parentNode!=null) {
123            if(droppedNode.equals(parentNode)) {
124                //�e�m�[�h��q�m�[�h�Ƀh���b�v���悤�Ƃ��Ă���
125                rejectDrag(dtde);
126                return;
127            }
128
129            parentNode = (DefaultMutableTreeNode)parentNode.getParent();
130        }
131        
132        //dropTargetNode 
133        dropTargetNode = targetNode; //(TreeNode) path.getLastPathComponent();
134        dtde.acceptDrag(dtde.getDropAction());
135        repaint();
136    }
137    private void rejectDrag(DropTargetDragEvent dtde) {

143    //@Override
144    public void drop(DropTargetDropEvent dtde) {
145        Object droppedObject;
146        try {
147            droppedObject = dtde.getTransferable().getTransferData(localObjectFlavor);
148        }catch(Exception e) {
149            e.printStackTrace();
150            dtde.dropComplete(false);
151            return;
152        }
153        DefaultTreeModel model = (DefaultTreeModel) getModel();
154        Point p = dtde.getLocation();
155        TreePath path = getPathForLocation(p.x, p.y);
156        if(path==null || !(droppedObject instanceof MutableTreeNode)) {
157            dtde.dropComplete(false);
158            return;
159        }
160        Itk.logInfo("drop path is " + path);
161        MutableTreeNode droppedNode       = (MutableTreeNode) droppedObject;
162        DefaultMutableTreeNode targetNode = (DefaultMutableTreeNode) path.getLastPathComponent();
163        DefaultMutableTreeNode parentNode = (DefaultMutableTreeNode) targetNode.getParent();
164        if(targetNode.equals(droppedNode)) {
165            //
166            dtde.dropComplete(false);
167            return;
168        }
169        dtde.acceptDrop(DnDConstants.ACTION_MOVE);
170        model.removeNodeFromParent(droppedNode);
171        Itk.logDebug("now targetNode.getAllowsChildren()== " +targetNode.getAllowsChildren());
172        if(parentNode!=null &&!targetNode.getAllowsChildren()) {
173                //parent is 
174            model.insertNodeInto(droppedNode, parentNode, parentNode.getIndex(targetNode));
175        }else{
176                // parent is LeafNode
177            model.insertNodeInto(droppedNode, targetNode, targetNode.getChildCount());
178        }
179        dtde.dropComplete(true);
180    }

184    static class RJLTransferable implements Transferable {
185        Object object;
186        public RJLTransferable(Object o) {
187            object = o;
188        }
189        //@Override
190        public Object getTransferData(DataFlavor df) throws UnsupportedFlavorException, IOException {
191            if(isDataFlavorSupported(df)) {
192                return object;
193            }else{
194                throw new UnsupportedFlavorException(df);
195            }
196        }
197        //@Override
198        public boolean isDataFlavorSupported(DataFlavor df) {
199            //return (df.getHumanPresentableName().equals(NAME));
200            return (df.equals(localObjectFlavor));
201        }
202        //@Override
203        public DataFlavor[] getTransferDataFlavors() {
204            return supportedFlavors;
205        }
206    }

208    // custom renderer
209    class DnDTreeCellRenderer extends DefaultTreeCellRenderer {
210
211                private boolean isTargetNode;
212        private boolean isTargetNodeLeaf;
213        //private Insets lastItemInsets;
214        Icon node_icon, link_icon, agent_icon, symlink_icon;
215
216
217        public DnDTreeCellRenderer() {
218            super();
219            
220            node_icon = new ImageIcon(getClass().getResource("/img/node_icon.png"));
221            link_icon = new ImageIcon(getClass().getResource("/img/link_icon.png"));
222            agent_icon = new ImageIcon(getClass().getResource("/img/agent_icon.png"));
223            symlink_icon = new ImageIcon(getClass().getResource("/img/symlink_icon.png"));
224        }
225
226        @Override
227        public Component getTreeCellRendererComponent(JTree tree, Object value,
228                                                      boolean isSelected, boolean isExpanded, boolean isLeaf,
229                                                      int row,
230                                                      boolean hasFocus) {
231                super.getTreeCellRendererComponent(tree, value, isSelected, isExpanded, isLeaf, row, hasFocus);
232            isTargetNode = (value == dropTargetNode);
233            isTargetNodeLeaf = (isTargetNode && ((TreeNode)value).getAllowsChildren());
234
235            /* for ob_node */
236            OBNode ob_node = null;
237            if (value instanceof OBNode) {
238                ob_node =  (OBNode)(value);     
239                        } else {
240                                return this;
241                        }
242
243            if (isLeaf) {
244                if (ob_node.getNodeType() == OBNode.NType.NODE) {
245                        setIcon(node_icon);
246                } else if (ob_node.getNodeType() == OBNode.NType.LINK) {
247                        setIcon(link_icon);
248                } else if (ob_node.getNodeType() == OBNode.NType.AGENT) {
249                        setIcon(agent_icon);
250                } else if (ob_node.getNodeType() == OBNode.NType.SYMLINK) {
251                        setIcon(symlink_icon);
252                }
253            } else {
254                /* no op */
255            }
256            String tooltip = ob_node.getHintString();
257            setToolTipText(tooltip);
258            return this;
259        }
260
261
262        @Override
263        public void paintComponent(Graphics g) {
264                super.paintComponent(g);
265            if(isTargetNode) {
266                g.setColor(Color.BLACK);
267                if(!isTargetNodeLeaf) {
268                    g.drawLine(0, 0, getSize().width, 0);
269                }else{
270                    g.drawRect(0, 0, getSize().width-1, getSize().height-1);
271                }
272            }
273        }
