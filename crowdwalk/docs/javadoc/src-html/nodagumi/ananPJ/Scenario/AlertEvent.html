023//============================================================
024/**
025 * Alert Event (EVACUATE)。
026 * 指定した場所のエージェントにメッセージを伝える。
027 * <pre>
028 *  { "type" : "Alert",
029 *    "atTime" : __Time__,
030 *    ("placeTag" : __Tag__, |
031 *     "placeId" : __Int__,)
032 *    ("onoff" : ( true | false ),)?
033 *    "message" : __String__}
034 *
035 *  __Time__ ::= "hh:mm:ss"
036 * </pre>
037 * "placeTag" もしくは "placeId" のいずれかを指定する。
038 * {@code "message"} の利用は、
039 * {@link nodagumi.ananPJ.Agents.RationalAgent RationalAgent} の
040 * {@link nodagumi.ananPJ.Agents.Think.ThinkFormulaAgent#call_listenAlert listenAlert} を
041 * 参照。
042 * <p>
043 * [2015.01.21 I.Noda] 現状では避難だけだが、
044 * いろいろな指示（エージェントの意思変更）・情報提供(エージェントへの条件付与)
045 * できるようにしたほうが良い。
046 * <p>
047 * [2015.0216 I.Noda] 上記を実装。
048 */
049public class AlertEvent extends PlacedEvent {
050    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
051    /**
052     * 情報を示すフラグ
053     */
054    public Term message = null ;
055
056    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
057    /**
058     * 正操作か逆操作かどうか
059     * onoff == true ならば、Alert をセット
060     * そうでなければ、Alert を clear する。
061     */
062    public boolean onoff = true ;
063
064    //----------------------------------------
065    /**
066     * JSON Term による setup
067     */
068    public void setupByJson(Scenario _scenario,
069                            Term eventDef) {
070        super.setupByJson(_scenario, eventDef) ;
071
072        message = eventDef.getArgTerm("message") ;
073        if(eventDef.hasArg("onoff")) {
074            onoff = eventDef.getArgBoolean("onoff") ;
075        } else {
076            onoff = true ;
077        }
078    }
079
080    //----------------------------------------
081    /**
082     * 終了イベント発生処理
083     * @param currentTime : 現在の絶対時刻
084     * @param map : 地図データ
085     * @return : true を返す。
086     */
087    @Override
088    public boolean occur(SimTime currentTime, NetworkMap map) {
089        return occur(currentTime, map, !onoff) ;
090    }
091
092    //----------------------------------------
093    /**
094     * 終了イベント発生逆処理
095     * @param currentTime : 現在の絶対時刻
096     * @param map : 地図データ
097     * @return : true を返す。
098     */
099    @Override
100    public boolean unoccur(SimTime currentTime, NetworkMap map) {
101        return occur(currentTime, map, onoff) ;
102    } ;
103
104    //----------------------------------------
105    /**
106     * 終了イベント発生処理
107     * @param currentTime : 現在の絶対時刻
108     * @param map : 地図データ
109     * @param inverse : 逆操作かどうか
110     * @return : true を返す。
111     */
112    public boolean occur(SimTime currentTime, NetworkMap map, boolean inverse) {
113        for(MapLink link : map.getLinks()) {
114            if(checkTagOrId(link)) {
115                link.addAlertMessage(message, currentTime, !inverse) ;
116                Itk.logInfo("AlertEvent",onoff,link,message, currentTime) ;
117            }
118        }
119        return true ;
120    }
121} // class AlertEvent
