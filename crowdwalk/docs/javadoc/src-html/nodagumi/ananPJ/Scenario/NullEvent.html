024/**
025 * Null Event。
026 * an event to do nothing.
027 * All attributes are ignored.
028 * So, it can be used as an commented event.
029 * <pre>
030 *  { "type" : "None",
031 *    ... }
032 * </pre>
033 */
034public class NullEvent extends EventBase {
035    //----------------------------------------
036    /**
037     * Null イベント発生処理
038     * 実は何もしない
039     * @param currentTime : 現在の絶対時刻
040     * @param map : 地図データ
041     * @return : true を返す。
042     */
043    @Override
044    public boolean occur(SimTime currentTime, NetworkMap map) {
045        return true ;
046    }
047
048    //----------------------------------------
049    /**
050     * Start イベント発生逆処理
051     * @param currentTime : 現在の絶対時刻
052     * @param map : 地図データ
053     * @return : true を返す。
054     */
055    @Override
056    public boolean unoccur(SimTime currentTime, NetworkMap map) {
057        return true ;
058    }
059} // class NullEvent
