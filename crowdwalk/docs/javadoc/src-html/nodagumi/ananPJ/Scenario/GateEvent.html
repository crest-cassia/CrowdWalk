023/**
024 * Gate Event
025 */
026abstract public class GateEvent extends PlacedEvent {
027    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
028    /**
029     * ゲートを示す tag
030     * デフォルトでは、placeTag と同じもの。
031     */
032    public Term gateTag = null ;
033    
034    //----------------------------------------
035    /**
036     * JSON Term による setup
037     */
038    public void setupByJson(Scenario _scenario,
039                            Term _eventDef) {
040        super.setupByJson(_scenario, _eventDef) ;
041
042        gateTag = _eventDef.getArgTerm("gateTag") ;
043        if(gateTag == null) {
044            gateTag = placeTag ;
045        }
046    }
047
048    //----------------------------------------
049    /**
050     * CSV による setup
051     */
052    @Override
053    public void setupByCsvColumns(Scenario _scenario,
054                                  ShiftingStringList columns) {
055        super.setupByCsvColumns(_scenario, columns) ;
056
057        gateTag = placeTag ;
058    }
059
060    //----------------------------------------
061    /**
062     * 文字列化 後半
063     */
064    public String toStringTail() {
065        return (super.toStringTail() + "," + "gate=" + gateTag);
066    }
067} // class GateEvent
