013package nodagumi.ananPJ.Scenario;
014
015import nodagumi.ananPJ.NetworkMap.NetworkMap;
016import nodagumi.ananPJ.NetworkMap.Link.*;
017import nodagumi.ananPJ.NetworkMap.Node.*;

023//============================================================
024/**
025 * Set Tag Event (SET/REMOVE)。
026 * タグをリンクに追加/除去する。
027 * <pre>
028 *  { "type" : ( "SetTag" | "AddTag" | "RemoveTag" ),
029 *    "atTime" : __Time__,
030 *    ("placeTag" : __Tag__, |
031 *     "placeId" : __Int__,)
032 *    ("onoff" : ( true | false ),)?
033 *    "noticeTag" : __Tag__}
034 *
035 *  __Time__ ::= "hh:mm:ss"
036 * </pre>
037 * "placeTag" もしくは "placeId" のいずれかを指定する。
038 */
039public class SetTagEvent extends PlacedEvent {
040    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
041    /**
042     * 付加・削除するタグ
043     */
044    public Term noticeTag = null ;
045
046    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
047    /**
048     * 付加・削除の識別
049     */
050    public boolean onoff = true ;
051
052    //----------------------------------------
053    /**
054     * JSON Term による setup
055     */
056    public void setupByJson(Scenario _scenario,
057                            Term eventDef) {
058        super.setupByJson(_scenario, eventDef) ;
059
060        noticeTag = eventDef.getArgTerm("noticeTag") ;
061        String eventType = eventDef.getArgString("type") ;
062        if(eventType.equals("AddTag")) {
063            onoff = true ;
064        } else if (eventType.equals("RemoveTag")) {
065            onoff = false ;
066        } else if(eventDef.hasArg("onoff")) {
067            onoff = eventDef.getArgBoolean("onoff") ;
068        } else {
069            onoff = true ;
070        }
071    }
072
073    //----------------------------------------
074    /**
075     * CSV による setup
076     */
077    @Override
078    public void setupByCsvColumns(Scenario _scenario,
079                                  ShiftingStringList columns) {
080        super.setupByCsvColumns(_scenario, columns) ;
081
082        String command = columns.nth(3) ;
083        String subcoms[] = command.split(":") ;
084        if(subcoms.length < 2 || subcoms.length > 2 ||
085           subcoms[1].length() == 0) {
086            Itk.logWarn("Strange commands for SetTag event.") ;
087            Itk.logWarn_("columns", columns) ;
088        } else {
089            noticeTag = new Term(subcoms[1], true) ;
090        }
091        if(subcoms[0].equals("SET")) {
092            onoff = true ;
093        } else if(subcoms[0].equals("REMOVE")) {
094            onoff = false ;
095        } else {
096            Itk.logWarn("Strange commands for SetTag event.") ;
097            Itk.logWarn_("columns", columns) ;
098        }
099    }
100
101    //----------------------------------------
102    /**
103     * タグ操作イベント発生処理
104     * @param currentTime : 現在の絶対時刻
105     * @param map : 地図データ
106     * @return : true を返す。
107     */
108    @Override
109    public boolean occur(SimTime currentTime, NetworkMap map) {
110        return occur(currentTime, map, false) ;
111    }
112
113    //----------------------------------------
114    /**
115     * タグ操作イベント発生逆処理
116     * @param currentTime : 現在の絶対時刻
117     * @param map : 地図データ
118     * @return : true を返す。
119     */
120    @Override
121    public boolean unoccur(SimTime currentTime, NetworkMap map) {
122        return occur(currentTime, map, true) ;
123    }
124    //----------------------------------------
125    /**
126     * タグ操作イベント発生処理(generic)
127     * @param currentTime : 現在の絶対時刻
128     * @param map : 地図データ
129     * @param inverse : 逆操作かどうか
130     * @return : true を返す。
131     */
132    public boolean occur(SimTime currentTime, NetworkMap map, boolean inverse) {
133        for(MapLink link : map.getLinks()) {
134            if(checkTagOrId(link)) {
135                if(onoff ^ inverse) {
136                    link.addTag(noticeTag.getString()) ;
137                } else {
138                    link.removeTag(noticeTag.getString()) ;
139                }
140            }
141        }
142        for(MapNode node : map.getNodes()) {
143            if(checkTagOrId(node)) {
144                if(onoff ^ inverse) {
145                    node.addTag(noticeTag.getString()) ;
146                } else {
147                    node.removeTag(noticeTag.getString()) ;
148                }
149            }
150        }
151        return true ;
152    }
153
154    //----------------------------------------
155    /**
156     * 文字列化 後半
157     */
158    public String toStringTail() {
159        return (super.toStringTail() +
160                "," + "notice=" + noticeTag +
161                "," + "on=" + onoff);
162    }
163} // class SetTagEvent
