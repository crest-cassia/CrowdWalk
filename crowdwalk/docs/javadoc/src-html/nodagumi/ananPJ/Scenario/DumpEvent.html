037/**
038 * This event cause to interrupt simulation and make a log dump.
039 * <pre>
040 *  { "type" : "Dump" 
041 *    "atTime" : __Time__,
042 *    "format" : "GenerationFile",
043 *    "filename" : __FileName__}
044 *
045 *  __Time__ ::= "hh:mm:ss"
046 * </pre>
047 */
048public class DumpEvent extends EventBase {
049    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
050    /**
051     * dump file name
052     */
053    public String filename = null ;
054
055    //============================================================
056    //============================================================
057    /**
058     * Enum for format of dump file
059     */
060    static public enum Format {
061        GenerationFile
062    }
063    //============================================================
064    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
065    /**
066     * Lexicon for format of dump file
067     */
068    static public Lexicon formatLexicon = new Lexicon() ;
069    static {
070        formatLexicon.registerEnum(Format.class) ;
071    }
072    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
073    /**
074     * Format Type
075     */
076    public Format format ;
077
078    //------------------------------------------------------------
079    /**
080     * JSON Term による setup.
081     * format と filename を設定する。
082     */
083    public void setupByJson(Scenario _scenario,
084                            Term eventDef) {
085        super.setupByJson(_scenario, eventDef) ;
086        
087        format = (Format)formatLexicon.lookUp(eventDef.getArgString("format")) ;
088        if(format == null) {
089            Itk.logError("Wrong format in DumpEvent:", eventDef) ;
090            Itk.quitByError() ;
091        }
092
093        String bareFilename = eventDef.getArgString("filename") ;
094        if(bareFilename == null) {
095            Itk.logError("Wrong dump filename in DumpEvent:", eventDef) ;
096            Itk.quitByError() ;
097        }
098        filename =
099            getScenario().getProperties()
100            .furnishPropertiesDirPath(bareFilename, false, false) ;
101    }
102
103
104    //------------------------------------------------------------
105    /**
106     * Dump イベント発生処理。
107     * エージェントの状態をダンプする。
108     * @param currentTime : 現在の絶対時刻
109     * @param map : 地図データ
110     * @return : true を返す。
111     */
112    @Override
113    public boolean occur(SimTime currentTime, NetworkMap map) {
114        switch(format){
115        case GenerationFile:
116            dumpInGenerationFileFormat(currentTime) ;
117            break ;
118        default:
119            Itk.logError("unknown dump format:", format) ;
120            Itk.quitByError() ;
121        }
122        return true ;
123    }
124
125    //------------------------------------------------------------
126    /**
127     * Dump イベント発生逆処理。
128     * 何もしない。
129     * @param currentTime : 現在の絶対時刻
130     * @param map : 地図データ
131     * @return : true を返す。
132     */
133    @Override
134    public boolean unoccur(SimTime currentTime, NetworkMap map) {
135        return true ;
136    }
137
138    //------------------------------------------------------------
139    /**
140     * Dump in GenerationFile format
141     */
142    public void dumpInGenerationFileFormat(SimTime currentTime) {
143        Term ruleList = Term.newArrayTerm() ;
144
145        dumpRuleTermForRemainingRules(currentTime, ruleList) ;
146        dumpRuleTermForWalkingAgents(currentTime, ruleList) ;
147
148        outputDumpFileInGenerationFileFormat(ruleList) ;
149    }
150    
151    //------------------------------------------------------------
152    /**
153     * output dump file
154     */
155    private void outputDumpFileInGenerationFileFormat(Term ruleList) {
156        try {
157            File dumpFile = new File(filename) ;
158            FileWriter writer = new FileWriter(dumpFile) ;
159            writer.write("#{ \"version\" : 2}\n") ;
160            writer.write(ruleList.toJson(true)) ;
161            writer.close() ;
162        } catch (IOException ex){
163            Itk.dumpStackTrace() ;
164            Itk.logError("IOError",ex) ;
165            Itk.quitByError() ;
166        }
167    }
168        
169    //------------------------------------------------------------
170    /**
171     * generate Dump term for remaining generation rules.
172     */
173    private Term dumpRuleTermForRemainingRules(SimTime currentTime,
174                                               Term ruleList) {
175        AgentHandler handler = getAgentHandler() ;
176        AgentFactoryList factoryList = handler.getAgentFactoryList() ;
177
178        // generate rule
179        for(String ruleName :
180                factoryList.getAgentFactoryConfigNameTable().keySet()) {
181            AgentFactoryConfig factoryConfig =
182                factoryList.getAgentFactoryConfigByName(ruleName) ;
183
184            Term rule = factoryConfig.toTermForRemainingAgents(currentTime) ;
185
186            if(rule != null) {
187                ruleList.addNth(rule) ;
188            }
189        }
190        return ruleList ;
191    }
192        
193    //------------------------------------------------------------
194    /**
195     * generate Dump term for walking agents in individual gen rule format
196     * in GenerationFile format.
197     */
198    private Term dumpRuleTermForWalkingAgents(SimTime currentTime,
199                                              Term ruleList) {
200        AgentHandler handler = getAgentHandler() ;
201
202        // collect agents by each rule
203        HashMap<String, ArrayList<AgentBase>> agentTable =
204            new HashMap<String, ArrayList<AgentBase>>() ;
205
206        for(AgentBase agent : handler.getWalkingAgentCollection()) {
207            String ruleName = agent.getFactory().getRuleName() ;
208            if(!agentTable.containsKey(ruleName)) {
209                agentTable.put(ruleName, new ArrayList<AgentBase>()) ;
210            }
211            agentTable.get(ruleName).add(agent) ;
212        }
213
214        // generate rules
215        for(String ruleName : agentTable.keySet()) {
216            AgentFactoryConfig factoryConfig =
217                handler.getAgentFactoryConfigByName(ruleName) ;
218            Term rule =
219                dumpTermForAgentsInOneRule(factoryConfig,
220                                           agentTable.get(ruleName),
221                                           currentTime) ;
222            ruleList.addNth(rule) ;
223        }
224
225        return ruleList ;
226    }
227
228    //------------------------------------------------------------
229    /**
230     * dump agents in one rule
231     */
232    private Term dumpTermForAgentsInOneRule(AgentFactoryConfig factoryConfig,
233                                            ArrayList<AgentBase> agentList,
234                                            SimTime currentTime) {
235        Term indivList = Term.newArrayTerm() ;
236        for(AgentBase agent : agentList) {
237            Term agentConf =
238                dumpTermForOneAgent(agent, currentTime) ;
239            indivList.addNth(agentConf) ;
240        }
241
242        Term rule = factoryConfig.toTermForWalkingAgents(indivList,
243                                                         currentTime,
244                                                         "_dumped_") ;
245
246        return rule ;
247    }
248    
249    //------------------------------------------------------------
250    /**
251     * dump one agent
252     */
253    private Term dumpTermForOneAgent(AgentBase agent, SimTime currentTime) {
254        Term agentConf = 
255            agent.dumpTermForIndividualConfig(currentTime) ;
256        return agentConf ;
257    }
258    
259    //------------------------------------------------------------
260    /**
261     * 文字列化 後半
262     */
263    public String toStringTail() {
264        return (super.toStringTail() +
265                "," + "filename=" + filename +
266                "," + "format=" + format);
267    }

272} // class DumpEvent
