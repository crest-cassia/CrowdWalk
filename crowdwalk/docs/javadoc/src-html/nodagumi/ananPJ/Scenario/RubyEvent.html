014package nodagumi.ananPJ.Scenario;
015
016import nodagumi.ananPJ.NetworkMap.NetworkMap;
017import nodagumi.ananPJ.Scenario.Scenario;
018import nodagumi.ananPJ.misc.SimTime;
019
020import nodagumi.Itk.* ;
021
022//======================================================================
023/**
024 * This event cause to call RubyEvent object defined by users.
025 * <pre>
026 *  { "type" : "Ruby" 
027 *    "atTime" : __Time__,
028 *    "rubyClass" : __EventClassInRuby__,
029 *    __otherItem__:__Value__,
030 *  }
031 *  __Time__ ::= "hh:mm:ss"
032 *  __EventClassInRuby__ : Ruby側の対応するクラス。
033 * </pre>
034 * <p>
035 * __EventClassInRuby__ は、RubyEventBase クラスの子クラスとして
036 * 読み込まれる Ruby プログラムの中で定義されている必要がある。
037 * "atTime" で指定した時刻に、このイベントの occur() が呼び出される。
038 * </p><p>
039 * Sample:
040 * "./sample/generatedTown/gridTown03.scnr.json" および
041 * "./sample/generatedTown/SampleEvent.rb". 
042 */
043public class RubyEvent extends EventBase {
044    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
045    /**
046     * Ruby Engine
047     */
048    public ItkRuby rubyEngine = null ;
049    
050    /**
051     * Ruby Object
052     */
053    public Object rubyEvent = null ;
054    
055    /**
056     * ルビークラス名
057     */
058    public String rubyClass ;
059
060    //------------------------------------------------------------
061    /**
062     * JSON Term による setup.
063     * format と filename を設定する。
064     */
065    public void setupByJson(Scenario _scenario,
066                            Term _eventDef) {
067        super.setupByJson(_scenario, _eventDef) ;
068        
069        rubyClass = eventDef.getArgString("rubyClass") ;
070        if(rubyClass == null) {
071            Itk.logError("ruby class is not specified in RubyEvent.",
072                         _eventDef) ;
073            Itk.quitByError() ;
074        }
075        Itk.logDebug("RubyEvent:rubyClass", rubyClass) ;
076
077    }
078
079    //------------------------------------------------------------
080    /**
081     * Ruby Engine の設定
082     */
083    public void setupRubyEngine(ItkRuby _rubyEngine) {
084        if(_rubyEngine == null) {
085            Itk.logError("RubyEvent require Ruby Engine") ;
086            Itk.quitByError() ;
087        }
088        
089        rubyEngine = _rubyEngine ;
090
091        rubyEvent =
092            rubyEngine.newInstanceOfClass(rubyClass, this) ;
093    }
094
095    //------------------------------------------------------------
096    /**
097     * Ruby イベント発生処理。
098     * 指定されたRubyClassのインスタンスを生成し、呼び出す。
099     * @param currentTime : 現在の絶対時刻
100     * @param map : 地図データ
101     * @return : true を返す。
102     */
103    @Override
104    public boolean occur(SimTime currentTime, NetworkMap map) {
105        /* 初回は、ruby のセットアップ */
106        if(rubyEvent == null) {
107            setupRubyEngine(getSimulator().getRubyEngine()) ;
108        }
109        
110        Object ret =
111            rubyEngine.callMethod(rubyEvent, "occur",
112                                  currentTime, map) ;
113        return (boolean)ret ;
114    }
115
116    //------------------------------------------------------------
117    /**
118     * Ruby イベント発生逆処理。
119     * 何もしない。
120     * @param currentTime : 現在の絶対時刻
121     * @param map : 地図データ
122     * @return : true を返す。
123     */
124    @Override
125    public boolean unoccur(SimTime currentTime, NetworkMap map) {
126        return true ;
127    }
128
129    //------------------------------------------------------------
130    /**
131     * 文字列化 後半
132     */
133    public String toStringTail() {
134        return (super.toStringTail() +
135                "," + "rubyClass=" + rubyClass) ;
136    }
137    //============================================================
138    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
139    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
140    //------------------------------------------------------------
141} // class RubyEvent
