013package nodagumi.ananPJ.Scenario;
014
015import nodagumi.ananPJ.NetworkMap.NetworkMap;
016import nodagumi.ananPJ.NetworkMap.Link.*;
017import nodagumi.ananPJ.NetworkMap.Node.*;
018import nodagumi.ananPJ.Scenario.Scenario;
019import nodagumi.ananPJ.Simulator.EvacuationSimulator;
020import nodagumi.ananPJ.Simulator.AgentHandler;
021import nodagumi.ananPJ.misc.SimTime;
022import nodagumi.ananPJ.misc.SimClock;
023
024import nodagumi.Itk.* ;
025
026//============================================================
027/**
028 * EventBase class
029 */
030abstract public class EventBase {
031    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
032    /**
033     * 所属するシナリオ。
034     */
035    public Scenario scenario = null ;
036
037    /**
038     * その他、拡張用の情報。
039     * シナリオで与えられる定義情報。json のみ対応。
040     */
041    public Term eventDef = null ;
042
043    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
044    /**
045     * ID。
046     * シナリオ内のイベント相互で参照するためのID。
047     * null でも良いが、他から参照できない。
048     */
049    public String id = null ;
050
051    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
052    /**
053     * イベント発生時刻文字列
054     */
055    public String atTimeString = null ;
056
057    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
058    /**
059     * イベント発生時刻 (絶対時刻)
060     */
061    public SimTime atTime = null ;
062
063    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
064    /**
065     * コメント
066     */
067    public String comment = null ;
068
069    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
070    /**
071     * 過去の（既に完了した）ものか？
072     */
073    public boolean isCompleted = false ;
074
075    //----------------------------------------
076    /**
077     * JSON Term による setup
078     */
079    public void setupByJson(Scenario _scenario,
080                            Term _eventDef) {
081        scenario = _scenario ;
082        eventDef = _eventDef ;
083
084        id = eventDef.getArgString("id") ;
085        atTimeString = eventDef.getArgString("atTime") ;
086    }
087
088    //----------------------------------------
089    /**
090     * CSV による setup
091     */
092    public void setupByCsvColumns(Scenario _scenario,
093                                  ShiftingStringList columns) {
094        scenario = _scenario ;
095
096        if(columns.nth(0).length() > 0) {
097            id = columns.nth(0) ;
098        }
099
100        atTimeString = columns.nth(4) ;
101    }
102
103    //----------------------------------------
104    /**
105     * originClock によるイベント発生時刻の設定。
106     */
107    public void setupAtTime(SimClock originClock) {
108        atTime = originClock.newSimTimeByString(atTimeString) ;
109    }
110
111    //------------------------------------------------------------
112    //  accessors
113    //------------------------------------------------------------
114    /**
115     * get scenario
116     */
117    public Scenario getScenario() {
118        return scenario ;
119    }
120
121    //----------------------------------------
122    /**
123     * get simulator
124     */
125    public EvacuationSimulator getSimulator() {
126        return getScenario().getSimulator() ;
127    }
128    
129    //----------------------------------------
130    /**
131     * get agent handler
132     */
133    public AgentHandler getAgentHandler() {
134        return getSimulator().getAgentHandler() ;
135    }
136    
137    //----------------------------------------
138    /**
139     * get event def
140     */
141    public Term getEventDef() {
142        return eventDef ;
143    }
144
145    //----------------------------------------
146    /**
147     * id を持つかどうか
148     * @return 持てば true
149     */
150    public boolean hasId() {
151        return id != null ;
152    }
153
154    //----------------------------------------
155    /**
156     * 発生チェックと実行(以前の checkIfHappend)
157     * @param currentTime : 現在の絶対時刻
158     * @return 実施したら true
159     */
160    public boolean tryOccur(SimTime currentTime, NetworkMap map) {
161        if(shouldOccurAt(currentTime)) {
162            isCompleted = occur(currentTime, map) ;
163            return true ;
164        } else {
165            return false ;
166        }
167    }
168
169    //----------------------------------------
170    /**
171     * 発生時刻チェック
172     * @param currentTime : 現在の絶対時刻
173     * @return 発生時刻なら true
174     */
175    public boolean shouldOccurAt(SimTime currentTime) {
176        return !isCompleted && currentTime.isAfterOrAt(atTime) ;
177    }
178
179    //----------------------------------------
180    /**
181     * イベント発生処理 (以前の setEnabled(true))
182     * @param currentTime : 現在の絶対時刻
183     * @param map : 地図データ
184     * @return : 完了したら true を返す。false を返すと、延々呼び出される。
185     */
186    abstract public boolean occur(SimTime currentTime, NetworkMap map) ;
187
188    //----------------------------------------
189    /**
190     * イベント発生逆処理 (以前の setEnabled(false))
191     * @param currentTime : 現在の絶対時刻
192     * @param map : 地図データ
193     * @return : 完了したら true を返す。
194     */
195    abstract public boolean unoccur(SimTime currentTime, NetworkMap map) ;
196
197    //----------------------------------------
198    /**
199     * 文字列化
200     */
201    public String toString() {
202        return (getClass().getSimpleName() +
203                "[" + id + ":" +
204                "," + "@" + atTimeString +
205                toStringTail() + "]") ;
206    }
207
208    //----------------------------------------
209    /**
210     * 文字列化 後半
211     */
212    public String toStringTail() {
213        return "" ;
214    }
215
216} // class EventBase
