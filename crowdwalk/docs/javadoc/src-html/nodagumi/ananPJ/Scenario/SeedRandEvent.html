014package nodagumi.ananPJ.Scenario;
015
016import nodagumi.ananPJ.NetworkMap.NetworkMap;
017import nodagumi.ananPJ.Scenario.Scenario;
018import nodagumi.ananPJ.misc.SimTime;
019
020import nodagumi.Itk.* ;

022//======================================================================
023/**
024 * This event cause to set random seed.
025 * <pre>
026 *  { "type" : "SeedRand" 
027 *    "atTime" : __Time__,
028 *    "seed" : __Integer__,
029 *    "timing" : __Timing__
030 *  }
031 *  
032 *  __Timing__ ::= "atEvent" | "afterGeneration"
033 * </pre>
034 * <p>
035 * "seed" で指定した値を CrowdWalkSimulator の random にセットする。
036 * </p><p>
037 * "Dump" イベントなどで状態をセーブした後、再スタートで全く同じ現象を
038 * 発生させたい場合、
039 * "Dump" イベントの時刻（および再スタートする際のruleのstart time）
040 * の１秒後を、"atTime" に指定するする。
041 * "Dump" 側のシナリオと、再スタート側のシナリオ両方に同じように記述すること。
042 * </p><p>
043 * "timing" には、どの時点で seed を set するかを指定する。
044 * <ul>
045 *  <li> "atEvent" : このイベントが発生したタイミング。 </li>
046 *  <li> "afterGeneration" : 生成ルールの直後。Dump との動機を取る時に便利。 </li>
047 * </ul>
048 * デフォルトは "atEvent"。
049 * </p><p>
050 * サンプル：<br/>
051 * "./sample/generatedTown/gridTown02d.scnr.json" および
052 * "./sample/generatedTown/gridTown02ds.scnr.json". 
053 * </p>
054 */
055public class SeedRandEvent extends EventBase {
056    //============================================================
057    //============================================================
058    /**
059     * set seed するタイミング用の enum
060     */
061    static public enum Timing {
062        atEvent,
063        afterGeneration,
064        none } ;
065    
066    //============================================================
067    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
068    /**
069     * set seed するタイミング用の enum の Lexicon
070     */
071    static public Lexicon timingLexicon = new Lexicon() ;
072    static {
073        timingLexicon.registerEnum(Timing.class) ;
074    } ;
075    
076    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
077    /**
078     * seed value
079     */
080    public long seed ;
081    
082    /**
083     * seed value
084     */
085    public Timing timing ;
086    
087    //------------------------------------------------------------
088    /**
089     * JSON Term による setup.
090     * format と filename を設定する。
091     */
092    public void setupByJson(Scenario _scenario,
093                            Term _eventDef) {
094        super.setupByJson(_scenario, _eventDef) ;
095    }
096
097    //------------------------------------------------------------
098    /**
099     * SeedRand イベント発生処理。
100     * @param currentTime : 現在の絶対時刻
101     * @param map : 地図データ
102     * @return : true を返す。
103     */
104    @Override
105    public boolean occur(SimTime currentTime, NetworkMap map) {
106        if(eventDef.hasArg("seed")) {
107            seed = eventDef.getArgInt("seed") ;
108        } else {
109            Itk.logError("SeedRandEvent", "must specify 'seed' value.") ;
110            Itk.quitByError() ;
111        }
112        
113        if(eventDef.hasArg("timing")) {
114            String timingVal = eventDef.getArgString("timing") ;
115            timing = (Timing)(timingLexicon.lookUp(timingVal)) ;
116        } else {
117            timing = Timing.atEvent ;
118        }
119
120        switch(timing) {
121        case atEvent:
122            getSimulator().getRandom().setSeed(seed) ;
123            break ;
124        case afterGeneration:
125            getSimulator().reserveDelayedSetSeed(timing, seed) ;
126            break ;
127        default:
128            Itk.logError("unknown timing value.", eventDef) ;
129            Itk.quitByError() ;
130        }
131        
132        return true ;
133    }
134
135    //------------------------------------------------------------
136    /**
137     * SeedRand イベント発生逆処理。
138     * 何もしない。
139     * @param currentTime : 現在の絶対時刻
140     * @param map : 地図データ
141     * @return : true を返す。
142     */
143    @Override
144    public boolean unoccur(SimTime currentTime, NetworkMap map) {
145        return true ;
146    }
147
148    //------------------------------------------------------------
149    /**
150     * 文字列化 後半
151     */
152    public String toStringTail() {
153        return (super.toStringTail() +
154                "," + "seed=" + seed) ;
155    }
156    //============================================================
157    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
158    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
159    //------------------------------------------------------------
160} // class RubyEvent
