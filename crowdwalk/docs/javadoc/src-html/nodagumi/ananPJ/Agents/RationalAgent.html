014package nodagumi.ananPJ.Agents;
015
016import java.util.Random;
017import java.util.HashMap;
018
019import nodagumi.ananPJ.NetworkMap.Link.*;
020import nodagumi.ananPJ.NetworkMap.Node.*;
021import nodagumi.ananPJ.Agents.AgentBase;
022import nodagumi.ananPJ.Agents.BustleAgent ;
023import nodagumi.ananPJ.Agents.Think.ThinkEngine;
024import nodagumi.ananPJ.misc.SimTime;
025
026import nodagumi.Itk.*;
027
028//======================================================================
029/**
030 * 理性のある、ルールに基づいて行動するエージェント
031 *
032 * <h3> config, fallbackResources に書ける設定 </h3>
033 * {@link BustleAgent} に加えて、
034 * <pre>
035 *  {
036 *    "margin" : __double__ // 距離コストに重畳するノイズの大きさ。一様乱数。
037 *    "rule" : __ThinkRule__ // 思考ルール
038 * }
039 * </pre>
040 * 思考ルールの記述法については {@link nodagumi.ananPJ.Agents.Think.ThinkEngine ThinkEngine} 参照。
041 */
042public class RationalAgent extends BustleAgent {
043    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
044    /**
045     * クラス名。
046     * ClassFinder でも参照できるようにしておく。
047     */
048    public static String typeString = "RationalAgent" ;
049    public static String getTypeName() { return typeString ;}
050
051    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
052    /**
053     * 気まぐれ度合い（cost への random の度合い）の規定値
054     */
055    static final public double Fallback_CapriciousMargin = 200.0 ;
056
057    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
058    /**
059     * 気まぐれ度合い（cost への random の度合い）
060     */
061    public double capriciousMargin = Fallback_CapriciousMargin ;
062
063    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
064    /**
065     * その場Uターンをするかどうかのフラグ。
066     * これが true になると、
067     * calcSpeed() で speed = 0.0 とし、
068     * advanceNextPlace で、対向レーンに移る。
069     */
070    private boolean turnAroundP = false ;
071
072    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
073    /**
074     * 推論エンジン
075     */
076    public ThinkEngine thinkEngine = new ThinkEngine() ;
077
078    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
079    /**
080     * alert された message
081     */
082    public HashMap<Term, SimTime> alertedMessageTable =
083        new HashMap<Term, SimTime>() ;
084
085    //------------------------------------------------------------
086    // コンストラクタ
087    //------------------------------------------------------------
088    /**
089     * 引数なしconstractor。 ClassFinder.newByName で必要。
090     */
091    public RationalAgent(){}
092
093    //------------------------------------------------------------
094    /**
095     * Conf による初期化。
096     */
097    @Override
098    public void initByConf(Term conf, Term fallback) {
099        super.initByConf(conf, fallback) ;
100
101        capriciousMargin = getDoubleFromConfig("margin", capriciousMargin) ;
102
103        thinkEngine.setAgent(this) ;
104        thinkEngine.setRule(getTermFromConfig("rule", null)) ;
105    } ;
106
107    //------------------------------------------------------------
108    /**
109     * あるwayを選択した場合の目的地(_target)までのコスト。
110     * 正規のコストに、ランダム要素を加味する。
111     */
112    @Override
113    public double calcCostFromNodeViaLink(MapLink _link, MapNode _node, Term _target) throws TargetNotFoundException {
114        double cost = super.calcCostFromNodeViaLink(_link, _node, _target) ;
115        double noise = capriciousMargin * getRandomDouble() ;
116        return cost + noise;
117    }
118
119    //------------------------------------------------------------
120    // alertMessage
121    //------------------------------------------------------------
122    /**
123     * Alert 関係
124     */
125    public void alertMessage(Term message, SimTime alertTime) {
126        alertedMessageTable.put(message, alertTime) ;
127        Itk.logInfo("hear Alert Message", this, message, alertTime,
128                    currentPlace) ;
129    }
130
131    //------------------------------------------------------------
132    // 推論
133    //------------------------------------------------------------
134    /**
135     * シミュレーション各サイクルの前半に呼ばれる。
136     */
137    @Override
138    public void preUpdate(SimTime currentTime) {
139        this.currentTime = currentTime ;
140        thinkCycle() ;
141        super.preUpdate(currentTime) ;
142    }
143
144    //------------------------------------------------------------
145    /**
146     * 思考ルーチン
147     * 状態が変わる毎に呼ばれるべき。
148     */
149    public Term thinkCycle() {
150        return thinkEngine.think() ;
151    }
152
153    //------------------------------------------------------------
154    // Uターン関連
155    //------------------------------------------------------------
156    /**
157     * Uターンフラグを on
158     */
159    public boolean setTurnAround() {
160        return setTurnAround(true) ;
161    }
162    
163    /**
164     * Uターンフラグを on/off
165     */
166    public boolean setTurnAround(boolean _turnAroundP) {
167        return turnAroundP = _turnAroundP ;
168    }
169
170    //------------------------------------------------------------
171    /**
172     * 速度計算(turn around を考慮)
173     */
174    @Override
175    protected double calcSpeed(double previousSpeed, SimTime currentTime) {
176        if(turnAroundP) {
177            return 0.0 ;
178        } else {
179            return super.calcSpeed(previousSpeed, currentTime) ;
180        }
181    }
182    
183    //------------------------------------------------------------
184    /**
185     * 次の位置の計算(turn around を考慮)
186     */
187    @Override
188    protected boolean advanceNextPlace(double _speed, SimTime currentTime,
189                                       boolean stayOnLink) {
190        boolean _turnAroundP = turnAroundP ;
191        turnAroundP = false ;
192        if(_turnAroundP && canTurnAround()) {
193            nextPlace.set(currentPlace) ;
194            nextPlace.turnAround() ;
195            return false ;
196        } else {
197            return super.advanceNextPlace(_speed, currentTime, stayOnLink) ;
198        }
199    }
200
201    //------------------------------------------------------------
202    /**
203     * 現在地でturn around できるかどうか。
204     */
205    protected boolean canTurnAround() {
206        return currentPlace.canTurnAround() ;
207    }
208    
209} // class RationalAgent
