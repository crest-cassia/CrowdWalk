017/**
018 * エージェント生成機構。(fromNode)
019 */
020public class AgentFactoryFromNode extends AgentFactory {
021    MapNode startNode;
022
023    //------------------------------------------------------------
024    /**
025     *  Config によるコンストラクタ
026     */
027    public AgentFactoryFromNode(AgentFactoryConfig config,
028                                OBNode _startPlace,
029                                int _totalInFactory,
030                                Random _random) {
031        super(config, _startPlace, _totalInFactory, _random) ;
032    }
033
034    //------------------------------------------------------------
035    /**
036     * Config による初期化
037     */
038    @Override
039    public void init(AgentFactoryConfig config,
040                     OBNode _startPlace,
041                     int _totalInFactory,
042                     Random random) {
043        super.init(config, _startPlace, _totalInFactory, random) ;
044        startNode = (MapNode)_startPlace ;
045    }
046
047    //------------------------------------------------------------
048    /**
049     * エージェントを初期位置に置く。
050     */
051    @Override
052    protected void placeAgent(AgentBase agent) {
053        agent.place(null, startNode, 0.0) ;
054    }
055
056    //------------------------------------------------------------
057    /**
058     * エージェント生成ルールの情報を文字列で返す。
059     * パネル表示用。
060     */
061    @Override
062    public String getStartInfo() {
063        return (startNode.getTagString() +
064                "(" +  startNode.ID + ")" +
065                " from " + getStartTime().getAbsoluteTime() +
066                " ("  + getTotalInFactory() +
067                " in " + getDuration() + " sec)");
068    }
069
070    //------------------------------------------------------------
071    /**
072     * エージェント生成の出発地点のオブジェクトを返す。
073     * パネル用。
074     */
075    @Override
076    public OBNode getStartObject() { return startNode; }
077
078}
