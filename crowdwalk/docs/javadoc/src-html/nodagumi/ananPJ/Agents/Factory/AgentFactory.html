077    static public void registerAgentClass(Class<?> agentClass) {
078        try {
079            classFinder.registerClassDummy(agentClass) ;
080            String aliasName =
081                (String)
082                classFinder.callMethodForClass(agentClass, "getTypeName", true) ;
083            classFinder.alias(aliasName, agentClass) ;
084        } catch (Exception ex) {
085            Itk.dumpStackTraceOf(ex) ;
086            Itk.logError("cannot process registerAgentClass()") ;
087            Itk.logError_("agentClass",agentClass) ;
088        }
089    }

300    /**
301     * エージェント生成
302     */
303    public void tryUpdateAndGenerate(EvacuationSimulator simulator,
304                                     SimTime currentTime,
305                                     List<AgentBase> agents) {
306        if (!isEnabled()) return;
307
308        if (isFinished(currentTime)) {
309            enabled = false;
310            return;
311        }
312
313        if (currentTime.isBefore(getStartTime())) {
314            /* not yet time to start generating agents */
315            return;
316        }
317
318        // 生成するエージェントの数を求める。
319
320        int nAgentsToGen = 0 ;
321        if(config.ruleType == AgentFactoryList.RuleType.INDIVIDUAL) {
322            nAgentsToGen =
323                getIndividualConfigList().remainSizeBefore(currentTime) ;
324        } else {
325            nAgentsToGen = getNAgentsRemain() ;
326            double duration_left =
327                getStartTime().getAbsoluteTime() + getDuration()
328                - currentTime.getAbsoluteTime() ;
329            if (duration_left > 0) {
330                double r
331                    = ((double)(nAgentsToGen) /
332                       duration_left * currentTime.getTickUnit()) ;
333                // tkokada: free timeScale update
334                if (((int) r) > nAgentsToGen)
335                    r = nAgentsToGen;
336                nAgentsToGen = (int)r;
337                if (random.nextDouble() < r - (double)nAgentsToGen) {
338                    nAgentsToGen++;
339                }
340            }
341        }
342        /* else, no time left, must generate all remains */
343
344        // fallbacks
345        Term fallbackForAgent = getFallbackForAgent() ;
346
347        /* [I.Noda] ここで Agent 生成 */
348        for (int i = 0; i < nAgentsToGen; ++i) {
349            nAgentsGenerated++;
350            launchAgent(getAgentClassName(),
351                        simulator,
352                        currentTime,
353                        agents,
354                        fallbackForAgent) ;
355        }
356    }

359    /**
360     * agent 一人を生成。
361     * @param agentClassName : エージェントクラス名。
362     */
363    public AgentBase launchAgent(String agentClassName,
364                                 EvacuationSimulator simulator,
365                                 SimTime currentTime,
366                                 List<AgentBase> agents,
367                                 Term fallbackForAgent) {
368        AgentBase agent = null ;
369        try {
370            agent = newAgentByName(agentClassName) ;
371        } catch (Exception ex) {
372            Itk.logError("class name not found") ;
373            Itk.logError_("agentClassName", agentClassName) ;
374            Itk.quitWithStackTrace(ex);
375        }
376        try {
377            agent.init(random, simulator, this, currentTime,
378                       fallbackForAgent) ;
379        } catch (Exception ex) {
380            Itk.logError("initialization error for agent") ;
381            Itk.logError_("agent=", agent) ;
382            Itk.logError_("agentConfig=", getAgentConf()) ;
383            Itk.logError_("fallback=", fallbackForAgent) ;
384            Itk.quitWithStackTrace(ex) ;
385        }
386
387        agents.add(agent);
388        placeAgent(agent); // この時点では direction が 0.0 のため、add_agent_to_lane で agent は登録されない
389        setupAgentByIndividualConfig(agent) ;
390        agent.prepareForSimulation() ;
391        agent.getCurrentLink().agentEntersOnLane(agent); // ここで add_agent_to_lane させる
392
393        return agent ;
394    }

487    /**
488     * planned route の中で、経由地点の tag を取り出す。
489     * directive の中の経由場所tagも Agent Class に応じて取り出す。
490     */
491    public ArrayList<Term> getNakedPlannedRoute() {
492        ArrayList<Term> routeTags = new ArrayList<Term>();
493
494        for(Term candidate : getPlannedRoute()) {
495            if(isKnownDirectiveInAgentClass(candidate)) {
496                pushPlaceTagInDirectiveByAgentClass(candidate, routeTags, routeTags) ;
497            } else {
498                routeTags.add(candidate);
499            }
500        }
501        return routeTags;
502    }
