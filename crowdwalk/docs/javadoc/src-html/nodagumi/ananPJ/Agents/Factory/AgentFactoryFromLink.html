018/**
019 * エージェント生成機構。(fromLink)
020 */
021public class AgentFactoryFromLink extends AgentFactory {
022    MapLink startLink;
023
024    //------------------------------------------------------------
025    /**
026     *  Config によるコンストラクタ
027     */
028    public AgentFactoryFromLink(AgentFactoryConfig config,
029                                OBNode _startPlace,
030                                int _totalInFactory,
031                                Random _random) {
032        super(config, _startPlace, _totalInFactory, _random) ;
033    }
034
035    //------------------------------------------------------------
036    /**
037     * Config による初期化
038     */
039    @Override
040    public void init(AgentFactoryConfig config,
041                     OBNode _startPlace,
042                     int _totalInFactory,
043                     Random random) {
044        super.init(config, _startPlace, _totalInFactory, random) ;
045        startLink = (MapLink)_startPlace ;
046    }
047
048    @Override
049    protected boolean isFinished(SimTime currentTime) {
050        if (super.isFinished(currentTime) || startLink.isShutOff()) return true;
051        return false;
052    }
053
054    @Override
055    protected void placeAgent(AgentBase agent) {
056        agent.placeAtRandomPosition(startLink) ;
057        //startLink.agentEnters(agent);
058    }
059
060    //------------------------------------------------------------
061    /**
062     * エージェント生成ルールの情報を文字列で返す。
063     * パネル表示用。
064     */
065    @Override
066    public String getStartInfo() {
067        return (startLink.getTagString() +
068                "(" +  startLink.ID + ")" +
069                " on " + getStartTime().getAbsoluteTime() +
070                " ("  + getTotalInFactory() +
071                " in " + getDuration() + " sec)");
072    }
073
074    //------------------------------------------------------------
075    /**
076     * エージェント生成の出発地点のオブジェクトを返す。
077     * パネル用。
078     */
079    @Override
080    public OBNode getStartObject() { return startLink; }
081
082}
