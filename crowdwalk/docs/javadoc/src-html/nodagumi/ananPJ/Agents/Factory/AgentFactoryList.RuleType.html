040/**
041 * Generation Rule (= Agent Factory)のセット。
042 * Generate agents depending on a generation file.
043 * Generation File は、Version 0, 1, 2 の3種類。
044 * その区別は、先頭行に
045 *  <pre>
046 *    #{ "version" : 2 }
047 *  </pre>
048 * という形で記述して区別する。
049 * <p>
050 * Version 2 では、2行目以降に Generation Rule の配列で JSON 形式で記述される。
051 * つまり、
052 *  <pre>
053 *    #{ "version" : 2 }
054 *    [ __Generation_Rule__,
055 *      __Generation_Rule__,
056 *      ... ]
057 *  </pre>
058 * となる。
059 * 各 rule の、"name" の項目はルールの名前で、
060 * log 出力などに置いて、そのエージェントを生成したルールの識別に使われる。
061 * 省略すると、単に、rule の index (前から順につけられた番号の文字列)がつけられる。
062 * <code>__Generation_Rule__</code> には以下のバリエーションがある。
063 * <dl>
064 *   <dt>EACH</dt>
065 *     <dd>
066 *       指定したタグを持つすべてのリンクとノードから、
067 *       各々指定した数のエージェントが出発する。
068 *      <pre>
069 *        { "rule" : "EACH",
070 *          "name" : __RuleName__,
071 *          "ignore" : ("true" | "false")
072 *          "agentType" : { "className" : __AgentType__,
073 *                          (__AgentConfig__)* },
074 *          "total" : __Integer__,
075 *          "speedModel" : __SpeedModel__,
076 *          "startPlace" : __Tag__,
077 *          "conditions" : __Conditions__,
078 *          "startTime" : __Time__,
079 *          "duration" : __DurationInSec__,
080 *          "goal" : __Tag__,
081 *          "plannedRoute" : [ __Tag__, __Tag__, ... ]
082 *        }
083 *      </pre>
084 *       __Conditions__ は、生成されたエージェントに付与されるタグの配列。
085 *     </dd>
086 *   <dt>RANDOM</dt>
087 *     <dd>
088 *       指定したタグを持つすべてのリンクとノードから、
089 *       ランダムにエージェントが出発する。
090 *       エージェントの総数は指定した数(total)となる。
091 *      <pre>
092 *        { "rule" : "RANDOM",
093 *          "name" : __RuleName__,
094 *          "ignore" : ("true" | "false")
095 *          "agentType" : { "className" : __AgentType__,
096 *                          (__AgentConfig__)* },
097 *          "total" : __Integer__,
098 *          "speedModel" : __SpeedModel__,
099 *          "startPlace" : __Tag__,
100 *          "conditions" : __Conditions__,
101 *          "startTime" : __Time__,
102 *          "duration" : __DurationInSec__,
103 *          "goal" : __Tag__,
104 *          "plannedRoute" : [ __Tag__, __Tag__, ... ],
105 *          "individualConfig" : [ __IndivConfig__, __IndivConfig__, ...]
106 *        }
107 *      </pre>
108 *        __IndivConfig__ については、以下を参照。
109 *     </dd>
110 *   <dt>EACHRANDOM</dt>
111 *     <dd>
112 *       RANDOMと同じだが、タグを持つ各リンク・ノードからは、
113 *       最低1エージェントは出発する。
114 *      （エージェントの出発しないリンク・ノードはない）
115 *      <pre>
116 *        { "rule" : "EACHRANDOM",
117 *          "name" : __RuleName__,
118 *          "ignore" : ("true" | "false")
119 *          "agentType" : { "className" : __AgentType__,
120 *                          (__AgentConfig__)* },
121 *          "total" : __Integer__,
122 *          "maxFromEach" : __Integer__,
123 *          "speedModel" : __SpeedModel__,
124 *          "startPlace" : __Tag__,
125 *          "conditions" : __Conditions__,
126 *          "startTime" : __Time__,
127 *          "duration" : __DurationInSec__,
128 *          "goal" : __Tag__,
129 *          "plannedRoute" : [ __Tag__, __Tag__, ... ]
130 *        }
131 *      </pre>
132 *     </dd>
133 *   <dt>TIMEEVERY</dt>
134 *     <dd>
135 *       指定した時間区間のすべてのタイムステップに於いて、
136 *       指定したタグを持つすべてのリンクから、
137 *       各々指定した数のエージェントが出発する。
138 *       EACH を、時間区間全て分コピーしたものと同じ。
139 *      <pre>
140 *        { "rule" : "TIMEEVERY",
141 *          "name" : __RuleName__,
142 *          "ignore" : ("true" | "false")
143 *          "agentType" : { "className" : __AgentType__,
144 *                          (__AgentConfig__)* },
145 *          "total" : __Integer__,
146 *          "speedModel" : __SpeedModel__,
147 *          "startPlace" : __Tag__,
148 *          "conditions" : __Conditions__,
149 *          "startTime" : __Time__,
150 *          "duration" : __DurationInSec__,
151 *          "everyEndTime" : __Time__,
152 *          "everySeconds" : __IntervalInSec__,
153 *          "goal" : __Tag__,
154 *          "plannedRoute" : [ __Tag__, __Tag__, ... ]
155 *        }
156 *      </pre>
157 *     </dd>
158 *   <dt>INDIVIDUAL</dt>
159 *     <dd>
160 *       individualConfig によって、個別のパラメータ、開始時刻で
161 *       エージェントを生成される。
162 *       individualConfig によって指定されないものは、ルール共通の
163 *       パラメータが指定される。
164 *      <pre>
165 *        { "rule" : "INDIVIDUAL",
166 *          "name" : __RuleName__,
167 *          "ignore" : ("true" | "false")
168 *          "agentType" : { "className" : __AgentType__,
169 *                          (__AgentConfig__)* },
170 *          "speedModel" : __SpeedModel__,
171 *          "startPlace" : __Tag__,
172 *          "conditions" : __Conditions__,
173 *          "goal" : __Tag__,
174 *          "plannedRoute" : [ __Tag__, __Tag__, ... ],
175 *          "individualConfig" : [ __IndivConfig__, __IndivConfig__, ...]
176 *        }
177 *      </pre>
178 *       __IndivConfig__ は以下のようなJSON Object の形式。
179 *      <pre>
180 *         { "startTime" : "HH:MM:SS",
181 *           ... }
182 *      </pre>
183 *      追加で設定できる項目は、AgentClass によって異なる。以下を参照。
184 *      <ul>
185 *        <li> {@link nodagumi.ananPJ.Agents.AgentBase#setupByIndividualConfig AgentBase} </li>
186 *        <li> {@link nodagumi.ananPJ.Agents.WalkAgent#setupByIndividualConfig WalkAgent} </li>
187 *      </ul>
188 *     </dd>
189 *   <dt>RUBY</dt>
190 *     <dd>
191 *       Ruby script による generation を行う。
192 *       "class" で指定された Ruby のクラス(AgentFactoryBase class の派生クラス)
193 *       のインスタンスが生成される。
194 *       "config" に、このルール専用のパラメータをJSONで指定できる。
195 *      <pre>
196 *        { "rule" : "RUBY",
197 *          "name" : __RuleName__,
198 *          "ruleClass" : __RubyClassNameInString__,
199 *          "config" : __Config__
200 *        }
201 *      </pre>
202 *     </dd>
203 * </dl>
204 * {@code "agentType"} の {@code "className"} および {@code "config"} については、
205 * {@link nodagumi.ananPJ.Agents.Factory.AgentFactory AgentFactory} で説明されている
206 * エージェントクラスとその設定パラメータを参照のこと。
207 * <p>
208 * Version 0, 1 の形式は以下の通り。
209 * <pre>
210 * format of generation file of one line:
211 * [RULE_STRING,][AgentClass,AgentConf,]TAG,START_TIME,DURATION,TOTAL,EXIT_TAG[,ROUTE...]
212 * TAG,START_TIME,DURATION,TOTAL[,EXIT_TAG,NAVIGATED_LINK_TAG]*
213 *  (memo: [AgentClass,AgentConf,] Part is only for Ver.1 format.
214 *  (memo: STAFF は、2014.12.24 に排除することに決定。)
215 *
216 * descriptions:
217 *  RULE_STRING:    EACH or RANDOM or EACHRANDOM 
218 *  AgentClass:     class name of agents (short name or full path name)
219 *  AgentConf:      configuration for the agents. JSON format string
220 *  TAG:            agents are generated on the links or nodes with this tag.
221 *  START_TIME:     starting time which agents are generated
222 *  TOTAL:          total number of generated agents
223 *  DURATION:       duration time to finish generating agents from START_TIME
224 *  EXIT_TAG:       set the goal of generated agents
225 *  ROUTE:          routing point
226 *  NAVIGATED_LINK_TAG:
227 *                  navigated link which agent meets with the staff
228 * example1) EACH,LINK_TAG_1,14:00:00,10,1,EXIT_TAG_2
229 * example2) RANDOM,LINK_TAG_2,09:00:00,1,10,EXIT_TAG_3
230 * example3) EACHRANDOM,LINK_TAG_3,23:44:12,10,140,1,EXIT_TAG4,STATION
231 * example6) TIMEEVERY,NaiveAgent,"{}",LINK_TAG_1,18:00:00,18:00:00,60,60,100,LANE,EXIT_1,EXIT_2,EXIT_3
232 * </pre>
233 */

383    /**
384     * 設定解析ルーチン
385     */
386    public void scanFile(final String filename,
387                         NetworkMap map,
388                         boolean display)
389        throws Exception
390    {
391        BufferedReader br = null;
392        try {
393            br = new BufferedReader(new FileReader(filename));
394        } catch (IOException e) {
395            Itk.logError("Fail to open a generation file.", e.toString()) ;
396            return;
397        }
398
399        // モードラインの読み込みを試す。
400        // 呼んだ後、read pointer は先頭へ。
401        tryScanModeLine(br) ;
402
403        switch(fileFormat) {
404        case Ver0:
405        case Ver1:
406            scanCsvFile(br, map) ;
407            Itk.logInfo("Load Generation File (CSV)", filename) ;
408            break ;
409        case Ver2:
410            scanJsonFile(br, map) ;
411            Itk.logInfo("Load Generation File (JSON)", filename) ;
412            break ;
413        default:
414            Itk.logError("Unknown Format Version",
415                         fileFormat.toString(), "(file=" + filename + ")") ;
416        }
417        return ;
418    }

421    /**
422     * try to check mode line
423     * [example]
424     *   # { 'version' : '1' }
425     * @param reader 最初の行を含む Reader
426     * @return modelineの形式であれば true を返す。
427     */
428    public boolean tryScanModeLine(BufferedReader reader) {
429        if(!reader.markSupported()) {
430            Itk.logWarn("This reader does not support mark()", reader) ;
431            return false ;
432        } else {
433            try {
434                reader.mark(BufferedReadMax) ;
435                String line = reader.readLine() ;
436                if(line == null) {
437                    Itk.logWarn("This file is empty:" + reader) ;
438                    return false ;
439                } else {
440                    boolean scanned = scanModeLine(line) ;
441                    if(!scanned) reader.reset() ;
442                    return scanned ;
443                }
444            } catch (Exception ex) {
445                Itk.dumpStackTraceOf(ex) ;
446                Itk.logError("something wrong to set mark for:" + reader) ;
447                Itk.logError_("BufferedReadMax", BufferedReadMax) ;
448                return false ;
449            }
450        }
451    }
452    static private int BufferedReadMax = 1024 ; // 最大の1行のサイズ
453
454    //------------------------------------------------------------
455    /**
456     * mode line check
457     * [example]
458     *   # { 'version' : '1' }
459     * @param modeline 最初の行
460     * @return modelineの形式であれば true を返す。
461     */
462    public boolean scanModeLine(String modeline) {
463        if(modeline.startsWith("#")) {
464            // 先頭の '#' をカット
465            String modeString = modeline ;
466            while(modeString.startsWith("#")) {
467                modeString = modeString.substring(1) ;
468            }
469            // のこりを JSON として解釈
470            modeMap = (Map<String, Object>)JSON.decode(modeString) ;
471            String versionString = modeMap.get("version").toString() ;
472            if(versionString != null && versionString.equals("2")) {
473                fileFormat = FileFormat.Ver2 ;
474            } else if(versionString != null && versionString.equals("1")) {
475                fileFormat = FileFormat.Ver1 ;
476            } else {
477                fileFormat = FileFormat.Ver0 ;
478            }
479            return true ;
480        } else {
481            fileFormat = FileFormat.Ver0 ;
482            return false ;
483        }
484    }

487    /**
488     * 設定解析ルーチン (CSV file) (Ver.0, Ver.1 file format)
489     */
490    public void scanCsvFile(BufferedReader br,
491                            NetworkMap map)
492        throws Exception
493    {
494        String line = null;
495        try {
496            // 各行をCSVとして解釈するためのパーザ
497            // quotation にはシングルクォート(')を用いる。
498            // これは、JSON の文字列がダブルクォートのため。
499            //[2014.12.23 I.Noda] csvParser は、ShiftingStringList の中へ。
500            ShiftingStringList.setCsvSpecialChars(',','\'','\\') ;
501
502            int ruleCount = 1 ;
503            while ((line = br.readLine()) != null) {
504                //一行解析
505                AgentFactoryConfig factoryConfig =
506                    scanCsvFileOneLine(line, map, ruleCount) ;
507
508                if(factoryConfig == null) continue ;
509                ruleCount++ ;
510
511                // 経路情報に未定義のタグが使用されていないかチェックする
512                checkPlannedRouteInConfig(map, factoryConfig, line) ;
513
514                // ここから、エージェント生成が始まる。
515                addFactoriesByConfig(map, factoryConfig) ;
516            }
517        } catch (Exception e) {
518            Itk.logError("Error in agent generation.", e.getMessage()) ;
519            Itk.logError_("line", line) ;
520            Itk.dumpStackTraceOf(e);
521        }
522
523        // 経路情報に未定義のタグが使用されていたら例外を発生させる
524        raiseExceptionRouteDefinitionError() ;
525    }

528    /**
529     * scan one line of CSV file
530     */
531    private AgentFactoryConfig scanCsvFileOneLine(String line,
532                                                  NetworkMap map,
533                                                  int ruleCount)
534        throws IOException
535    {
536        //コメント行読み飛ばし
537        if (line.startsWith("#")) return null ;
538        if (line.startsWith(",")) return null ;
539
540        // カラムに分割
541        // [2014/12/15 I.Noda]
542        // CSV Parser を使うように変更。
543        // さらに、ShiftingColumns を使うよにする。
544        //String items[] = line.split(",");
545        //String items[] = csvParser.parseLine(line) ;
546        //int index = 0;
547        ShiftingStringList columns =
548            ShiftingStringList.newFromCsvRow(line) ;
549
550        // 行の長さチェック
551        if(fileFormat == FileFormat.Ver1) {
552            if (columns.length() < 7 && columns.length() != 4) {
553                Itk.logWarn("malformed line", line);
554                return null ;
555            }
556        } else {
557            if (columns.length() < 5 && columns.length() != 2) {
558                Itk.logWarn("malformed line", line);
559                return null ;
560            }
561        }
562
563        // check rule strings
564        /* [2014.12.20 I.Noda] should obsolete
565         * ruleType が指定されないことがあるのか？
566         * これは、特に CSV の場合、致命的なバグの原因となる。
567         */
568        RuleType ruleType = (RuleType)ruleLexicon.lookUp(columns.top()) ;
569        if (ruleType == null)
570            // if no rule tag, default tag "EACH" is applied.
571            ruleType = RuleType.EACH ;
572        else
573            columns.shift() ;
574
575        // LINER_GENERATE_AGENT_RATIO の場合、
576        // lga_ratio が次に来る。
577        // で、読み込んだら次の行。（エージェント生成しない）
578        if (ruleType == RuleType.LINER_GENERATE_AGENT_RATIO) {
579            double lga_ratio = 0;
580            lga_ratio = Double.parseDouble(columns.get()) ;
581            if (lga_ratio > 0)
582                liner_generate_agent_ratio = lga_ratio;
583            return null ;
584        }
585
586        // 生成条件の格納用データ
587        AgentFactoryConfig factoryConfig = newConfigByRuleType(ruleType) ;
588
589        // 生成の設定情報を以下にまとめて保持。
590        factoryConfig.originalInfo = line ;
591        
592        // 生成ルールの名前。CSVの場合は、順序の番号の文字列。
593        factoryConfig.ruleName = ("" + ruleCount) ;
594
595        /* [I.Noda] Ver1 以降は、ruleType の直後はエージェントクラス名 */
596        if(fileFormat == FileFormat.Ver1) {
597            factoryConfig.agentClassName = columns.nth(0) ;
598            factoryConfig.agentConf = Term.newByJson(columns.nth(1)) ;
599            columns.shift(2) ;
600        } else {
601            /* [2014.12.29 I.Noda]
602             * agentClassName を埋めておかないと、directive の処理が出来ない。
603             * なので、AgentFactory に指定する規定値を埋める。
604             */
605            factoryConfig.agentClassName = AgentFactory.DefaultAgentClassName ;
606        }
607
608        // read start link
609        // もし start link の解析に失敗したら、次の行へ。
610        if(! scanStartLinkTag(Itk.intern(columns.get()), map, factoryConfig))
611            return null ;
612
613        // 出発時刻
614        try {
615            factoryConfig.startTime = new SimTime(columns.get()) ;
616        } catch(Exception ex) {
617            return null ;
618        }
619
620        // TIMEEVERYの場合は、出発時刻間隔
621        if (ruleType == RuleType.TIMEEVERY) {
622            try {
623                ((AgentFactoryConfig_TimeEvery)factoryConfig).everyEndTime =
624                    new SimTime(columns.get()) ;
625            } catch(Exception ex) {
626                return null ;
627            }
628            ((AgentFactoryConfig_TimeEvery)factoryConfig).everySeconds =
629                Integer.parseInt(columns.get()) ;
630        }
631
632        // duration
633        factoryConfig.duration = Double.parseDouble(columns.get()) ;
634
635        // total number of generated agents
636        factoryConfig.total = Integer.parseInt(columns.get());
637        if (liner_generate_agent_ratio != 1) {
638            Itk.logInfo("use liner_generate_agent_ratio",
639                        "factoryConfig.total:", factoryConfig.total,
640                        "ratio:", liner_generate_agent_ratio);
641            factoryConfig.total =
642                (int)(factoryConfig.total * liner_generate_agent_ratio);
643        }
644
645        // speed model
646        /* [2014.12.20 I.Noda] should obsolete
647         * speed_model を指定しないことはあるのか？
648         * 少なくとも CSV で指定する場合、
649         * 混乱の原因以外の何物でもない
650         * 一番問題なのは、speed_model に相当するタグでなければ、
651         * columns を shift しないところ。
652         */
653        factoryConfig.speedModel =
654            (SpeedCalculationModel)speedModelLexicon.lookUp(columns.top()) ;
655        if(factoryConfig.speedModel == null) {
656            factoryConfig.speedModel = getFallbackSpeedModel() ;
657        } else {
658            columns.shift() ;
659        }
660
661        // EACHRANDOM
662        if (factoryConfig.ruleType == RuleType.EACHRANDOM) {
663            ((AgentFactoryConfig_EachRandom)factoryConfig).maxFromEachPlace =
664                Integer.parseInt(columns.get()) ;
665        }
666
667        // goal を scan
668        factoryConfig.goal = new Term(columns.top(), true) ;
669
670        // ゴールより後ろの読み取り。
671        if(!scanRestColumns(columns, map, factoryConfig))
672            return null ;
673
674        return factoryConfig ;
675    }

705     */
706    static private Pattern startpat = Pattern.compile("(.+)\\((.+)\\)");

709    /**
710     * start_link_tag の解析
711     * start_link_tag は、
712     *    Tag | Tag(Cond;Cond;...) 
713     * という形式らしい。
714     */
715    public boolean scanStartLinkTag(String start_link_tag,
716                                    NetworkMap map,
717                                    AgentFactoryConfig factoryConfig) {
718        Matcher tag_match = startpat.matcher(start_link_tag);
719        if (tag_match.matches()) {
720            start_link_tag = tag_match.group(1);
721            factoryConfig.conditions = tag_match.group(2).split(";");
722        }
723
724        factoryConfig.startLinkTag = start_link_tag ;
725
726        /* get all links with the start_link_tag */
727        map.getLinks().findTaggedLinks(start_link_tag,
728                                       factoryConfig.startLinks) ;
729
730        /* get all nodes with the start_link_tag */
731        map.getNodes().findTaggedNodes(start_link_tag,
732                                       factoryConfig.startNodes) ;
733
734        if (factoryConfig.startLinks.size() == 0 &&
735            factoryConfig.startNodes.size() == 0) {
736            Itk.logError("no matching start:" + start_link_tag);
737            return false ;
738        } else {
739            return true ;
740        }
741    }

744    /**
745     * 残りの column の読み込み
746     */
747    private boolean scanRestColumns(ShiftingStringList columns,
748                                    NetworkMap map,
749                                    AgentFactoryConfig factoryConfig) {
750        factoryConfig.plannedRoute = new ArrayList<Term>();
751
752        // goal and route plan
753        //String goal = items[index];
754        if (factoryConfig.goal.isNull()) {
755            Itk.logWarn("no matching link", "" + columns.top() +
756                        " while reading agent generation rule.");
757            return false ;
758        }
759        columns.shift() ;
760        while(!columns.isEmpty()) {
761            String tag = Itk.intern(columns.get()) ;
762            if (tag != null &&
763                !tag.equals("")) {
764                Term tagTerm = 
765                    tryScanDirectiveAndMakeTerm(tag, columns) ;
766                factoryConfig.plannedRoute.add(tagTerm) ;
767            }
768        }
769        return true ;
770    }

773    /**
774     * WAIT directive の解釈
775     * [2014.12.29 I.Noda]
776     * ここだけからはどうしても、WAIT_* 系の処理を、
777     * WaitRunningAroundAgent に局所化出来ない。
778     * CSV である限り、括弧"()"の位置とか、
779     * 一般的に扱う処理方法が見当たらない。
780     */
781    private Term tryScanDirectiveAndMakeTerm(String head,
782                                             ShiftingStringList columns) {
783        try {
784            Matcher matchFull = 
785                WaitDirective.FullPattern.matcher(head) ;
786            if(matchFull.matches()) {
787                // 引数込みですでにheadに含まれている。
788                // directive 用の Term に変換する。
789                WaitDirective directive =
790                    WaitDirective.scanDirective(head) ;
791                if(directive != null) {
792                    return directive.toTerm() ;
793                } else {
794                    return new Term(head, true) ;
795                }
796            }
797            Matcher matchHead =
798                WaitDirective.HeadPattern.matcher(head) ;
799            if(matchHead.matches()) {
800                // CSV 解釈で、カンマで引数が分断されている場合。
801                String fullForm = 
802                    head + "," + columns.nth(0) + "," + columns.nth(1) ;
803                Matcher matchFull2 = 
804                    WaitDirective.FullPattern.matcher(fullForm) ;
805                if(matchFull2.matches()) {
806                    String directive = matchFull2.group(1) ;
807                    WaitDirective.Type wait =
808                        (WaitDirective.Type)
809                        WaitDirective.lexicon.lookUp(directive) ;
810                    if(wait != null) {
811                        //正しい 3引数 wait directive は、まとめたものを返す。
812                        columns.shift(2) ;
813                        // 再帰呼び出し
814                        return tryScanDirectiveAndMakeTerm(fullForm,
815                                                           columns) ;
816                    }
817                }
818                // ここで head に match しているとするならおかしい。
819                Itk.logWarn("strange tag form in planned route:" + head) ;
820            }
821            // それ以外の場合は、もとの head をTerm化して返す。
822            return new Term(head, true) ;
823        } catch (Exception ex) {
824            Itk.dumpStackTraceOf(ex) ;
825            return new Term(head, true) ;
826        }
827    }

830    /**
831     * 設定解析ルーチン (JSON file) (Ver.2 file format)
832     */
833    public void scanJsonFile(BufferedReader br,
834                             NetworkMap map)
835        throws Exception
836    {
837        Term json = Term.newByScannedJson(JSON.decode(br),true) ;
838        if(json.isArray()) {
839            int ruleCount = 1 ;
840            for(Object _item : json.getArray()) {
841                Term item = (Term)_item ;
842                if(item.isObject()) {
843                    AgentFactoryConfig factoryConfig = 
844                        scanJsonFileOneItem(item, map, ruleCount) ;
845
846                    if(factoryConfig == null) continue ;
847                    ruleCount++ ;
848
849                    // 経路情報に未定義のタグが使用されていないかチェックする
850                    checkPlannedRouteInConfig(map, factoryConfig,
851                                              item.toJson()) ;
852
853                    // ここから、エージェント生成が始まる。
854                    addFactoriesByConfig(map, factoryConfig) ;
855                } else {
856                    Itk.logError("wrong json for generation rule:",
857                                 item.toJson()) ;
858                    continue ;
859                }
860            }
861        } else {
862            Itk.logError("wrong json for generation file:",json.toJson()) ;
863            throw new Exception("wrong json for generation file:" +
864                                json.toJson()) ;
865        }
866    }

870     * 設定解析ルーチン (JSON one line) (Ver.2 file format)
871     */
872    public AgentFactoryConfig scanJsonFileOneItem(Term json,
873                                                  NetworkMap map,
874                                                  int ruleCount)
875    {
876        // ignore が true なら、項目を無視する。
877        if(json.getArgBoolean("ignore")) { return null ; }
878
879        // rule
880        RuleType ruleType = 
881            (RuleType)ruleLexicon.lookUp(json.getArgString("rule")) ;
882        AgentFactoryConfig factoryConfig = newConfigByRuleType(ruleType) ;
883
884        // ruleName を一旦与えておく。
885        factoryConfig.ruleName = ("" + ruleCount) ;
886
887        // scan
888        factoryConfig.scanJson(this, json, map) ;
889
890        // name の uniqness check
891        checkUniqueNameRule(factoryConfig) ;
892
893        return factoryConfig ;
894
895    }

898    /**
899     * ruleName の uniqness check して、テーブルへ。
900     * 名前が変更されたら true を返す。
901     */
902    private boolean checkUniqueNameRule(AgentFactoryConfig factoryConfig) {
903        String ruleName = factoryConfig.ruleName ;
904        String suffix = "" ;
905        int i = 0 ;
906        while(factoryConfigNameTable.containsKey(ruleName + suffix)) {
907            i++ ;
908            suffix = "_" + i ;
909        }
910        ruleName = ruleName + suffix ;
911        factoryConfig.ruleName = ruleName ;
912        factoryConfigNameTable.put(ruleName, factoryConfig) ;
913
914        return (i != 0) ;
915    }

933    /**
934     * 経路情報に未定義のタグが使用されていたらその内容を返す
935     */
936    public ArrayList<String> checkPlannedRoute(String agentClassName,
937                                               NetworkMap map,
938                                               List<Term> plannedRoute) {
939        ArrayList<Term> linkTags = new ArrayList();
940        ArrayList<Term> nodeTags = new ArrayList();
941        int index = 0;
942        while (index < plannedRoute.size()) {
943            Term candidate = plannedRoute.get(index);
944
945            if(AgentFactory
946               .isKnownDirectiveInAgentClass(agentClassName, candidate)) {
947                AgentFactory
948                    .pushPlaceTagInDirectiveByAgentClass(agentClassName,
949                                                         candidate,
950                                                         nodeTags,
951                                                         linkTags) ;
952            } else {
953                nodeTags.add(candidate);
954            }
955            index += 1 ;
956        }
957
958        ArrayList<String> result = new ArrayList();
959        for (Term nodeTag : nodeTags) {
960            boolean found = false;
961            for (MapNode node : map.getNodes()) {
962                if (node.hasTag(nodeTag)){
963                    found = true;
964                    break;
965                }
966            }
967            if (! found) {
968                result.add("Undefined Node Tag: " + nodeTag);
969            }
970        }
971        for (Term linkTag : linkTags) {
972            if (! map.getLinks().tagExistP(linkTag.getString()))
973                result.add("Undefined Link Tag: " + linkTag);
974        }
975        return result;
976    }

979    /**
980     * 経路情報に未定義のタグが使用されていたら例外を発生させる
981     * エラー情報は、checkPlannedRouteInConfig() で調べて蓄えられている。
982     */
983    private void raiseExceptionRouteDefinitionError()
984        throws Exception
985    {
986        if (! definitionErrors.isEmpty()) {
987            StringBuilder errorMessage = new StringBuilder();
988            Iterator it = definitionErrors.entrySet().iterator();
989            while (it.hasNext()) {
990                Map.Entry<String, ArrayList<String>> entry =
991                    (Map.Entry)it.next();
992                String _line = entry.getKey();
993                ArrayList<String>messages = entry.getValue();
994                errorMessage.append("line: ").append(_line).append("\n");
995                for (String message: messages) {
996                    errorMessage.append("    ").append(message).append("\n");
997                }
998            }
999            throw new Exception(errorMessage.toString());
1000        }
1001    }
