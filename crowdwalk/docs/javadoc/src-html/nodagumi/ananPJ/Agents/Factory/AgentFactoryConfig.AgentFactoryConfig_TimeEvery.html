165    static public AgentFactoryConfig newByRuleType(RuleType ruleType) {
166            
167        AgentFactoryConfig factoryConfig = null;
168        switch(ruleType) {
169        case EACHRANDOM:
170            factoryConfig = new AgentFactoryConfig_EachRandom() ;
171            break ;
172        case TIMEEVERY:
173            factoryConfig = new AgentFactoryConfig_TimeEvery() ;
174            break ;
175        case RUBY:
176            factoryConfig = new AgentFactoryConfig_Ruby() ;
177            break ;
178        case EACH:
179        case RANDOM:
180        case INDIVIDUAL:
181            factoryConfig = new AgentFactoryConfig() ;
182            break ;
183        default:
184            Itk.logError("AgentGenerationFile include invalid rule type:",
185                         ruleType) ;
186            return null ;
187        }
188        factoryConfig.ruleType = ruleType ;
189        return factoryConfig ;
190    }

205    /**
206     * JSONへの変換用のTerm変換
207     */
208    public Term toTerm() {
209        Term jTerm = new Term() ;
210        
211        { // agentType
212            if(agentConf != null) { // new gen style.
213                Term agentType = agentConf.clone() ;
214                SetupFileInfo.clearFallback(agentType) ;
215                jTerm.setArg("agentType", agentType) ;
216            } else { // old stype gen file
217                Term agentType = new Term() ;
218                agentType.setArg("className", agentClassName) ;
219                jTerm.setArg("agentType", agentType) ;
220            }
221        }
222        jTerm.setArg("conditions",conditions);
223        jTerm.setArg("goal",goal);
224        jTerm.setArg("plannedRoute",plannedRoute) ;
225        jTerm.setArg("startTime",startTime.getAbsoluteTimeString()) ;
226        jTerm.setArg("duration",duration) ;
227        jTerm.setArg("total",total) ;
228        jTerm.setArg("speedModel", speedModel) ;
229        jTerm.setArg("name", ruleName) ;
230        
231        jTerm.setArg("individualConfig", individualConfigList.toTerm()) ;
232
233        jTerm.setArg("rule",
234                     AgentFactoryList.ruleLexicon
235                     .lookUpByMeaning(ruleType).get(0));
236        if(startLinkTag != null)
237            jTerm.setArg("startPlace", startLinkTag) ;
238        jTerm.setArg("speedModel", 
239                     AgentFactoryList.speedModelLexicon
240                     .lookUpByMeaning(speedModel).get(0));
241
242        return jTerm ;
243    }

247     * 未だ生成していないエージェントについてのConfig.
248     * DumpEvent で用いる。
249     */
250    public Term toTermForRemainingAgents(SimTime currentTime) {
251        int remainN = 0 ;
252        for(AgentFactory factory : getAgentFactoryList()) {
253            remainN += factory.getNAgentsRemain() ;
254        }
255        
256        if(remainN > 0) {
257            Term rule = toTerm() ;
258            if(startTime.isBefore(currentTime)) {
259                rule.setArgWithBackup("startTime",
260                                      currentTime.getAbsoluteTimeString(),
261                                      TermArgBackupPrefix) ;
262                double _duration = (duration -
263                                    startTime.calcDifferenceTo(currentTime)) ;
264                rule.setArgWithBackup("duration", _duration,
265                                      TermArgBackupPrefix) ;
266            }
267            rule.setArgWithBackup("total", remainN, TermArgBackupPrefix) ;
268
269            return rule ;
270        } else {
271            return null ;
272        }
273    }

304    /**
305     * JSON Object からパラメータ設定
306     */
307    public AgentFactoryConfig scanJson(AgentFactoryList factoryList,
308                                       Term json,
309                                       NetworkMap map) {
310        originalInfo = json.toJson() ;
311
312        // rule name。
313        // ruleName には、事前に、ルールの順番の番号がうめられている。
314        String _ruleName = json.getArgString("name") ;
315        if(_ruleName != null) { ruleName = _ruleName ; }
316
317        // agentType & className ;
318        Term agentType = json.getArgTerm("agentType") ;
319        agentClassName = agentType.getArgString("className") ;
320        agentConf = agentType ;
321
322        // startPlace
323        if(!factoryList.scanStartLinkTag(json.getArgString("startPlace"),
324                                         map, this))
325            return null ;
326
327        // conditions
328        Term _conditions = json.getArgTerm("conditions") ;
329        if(_conditions != null) {
330            if(!_conditions.isArray()) {
331                Itk.logError("'conditions' in generation file should be an array of tags.",
332                             "conditions=", _conditions) ;
333                return null ;
334            }
335            int l = _conditions.getArraySize() ;
336            conditions = new String[l] ;
337            for(int i = 0 ; i < l ; i++) {
338                conditions[i] = _conditions.getNthString(i) ;
339            }
340        }
341
342        // individualConfig
343        individualConfigList =
344            new IndividualConfigList(json.getArgTerm("individualConfig")) ;
345
346        // INDIVIDUAL の時は、いろいろ別扱い。
347        if(ruleType == RuleType.INDIVIDUAL) {
348            individualConfigList.sortByStartTime() ;
349            String firstTimeStr =
350                individualConfigList.peekFirst().getArgString("startTime") ;
351            String lastTimeStr =
352                individualConfigList.peekLast().getArgString("startTime") ;
353            startTime = new SimTime(firstTimeStr) ;
354            duration = startTime.calcDifferenceTo(new SimTime(lastTimeStr));
355            total = individualConfigList.size() ;
356        } else {
357            // startTime
358            try {
359                startTime = new SimTime(json.getArgString("startTime")) ;
360            } catch(Exception ex) {
361                return null ;
362            }
363
364            // duration
365            duration = json.getArgDouble("duration") ;
366
367            // total
368            total = json.getArgInt("total") ;
369            if (factoryList.liner_generate_agent_ratio > 0) {
370                total =
371                    (int)(total * factoryList.liner_generate_agent_ratio);
372                Itk.logInfo("Agent Population (JSON)", total);
373            }
374        }
375                
376        // speedModel
377        scanJsonForSpeedModel(factoryList, json) ;
378
379        // goal
380        goal = json.getArgTerm("goal") ;
381
382        // plannedRoute
383        Term plannedRouteTerm = json.getArgTerm("plannedRoute") ;
384        plannedRoute = (plannedRouteTerm == null ?
385                        new ArrayList<Term>() :
386                        plannedRouteTerm.<Term>getTypedArray()) ;
387            
388        return this ;
389    }

436    /**
437     * EACH 用生成ルーチン
438     * 各々の link, node で total 個ずつのエージェントが生成。
439     */
440    protected void addFactoriesForEach(AgentFactoryList factoryList) {
441        for (final MapLink startLink : startLinks) {
442            AgentFactory factory =
443                new AgentFactoryFromLink(this,
444                                         startLink,
445                                         this.total,
446                                         factoryList.random) ;
447            registerAgentFactory(factory, factoryList) ;
448        }
449        for (final MapNode startNode : startNodes) {
450            AgentFactory factory =
451                new AgentFactoryFromNode(this,
452                                         startNode,
453                                         this.total,
454                                         factoryList.random) ;
455            registerAgentFactory(factory, factoryList) ;
456        }
457    }

460    /**
461     * RANDOM 用生成ルーチン
462     * 指定された link, node において、
463     * 合計で total 個のエージェントが生成。
464     */
465    protected void addFactoriesForRandom(AgentFactoryList factoryList) {
466        int links_size = this.startLinks.size();
467        int size = links_size + this.startNodes.size();// linkとnodeの合計
468        int[] chosen_links = new int[this.startLinks.size()];
469        int[] chosen_nodes = new int[this.startNodes.size()];
470        for (int i = 0; i < this.total; i++) {
471            int chosen_index = factoryList.random.nextInt(size);
472            if (chosen_index + 1 > links_size)
473                chosen_nodes[chosen_index - links_size] += 1;
474            else
475                chosen_links[chosen_index] += 1;
476        }
477        for (int i = 0; i < this.startLinks.size(); i++) {
478            if (chosen_links[i] > 0) {
479                AgentFactory factory =
480                    new AgentFactoryFromLink(this,
481                                             this.startLinks.get(i),
482                                             chosen_links[i],
483                                             factoryList.random) ;
484                registerAgentFactory(factory, factoryList) ;
485            }
486        }
487        for (int i = 0; i < this.startNodes.size(); i++) {
488            if (chosen_nodes[i] > 0) {
489                AgentFactory factory =
490                    new AgentFactoryFromNode(this,
491                                             this.startNodes.get(i),
492                                             chosen_nodes[i],
493                                             factoryList.random) ;
494                registerAgentFactory(factory, factoryList) ;
495            }
496        }
497    }

501     * 生成ルール情報格納用クラス(EachRandom 用)
502     */
503    static public class AgentFactoryConfig_EachRandom 
504        extends AgentFactoryConfig {
505        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
506        /**
507         * 各リンク・ノードにおける発生上限数
508         */
509        public int maxFromEachPlace = 0 ;
510
511        //----------------------------------------
512        /**
513         * JSON Object への変換
514         */
515        public Term toTerm(){
516            Term jTerm = super.toTerm() ;
517
518            jTerm.setArg("maxFromEach",maxFromEachPlace) ;
519
520            return jTerm ;
521        }
522
523        //----------------------------------------
524        /**
525         * JSON Object からパラメータ設定
526         */
527        public AgentFactoryConfig scanJson(AgentFactoryList factoryList,
528                                           Term json,
529                                           NetworkMap map) {
530            AgentFactoryConfig r = super.scanJson(factoryList, json, map) ;
531            
532            if(r == null) {
533                return null ;
534            }
535
536            maxFromEachPlace = json.getArgInt("maxFromEach") ;
537
538            return this ;
539        }
540
541        //----------------------------------------
542        /**
543         * Factory を追加。
544         * EACH RANDOM 用生成ルーチン
545         * RANDOM に、1箇所での生成数の上限を入れたもの。
546         * 合計で total 個のエージェントが生成。
547         */
548        public void addFactories(AgentFactoryList factoryList,
549                                 NetworkMap map) {
550
551            int maxFromEachPlace = this.maxFromEachPlace ;
552            int total = this.total ;
553
554            int links_size = this.startLinks.size() ;
555            int nodes_size = this.startNodes.size() ;
556            int size = links_size + nodes_size ; // linkとnodeの合計
557            int[] chosen_links = new int[this.startLinks.size()];
558            int[] chosen_nodes = new int[this.startNodes.size()];
559
560            /* [2014.12.24 I.Noda]
561             * アルゴリズムがあまりにまずいので、修正。
562             */
563            if(total > 0) {
564                int population = 0 ;
565                //とりあえず、maxFromEachPlace で埋める。
566                for(int i = 0 ; i < links_size ; i++) {
567                    chosen_links[i] = maxFromEachPlace ;
568                    population += maxFromEachPlace ;
569                }
570                for(int i = 0 ; i < nodes_size ; i++) {
571                    chosen_nodes[i] = maxFromEachPlace ;
572                    population += maxFromEachPlace ;
573                }
574                //減らしていく。
575                while(population > total){
576                    int chosen_index = factoryList.random.nextInt(size) ;
577                    if(chosen_index < links_size) {
578                        if(chosen_links[chosen_index] > 0) {
579                            chosen_links[chosen_index] -= 1 ;
580                            population -= 1 ;
581                        }
582                    } else {
583                        if(chosen_nodes[chosen_index - links_size] > 0) {
584                            chosen_nodes[chosen_index - links_size] -= 1;
585                            population -= 1 ;
586                        }
587                    }
588                }
589            }
590
591            for (int i = 0; i < this.startLinks.size(); i++) {
592                if (chosen_links[i] > 0) {
593                    AgentFactory factory =
594                        new AgentFactoryFromLink(this,
595                                                 this.startLinks.get(i),
596                                                 chosen_links[i],
597                                                 factoryList.random) ;
598                    registerAgentFactory(factory, factoryList) ;
599                }
600            }
601            for (int i = 0; i < this.startNodes.size(); i++) {
602                if (chosen_nodes[i] > 0) {
603                    AgentFactory factory =
604                        new AgentFactoryFromNode(this,
605                                                 this.startNodes.get(i),
606                                                 chosen_nodes[i],
607                                                 factoryList.random) ;
608                    registerAgentFactory(factory, factoryList) ;
609                }
610            }
611        }
612    } // end class AgentFactoryConfig_EachRandom

616     * 生成ルール情報格納用クラス(TimeEvery 用)
617     */
618    static public class AgentFactoryConfig_TimeEvery
619        extends AgentFactoryConfig {
620        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
621        /**
622         * 生成の終了時刻
623         */
624        public SimTime everyEndTime = null ;
625
626        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
627        /**
628         * 生成のインターバル
629         */
630        public int everySeconds = 0 ;
631
632        //----------------------------------------
633        /**
634         * JSON Object への変換
635         */
636        public Term toTerm(){
637            Term jTerm = super.toTerm() ;
638
639            jTerm.setArg("everyEndTime",everyEndTime.getAbsoluteTimeString());
640            jTerm.setArg("everySeconds",everySeconds) ;
641
642            return jTerm ;
643        }
644
645        //----------------------------------------
646        /**
647         * JSON Object からパラメータ設定
648         */
649        public AgentFactoryConfig scanJson(AgentFactoryList factoryList,
650                                           Term json,
651                                           NetworkMap map) {
652            AgentFactoryConfig r = super.scanJson(factoryList, json, map) ;
653            
654            if(r == null) {
655                return null ;
656            }
657
658            // everyEndTime と everySecond の設定。
659            try {
660                String endTimeStr = json.getArgString("everyEndTime") ;
661                everyEndTime = new SimTime(endTimeStr) ;
662            } catch(Exception ex) {
663                return null ;
664            }
665            everySeconds = json.getArgInt("everySeconds") ;
666
667            return this ;
668        }
669
670        //----------------------------------------
671        /**
672         * Factory を追加。
673         * TIME EVERY 用生成ルーチン
674         * [2014.12.24 I.Noda]
675         * GOAL の部分の処理は他と同じはずなので、
676         * 特別な処理をしないようにする。
677         * 合計で (total * 生成回数) 個のエージェントが生成。
678         */
679        public void addFactories(AgentFactoryList factoryList,
680                                 NetworkMap map) {
681            
682            SimTime every_end_time = this.everyEndTime ;
683            double every_seconds = (double)this.everySeconds ;
684            int total = this.total ;
685
686            // [I.Noda] startTime も特別な意味
687            SimTime _startTime = this.startTime ;
688            this.startTime = null ;
689
690            SimTime step_time = _startTime.newSimTime() ;
691            /* let's assume start & goal & plannedRoute candidates
692             * are all MapLink!
693             */
694
695            while (step_time.isBeforeOrAt(every_end_time)) {
696                for (int i = 0; i < total; i++) {
697                    this.startTime = step_time.newSimTime() ;
698                    this.total = 1 ;
699                    if(this.startLinks.size() > 0) {
700                        MapLink start_link =
701                            this.startLinks.chooseRandom(factoryList.random) ;
702                        AgentFactory factory =
703                            new AgentFactoryFromLink(this,
704                                                     start_link,
705                                                     this.total,
706                                                     factoryList.random) ;
707                        registerAgentFactory(factory, factoryList) ;
708                    } else if (this.startNodes.size() > 0) {
709                        MapNode start_node = 
710                            this.startNodes.chooseRandom(factoryList.random) ;
711                        AgentFactory factory =
712                            new AgentFactoryFromNode(this,
713                                                     start_node,
714                                                     this.total,
715                                                     factoryList.random) ;
716                        registerAgentFactory(factory, factoryList) ;
717                    } else {
718                        Itk.logError("no starting place for generation.") ;
719                        Itk.logError_("config",this) ;
720                    }
721                }
722                step_time.advanceSec(every_seconds) ;
723            }
724            // retrieve backup startTime.
725            this.startTime = _startTime ; 
726        }
727    } // end class AgentFactoryConfig_TimeEvery
