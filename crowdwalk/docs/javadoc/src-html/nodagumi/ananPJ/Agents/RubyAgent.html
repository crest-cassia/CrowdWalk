033/**
034 * 制御に、Ruby の script を呼び出すエージェント。
035 *
036 * <h3> config, fallbackResources に書ける設定 </h3>
037 * {@link RationalAgent} に加えて、
038 * <pre>
039 *  {
040 *    "rubyAgentClass" : __String__ // Ruby 内でのクラス名
041 * }
042 * </pre>
043 * 上記で指定されたクラスのインスタンスが生成され、java 内で以下のメソッドが
044 * 呼ばれた時に、ruby 側の対応する同名のメソッドに制御が移る。
045 * <UL>
046 *  <LI> {@code preUpdate(double time)} : update cycle の前半。次の位置を決める。
047 *  <LI> {@code update(double time)} : update cycle の後半。実際に動かす。
048 *  <LI> {@code calcCostFromNodeViaLink(...)} : 分岐点で、各経路の目的地までのコストを計算。
049 *       これを元に、最小コストの経路を選ぶ。
050 * </UL>
051 *
052 * このRubyのクラスは、RubyAgentBase クラスを継承しているべきである。
053 */

095     * 各メソッドに於いてルビー側のメソッドを呼び出すかどうかの
096     * チェックを行うフィルタのテーブル。
097     * フィルタは文字列の配列であり、各要素(triggerに対応)の値は、
098     * trigger される(ruby を呼び出す) 場合には、ruby のメソッド名が入っている。
099     * そうでない場合は、null が入っている。
100     * ルビー側のクラスごとに定義する。
101     * 構造は、(ruby 風に書くと)
102     * <pre>{@code
103     *    triggerFilterTable = { 
104     *              "RubyClassName0" => [null, "update", null, null],
105     *              "RubyClassName1" => ["preUpdate", "update", null, null],
106     *              ... }
107     * }</pre>
108     */
109    static protected HashMap<String, ArrayList<String>> triggerFilterTable
110        = new HashMap<String, ArrayList<String>>() ;

163    /**
164     * Conf による初期化。
165     * init() より先にこちらが実行されるので注意。
166     */
167    @Override
168    public void initByConf(Term conf, Term fallback) {
169        super.initByConf(conf, fallback) ;
170
171        rubyEngine = handler.getSimulator().getRubyEngine() ;
172        if(rubyEngine == null) {
173            Itk.logError("ruby engine is not available.",
174                         "should specify 'use_ruby' property to be 'true'.") ;
175            Itk.quitByError() ;
176        }
177        
178        rubyAgentClass = getStringFromConfig("rubyAgentClass", rubyAgentClass) ;
179        rubyAgent = rubyEngine.newInstanceOfClass(rubyAgentClass, this,
180                                                  conf, fallback) ;
181        setupTriggerFilter(rubyAgentClass) ;
182    } ;

184    //------------------------------------------------------------
185    /**
186     * 指定した ruby の AgentClass のtriggerFilter を取得。
187     * もし新しいものであれば、新たに作る。
188     * @param rubyClassName : ruby での AgentClass 名
189     */
190    protected void setupTriggerFilter(String rubyClassName) {
191        if(!triggerFilterTable.containsKey(rubyClassName)) {
192            triggerFilterTable.put(rubyClassName,
193                                   prepareTriggerFilterFor(rubyClassName)) ;
194        }
195        triggerFilter = triggerFilterTable.get(rubyClassName) ;
196    }
197
198    /**
199     * 指定した ruby の AgentClass のtriggerFilter を作成。
200     * @param rubyClassName : ruby での AgentClass 名
201     * @return 対応する triggerFilter
202     */
203    private ArrayList<String> prepareTriggerFilterFor(String rubyClassName){
204        ArrayList<String> filter = new ArrayList<String>() ;
205        for(TriggerEntry trigger : TriggerEntryList) {
206            // 配列を拡大。
207            while(filter.size() <= trigger.ordinal()) {
208                filter.add(null) ;
209            }
210            String methodName = trigger.toString() ;
211            String script = String.format("%s.checkTriggerFilter('%s')",
212                                          rubyClassName, methodName) ;
213            boolean triggered = rubyEngine.evalBoolean(script) ;
214            if(triggered) {
215                filter.set(trigger.ordinal(), methodName) ;
216            }
217        }
218        return filter ;
219    }

221    /**
222     * トリガされているかのチェック。
223     * @param trigger : ruby での トリガ。
224     * @return trigger されていれば true
225     */
226    protected String triggeredMethod(TriggerEntry trigger) {
227        return triggerFilter.get(trigger.ordinal()) ;
228    }

255        String tagString ;
256        if(tag instanceof Term) {

258        } else if(tag instanceof String) {
259            tagString = (String)tag ;
260        } else {
261            tagString = tag.toString() ;
262        }
263        return getCurrentLink().hasTag(tagString) ;
264    }

269    /**
270     * シミュレーション各サイクルの前半に呼ばれる。
271     */
272    @Override
273    public void preUpdate(SimTime currentTime) {
274        this.currentTime = currentTime ;
275        String rubyMethod = triggeredMethod(TriggerEntry.preUpdate) ;
276        if(rubyMethod != null) {
277            rubyEngine.callMethod(rubyAgent, rubyMethod) ;
278        } else {
279            super_preUpdate() ;
280        }
281    }

296    public boolean update(SimTime currentTime) {
297        this.currentTime = currentTime ;
298        String rubyMethod = triggeredMethod(TriggerEntry.update) ;

301        } else {

316     * あるlinkを選択した場合の目的地(_target)までのコスト。
317     */
318    @Override
319    public double calcCostFromNodeViaLink(MapLink _link, MapNode _node, Term _target)
320        throws TargetNotFoundException {
321        String rubyMethod =
322            triggeredMethod(TriggerEntry.calcCostFromNodeViaLink) ;
323        if(rubyMethod != null) {
324            return rubyEngine.callMethodDouble(rubyAgent, rubyMethod,
325                                               _link, _node, _target) ;
326        } else {
327            return super_calcCostFromNodeViaLink(_link, _node, _target) ;
328        }
329    }

335                                                Term _target)

337        return super.calcCostFromNodeViaLink(_link, _node, _target) ;
