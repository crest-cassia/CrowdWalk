088    /**
089     * WAITING の処理
090     * @return もしここで preUpdate から return するなら true。
091     */
092    protected boolean doWait(Term how, SimTime currentTime) {
093        if (how.equals("SCATTER")) {
094            waiting = true;
095            scatter(currentTime);
096            return true;
097        } else if (how.equals("PACK")) {
098            waiting = true;
099            pack(currentTime);
100            return true;
101        } else {
102            Itk.logWarn("how to move not stated!", this);
103            return false ;
104        }
105    }

115    /**
116     * WAIT_FOR/WAIT_UNTIL 中のエージェントをレーンごとに均等な間隔で配置する。
117     * ※制限事項
118     * ・同じレーン上に scatter 以外のエージェントが存在すると間隔が詰まり均等にならなくなる。
119     * ・対向流が発生すると laneWidth が変化するため均等な間隔にならない事がある。
120     */
121    protected boolean scatter(SimTime currentTime) {
122        ArrayList<AgentBase> agents = currentPlace.getLane();
123        int laneWidth = currentPlace.getLaneWidth();
124        double space = currentPlace.getLinkLength() / ((agents.size() - 1) / laneWidth + 2);
125
126        // scatter メソッドが適用できないほど超過密状態の場合には pack を使用する
127        if (space <= minDistanceBetweenAgents) {
128            return pack(currentTime);
129        }
130
131        int index = currentPlace.getIndexFromHeadingInLane(this) ;
132        double stopPosition = currentPlace.getLinkLength() - space * ((index / laneWidth) + 1);
133        double d = stopPosition - getAdvancingDistance();
134
135        // エージェントが停止予定位置より先にいた場合はその位置に留まらせる
136        if (d < 0.0) d = 0.0; 
137
138        // d が speed より小さい時、d に基づきスピードを決める。 
139        double previousSpeed = speed ;
140        speed = calcSpeed(speed, currentTime);
141        if (d < speed * currentTime.getTickUnit()) {
142            speed = d / currentTime.getTickUnit() ;
143        }
144        checkSpeedChange(previousSpeed, speed) ;
145
146        return advanceNextPlace(speed, currentTime, true) ;
147    }

149    //----------------------------------------------------------------------
150    /**
151     * できるだけ過密に並ぶ
152     */
153    protected boolean pack(SimTime currentTime) {
154        double previousSpeed = speed;
155        speed = calcSpeed(speed, currentTime);
156
157        // currentLink 内に収まるスピードにする
158        double deltaDistance = speed * currentTime.getTickUnit() ;
159        if (currentPlace.isBeyondLinkWithAdvance(deltaDistance)) {
160            double distance = currentPlace.getLinkLength() * Place.AlmostOne
161                - currentPlace.getAdvancingDistance();
162            speed = distance / currentTime.getTickUnit();
163        }
164
165        checkSpeedChange(previousSpeed, speed);
166
167        return advanceNextPlace(speed, currentTime, true) ;
168    }

170    static final double NOT_WAITING = -100.0;
171    private double wait_time = NOT_WAITING;
172    private double wait_time_start = NOT_WAITING;
173
174    //------------------------------------------------------------
175    /**
176     * cycle 前処理。
177     * Wait directive の処理を呼び出す。
178     */
179    @Override
180    public void preUpdate(SimTime currentTime) {
181        waiting = false;
182        if (currentTime.isBeforeOrAt(generatedTime) || routePlan.isEmpty()) {
183            //生成前かroutePlanがすでに空なら
184            // do nothing
185        } else {
186            // WAIT directive かどうかのチェック。
187            Term tag = routePlan.top() ;
188            WaitDirective.Type waitType = WaitDirective.isDirective(tag) ;
189            if(waitType != null) {
190                Term target = tag.getArgTerm("target") ;
191                Term how = tag.getArgTerm("how") ;
192                Term until = tag.getArgTerm("until") ;
193
194                try {
195                    switch(waitType) {
196                    case WAIT_UNTIL:
197                        if(doWaitUntil(target, how, until, currentTime)) return ;
198                        break ;
199                    case WAIT_FOR:
200                        if(doWaitFor(target, how, until, currentTime)) return ;
201                        break ;
202                    }
203                } catch (Exception ex) {
204                    Itk.dumpStackTraceOf(ex) ;
205                }
206            }
207        }
208        super.preUpdate(currentTime);
209    }

211    //------------------------------------------------------------
212    /**
213     * WAIT_UNTIL の場合の preUpdate 処理
214     * @return 処理が終わればすぐに return すべきかどうか。
215     *         (true なら super のpreUpdateを呼ばない)
216     */
217    protected boolean doWaitUntil(Term target, Term how, Term until,
218                                  SimTime currentTime) {
219        if (currentPlace.getLink().hasTag(target)) {
220            // until は、待っているリンクで生じるイベント
221            // until イベントが生じていたら、WAIT解除
222            if (currentPlace.getLink().hasTag(until.getString())) {
223                routePlan.shift() ;
224                return false ;
225            } else {
226                return doWait(how, currentTime) ;
227            }
228        } else {
229            return false ;
230        }
231    }
232
233    //------------------------------------------------------------
234    /**
235     * WAIT_FOR の場合の preUpdate 処理
236     * @return 処理が終わればすぐに return すべきかどうか。
237     *         (super のpreUpdateがいらないかどうか)
238     */
239    protected boolean doWaitFor(Term target, Term how, Term until,
240                                SimTime currentTime) {
241        // until は待つ時間(?)
242        if (currentPlace.getLink().hasTag(target)) {
243            // 待ち始めた時刻と待ち時間を記録(初回のみ)
244            if (wait_time == NOT_WAITING) {
245                wait_time = until.getDouble() ;
246                wait_time_start = currentTime.getRelativeTime() ;
247            }
248
249            // wait_time (untilに記録) 時間過ぎたらWAIT解除
250            if (currentTime.getRelativeTime() - wait_time_start > wait_time) {
251                wait_time = NOT_WAITING;
252                routePlan.shift() ;
253            } else {
254                boolean returnP = doWait(how, currentTime) ;
255                if(returnP) return true;
256            }
257        } else if (wait_time != NOT_WAITING) {
258            Itk.logWarn("Pushed Out", "agent " + ID + " pushed out from " + target);
259            wait_time = NOT_WAITING;
260            routePlan.shift() ;
261        }
262        return false ;
263    }

276    @Override
277    public ArrayList<Term> getPlannedRoute() {
278        ArrayList<Term> goal_tags = new ArrayList<Term>();
279
280        int delta = 0;
281        while (delta < routePlan.length()) {
282            Term candidate = routePlan.top(delta) ;
283            /* [2014.12.27 I.Noda]
284             * 読み込み時点で、directive はすでに1つのタグに集約されているはず。
285             * (in "AgentGenerationFile.java")
286             */
287            WaitDirective.Type type = 
288                WaitDirective.isDirective(candidate) ;
289            if(type != null) {
290                goal_tags.add(candidate.getArgTerm("target")) ;
291            } else {
292                goal_tags.add(candidate);
293            }
294            delta++ ;
295        }
296        return goal_tags;
297    }

334    /**
335     * WAIT directive 解釈
336     */
337    static public class WaitDirective {
338        //==============================
339        //::::::::::::::::::::::::::::::
340        /**
341         * enum for WAIT directive
342         */
343        static public enum Type {
344            WAIT_FOR,
345            WAIT_UNTIL
346        }
347        //::::::::::::::::::::::::::::::
348        /**
349         * Lexicon for WAIT directive
350         */
351        static public Lexicon lexicon = new Lexicon() ;
352        static {
353            // Rule で定義された名前をそのまま文字列で Lexicon を
354            // 引けるようにする。
355            // 例えば、 WaitDirective.WAIT_FOR は、"WAIT_FOR" で引けるようになる。
356            lexicon.registerEnum(Type.class) ;
357        }
358
359        //==============================
360        //------------------------------
361        static public Type isDirective(Term term) {
362            return (Type)lexicon.lookUp(term.getHeadString()) ;
363        }
364        //==============================
365        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
366        /**
367         * route 中の WAIT 命令の解釈パターン
368         */
369        static public Pattern FullPattern =
370            Pattern.compile("(\\w+)\\((\\w+),(\\w+),(\\w+)\\)") ;
371        static public Pattern HeadPattern =
372            Pattern.compile("(\\w+)\\((\\w+)") ;
373        static public Pattern TailPattern =
374            Pattern.compile("(\\w+)\\)") ;
375
376        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
377        /**
378         * route 中の WAIT 命令の解釈パターン
379         */
380        public Type type ;
381        public String head ;
382        public String target ; // arg1
383        public String how ;    // arg2
384        public String untilStr ; // arg3
385
386        //------------------------------
387        public WaitDirective(Type _type, String _head, String _target, 
388                             String _how, String _untilStr) {
389            type = _type ;
390            head = _head ;
391            target = _target ;
392            how = _how ;
393            untilStr = _untilStr ;
394        }
395
396        //------------------------------
397        public Term targetTerm() {
398            return new Term(target, false) ;
399        }
400
401        //------------------------------
402        private static final String Lex_target = Itk.intern("target") ;
403        private static final String Lex_how = Itk.intern("how") ;
404        private static final String Lex_until = Itk.intern("until") ;
405        
406        //------------------------------
407        public Term toTerm() {
408            Term term = new Term(head, false) ;
409            term.setArg(Lex_target, new Term(target, false)) ;
410            term.setArg(Lex_how, new Term(how, false)) ;
411            switch(type) {
412            case WAIT_UNTIL:
413                term.setArg(Lex_until, new Term(untilStr, false)) ;
414                break ;
415            case WAIT_FOR:
416                term.setArg(Lex_until,
417                            new Term(Double.parseDouble(untilStr))) ;
418                break ;
419            }
420
421            return term ;
422        }
423
424        //==============================
425        //------------------------------
426        /**
427         * atom String 状態の wait directive を解析する。
428         * <pre>
429         *   _directive_ ::= _head_(_target_,_how_,_until_)
430         *   _head_      ::= "WAIT_UNTIL" | "WAIT_FOR"
431         *   _how_       ::= ???
432         *   _until_     ::= _eventTag_ | _duration_
433         * </pre>
434         */
435        static public WaitDirective scanDirective(Term directive) {
436            return scanDirective(directive.getString()) ;
437        }
438        //==============================
439        //------------------------------
440        static public WaitDirective scanDirective(String directive) {
441            Matcher matchFull = FullPattern.matcher(directive) ;
442            if(! matchFull.matches()) {
443                return null ;
444            }
445
446            String head = Itk.intern(matchFull.group(1)) ;
447            Type waitType = (Type)lexicon.lookUp(head) ;
448            if(waitType == null) {
449                return null ;
450            }
451
452            WaitDirective _directive = 
453                new WaitDirective(waitType, head,
454                                  Itk.intern(matchFull.group(2)),
455                                  Itk.intern(matchFull.group(3)),
456                                  Itk.intern(matchFull.group(4))) ;
457
458            return _directive ;
459        }
460    }
