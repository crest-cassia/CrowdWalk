051    static public boolean registerFormulas(Lexicon lexicon)
052    {
053        try {
054            Class currentClass =
055                new Object(){}.getClass().getEnclosingClass() ;
056            singleton =
057                (ThinkFormula)(currentClass.newInstance()) ;
058        } catch(Exception ex) {
059            Itk.logError("wrong class definition") ;
060            Itk.dumpStackTraceOf(ex) ;
061        }
062
063        ThinkFormula.register("getFallback",    singleton, false, lexicon) ;
064        ThinkFormula.register("getParam",       singleton, false, lexicon) ;
065        ThinkFormula.register("setParam",       singleton, false, lexicon) ;
066        ThinkFormula.register("agentHasTag",    singleton, false, lexicon) ;
067        ThinkFormula.register("placeHasTag",    singleton, false, lexicon) ;

069        ThinkFormula.register("addPlaceTag",    singleton, false, lexicon) ;
070        ThinkFormula.register("removeAgentTag", singleton, false, lexicon) ;

072        ThinkFormula.register("listenAlert",    singleton, false, lexicon) ;
073        ThinkFormula.register("saveAlert",      singleton, false, lexicon) ;
074        ThinkFormula.register("announceAlert",  singleton, false, lexicon) ;

094     *   <li>{@link #call_agentHasTag "agentHasTag"}</li>
095     *   <li>{@link #call_placeHasTag "placeHasTag"}</li>
096     *   <li>{@link #call_addAgentTag "addAgentTag"}</li>

111    @Override
112    public Term call(String head, Term expr,
113                     ThinkEngine engine, Object env) {
114        if(head.equals("getFallback")) {
115            return call_getFallback(head, expr, engine, env) ;
116        } else if(head.equals("getParam")) {
117            return call_getParam(head, expr, engine, env) ;
118        } else if(head.equals("setParam")) {
119            return call_setParam(head, expr, engine, env) ;
120        } else if(head.equals("agentHasTag")) {
121            return call_agentHasTag(head, expr, engine, env) ;
122        } else if(head.equals("placeHasTag")) {
123            return call_placeHasTag(head, expr, engine, env) ;
124        } else if(head.equals("addAgentTag")) {
125            return call_addAgentTag(head, expr, engine, env) ;
126        } else if(head.equals("addPlaceTag")) {
127            return call_addPlaceTag(head, expr, engine, env) ;
128        } else if(head.equals("removeAgentTag")) {
129            return call_removeAgentTag(head, expr, engine, env) ;
130        } else if(head.equals("removePlaceTag")) {
131            return call_removePlaceTag(head, expr, engine, env) ;
132        } else if(head.equals("listenAlert")) {
133            return call_listenAlert(head, expr, engine, env) ;
134        } else if(head.equals("saveAlert")) {
135            return call_saveAlert(head, expr, engine, env) ;
136        } else if(head.equals("announceAlert")) {
137            return call_announceAlert(head, expr, engine, env) ;
138        } else if(head.equals("clearAlert")) {
139            return call_clearAlert(head, expr, engine, env) ;
140        } else if(head.equals("clearAllAlert")) {
141            return call_clearAllAlert(head, expr, engine, env) ;
142        } else if(head.equals("changeGoal")) {
143            return call_changeGoal(head, expr, engine, env) ;
144        } else if(head.equals("clearPlannedRoute")) {
145            return call_clearPlannedRoute(head, expr, engine, env) ;
146        } else if(head.equals("insertRoute")) {
147            return call_insertRoute(head, expr, engine, env) ;
148        } else if(head.equals("turnAround")) {
149            return call_turnAround(head, expr, engine, env) ;
150        } else if(head.equals("clearNextLinkCache")) {
151            return call_clearNextLinkCache(head, expr, engine, env) ;
152        } else {
153            Itk.logWarn("unknown expression", "expr=", expr) ;
154            return Term_Null ;
155        }
156    }

177    /**
178     * 推論(getParam)。
179     * <pre>
180     * { "" : "getParam",
181     *   "name" : _nameOfValue_,
182     * }
183     *  _nameOfValue_ ::= "currentTime" | "agentId" | "linkId" |
184     *                    "speed" | "emptySpeed" | "goal" |
185     *                    "triageLevelInt" | "triageLevelName"
186     * </pre>
187     */
188    public Term call_getParam(String head, Term expr,
189                              ThinkEngine engine, Object env) {
190        String name = expr.getArgString("name") ;
191        RationalAgent agent = engine.getRationalAgent() ;
192
193        if(name.equals("currentTime")) {
194            return new Term(agent.currentTime.getRelativeTime()) ;
195        } else if(name.equals("agentId")) {
196            return new Term(agent.ID, false) ;
197        } else if(name.equals("linkId")) {
198            return new Term(agent.getCurrentLink().ID, false) ;
199        } else if(name.equals("speed")) {
200            return new Term(agent.getSpeed()) ;
201        } else if(name.equals("emptySpeed")) {
202            return new Term(agent.getEmptySpeed()) ;
203        } else if(name.equals("goal")) {
204            return agent.getGoal() ;
205        } else if(name.equals("triageLevelInt")) {
206            return new Term(agent.getTriageInt()) ;
207        } else if(name.equals("triageLevelName")) {
208            return new Term(agent.getTriageName(), false) ;
209        } else {
210            Itk.logError("unknown parameter name for getParam.", "name=",name) ;
211            return Term_Null ;
212        }
213    }

382        Term message = expr.getArgTerm("message") ;
383        RationalAgent agent = ((ThinkEngine)env).getRationalAgent() ;

385        SimTime alertTime =
386            ((ThinkEngine)env).getAlertedMessageTable().get(message) ;

388            return Term_True ;
389        } else {
390            return Term_False ;

395    /**
396     * 推論(saveAlert)。
397     * 現時刻に、ある {@code message} を聴いたことにする。
398     * <pre>
399     * { "" : "saveAlert",
400     *   "message" : _alertMessage_,
401     *  ("redundant" : _Boolean_) }
402     * _alertMessage_ ::= _String_
403     * </pre>
404     * "redundant" は、重複を許すかどうか。true だと重複の追加を許す。
405     * デフォルトは false。
406     *
407     * @return Alert 追加に成功したら Term_True を返す。
408     */
409    public Term call_saveAlert(String head, Term expr,
410                               ThinkEngine engine, Object env) {
411        Term message = expr.getArgTerm("message") ;
412        boolean redundant = expr.getArgBoolean("redundant") ;
413        SimTime alertTime =
414            ((ThinkEngine)env).getAlertedMessageTable().get(message) ;
415
416         if(redundant || (alertTime == null)) {
417             ((ThinkEngine)env)
418             .getAlertedMessageTable()
419             .put(message,
420                  ((ThinkEngine)env).getAgent().currentTime) ;
421            return ThinkFormula.Term_True ;
422        } else {
423            return ThinkFormula.Term_False ;
424        }
425    }

519     * 推論(insert route)。
520     * これからのルートの先頭にサブゴールを追加。
521     * <pre>
522     * { "" : "insertRoute",
523     *   "route" : _route_ }
524     * _route_ ::= _tag_ || [_tag_, _tag_, ...]
525     * </pre>
526     */
527    public Term call_insertRoute(String head, Term expr,
528                                 ThinkEngine engine, Object env) {
529        Term route = expr.getArgTerm("route") ;
530        if(route.isArray()) {
531            for(int i = route.getArray().size() ; i > 0 ; i--) {
532                ((ThinkEngine)env).getAgent().insertRouteTagSafely(route.getNthTerm(i-1)) ;
533            }
534        } else {
535            ((ThinkEngine)env).getAgent().insertRouteTagSafely(route) ;
536        }
537        return Term_True ;
538    }
