039    /**
040     * 代表インスタンス
041     * (複数あっても無駄なので、単一である方が良い)
042     * registerFormulas で登録。
043     */
044    static ThinkFormula singleton ;
045
046    //============================================================
047    //------------------------------------------------------------
048    /**
049     * 登録
050     */
051    static public boolean registerFormulas(Lexicon lexicon)
052    {
053        try {
054            Class currentClass =
055                new Object(){}.getClass().getEnclosingClass() ;
056            singleton =
057                (ThinkFormula)(currentClass.newInstance()) ;
058        } catch(Exception ex) {
059            Itk.logError("wrong class definition") ;
060            Itk.dumpStackTraceOf(ex) ;
061        }
062
063        ThinkFormula.register("null",   singleton, false, lexicon) ;
064        ThinkFormula.register("quote",  singleton, false, lexicon) ;
065        ThinkFormula.register("log",    singleton, false, lexicon) ;
066
067        return true ;
068    }
069
070    //------------------------------------------------------------
071    /**
072     * 呼び出し.
073     * <pre>
074     * { "" : "null" } || null
075     * </pre>
076     * あるいは以下のフォーマット
077     * <ul>
078     *   <li>{@link #call_quote "quote"}</li>
079     *   <li>{@link #call_log "log"}</li>
080     * </ul>
081     */
082    @Override
083    public Term call(String head, Term expr,
084                     ThinkEngine engine, Object env) {
085        if(head.equals("null")) {
086            return Term_Null ;
087        } else if(head.equals("quote")) {
088            return call_quote(head, expr, engine, env) ;
089        } else if(head.equals("log")) {
090            return call_log(head, expr, engine, env) ;
091        } else {
092            Itk.logWarn("unknown expression", "expr=", expr) ;
093            return Term_Null ;
094        }
095    }
096

099     * quote。
100     * <pre>

102     *    "value": _Term_ }
103     * </pre>
104     * _Term_ を評価せずに返す。
105     */
106    public Term call_quote(String head, Term expr,
107                           ThinkEngine engine, Object env) {
108            return expr.getArgTerm("value") ;
109    }

111    //------------------------------------------------------------
112    /**
113     * 推論(log)。
114     * <pre>
115     *   {"":"log",
116     *    "special": ("alertMessages" | "tags" | ...),
117     *    ["level" : ("Trace" | "Debug" | "Info" | "Warn" | "Error" | "Fatal"}]
118     *   }
119     *  OR
120     *   {"":"log",
121     *    "tag": _StringForTag_,
122     *    "value": _expr_,
123     *    ["level" : ("Trace" | "Debug" | "Info" | "Warn" | "Error" | "Fatal"}]
124     *   }
125     * </pre>
126     * "special":"alertMessages" の場合は、
127     * エージェントが取得している alert message のリストが表示される。
128     * "special":"tags" の場合は、
129     * エージェントが保持している tag のリストが表示される。
130     * "level" が省略された場合、規定値は "Info"。
131     */
132    public Term call_log(String head, Term expr,
133                         ThinkEngine engine, Object env) {
134        Term special = expr.getArgTerm("special") ;
135        Itk.LogLevel level = getLogLevel(expr.getArgTerm("level")) ; 
136
137        if(special != null) {
138            if(special.equals("alertMessages")) {
139                call_logAlertMessages(level, engine, env) ;
140            } else if(special.equals("tags")) {
141                call_logTags(level, engine, env) ;
142            } else {
143                Itk.logError("unknown log special:", expr) ;
144            }
145        } else {
146            String tag = expr.getArgString("tag") ;
147            Term value = expr.getArgTerm("value") ;
148            Term result = ((value != null) ? engine.think(value, env) : null) ;
149
150            if(result == null) {
151                logGenericWithLevel(engine, level, true, tag) ;
152            } else {
153                logGenericWithLevel(engine, level, true, tag, ":", result) ;
154            }
155        }
156        return ThinkFormula.Term_True ;
157    }
158
159    //------------------------------------------------------------
160    /**
161     * 推論(log alertMessages)。
162     */
163    public void call_logAlertMessages(Itk.LogLevel level,
164                                      ThinkEngine engine, Object env) {
165        if(engine.isNullAgent()) {
166            logGenericWithLevel(engine, level, true, "no agent") ;
167        } else {
168            logGenericWithLevel(engine, level, true, "alertMessages",
169                                "time=", engine.getAgent().currentTime) ;
170            for(Map.Entry<Term, SimTime> entry :
171                    engine.getAlertedMessageTable().entrySet()) {
172                Term message = entry.getKey() ;
173                SimTime alertTime = entry.getValue() ;
174                logGenericWithLevel(engine, level, true, message, alertTime) ;
175            }
176            logGenericWithLevel(engine, level, true, "-----------------") ;
177        }
178    }
179
180    //------------------------------------------------------------
181    /**
182     * 推論(log tags)。
183     */
184    public void call_logTags(Itk.LogLevel level,
185                             ThinkEngine engine, Object env) {
186        if(engine.isNullAgent()) {
187            logGenericWithLevel(engine, level, false, "(null agent)") ;
188        } else {
189            logGenericWithLevel(engine, level, false,
190                                "tags=", engine.getAgent().getTags()) ;
191        }
192    }
193

195    /**
196     * ログ出力(level付き)
197     */
198    public void logGenericWithLevel(ThinkEngine engine,
199                                    Itk.LogLevel level,
200                                    boolean contP,
201                                    Object... objects) {
202        engine.logInThinkWithLevel(level, contP, objects) ;
203    }
204
205    //------------------------------------------------------------
206    /**
207     * ログ出力 level 解析
208     */
209    public Itk.LogLevel getLogLevel(Term term) {
210        if(term == null) { // default
211            return Itk.LogLevel.Info ;
212        } else {
213            return Itk.getLogLevel(term.getString()) ;
214        }
215    }
