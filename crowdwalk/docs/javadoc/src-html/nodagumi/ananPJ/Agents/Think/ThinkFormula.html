026/**
027 * 思考を表す式の処理系。
028 * 組み込まれている Formula リスト
029 * <ul>
030 *  <li>{@link ThinkFormulaLogical#call Logical: 論理形式および実行制御}</li>
031 *  <li>{@link ThinkFormulaArithmetic#call Arithmetic: 数値処理など}</li>
032 *  <li>{@link ThinkFormulaAgent#call Agent: エージェント制御関係}</li>
033 *  <li>{@link ThinkFormulaMisc#call Misc: その他}</li>
034 * </ul>
035 *
036 * <h3>新しい Formula の作成方法 </h3>
037 * <a name="addToExistingThinkFormula"></a>
038 * <h4> すでにある ThinkFormula*** への追加 </h4>
039 * 新しい Formula の head 名を、{@code "fooBar"} とする。
040 * また、それに対応するメソッド名を、{@code call_fooBar()} とする。
041 * <ol>
042 *   <li> {@code static public boolean registerFormulas()} の定義の中に、
043 *        {@code ThinkFormula.register("fooBar", singleton);} を追加。
044 *   </li>
045 *   <li> {@code public Term call(...)} のところの JavaDoc の記述に、
046 *        {@code <li>{@link #call_fooBar "fooBar"}</li>} を追加。
047 *        (javadoc でちゃんと参照できるようにしておくため。)
048 *   </li>
049 *   <li> {@code public Term call(...)} の定義の {@code if / else if} の並び
050 *        のところで、
051 *        {@code else if(head.equals("fooBar")) { return call_fooBar(head, expr, engine);}} 
052 *        を追加する。
053 *   </li>
054 *   <li> {@code public Term call_fooBar(...)} を定義する。
055 *        このメソッドの引数は、
056 *        <ul>
057 *          <li> {@code String head}: "fooBar" が渡ってくるはず。 </li>
058 *          <li> {@code Term expr}: formula そのものが渡される。
059 *               formula の引数などをアクセスする場合には、
060 *               {@code expr.getArgInt("argName")} などを呼び出す。
061 *          </li>
062 *          <li> {@code ThinkEngine engine}: formula を処理する ThinkEngine。
063 *               再帰的な呼び出しの場合は、{@code engine.think(...)} を
064 *               使う。
065 *               対象となっているエージェントは、{@code engine.getAgent()} で
066 *               参照できる。
067 *          </li>
068 *        </ul>
069 *        このメソッドの戻り値は、そのまま formula の戻り値として扱われる。
070 *   </li>
071 * </ol>
072 * 
073 * <h4> 新たに ThinkFormula*** を作成する場合 </h4>
074 * 新たに作るものを、{@code ThinkFormulaBazBaz.java} とする。
075 * <ol>
076 *   <li> {@code ThinkFormulaMisc.java} などを参考に
077 *        {@code ThinkFormulaBazBaz.java} の雛形を作る。
078 *   </li>
079 *   <li> {@code ThinkFormulaBazBaz.java} の雛形で必須のなのは、
080 *        代表インスタンスの {@code static ThinkFormula singleton;} と、
081 *        {@code public boolean registerFormulas(){...}}。
082 *   </li>
083 *   <li> <a href="#addToExistingThinkFormula">「すでにある ThinkFormula*** への追加」</a> 
084 *        に従い、formulaを追加。
085 *   </li>
086 *   <li> {@code "ThinkFormula.java"} の {@code static public Lexicon lexicon} の
087 *        宣言の後にある {@code static { ... }} 中に、
088 *        {@code ThinkFormulaBazBaz.registerFormulas();} を追加。
089 *   </li>
090 *   <li> さらに、{@code abstract public class ThinkFormula} の javadoc のところに、
091 *        {@code <li>{@link ThinkFormulaBazBaz#call BazBaz: ???}</li>} を追加。
092 *   </li>
093 * </ol>
094 */
095abstract public class ThinkFormula {
096
097    //============================================================
098    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
099    /**
100     * Formula Lexicon
101     */
102    static public Lexicon lexicon = new Lexicon() ;
103
104    //============================================================
105    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
106    /**
107     * 配列型 Form を許すかどうかの Lexicon
108     */
109    static public Lexicon lexiconForArrayForm = new Lexicon() ;
110
111    //============================================================
112    //============================================================
113    static {
114        ThinkFormulaMisc.registerFormulas(ThinkFormula.lexicon) ;
115        ThinkFormulaLogical.registerFormulas(ThinkFormula.lexicon) ;
116        ThinkFormulaArithmetic.registerFormulas(ThinkFormula.lexicon) ;
117        ThinkFormulaAgent.registerFormulas(ThinkFormula.lexicon) ;
118    }
119
120    //============================================================
121    //------------------------------------------------------------
122    /**
123     * Formula を登録
124     * @param name : form の head.
125     * @param formula : 登録する formula のインスタンス.
126     * @param permitArrayForm : 配列型の form を許すかどうか.
127     */
128    static public void register(String name, ThinkFormula formula,
129                                boolean permitArrayForm,
130                                Lexicon _lexicon) {
131        _lexicon.register(name, formula) ;
132        if(permitArrayForm) {
133            lexiconForArrayForm.register(name, Boolean.TRUE) ;
134        }
135    }
136
137    //============================================================
138    //------------------------------------------------------------
139    /**
140     * 配列型 form を許す head かどうか？
141     * @param name : form の head.
142     * @return 許すなら true。
143     */
144    static public boolean doesPermitArrayForm(String name) {
145        Boolean permit = (Boolean)lexiconForArrayForm.lookUp(name) ;
146        return Boolean.TRUE.equals(permit) ;
147    }
148    
149    //============================================================
150    // 特殊 Term
151    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
152    /**
153     * null
154     */
155    static public Term Term_Null = new Term() ;
156
157    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
158    /**
159     * true
160     */
161    static public Term Term_True = new Term(true, false) ;
162
163    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
164    /**
165     * false
166     */
167    static public Term Term_False = new Term(false, false) ;
168
169    //============================================================
170    //------------------------------------------------------------
171    /**
172     * boolean term.
173     */
174    static public Term booleanTerm(boolean v) {
175        return (v ? Term_True : Term_False) ;
176    }
177
178
179    //------------------------------------------------------------
180    /**
181     * コンストラクタ
182     */
183    public ThinkFormula(){}
184
185    //------------------------------------------------------------
186    /**
187     * 呼び出し
188     */
189    abstract public Term call(String head, Term expr,
190                              ThinkEngine engine, Object env) ;
191
192    //------------------------------------------------------------
193    /**
194     * False かどうかのチェック
195     */
196    public boolean checkFalse(Term expr) {
197        return (expr == null ||
198                expr.equals(Term_Null) ||
199                expr.equals(Term_False)) ;
200    }
201
202    //------------------------------------------------------------
203    /**
204     * _expr_ から引数の取り出し。
205     * _expr_ が Object 型 headedTerm の場合は、slot の arg を、
206     * _expr_ が Array 型 headedTerm の場合は、nth を、
207     * 返す。
208     */
209    public Term getArgFromExpr(Term expr, String slot, int n) {
210        if(expr.isArray()) {
211            if(expr.getArraySize() > n) {
212                return expr.getNthTerm(n) ;
213            } else {
214                return null ;
215            }
216        } else if(expr.isObject()) {
217            return expr.getArgTerm(slot) ;
218        } else {
219            return null ;
220        }
221    }
222
223    //------------------------------------------------------------
224    /**
225     * _expr_ から配列引数の取り出し。
226     * _expr_ が Object 型 headedTerm の場合は、slot の arg を、
227     * _expr_ が Array 型 headedTerm の場合は、_expr_ そのものを
228     * 返す。
229     */
230    public Term getArrayArgOrItself(Term expr, String slot) {
231        if(expr.isArray()) {
232            return expr ;
233        } else if(expr.isObject()) {
234            Term body = expr.getArgTerm(slot) ;
235            if(body.isArray()) {
236                return body ;
237            } else {
238                return null ;
239            }
240        } else {
241            return null ;
242        }
243    }
244
245    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
246    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
247    //------------------------------------------------------------
248    //============================================================
249} // class ThinkFormula
