406    /**
407     * シミュレーション準備 (from Link)
408     */
409    public void prepareForSimulation_FromLink() {
410        // 仮に、forwardDirection と仮定。
411        MapNode fromNode = currentPlace.getFromNode();
412        currentPlace.setEnteringNode(fromNode) ;
413        double costOfForwardDirection =
414            calcCostFromPlace(currentPlace, routePlan) ;
415
416        // backwardDirection に変更。
417        currentPlace.turnAround() ;
418        double costOfBackwardDirection =
419            calcCostFromPlace(currentPlace, routePlan) ;
420
421        //もし forward の方が低コストなら、再度 turnAround
422        if(costOfForwardDirection < costOfBackwardDirection) {
423            currentPlace.turnAround() ;
424        } else {
425            // do nothing
426        }
427    }

430    /**
431     * シミュレーション準備 (from Node)
432     */
433    public void prepareForSimulation_FromNode() {
434        currentPlace.setAdvancingDistance(0.0) ;
435        MapNode startNode = currentPlace.getEnteringNode() ;
436        double bestDist = Double.MAX_VALUE ;
437        MapLink bestLink = null ;
438        for(MapLink link : startNode.getLinks()) {
439            currentPlace.setLink(link) ;
440            double dist = calcCostFromPlace(currentPlace, routePlan) ;
441            if(dist < bestDist) {
442                bestLink = link ;
443                bestDist = dist ;
444            }
445        }
446        // 進行可能なリンクが一つも見つからない場合はエージェントをスタックさせる
447        if(bestLink == null) {
448            Itk.logError("currentPlace has no link for routePlan.") ;
449            Itk.logError_("currentPlace", currentPlace) ;
450            Itk.logError_("routePlan", routePlan) ;
451            finalizeEvacuation(SimTime.Zero, false, true) ;
452        }
453        currentPlace.setLink(bestLink) ;
454    }

495     */
496    private void calcSwingWidth() {
497        /* agent drawing */
498        //TODO should no call here, as lane should be set up properly
499        MapLink currentLink = currentPlace.getLink() ;
500
501        /* [2015.07.04 I.Noda]
502         * おそらくここでいちいち setup_lanes() を呼ぶのはおかしい。
503         * MapLink の update() で処理することにする。
504         */
505        //currentLink.setup_lanes();
506
507        int w = currentPlace.getLaneWidth() ;
508        int index = currentPlace.getIndexInLane(this) ;
509        if (isBackwardDirection())
510            index = currentPlace.getLane().size() - index;
511
512        if (isForwardDirection()) {
513            if (index >= 0) {
514                swing_width = (2.0 * ((currentPlace.getLane().size() - index) % w)) / currentLink.getWidth() - 1.0;
515            }  else {
516                swing_width = 0.0;
517            }
518        } else {
519            if (index >= 0) {
520                swing_width = 1.0 - (2.0 * (index % w)) / currentLink.getWidth();
521            }  else {
522                swing_width = 0.0;
523            }
524        }
525    }

549    /**
550     * currentPlace を nextPlace まで進める。
551     * @return 避難完了もしくはスタックした場合に true。それ以外はfalse。
552     */
553    protected boolean moveToNextPlace(SimTime currentTime) {
554        currentPlace.set(nextPlace) ;
555        while (!currentPlace.isOnLink()) {
556
557            /* もし、リンクを通り過ぎていて、そのリンクの終点が goal なら
558             * 避難完了 */
559            if ((isPlannedRouteCompleted() || isRestAllRouteDirective()) &&
560                currentPlace.getHeadingNode().hasTag(goal)){
561                recordTrail(currentTime, currentPlace, null) ;
562                finalizeEvacuation(currentTime, true, false) ;
563                return true;
564            }
565
566            chooseNextLinkCache.forced = true;
567            MapLink nextLink
568                = chooseNextLink(currentTime, currentPlace, routePlan, true) ;
569            chooseNextLinkCache.forced = true;
570
571            // 進行可能なリンクが見つからなければスタックさせる
572            if (nextLink == null) {
573                Itk.logInfo("Agent stuck",
574                            String.format("%s ID: %s, time: %.1f, linkID: %s",
575                                          getTypeName(), this.ID,
576                                          currentTime.getRelativeTime(),
577                                          currentPlace.getLink().ID)) ;
578                finalizeEvacuation(currentTime, false, true) ;
579                return true;
580            }
581
582            tryToPassNode(currentTime, routePlan, nextLink) ;
583
584            /* もし、渡った先のリンクがゴールなら、避難完了 */
585            if ((isPlannedRouteCompleted() || isRestAllRouteDirective()) &&
586                currentPlace.getLink().hasTag(goal)){
587                finalizeEvacuation(currentTime, true, false) ;
588                return true ;
589            }
590        }
591
592        // 進行可能なリンクが見つからず停止している状態ならばスタックさせる
593        if (speed == 0.0) {
594            MapLinkTable nextLinkList
595                = currentPlace.getHeadingNode().getUsableLinkTable();
596            if (nextLinkList.size() == 0) {
597                Itk.logInfo("Agent stuck", 
598                            String.format("%s ID: %s, time: %.1f, linkID: %s",
599                                          getTypeName(), this.ID,
600                                          currentTime.getRelativeTime(),
601                                          currentPlace.getLink().ID)) ;
602                finalizeEvacuation(currentTime, false, true) ;
603                return true;
604            }
605        }
606
607        return false;
608    }

612     * update
613     */
614    @Override
615    public boolean update(SimTime currentTime) {
616        /* [2015.01.10 I.Noda] 生成前なら処理しない */
617        if (currentTime.isBefore(generatedTime)) {
618            return false;
619        }
620
621        if ((isPlannedRouteCompleted() || isRestAllRouteDirective()) &&
622            getPrevNode().hasTag(goal)){
623            finalizeEvacuation(currentTime, true, false) ;
624            return true;
625        }

684    /**
685     * 前方にいるエージェントまでの距離計算
686     */
687    private double calcDistanceToPredecessor(SimTime currentTime) {
688        //前方のエージェントまでの距離の作業変数
689        double distToPredecessor = -currentPlace.getAdvancingDistance();
690        /* [2015.01.10 I.Noda]
691         * 余裕を持って探索するため、長めにとってみる。
692         */
693        double maxDistance
694            = (personalSpace + emptySpeed) * (currentTime.getTickUnit() + 1.0) ;
695
696        //前方のエージェントを探している場所
697        Place workingPlace = currentPlace.duplicate() ;
698        workingPlace.makeAdvance(maxDistance) ;
699
700        // 現在のリンク中での相対位置
701        int indexInLane = workingPlace.getIndexInLane(this) ;
702        //作業用の routePlan
703        RoutePlan workingRoutePlan = routePlan.duplicate() ;
704
705        while (workingPlace.getAdvancingDistance() > 0) {
706            ArrayList<AgentBase> agents = workingPlace.getLane() ;
707            int currentWidth = workingPlace.getLaneWidth() ;
708            int predecessorIndex = indexInLane + currentWidth ;
709            if(agents.size() > 0 && predecessorIndex < agents.size()) {
710                // 現在のworkingPlace に前の人がいる場合
711                // indexが負の場合は、最後尾の人が直前の人
712                if(predecessorIndex < 0) predecessorIndex = 0 ;
713                distToPredecessor +=
714                    agents.get(predecessorIndex).currentPlace.getAdvancingDistance() ;
715                break ;
716            } else {
717                //次以降のリンクに前の人がいる場合
718                distToPredecessor += workingPlace.getLinkLength() ;
719                indexInLane -= agents.size() ;
720                MapLink nextLink =
721                    chooseNextLinkBody(currentTime, workingPlace,
722                                       workingRoutePlan, true) ;
723                if (nextLink == null) {
724                    break;
725                }
726                workingPlace.transitTo(nextLink) ;
727            }
728        }
729        //最大を超えていれば、それで頭打ち
730        if(distToPredecessor > maxDistance) {
731            distToPredecessor = maxDistance ;
732        }
733
734        return distToPredecessor ;
735    }

739     * lane および plain による速度計算

741    private double calcSpeedBody(double previousSpeed, SimTime currentTime) {
742            /* base speed */
743        double baseSpeed =
744            currentPlace.getLink().calcEmptySpeedForAgent(emptySpeed,
745                                                          this, currentTime) ;
746        //自由速度に向けた加速
747        accel = calcAccel(baseSpeed, previousSpeed, currentTime) ;
748
749        //時間積分
750        double deltaSpeed = accel * currentTime.getTickUnit();
751        double _speed = previousSpeed + deltaSpeed ;
752
753        //速度幅制限
754        if (_speed > baseSpeed) {
755            _speed = baseSpeed ;
756        } else if (_speed < 0) {
757            _speed = 0;
758        }
759
760        //出口直前の場合で最前列の場合は、最大速にしておく。
761        int w = currentPlace.getLaneWidth() ;
762        int indexInLane = currentPlace.getIndexFromHeadingInLane(this) ;
763        if (indexInLane < w && currentPlace.getHeadingNode().hasTag(goal)) {
764            _speed = baseSpeed;
765        }
766
767        return _speed ;
768    }

771    /**
772     * lane および plain による速度計算
773     */
774    protected double calcAccel(double baseSpeed, double previousSpeed,
775                             SimTime currentTime) {
776        // 自由速度に向けた加速
777        double _accel = A_0 * (baseSpeed - previousSpeed) ;
778
779        // social force による減速
780        switch (getSpeedModel()) {
781        case LaneModel:
782            double distToPredecessor = calcDistanceToPredecessor(currentTime) ;
783            _accel += calcSocialForce(distToPredecessor) ;
784            break;
785        case PlainModel:
786        case CrossingModel:
787            double lowerBound =
788                -((baseSpeed / currentTime.getTickUnit()) + _accel) ;
789            _accel += accumulateSocialForces(currentTime, lowerBound) ;
790            break;
791        default:
792            Itk.logError("Unknown Speed Model") ;
793            Itk.logError_("speedModel",getSpeedModel()) ;
794            break;
795        }
796        return _accel ;
797    }

800    /**
801     * 前方エージェントからの social force を集める
802     * social force は、進行方向に沿った要素のみを扱う。
803     * 横方向距離は、距離計算の際に用いる。
804     * 横方向距離の計算は以下の通り。
805     *  順方向：エージェントから数えて n 番目のエージェントは、
806     *          ((w - (n % w)) % w) * u 横にいるとする。
807     *          つまり、w(レーン幅) = 3 の時、
808     *          1,2,3,4,5 番目のエージェントの横ずれ幅は、
809     *          2u, 1u, 0, 2u, 1u となる。
810     *  逆方向：エージェントから数えて n 番目のエージェントは、
811     *          ((w - (n % w)) % w + 1) * u 横にいるとする。
812     *          つまり、w(レーン幅) = 3 の時、
813     *          1,2,3,4,5 番目のエージェントの横ずれ幅は、
814     *          3u, 2u, 1u, 3u, 2u となる。
815     * @param currentTime : 時刻
816     * @param lowerBound : 計算上の力の下限。余計な計算しないために。
817     * @return 力
818     */
819    private double accumulateSocialForces(SimTime currentTime, double lowerBound) {
820        //求める力
821        double totalForce = 0.0 ;
822
823        //探す範囲
824        double maxDistance
825            = (personalSpace + emptySpeed) * (currentTime.getTickUnit() + 1.0) ;
826
827        //作業用の場所と経路計画
828        Place workingPlace = currentPlace.duplicate() ;
829        RoutePlan workingRoutePlan = routePlan.duplicate() ;
830
831        //当該エージェントの位置（注目しているリンクからの相対位置）
832        double relativePos = workingPlace.getAdvancingDistance() ;
833
834        //探索範囲終端まで場所を進めておく
835        workingPlace.makeAdvance(maxDistance) ;
836
837        //当該エージェントからのカウント
838        int count = 0 ; //(順方向)
839        int countOther = 0 ; //(逆方向)
840
841        //探索開始
842        while(workingPlace.getAdvancingDistance() > 0) {
843            if(totalForce < lowerBound) { break ; }
844
845            //逆方向探索
846            ArrayList<AgentBase> otherLane = workingPlace.getOtherLane() ;
847            int laneWidthOther = workingPlace.getOtherLaneWidth() ;
848            double linkLength = workingPlace.getLinkLength() ;
849            double insensitivePos = 0.0 ;
850            for(int i = 0 ; i < otherLane.size() ; i++) {
851                if(totalForce < lowerBound) { break ; }
852                AgentBase agent = otherLane.get(otherLane.size() - i - 1);
853                double agentPos = linkLength - agent.getAdvancingDistance() ;
854                if(agentPos > workingPlace.getAdvancingDistance()) {
855                    // 探索範囲外
856                    break ; // for からの脱出
857                } else if(agentPos <= relativePos) {
858                    // 当該エージェントの後方なので無視
859                    continue  ; // 次の for へ
860                } else if(agentPos <= insensitivePos) {
861                    // 直前のエージェントに近すぎる。（超過密状態用）
862                    continue ; // 次の for へ
863                } else {
864                    countOther++ ;
865                    double dx = agentPos - relativePos ;
866                    double dy =
867                        widthUnit_OtherLane *
868                        (((laneWidthOther - (countOther % laneWidthOther))
869                          % laneWidthOther) + 1) ;
870                    double force = calcSocialForceToHeading(dx,dy) ;
871                    totalForce += force ;
872                    if(countOther % laneWidthOther == 0) {
873                        insensitivePos = (agentPos +
874                                          insensitiveDistanceInCounterFlow) ;
875                    }
876                }
877            }
878
879            //順方向探索
880            ArrayList<AgentBase> sameLane = workingPlace.getLane() ;
881            int laneWidth = workingPlace.getLaneWidth() ;
882            boolean myTernIsOver = false ;
883            for(AgentBase agent : sameLane) {
884                if(totalForce < lowerBound) { break ; }
885                double agentPos = agent.getAdvancingDistance() ;
886                if(agent == this) {
887                    myTernIsOver = true ;
888                    continue ;
889                } else if(agentPos > workingPlace.getAdvancingDistance()) {
890                    // 探索範囲外
891                    break ; // for からの脱出
892//              } else if(agentPos <= relativePos) {
893                } else if(agentPos < relativePos) { // 同一の場合は次で判断
894                    // 当該エージェントの後方なので無視
895                    continue ; // 次の for へ
896                } else if(agentPos == relativePos && myTernIsOver) {
897                    // 同一位置で、かつ、自分より順番が後ろの場合は無視。
898                    // （つまり、同一位置で自分が先頭の時のみ、影響受けない）
899                    continue ;
900                } else {
901                    count++ ;
902                    double dx = agentPos - relativePos ;
903                    double dy =
904                        (widthUnit_SameLane *
905                         ((laneWidth - (count % laneWidth)) % laneWidth)) ;
906                    double force = calcSocialForceToHeading(dx,dy) ;
907                    totalForce += force ;
908                }
909            }
910            //次のリンクへ進む準備。
911            relativePos -= workingPlace.getLinkLength() ;
912            MapLink nextLink =
913                chooseNextLinkBody(currentTime, workingPlace,
914                                   workingRoutePlan, true) ;
915            if (nextLink == null) {
916                break;
917            }
918            //（直前のターンで）次のノードを交差して渡っている人の影響
919            if(getSpeedModel() == SpeedCalculationModel.CrossingModel) {
920                totalForce +=
921                    calcNodeCrossingForce(currentTime,
922                                          workingPlace.getLink(),
923                                          nextLink,
924                                          workingPlace.getHeadingNode(),
925                                          -relativePos) ;
926            }
927
928            // 次のリンクへ乗り移り。
929            workingPlace.transitTo(nextLink) ;
930        }
931
932        return totalForce ;
933    }

936    /**
937     * social force
938     * @param dist : 他のエージェントまでの距離
939     * @return 力
940     */
941    protected double calcSocialForce(double dist) {
942        return - emptySpeed * A_1  * Math.exp(A_2 * (personalSpace - dist)) ;
943    }

946    /**
947     * social force のうち、進行方向に沿った力のみを計算する。
948     * @param dx : 進行方向に沿った距離
949     * @param dy : 横方向の距離
950     * @return x 方向の力
951     */
952    protected double calcSocialForceToHeading(double dx, double dy) {
953        double dist = Math.sqrt(dx * dx + dy * dy) ;
954        double force = calcSocialForce(dist) ;
955        if(dist == 0.0) {
956            return force ;
957        } else {
958            return force * (dx / dist) ;
959        }
960    }

963    /**
964     * 向かっている交差点の交差する人流から受ける social force。
965     * @param currentTime : 現在時刻。
966     * @param node : 交差点。
967     * @param distance : 交差点までの距離
968     * @return social force。
969     */
970    public double calcNodeCrossingForce(SimTime currentTime,
971                                        MapLink fromLink,
972                                        MapLink toLink,
973                                        MapNode node,
974                                        double distance) {
975        RingBuffer<MapNode.PassingAgentRecord> buffer =
976            node.getPassingAgentRecordBuffer() ;
977        // 前方、バッファがなければ、まだ誰も交差してないので、
978        // おしまい。
979        if( buffer == null) { return 0.0 ; } 
980
981        // バッファを順にチェック。
982        double force = 0.0 ; /* force の小計 */
983        double dw = MapLink.dWidth ; /* 道幅を考慮するときの、
984                                          距離の減少分 */
985        for(MapNode.PassingAgentRecord record : buffer) {
986            if(currentTime.calcDifferenceFrom(record.time)
987               > nodeCrossingForceTimeMargin) {
988                // 時間マージンを過ぎていたら、それ以降無視。
989                break ;
990            }
991            if(record.isCrossing(fromLink, toLink)) {
992                double dCross = 
993                    ((record.fromLink.getWidth() + record.toLink.getWidth()) /
994                     2.0) ; /* 交差する道の両幅の平均 */
995                /* 交差分だけ幅減少 */
996                dCross -= dw ;
997                dCross /= nodeCrossingForceFactor ;
998                dw += MapLink.dWidth ; /* 交差した分を増やす */
999                if(dCross < 0.0) { dCross = 0.0 ; } ; /* 負の値は避ける */
1000                double dist = distance + dCross ;
1001                force += calcSocialForce(dist) ;
1002            }
1003        }
1004        return force ;
1005    }

1012    /**
1013     * try to pass a node, and enter next link
1014     * change: navigationReason, route, prev_node, previous_link,
1015     * current_link, position, evacuated, link.agentExists
1016     */
1017    protected boolean tryToPassNode(SimTime currentTime,
1018                                    RoutePlan workingRoutePlan,
1019                                    MapLink nextLink) {
1020        /* [2014.12.19 I.Noda]
1021         * NaiveAgent への経路記録の入り口のため,
1022         * recordTrail を導入。
1023         */
1024        recordTrail(currentTime, currentPlace, nextLink) ;
1025
1026        MapNode passingNode = currentPlace.getHeadingNode();
1027        MapLink previousLink = currentPlace.getLink() ;
1028        Direction direction_orig = currentPlace.getDirection() ;
1029
1030        /* agent exits the previous link */
1031        currentPlace.getLink().agentExits(this);
1032        // passingNode から出て行ったエージェント数のカウント
1033        currentPlace.getLink().incrementPassCounter(passingNode, false);
1034
1035        /* transit to new link */
1036        currentPlace.transitTo(nextLink) ;
1037        calcNextTarget(passingNode, workingRoutePlan, false) ;
1038        /* register agent to new link */
1039        currentPlace.getLink().agentEnters(this);
1040        // passingNode から入ってきたエージェント数のカウント
1041        currentPlace.getLink().incrementPassCounter(passingNode, true);
1042        /* record agent pass the passing node */
1043        passingNode.recordPassingAgent(currentTime, this,
1044                                       previousLink, nextLink) ;
1045
1046        //update_swing_flag = true;
1047        //2011年6月7日修正
1048        /*
1049         * この部分の修正では、歩行者がリンクの変更を伴う移動をおこなう場合に
1050         * swing_width を変更するか、しないかの処理を変更しています。
1051         * 従来はリンクの変更をする場合には、必ずswing_widthを更新していました。
1052         * そのため、ある歩行者がリンクに流入する際に、連続する歩行者が同一のレーンを移動するという
1053         * 不自然な描画が発生しています。(本来は異なるレーンで平行して移動するように描画されるべき)
1054         * 修正後は、
1055         * 条件分岐文1　現在のリンクと移動先のリンクの幅が同じ(レーン数が同じ)、他のリンクの流入出がない、移動先のリンクが出口を含む
1056         * 条件分岐文2　現在のリンクと移動先のリンクの幅が同じ(レーン数が同じ)、他のリンクの流入出がない
1057         * という二つの条件を満たした場合、swing_width の更新をおこないません。
1058         * この修正によって、swing_width が更新されないため、不自然な描画の発生は防がれています。
1059         */
1060        if (currentPlace.getLink().getWidth() == previousLink.getWidth() &&
1061            passingNode.getUsableLinkTable().size() == 2) {
1062            if (direction_orig != getDirection()) { swing_width *= -1; }
1063            update_swing_flag = false;
1064        } else {
1065            update_swing_flag = true;
1066        }
1067
1068        return true;
1069    }

1162    /**
1163     * ノードにおいて、次の道を選択するルーチン
1164     */
1165    protected MapLink chooseNextLinkBody(SimTime currentTime,
1166                                         Place passingPlace,
1167                                         RoutePlan workingRoutePlan,
1168                                         boolean on_node) {
1169        // 前回の呼び出し時と同じ結果になる場合は不要な処理を回避する
1170        MapLink nextLink
1171            = chooseNextLinkCache.getResultInCache(this, passingPlace) ;
1172        if(nextLink != null) { return nextLink ; }
1173
1174        chooseNextLinkCache.recordSituation(this, passingPlace) ;
1175
1176        final Term nextTarget =
1177            calcNextTarget(passingPlace.getHeadingNode(),
1178                           workingRoutePlan, on_node) ;
1179        navigationReason.clear().add("for").add(nextTarget).add("\n");
1180
1181        MapLinkTable candidateLinkList =
1182            passingPlace.getHeadingNode().getUsableLinkTable();
1183        double nextLinkCost = Double.MAX_VALUE;
1184        MapLinkTable nextLinksWithSameCost = null;
1185
1186        for (MapLink link : candidateLinkList) {
1187            /* ゴールもしくは経由点のチェック。あるいは、同じ道を戻らない */
1188            if (workingRoutePlan.isEmpty() && link.hasTag(goal)) {
1189                /* finishing up */
1190                nextLink = link ;
1191                nextLinkCost = 0.0 ;
1192                navigationReason.add("found goal").add(goal);
1193                break;
1194            } else if (link.hasTag(nextTarget)) {
1195                /* reached sub goal */
1196                nextLink = link ;
1197                nextLinkCost = 0.0 ;
1198                navigationReason.add("found mid-goal in").add(link) ;
1199                if(!isKnownDirective(workingRoutePlan.top())) {
1200                    // directive でない場合のみ shiftする。
1201                    // そうでなければ、directive の操作待ち。
1202                    workingRoutePlan.shift() ;
1203                }
1204                break;
1205            }
1206
1207            // 現在の link を選択した場合の nextTarget までのコスト計算
1208            try {
1209                double cost =
1210                    calcCostFromNodeViaLink(link,
1211                                            passingPlace.getHeadingNode(),
1212                                            nextTarget) ;
1213                if (cost < nextLinkCost) { // 最小cost置き換え
1214                    nextLink = link ;
1215                    nextLinkCost = cost;
1216                    nextLinksWithSameCost = null ;
1217                } else if (cost == nextLinkCost) { // 最小コストが同じ時の処理
1218                    if (nextLinksWithSameCost == null) {
1219                        nextLinksWithSameCost = new MapLinkTable();
1220                        nextLinksWithSameCost.add(nextLink) ;
1221                    }
1222                    nextLinksWithSameCost.add(link) ;
1223                }
1224            } catch(TargetNotFoundException e) {
1225                // この way_candidate からは next_target にたどり着けない
1226                Itk.logTrace(e.getMessage());
1227                continue;
1228            }
1229        }
1230
1231        if (nextLinksWithSameCost != null && nextLinksWithSameCost.size()>0) {
1232            nextLink
1233                = nextLinksWithSameCost.get(random.
1234                                           nextInt(nextLinksWithSameCost.size())) ;
1235        }
1236
1237        if (nextLink != null) {
1238            navigationReason
1239                .add("\n -> chose")
1240                .add(nextLink.getOther(passingPlace.getHeadingNode())) ;
1241        }
1242
1243        chooseNextLinkCache.recordResult(nextLink) ;
1244
1245        return nextLink;
1246    }

1259    /**
1260     * 指定された RoutePlan で、次のターゲットを得る。
1261     * @param node : このノードにおけるターゲットを探す。
1262     * @param workingRoutePlan : 指定された RoutePlan。shiftする可能性がある。
1263     * @return : workingRoutePlan の index 以降の次のターゲット、もしくは goal。
1264     * [2014.12.30 I.Noda] analysis
1265     * 次に来る、hint に記載されている route target を取り出す。
1266     * 今、top の target が現在のノードのタグにある場合、
1267     * route は１つ進める。
1268     */
1269    protected Term calcNextTarget(MapNode node,
1270                                  RoutePlan workingRoutePlan,
1271                                  boolean on_node) {
1272        if (on_node && !workingRoutePlan.isEmpty() &&
1273            node.hasTag(workingRoutePlan.top())){
1274            /* [2015.01.10 I.Noda] memo
1275             * on_node で、そのノードが subgoal なら、
1276             * routePlan をシフトして再度探索。
1277             * 2つ以上 subgoal を消化する場合を考え、再帰呼び出し。
1278             */
1279            workingRoutePlan.shift() ;
1280            return calcNextTarget(node, workingRoutePlan, on_node) ;
1281        } else {
1282            /* [2015.01.10 I.Noda] memo
1283             * routePlan のトップが、現在のノードの hint に入っているかどうか
1284             * 確認する。
1285             * 疑問：routePlan に directive が入っていても大丈夫か？
1286             */
1287            while (!workingRoutePlan.isEmpty()) {
1288                Term subgoal = nakedTargetFromRoutePlan(workingRoutePlan) ;
1289                if (node.hasTag(subgoal)) {
1290                    workingRoutePlan.shift() ;
1291                } else {
1292                    confirmCheckedRouteKey(subgoal, true) ;
1293                    if (node.hasHint(mentalMode, subgoal)) {
1294                        return subgoal;
1295                    } else {
1296                        Itk.logWarn("no sub-goal hint for " + subgoal);
1297                        workingRoutePlan.shift() ;
1298                    }
1299                }
1300            }
1301            return goal ;
1302        }
1303    }

1307     * ある _node においてあるwayを選択した場合の目的地(_target)までのコスト。
1308     * ここを変えると、経路選択の方法が変えられる。
1309     */
1310    public double calcCostFromNodeViaLink(MapLink _link, MapNode _node,
1311                                          Term _target)
1312        throws TargetNotFoundException
1313    {
1314        /* [2015.04.14 I.Noda]
1315         * もし新しい target なら、経路探査する。
1316         * [2016.01.31 I.Noda]
1317         * 本来なら、mentalMode ごとに探査すべきかもしれない。
1318         * そのためには、isCheckedRouteKey の拡張が必要。
1319         */
1320        confirmCheckedRouteKey(_target, true) ;
1321
1322        MapNode other = _link.getOther(_node);
1323        double cost = other.getDistance(mentalMode, _target) ;
1324        cost += _link.getMentalLength(mentalMode, _node);
1325        return cost ;
1326    }

1332     * @param _target : 次のターゲット。

1338        String targetTag = _target.getString() ;
1339        if(!getMap().isCheckedRouteKey(targetTag)) {
1340            if(_calcP) {
1341                Itk.logInfo("New Target", "find path.", "tag=", targetTag) ;
1342                getMap().calcGoalPathAllWithSync(targetTag) ;
1343            } else {
1344                Itk.logWarn("Unknown Target", "tag=", targetTag) ;
1345            }
1346            return false ;
1347        } else {
1348            return true ;
1349        }

1354     * あるplaceから現在のroutePlanの次の目的地までのコスト。
1355     * @param _place : 現在地を示す Place
1356     * @param _routePlan : 現在の経路計画。保存される。
1357     * @return コスト<br>次の目的地へのルートが見つからない場合は Double.MAX_VALUE を返す
1358     */
1359    protected double calcCostFromPlace(Place _place,
1360                                       final RoutePlan _routePlan) {
1361        RoutePlan workingRoutePlan = _routePlan.duplicate() ;
1362        Term target = calcNextTarget(_place.getHeadingNode(),
1363                                     workingRoutePlan,false) ;
1364        double costFromEnteringNode;
1365        try {
1366            costFromEnteringNode =
1367                calcCostFromNodeViaLink(_place.getLink(),
1368                                        _place.getEnteringNode(), target) ;
1369        } catch(TargetNotFoundException e) {
1370            Itk.logTrace(e.getMessage());
1371            return Double.MAX_VALUE;
1372        }
1373        double costFromPlace =
1374            costFromEnteringNode - _place.getAdvancingDistance() ;
1375        return costFromPlace ;
1376    }
