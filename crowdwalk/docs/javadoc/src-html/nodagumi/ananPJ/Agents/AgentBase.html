173    /**
174     * 初期化。constractorから分離。
175     * 注意：AgentBase の継承クラスでは、init() よりも initByConf() の方が
176     * 先に呼ばれてしまうことに注意。
177     */
178    public void init(Random _random, EvacuationSimulator simulator, 
179                     AgentFactory _factory, SimTime currentTime,
180                     Term fallback) {
181        super.init(null);
182        random = _random;
183        swing_width = getRandomDouble() * 2.0 - 1.0;
184        calcSwing();
185        // ObstructerBase のサブクラスのインスタンスを取得
186        obstructer = ObstructerBase.createAndInitialize(obstructerType, this) ;
187        //AgentFactory から移したもの
188        generatedTime = currentTime ;
189        setHandler(simulator.getAgentHandler()) ;
190
191        // 一旦、factory に制御を戻す。
192        // （ruby generation rule への対応のため）
193        _factory.initAgent(this, fallback) ;
194    }

196    //------------------------------------------------------------
197    /**
198     * Factory による初期化。
199     * Factory の方から一旦制御を戻す。
200     * @param _factory : Agent を生成する Agent Factory。
201     * @param fallback : fallback の値を収めたもの。
202     */
203    public void initByFactory(AgentFactory _factory, Term fallback) {
204        setFactory(_factory) ;
205        // set route
206        //setGoal(new Term(factory.goal, false));
207        setGoal(_factory.getGoal()) ; // 多分問題ないはず。[2017.04.22 I.Noda]
208        setPlannedRoute(_factory.clonePlannedRoute()) ;
209        // tag
210        for (final String tag : _factory.getTags()) {
211            addTag(tag);
212        }
213        // further setup by agentConf
214        initByConf(_factory.getAgentConf(), fallback) ;
215    }

720    /**
721     *
722     */
723    public String getNextCandidateString() {
724        if (isPlannedRouteCompleted()) {
725            return "";
726        } else {
727            Term term = routePlan.top();
728            if (term.isString()) {
729                return term.getHeadString();
730            } else if (term.isNull()) {
731                return "";
732            } else {
733                Term target = (Term)term.getArg("target");
734                if (target == null) {
735                    return "";
736                } else {
737                    return target.getHeadString();
738                }
739            }
740        }
741    }

744    /**
745     * evacuation の完了
746     * @param currentTime : 時刻
747     * @param onNode : true なら currentPlace.getHeadingNode() 上
748     *                 false なら currentPlace.getLink() 上
749     * @param stuck : スタックかどうか
750     */
751    protected void finalizeEvacuation(SimTime currentTime,
752                                      boolean onNode, boolean stuck) {
753        consumePlannedRoute() ;
754        setEvacuated(true, currentTime, stuck) ;
755        if(onNode) {
756            currentPlace.getHeadingNode().acceptEvacuatedAgent(this,
757                                                               currentTime) ;
758        }
759        if(currentPlace.isWalking()) {
760            lastPlace.set(currentPlace) ;
761            currentPlace.getLink().agentExits(this) ;
762            currentPlace.quitLastLink() ;
763        }
764        if (getMap() != null) {
765            getMap().getNotifier().agentEvacuated(this);
766        }
767    }

807     * 逆向きなどはちゃんと方向を直して扱う。
808     * [2020.02.14 I.Noda]
809     * もし、ID が null の場合、null がある方が前に進んでいるとみなす。
810     * 両方 null なら、同じ位置と見なす。
811     */
812    public int compareTo(AgentBase rhs) {
813        if(this == rhs) return 0 ;
814
815        double h1 = this.currentPlace.getAdvancingDistance() ;
816        double h2 = rhs.currentPlace.getAdvancingDistance() ;
817
818        if(h1 > h2) {
819            return 1 ;
820        } else if(h1 < h2) {
821            return -1 ;
822        } else if(ID == null && rhs.ID == null) {
823            return 0 ;
824        } else if(ID == null) {
825            return 1 ;
826        } else if(rhs.ID == null) {
827            return -1 ;
828        } else {
829            return ID.compareTo(rhs.ID) ;
830        }
831    }

921            return false;
922        }
923
924        int delta = 0 ;
925        while (delta < routePlan.length()) {
926            Term candidate = routePlan.top(delta);

928                return false ;
929            }

932        return true;

1016    /**
1017     * リンク上の表示上の横ずれ幅計算
1018     */
1019    protected void calcSwing() {
1020        MapLink currentLink = currentPlace.getLink() ;
1021        if (null == currentLink) {
1022            swing = new Vector3D(0.0, 0.0, 0.0);
1023            return;
1024        }
1025
1026        double scale = ((MapPartGroup)(currentLink.getParent())).getScale();
1027        double fwidth = currentLink.getWidth() / 2 / scale;
1028        double x1 = currentLink.getFrom().getX();
1029        double x2 = currentLink.getTo().getX();
1030        double y1 = currentLink.getFrom().getY();
1031        double y2 = currentLink.getTo().getY();
1032
1033        Vector3D v1 = new Vector3D(x2 - x1, y2 - y1, 0.0).normalize();
1034        Vector3D v2 = new Vector3D(0.0, 0.0, fwidth * swing_width);
1035        swing = Vector3D.crossProduct(v1, v2);
1036    }
