031/**
032 * 素朴なエージェント
033 * 各種エージェントクラスのベースにするもの。
034 *
035 * <h3> config, fallbackResources に書ける設定 </h3>
036 * {@link AwaitAgent} に加えて、
037 * <pre>
038 *  {
039 *    "trailCountStep" : __double__ // 一度通った道のカウントアップの幅。
040 * }
041 * </pre>
042 */
043public class NaiveAgent
044    extends AwaitAgent {
045    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
046    /**
047     * クラス名。
048     * ClassFinder でも参照できるようにしておく。
049     */
050    public static String typeString = "NaiveAgent" ;
051    public static String getTypeName() { return typeString ;}
052
053    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
054    /**
055     * 一度通った道に加えるカウントの大きさの規定値
056     */
057    static final public double Fallback_TrailCountStep = 1.0 ;
058
059    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
060    /**
061     * 一度通った道に加えるカウントの大きさの規定値
062     */
063    protected double trailCountStep = Fallback_TrailCountStep ;
064
065    //============================================================
066    /**
067     * 一度通った道の足跡テーブル
068     */
069    public class TrailCountTable {
070        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
071        /**
072         * 格納用テーブル
073         */
074        private HashMap<MapLink, HashMap<MapNode, Double>> table =
075            new HashMap<MapLink, HashMap<MapNode, Double>>() ;
076        
077        //------------------------------
078        /**
079         * アクセス
080         * @param fromNode 出発ノード
081         * @param toLink fromNodeからたどるリンク
082         */
083        public Double get(MapNode fromNode, MapLink toLink) {
084            HashMap<MapNode, Double> nodeTab = table.get(toLink) ;
085            if(nodeTab != null) {
086                Double count = nodeTab.get(fromNode) ;
087                if(count != null) {
088                    return count ;
089                } else {
090                    return 0.0 ;
091                }
092            } else {
093                return 0.0 ;
094            }
095        }
096
097        //------------------------------
098        /**
099         * 足跡を残す
100         * ここでは、単純に、通る毎にcountStepを足しているだけ。
101         * もし、減衰するフェロモンのように扱いたい場合は、
102         * 以下のようにする必要がある。
103         * <UL>
104         *  <LI> 開始からの時刻 t がわかっているとする。 </LI>
105         *  <LI> あるリンクの値が count とすると、
106         *       count = log(1+exp(count-t)) + t </LI>
107         *  <LI> 実際の値として用いるのは、
108         *       value = exp(count - t) </LI>
109         *  <LI> 上記のようにすれば、value は、毎ステップ (1/e) 倍に
110         *       していきつつ、使われると 1 増えるとしたのと同じになる</LI>
111         * </UL>
112         * @param fromNode 出発ノード
113         * @param toLink fromNodeからたどるリンク
114         * @param countStep 一歩の値
115         */
116        public Double add(MapNode fromNode, MapLink toLink, double countStep) {
117            HashMap<MapNode, Double> nodeTab = table.get(toLink) ;
118            if(nodeTab == null) {
119                nodeTab = new HashMap<MapNode, Double>() ;
120                table.put(toLink, nodeTab) ;
121            }
122
123            Double count = nodeTab.get(fromNode) ;
124            if(count == null) count = 0.0 ;
125
126            count += countStep ;
127            nodeTab.put(fromNode, count) ;
128
129            return count ;
130        }
131    } // class TrailCountTable
132
133    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
134    /**
135     * 一度通った道のカウント（アナログ値）
136     * 斉藤さんの実装を参考に再実装
137     */
138    public TrailCountTable trailCountTable = new TrailCountTable() ;
139
140    //------------------------------------------------------------
141    /**
142     * 引数なしconstractor。 ClassFinder.newByName で必要。
143     */
144    public NaiveAgent(){}
145
146    //------------------------------------------------------------
147    /**
148     * Conf による初期化。
149     */
150    @Override
151    public void initByConf(Term conf, Term fallback) {
152        super.initByConf(conf, fallback) ;
153
154        trailCountStep = getDoubleFromConfig("trailCountStep", trailCountStep) ;
155
156    } ;
157
158    //------------------------------------------------------------
159    /**
160     * あるlinkを選択した場合の目的地(_target)までのコスト。
161     * 正規のコストに、ランダム要素を加味する。
162     */
163    @Override
164    public double calcCostFromNodeViaLink(MapLink _link, MapNode _node,
165                                          Term _target) 
166        throws TargetNotFoundException {
167        return super.calcCostFromNodeViaLink(_link, _node, _target) ;
168    }
169
170    //------------------------------------------------------------
171    /**
172     * 最終決定したルート、足跡情報の記録
173     * [2014.12.19 I.Noda] tryToPassNode() より移動
174     */
175    @Override
176    protected void recordTrail(SimTime currentTime, Place passingPlace, 
177                               MapLink nextLink) {
178        super.recordTrail(currentTime, passingPlace, nextLink) ;
179
180        trailCountTable.add(passingPlace.getHeadingNode(),
181                            nextLink, trailCountStep) ;
182    }
183
184} // class NaiveAgent
