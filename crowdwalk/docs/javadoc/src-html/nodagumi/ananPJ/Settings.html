002package nodagumi.ananPJ;
003
004import java.io.BufferedReader;
005import java.io.BufferedWriter;
006import java.io.FileNotFoundException;
007import java.io.FileReader;
008import java.io.FileWriter;
009import java.io.IOException;
010import java.io.ObjectInputStream;
011import java.io.ObjectOutputStream;
012import java.util.TreeMap;
013
014import nodagumi.Itk.*;
015
016public class Settings extends TreeMap<String, String> {
017        String settingsFilename = "settings.ini";
018        static Settings settings = null;        
019
020        private Settings () {
021                super();
022        }
023        
024        public static Settings load(String s) {
025                if (settings != null) {
026                        if(!settings.settingsFilename.equals(s)) {
027                                Itk.logWarn("Settings is alread loaded from:",
028                                                        settings.settingsFilename) ;
029                                Itk.logWarn_("ignored file",s) ;
030                        }
031                        return settings;
032                }
033                settings = new Settings();
034
035                settings.settingsFilename = s;
036                try {
037                        FileReader fr = new FileReader (settings.settingsFilename);
038                        BufferedReader br = new BufferedReader (fr);
039                        
040                        String line = null;
041                        while ((line = br.readLine()) != null) {
042                                String elems[] = line.split("=");
043                                if (elems.length == 2) {
044                                        settings.put (elems[0], elems[1]);
045                                }
046                        }
047                        
048                        br.close ();
049                        fr.close ();
050                } catch (FileNotFoundException e) {
051                        /* No setting file, not a problem */
052                } catch (IOException e) {
053            Itk.dumpStackTraceOf(e);
054                }
055                return settings;
056        }
057        
058        public static void save () {
059                Itk.logInfo("Save Settings File", settings.settingsFilename) ;
060                try {
061                        FileWriter fw = new FileWriter (settings.settingsFilename);
062                        BufferedWriter bw = new BufferedWriter (fw);
063                        
064                        for (String key : settings.keySet()) {
065                                bw.write(key + "=" + settings.get(key) + "\n");
066                        }
067                        
068                        bw.close ();
069                        fw.close ();
070                } catch (IOException e) {
071            Itk.dumpStackTraceOf(e);
072                }
073        }
074        
075        public String get (String key, String defval) {
076                String val = get (key);
077                if (val == null) {
078                        val = defval;
079                        this.put(key, val);
080                }
081                return val;
082        }
083
084        public int get (String key, int defval) {
085                String val = get (key);
086                if (val == null) {
087                        val = "" + defval;
088                        this.put(key, val);
089                }
090                return Integer.parseInt(val);
091        }
092        
093        public void put (String key, int val) {
094                put (key, "" + val);
095        }
096
097        public static Settings getSettings() {
098                return settings;
099        }
100}
