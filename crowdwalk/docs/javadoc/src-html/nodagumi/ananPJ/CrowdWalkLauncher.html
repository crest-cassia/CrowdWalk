075    public static void defineOptions(Options options) {
076        options.addOption("c", "cui", false, "CUI モードでシミュレーションを開始する\nproperties-file の指定が必須");
077        options.addOption("g", "gui", false, "マップエディタウィンドウを開かずに GUI モードでシミュレーションを開始する\nproperties-file の指定が必須");
078        options.addOption("2", "use-2d-simulator", false, "2D GUI シミュレータを使用する");
079        options.addOption("h", "help", false, "この使い方を表示して終了する");
080        options.addOption(OptionBuilder.withLongOpt("log-level")
081            .withDescription("ログレベルを指定する\nLEVEL = Trace | Debug | Info | Warn | Error | Fatal")
082            .hasArg().withArgName("LEVEL").create("l"));
083        options.addOption("L", "legacy", false, "legacy モードにする\nPOLYGON | STRUCTURE タグをポリゴンの識別子として扱う");
084        options.addOption("N", "disable-no-hint-for-goal-log", false, "\"No hint for goal\" ログを出力しない");
085        options.addOption("o", "offline", false, "Internet への接続をおこなわない");
086        options.addOption(OptionBuilder.withLongOpt("tick")
087            .withDescription("tick 情報を FILE に出力する\nCUI モード時のみ有効")
088            .hasArg().withArgName("FILE").create("t"));
089        options.addOption(OptionBuilder.withLongOpt("fallback")
090                          .withDescription("fallback の先頭に追加する")
091                          .hasArg().withArgName("JSON").create("f"));
092        options.addOption("v", "version", false, "バージョン情報を表示して終了する");
093        options.addOption("V", "validate", false, "マップデータを検証する\n問題があればシミュレーションを実行せずに終了する");
094    }

096    /**
097     * コマンドラインオプションを解析して指定された処理を実行する
098     * @param args : main メソッドの args 引数
099     */
100    public static void parseCommandLine(String[] args, Options options) {
101        String propertiesFilePath = null;
102        ArrayList<String> fallbackStringList = new ArrayList<String>() ;
103
104        try {
105            CommandLine commandLine = new PosixParser().parse(options, args);
106            // ヘルプ表示オプションもしくはコマンドライン引数エラー
107            if (commandLine.hasOption("help") || commandLine.getArgs().length > 1) {
108                printHelp(options);
109                Itk.quitSafely() ;
110            }
111            // バージョン表示
112            if (commandLine.hasOption("version")) {
113                System.err.println(
114                    "CrowdWalk Version\n" +
115                    "----------------------------------------------------------------\n" +
116                    getVersion() +
117                    "----------------------------------------------------------------");
118                Itk.quitSafely() ;
119            }
120
121            // fallback への追加
122            if (commandLine.hasOption("fallback")) {
123                for(String fallback : commandLine.getOptionValues("fallback")) {
124                    fallbackStringList.add(fallback) ;
125                }
126            }
127
128            // プロパティファイルの指定あり
129            if (commandLine.getArgs().length == 1) {
130                propertiesFilePath = commandLine.getArgs()[0];
131            }
132
133            // ログレベルの指定
134            if (commandLine.hasOption("log-level")) {
135                setLogLevel(commandLine.getOptionValue("log-level"));
136            }
137
138            // 2D GUI シミュレータを使用する
139            use2dSimulator = commandLine.hasOption("use-2d-simulator");
140
141            // オフラインモード
142            offline = commandLine.hasOption("offline");
143
144            // "No hint for goal" ログを出力しない
145            CrowdWalkPropertiesHandler.setDisableNoHintForGoalLog(commandLine.hasOption("disable-no-hint-for-goal-log"));
146
147            // legacy モード
148            legacy = commandLine.hasOption("legacy");
149
150            // マップデータを検証する
151            CrowdWalkPropertiesHandler.setValidation(commandLine.hasOption("validate"));
152
153            // CUI モードで実行
154            if (commandLine.hasOption("cui")) {
155                if (propertiesFilePath == null) {
156                    printHelp(options);
157                    Itk.quitByError() ;
158                }
159                CuiSimulationLauncher launcher
160                    = launchCuiSimulator(propertiesFilePath, fallbackStringList);
161                // tick 情報の出力
162                if (launcher != null && commandLine.hasOption("tick")) {
163                    String tickFilePath = commandLine.getOptionValue("tick");
164                    saveTick(tickFilePath, launcher.simulator.currentTime);
165                }
166            }
167            // GUI モードで実行
168            else if (commandLine.hasOption("gui")) {
169                if (propertiesFilePath == null) {
170                    printHelp(options);
171                    Itk.quitByError() ;
172                }
173                launchGuiSimulator(propertiesFilePath, fallbackStringList);
174            }
175            // マップエディタの実行
176            else {
177                launchMapEditor(propertiesFilePath, fallbackStringList);
178            }
179        } catch (ParseException e) {
180            Itk.logError(e.getMessage());
181            printHelp(options);
182            Itk.quitByError() ;
183        } catch (Exception e) {
184            Itk.dumpStackTraceOf(e);
185            Itk.logError(e.getMessage());
186            Itk.quitByError() ;
187        }
188    }

197    /**
198     * ログレベルを設定する
199     */
200    public static void setLogLevel(String logLevelName) throws Exception {
201        for (Map.Entry<Itk.LogLevel, String> entry : Itk.LogTag.entrySet()) {
202            if (entry.getValue().equals(Itk.LogTagPrefix + logLevelName)) {
203                Itk.logLevel = entry.getKey();
204                return;
205            }
206        }
207        throw new Exception("Option error - ログレベル名が間違っています: "
208                            + logLevelName);
209    }

211    /**
212     * CUI シミュレータを実行する
213     */
214    public static CuiSimulationLauncher
215        launchCuiSimulator(String propertiesFilePath,
216                           ArrayList<String> commandLineFallbacks)
217    {
218        CuiSimulationLauncher launcher =
219            new CuiSimulationLauncher(propertiesFilePath,
220                                      commandLineFallbacks);
221        if (CrowdWalkPropertiesHandler.validation() && ! launcher.getMap().validate()) {
222            Itk.logError_("Failed to validate the map data.");
223            return null;
224        }
225        launcher.initialize();
226        launcher.start();
227        return launcher;
228    }
229
230    /**
231     * GUI シミュレータを実行する
232     */
233    public static GuiSimulationLauncher
234        launchGuiSimulator(String propertiesFilePath,
235                           ArrayList<String> commandLineFallbacks) throws Exception
236    {
237        settings = Settings.load(SETTINGS_FILE_NAME);
238        GuiSimulationLauncher launcher;
239        if (use2dSimulator) {
240            launcher = GuiSimulationLauncher.createInstance("GuiSimulationLauncher2D");
241        } else {
242            if (! GuiSimulationLauncher.isUsableClass("GuiSimulationLauncher3D")) {
243                throw new Exception("このビルドには 3D シミュレータ機能が組み込まれていません。");
244            }
245            launcher = GuiSimulationLauncher.createInstance("GuiSimulationLauncher3D");
246        }
247        launcher.init(propertiesFilePath, settings, commandLineFallbacks);
248        if (CrowdWalkPropertiesHandler.validation() && ! launcher.getMap().validate()) {
249            Itk.logWarn_("Failed to validate the map data.");
250            return null;
251        }
252        launcher.simulate();
253        return launcher;
254    }

256    /**
257     * マップエディタを実行する
258     */
259    public static void launchMapEditor(String propertiesFilePath,
260            ArrayList<String> commandLineFallbacks) throws Exception {
261        if (! offline) {
262            if (isInternetEnabled()) {
263                internetEnabled = true;
264            } else {
265                offline = true;
266            }
267        }
268        settings = Settings.load(SETTINGS_FILE_NAME);
269        MapEditor editor = new MapEditor(settings);
270        if (propertiesFilePath != null) {
271            editor.setPropertiesFromFile(propertiesFilePath, commandLineFallbacks);
272            if (! editor.loadNetworkMap()) {
273                Itk.quitByError() ;
274            }
275        } else {
276            editor.initProperties(commandLineFallbacks);
277            editor.initNetworkMap();
278        }
279        editor.show();
280    }

282    /**
283     * tick 情報をファイルに出力する
284     */
285    public static void saveTick(String filePath, SimTime currentTime) throws IOException {
286        PrintWriter writer = new PrintWriter(filePath);
287        if (filePath.toLowerCase().endsWith(".json")) {
288            writer.write("{\"tick\" : " + currentTime.getRelativeTime() + "}");
289        } else {
290            writer.println(currentTime.getRelativeTime());
291        }
292        writer.close();
293    }

299        HelpFormatter formatter = new HelpFormatter();

302        formatter.printHelp(commandLineSyntax, options);

312    /**
313     * インターネットにつながるか?
314     */
315    public static boolean isInternetEnabled() {
316        try {
317            Itk.logInfo("Trying HTTP connection to:",
318                        RESPONSE_CONFIRMATION_URL) ;
319            URL url = new URL(RESPONSE_CONFIRMATION_URL);
320            HttpURLConnection con = (HttpURLConnection)url.openConnection();
321            con.setRequestMethod("GET");
322            con.setConnectTimeout(1000);    // 接続タイムアウト: 1sec
323            con.setReadTimeout(1000);       // 読み取りタイムアウト: 1sec
324            con.connect();
325            if (con.getResponseCode() == -1) {
326                Itk.logWarn("HTTP connection got an unknown error") ;
327                Itk.logWarn_("HTTP Error", con.getResponseMessage()) ;
328                return false;
329            }
330            Itk.logInfo("HTTP connection succeeded.") ;
331        } catch (SocketTimeoutException e) {
332            Itk.logWarn("HTTP GET", e.getMessage());
333            return false;
334        } catch (IOException e) {
335            Itk.logWarn("HTTP GET", e.getMessage());
336            return false;
337        }
338        return true;
339    }

341    public static void main(String[] args) {
342        Options options = new Options();
343        defineOptions(options);
344        parseCommandLine(args, options);
345
346        // Settings の保存
347        Runtime.getRuntime().addShutdownHook(new Thread() {
348            public void run() {
349                if (settings != null) {
350                    // ディレクトリがマップファイルと異なる場合は危険なので保存しない
351                    String mapDir = settings.get("mapDir", "");
352                    for (String item : SETTING_PATH_ITEMS) {
353                        if (! settings.get(item + "Dir", "").equals(mapDir)) {
354                            settings.put(item + "Dir", "");
355                            settings.put(item + "File", "");
356                        }
357                    }
358
359                    Settings.save();
360                }
361            }
362        });
363    }
