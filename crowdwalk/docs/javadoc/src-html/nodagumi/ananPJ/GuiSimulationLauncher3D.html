037    /**

040    @Override
041    public void pauseRequest() {
042        pause();
043        Platform.runLater(() -> {
044            if (simulationFrame.getStartButton().isSelected()) {
045                simulationFrame.getStartButton().fire();
046            }
047        });
048    }

050    /**
051     * サイクル毎の画面描画
052     */
053    public void updateEveryTick(final SimTime currentTime) {
054        viewController.resume();
055        viewController.statusChanged("statusText", getStatusLine());
056        viewController.statusChanged("displayClock", currentTime);
057        viewController.statusChanged("evacuatedCount", simulator.getEvacuatedCountStatus());
058
059        if (pauseEnabled && ! paused && currentTime.getTickCount() == simulationFrame.getPauseTime().getTickCount()) {
060            pauseRequest();
061        }
062
063        if (isRecordSimulationScreen() || simulationFrame.isViewSynchronized()) {
064            final CountDownLatch latch = new CountDownLatch(1);
065            Platform.runLater(() -> {
066                simulationFrame.updateCamerawork(currentTime);
067                viewController.updatePanelView();
068                update_buttons();
069                // スクリーンショット
070                if (isRecordSimulationScreen()) {
071                    String path = String.format("%s/capture%06d.%s", screenshotDir, (int)currentTime.getTickCount(), screenshotImageType);
072                    simulationFrame.captureScreenShot(path, screenshotImageType);
073                }
074                latch.countDown();
075            });
076            // 画面の更新が完了するまで待つ
077            try {
078                latch.await();
079            } catch (InterruptedException e) {}
080        } else {
081            Platform.runLater(() -> {
082                simulationFrame.updateCamerawork(currentTime);
083                viewController.updatePanelView();
084                update_buttons();
085            });
086        }
087        // スクリーンショットを保存するためのスレッドが増えすぎない様にする
088        while (simulationFrame.getSaveThreadCount() > 16) {
089            try {
090                Thread.sleep(100);
091            } catch (InterruptedException e) {}
092        }
093        viewController.suspend();
094    }

096    /**
097     * 画面出力用properties設定
098     */
099    public void setPropertiesForDisplay() {
100        super.setPropertiesForDisplay();
101        try {
102            agentSize = properties.getDouble("agent_size_3d", agentSize);
103            if (agentSize < 0.1 || agentSize > 30.0) {
104                throw new Exception("Property error - 設定値が範囲(0.1～30.0)外です: agent_size or agent_size_3d:" + agentSize);
105            }
106        } catch(Exception e) {
107            Itk.logFatal("Property file error", e.getMessage());
108            Itk.quitByError() ;
109        }
110    }

133    /**
134     * エージェント表示の準備
135     */
136    public void setupAgentView() {
137        agentAppearances.clear();
138        for (HashMap parameters : loadAgentAppearance()) {
139            AgentAppearance3D appearance = new AgentAppearance3D(this, simulationFrame, parameters);
140            if (! appearance.isValidFor3D()) {
141                Itk.logFatal("3D view not defined", agentAppearanceFile);
142                Itk.quitByError() ;
143            }
144            agentAppearances.add(appearance);
145        }
146    }
147
148    /**
149     * ウィンドウとGUIを構築する
150     */
151    public void setupFrame() {
152        final int x = settings.get("simulatorPositionX", 0);
153        final int y = settings.get("simulatorPositionY", 0);
154        final GuiSimulationLauncher3D launcher = this;
155
156        JFXPanel fxPanel = new JFXPanel();  // JavaFX アプリケーションスレッドを起動するために必要
157        Platform.runLater(() -> {
158            simulationFrame = new SimulationFrame3D("Simulation Preview",
159                    simulationPanelWidth, simulationPanelHeight, launcher, properties, mapTiles);
160
161            viewController = new SimulationViewController3D(simulationFrame);
162            viewController.addNetworkMapPartsListener(map) ;
163            viewController.suspend();
164
165            // エージェント表示の準備
166            setupAgentView();
167            simulationFrame.getSimulationPanel().setAgentAppearances(agentAppearances);
168
169            simulationFrame.setX(x);
170            simulationFrame.setY(y);
171            simulationFrame.show();
172            Platform.setImplicitExit(false);
173        });
174    }
