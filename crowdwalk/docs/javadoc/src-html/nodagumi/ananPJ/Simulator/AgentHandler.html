460     * ruby (CrowdWalkWrapper) から呼び出し用。
461     * ruby 側では、CrowdWalkWrapperの継承クラスにおいて、
462     * setupSimulationLoggers() というメソッドを再定義する。その中で、
463     * <pre>
464     *   addMemberToAgentTrailLogFormatter("foo") {|agent, currentTime, handler|
465     *      ## ここで、ログに吐き出すデータ（文字列・数値・Array・Hash)を生成する
466     *      ## Ruby のプログラムを書く。
467     *      ## 以下は、その例
468     *      [1, 2, 3, "hogehoge",
469     *        {"a" {@code =>} 10,
470     *         "b" {@code =>} nil}] ;
471     *   }
472     * </pre>
473     * というものを好きなだけ並べる。
474     * この例では、"foo" というスロットに、
475     * 生成したデータ ([1,2,3,...]) が出力される。
476     * なお、このメソッドの name に、"foo" が格納されて呼び出される。
477     * 
478     * sample/generatedTown/SampleWrapper.rb 参照。
479     */
480    public void addMemberToAgentTrailLogFormatterForRuby(String name) {
481        agentTrailLogFormatter
482            .addMember(agentTrailLogFormatter.new Member(name){
483                    public Object value(AgentBase agent, Object timeObj,
484                                        Object agentHandlerObj) {
485                        AgentHandler handler = (AgentHandler)agentHandlerObj ;
486                        return
487                            handler
488                            .callRubyWrapper("callbackAgentTrailLogMember",
489                                             name,
490                                             agent, timeObj, agentHandlerObj) ;
491                    }}) ;
492    }

922    /**
923     * エージェントの preprocess 処理
924     */
925    private void preUpdateAgents(SimTime currentTime) {
926        if (true) {
927            synchronized (simulator) {
928                for (MapLink link : getEffectiveLinkSet()) {
929                    preUpdateAgentsOnLink(link, currentTime) ;
930                }
931            }
932        } else {
933            /* [2017.07.04 I.Noda]
934             * 試しに、マルチスレッドでの preUpdate を実装。
935             * 残念ながら、うまく動作しない。
936             * java.util.ConcurrentModificationException が発生する。
937             * WalkAgent.chooseNextLinkBody(WalkAgent.java:1161)。
938             * ここでの原因は、おそらく、MapNode.getUsableLinkTable()
939             * で排他処理していないせい。
940             * さらに、大して速くならない。8並列(paraN=8)としても、

942             * なので、ここのマルチスレッド化はしばらくあきらめ。
943             */
944            synchronized (simulator) {
945                Iterator<MapLink> linkIterator =
946                    getEffectiveLinkSet().iterator() ;
947                ArrayList<Thread> threadList = new ArrayList<Thread>() ;
948                int paraN = 8 ;
949                for(int i = 0 ; i < paraN ; i++) {
950                    Runnable worker = new Runnable(){
951                            public void run() {
952                                MapLink link = null;
953                                while(true) {
954                                    synchronized(linkIterator) {
955                                        if(!linkIterator.hasNext()) break ;
956                                        link = linkIterator.next() ;
957                                    }
958                                    preUpdateAgentsOnLink(link, currentTime) ;
959                                }
960                            }
961                        } ;
962                    Thread thread = new Thread(worker) ;
963                    threadList.add(thread) ;
964                    thread.start() ;
965                }
966                try {
967                    for(Thread thread : threadList) {
968                        thread.join() ;
969                    }
970                } catch (InterruptedException ex) {
971                    Itk.quitWithStackTrace(ex) ;
972                }
973            }
974        }
975    }

1275            }

1277                if (factory.isEnabled()) return false;

1883        if (! agent.isDead()) {
1884            if (! tagsForIndividualPedestriansLog.isEmpty() || ! regexpsForIndividualPedestriansLog.isEmpty()) {
1885                boolean applied = false;
1886                for (String tag : agent.getTags()) {

1893                            applied = true;

1895                        }

1930            double emptySpeed = SetupFileInfo.fetchFallbackDouble(fallback, "emptySpeed", 0.0);

1933                Itk.quitByError();
1934            }
1935
1936            checkpoints = Arrays.asList(_checkpoints.split("\\s*,\\s*"));
1937            checkpointPassingTimes = new HashMap();

1943                        counter++;

1945                        for (MapLink link : node.getLinks()) {
1946                            if (link.getLength() < emptySpeed) {
1947                                Itk.logError("can not setup Logger", "There is a link of a length that can passing in 1 step: " + checkpoint);
1948                                Itk.quitByError();
1949                            }
1950                        }
1951                    }
1952                }
1953                if (counter == 0) {
1954                    Itk.logError("can not setup Logger", "Checkpoint node does not exist: " + checkpoint);

1960

1962                    agentMovementHistoryLoggerFormatter.new Column(checkpoint) {
1963                        public String value(AgentBase agent, Object timeObj, Object agentHandlerObj) {
1964                            HashMap<String, SimTime> checkpointPassingTime = checkpointPassingTimes.get(agent);
1965                            if (checkpointPassingTime == null) {
1966                                return "";
1967                            }
1968                            SimTime passingTime = checkpointPassingTime.get(checkpoint);
1969                            return passingTime == null ? "" : "" + (int)passingTime.getRelativeTime();
1970                        }
1971                    }
1972                );
1973            }
1974        }
1975
1976        try {
1977            String agentHistoryPath =
1978                simulator.getProperties()
1979                .getFilePath("agent_movement_history_file", null, false);
1980            if (agentHistoryPath != null) {

1984        } catch(Exception e) {

1986            Itk.quitWithStackTrace(e) ;
1987        }
1988    }
