206    /**
207     * 遅延 set seed の予約。
208     * @param timing : set seed するタイミング。enum SeedRandEvent.Timing。
209     * @param seed : set する seed 値。
210     */
211    public void reserveDelayedSetSeed(SeedRandEvent.Timing timing, long seed) {
212        delayedSetSeedTiming = timing ;
213        delayedSetSeedValue = seed ;
214        Itk.logInfo("reserveDelayedSetSeed",
215                    "timing=" + timing, "seed=" + seed);
216    }
217    
218    /**
219     * 遅延 set seed のチェックと実行
220     * @param timing : try するタイミング。
221     *                 これが deleydSetSeedTiming と一致すれば、実行される。
222     * @return set seed が実行されたら、true。
223     */
224    public boolean tryDelayedSetSeed(SeedRandEvent.Timing timing) {
225        if(delayedSetSeedTiming == timing) {
226            random.setSeed(delayedSetSeedValue) ;
227            delayedSetSeedTiming = SeedRandEvent.Timing.none ;
228            delayedSetSeedValue = 0 ;
229            Itk.logInfo("succeed to tryDelayedSetSeed",
230                        "timing=" + timing, "seed=" + delayedSetSeedValue) ;
231            return true ;
232        } else {
233            return false ;
234        }
235    }

652    void buildMap () {
653        /* Nodes */
654        for (MapNode node : getNodes()) {
655            // do nothinkg for node
656        }

658        for (MapLink link : getLinks()) {

662        for (MapArea area : networkMap.getAreas()) {

664                if (area.intersectsLine(link.getLine2D())) {

666                }
667            }
668        }
669        /* 主観的距離計さんルールの設定 */
670        networkMap.setMentalMapRules(properties.getMentalMapRules()) ;
671    }

718    /**
719     * ルート探索の準備
720     */
721    private void buildRoutes() {
722        /* evacuation based on goal tags */
723        ArrayList<String> all_goal_tags = agentHandler.getAllGoalTags();
724
725        ArrayList<String> no_goal_list = new ArrayList<String>();
726        HashMap<String, CalcGoalPath> workers = new HashMap<String, CalcGoalPath>();
727        HashMap<String, Thread> threads = new HashMap<String, Thread>();
728
729        // 経路探索をゴールごとに別スレッドで実行
730        for (String goal_tag : all_goal_tags) {
731            CalcGoalPath worker = new CalcGoalPath(goal_tag);
732            Thread thread = new Thread(worker);
733            workers.put(goal_tag, worker);
734            threads.put(goal_tag, thread);
735            thread.start();
736        }
737        // スレッド終了を待ってno_goal_list更新
738        try {
739            for (String goal_tag : all_goal_tags) {
740                threads.get(goal_tag).join();
741                CalcGoalPath worker = workers.get(goal_tag);
742                if (!worker.goalCalculated) {
743                    no_goal_list.add(worker.goalTag);
744                }
745            }
746        } catch (InterruptedException e) {
747            Itk.quitWithStackTrace(e);
748        }
749
750        // チェック
751        checkMapDefectiveness(all_goal_tags, no_goal_list) ;
752
753        // 不要なメモリを速やかに解放する
754        // (メモリ消費量が多いほど実行速度が遅くなる傾向があるため)
755        workers = null;
756        threads = null;
757        System.gc();
758
759    }

771    /**
772     * 地図とゴールの完全性チェック。
773     * 初期ゴールまで辿り着かないノード・リンクのチェック。
774     * 「初期ゴール」とは、開始時点でゴールと認定されているタグのこと。
775     * @return ゴールに到達できないノードがあれば true。
776     */
777    private boolean checkMapDefectiveness(List<String> goalTagList,
778                                          List<String> phantomGoalTagList) {
779        boolean isDefective = false ;
780        // 目的地のないゴールのリスト。
781        if (phantomGoalTagList.size() > 0) {
782            Itk.logInfo("Found goal tags without places:", 
783                        phantomGoalTagList) ;
784            isDefective |= true ;
785        }
786
787        // ゴールにたどり着かないノードとリンクの数え上げ
788        ArrayList<String> unreachableGoalList = new ArrayList<String>() ;
789        ArrayList<Integer> unreachableNodeCount = new ArrayList<Integer>() ;
790        ArrayList<Integer> unreachableLinkCount = new ArrayList<Integer>() ;
791        for (String goal : goalTagList) {
792            int nodeCount = 0 ;
793            int linkCount = 0 ;
794            for (MapNode node : networkMap.getNodes()) {
795                if (node.getHint(NavigationHint.DefaultMentalMode, goal, false)
796                    == null) {
797                    nodeCount += 1 ;
798                    for (MapLink link : node.getUsableLinkTable()) {
799                        //  link.addTag("INVALID_ROUTE");
800                        linkCount += 1 ;
801                    }
802                }
803            }
804            if(nodeCount + linkCount > 0) {
805                unreachableGoalList.add(goal) ;
806                unreachableNodeCount.add(nodeCount) ;
807                unreachableLinkCount.add(linkCount) ;
808                isDefective |= true ;
809            }
810        }
811
812        // 数え上げ結果報告
813        if(unreachableGoalList.size() > 0) {
814            for (int i = 0; i < unreachableGoalList.size(); i++) {
815                Itk.logInfo("Unreachable",
816                            "no route to",
817                            unreachableGoalList.get(i),
818                            "from",
819                            unreachableNodeCount.get(i), "nodes and",
820                            unreachableLinkCount.get(i), "links.") ;
821            }
822        } else {
823            Itk.logInfo("Map Check",
824                        "all goals can be reachable from whole map.");
825        }
826
827        return isDefective ;
828    }

843    /**
844     * Rubyエンジン準備。
845     * properties file の "use_ruby" が true の場合に有効化。
846     * 初期化は以下の順に行われる。
847     * <OL>
848     *   <LI> CrowdWalk を起動したところの directory からの相対で、
849     *        ruby のライブラリのプログラムのあるところ (現状で "src/main/ruby")
850     *        を LOAD_PATH に追加。
851     *   </LI>
852     *   <LI> properties file で指定された load path を追加。
853     *   </LI>
854     *   <LI> init file をロード。デフォルトは、"initForCrowdWalk.rb"。
855     *        "src/main/ruby" に用意してある。
856     *   </LI>
857     *   <LI> current dir を、properties file があるところへ移す。
858     *   </LI>
859     *   <LI> properties file で指定した init script の実行。
860     *   </LI>
861     *   <LI> Ruby の CrowdWalkWrapper のクラスのインスタンスを作る。
862     *        "ruby_simulation_wrapper_class" でクラスを指定できる。
863     *   </LI>
864     * </OL>
865     * RubyAgent については、generation file にて指定する。
866     */
867    private void buildRubyEngine() {
868        try {
869            if(properties != null && properties.getBoolean("use_ruby", false)) {
870                Itk.logInfo("Ruby Engine", "loading...") ;
871                // engine
872                rubyEngine = new ItkRuby() ;
873                // default load path
874                String currentDir = (String)rubyEngine.getCurrentDirectory() ;
875                String libDir = currentDir + "/" + rubyLibDir ;
876                rubyEngine.pushLoadPath(libDir) ;
877                // default load path on jar
878                rubyEngine.pushLoadPath(RubyLibDirInJar) ;
879                // additional load path
880                rubyLoadPath = properties.getPathList("ruby_load_path", null) ;
881                if(rubyLoadPath != null) {
882                    for(String path : rubyLoadPath) {
883                        rubyEngine.pushLoadPath(path) ;
884                    }
885                }
886                Itk.logInfo("Ruby Engine",
887                            "LOAD_PATH=", rubyEngine.getLoadPaths());
888                // load init file
889                if(rubyInitFile != null) {
890                    Itk.logInfo("Ruby Engine",
891                                "load init_file:", rubyInitFile) ;
892                    rubyEngine.eval(String.format("require '%s'",rubyInitFile)) ;
893                }
894                // current dir を properties の directory へ。
895                rubyEngine
896                    .setCurrentDirectory(properties.getPropertiesDirAbs()) ;
897                Itk.logInfo("Ruby Engine",
898                            "current dir=", rubyEngine.eval("Dir::pwd")) ;
899                // init script
900                Term initScriptTerm =
901                    properties.getTerm("ruby_init_script", null) ;
902                if(initScriptTerm != null) {
903                    String initScript = "" ;
904                    if(initScriptTerm.isArray()) {
905                        for(Object item : initScriptTerm.getArray()) {
906                            initScript += "\n" ;
907                            initScript += ((Term)item).getString() ;
908                        }
909                    } else {
910                        initScript = initScriptTerm.getString() ;
911                    }
912                    
913                    Itk.logInfo("Ruby Engine", 
914                                "eval init script=", initScript) ;
915                    rubyEngine.eval(initScript) ;
916                }
917                // wrapperの作成
918                rubyWrapperClass =
919                    properties.getString("ruby_simulation_wrapper_class", null);
920                if(rubyWrapperClass != null) {
921                    rubyWrapper =
922                        rubyEngine.newInstanceOfClass(rubyWrapperClass, this) ;
923                }
924                // AgentFactoryByRuby 用の処理
925                agentHandler.setupAgentFactoryByRuby(rubyEngine) ;
926                
927                // irb 用処理
928                if(properties.getBoolean("use_irb", false)) {
929                    Itk.logInfo("Setup Irb Thread.") ;
930                    irbSemaphore = new Semaphore(1) ;
931                    irbThread = new Thread(() -> {
932                            this.irbRepLoop() ;
933                        }) ;
934                    irbAcquireSemaphore() ;
935                    irbThread.start() ;
936                }
937                
938                // final.
939                Itk.logInfo("Ruby Engine", "loading... Done.") ;
940            }
941        } catch (Exception ex) {
942            Itk.logError("Exception in buildRubyEngine") ;
943            Itk.quitWithStackTrace(ex) ;
944        }
945    }

948    // Irb setup
949    /**
950     * Irb Rep Loop.
951     */
952    private void irbRepLoop() {
953        try {
954            // wait until thread is ready.
955            while(irbSemaphore == null) {
956                Thread.yield() ;
957            }
958            // initialize
959            irbAcquireSemaphore() ;
960            rubyEngine.eval("require 'irb'") ;
961            rubyEngine.setVariable("$crowdwalk", this) ;
962            //        rubyEngine.eval("pp IRB.conf[:PROMPT]") ;
963            //        rubyEngine.eval("IRB.conf[:PROMPT][:DEFAULT][:PROMPT_I] = 'CrowdWalk> '") ;
964        
965            // run
966            rubyEngine.eval("IRB.start") ;
967            irbReleaseSemaphore() ;
968        } catch(Exception ex) {
969            // do nothing ;
970        }
971    }

993    /**
994     * irb semaphor release.
995     */
996    public void irbReleaseSemaphore() throws InterruptedException {
997        if(irbSemaphore != null) {
998            irbSemaphore.release() ;
999            Thread.yield() ;
1000            Thread.sleep(1) ;
1001        }
1002    }

1004    /**
1005     * irb yield thread.
1006     */
1007    public void irbYield() throws InterruptedException {
1008        if(irbThread != null) {
1009            irbReleaseSemaphore() ;
1010            irbAcquireSemaphore() ;
1011        }
1012    }

1014    /**
1015     * Irb wait N cycle. (called from Ruby)
1016     */
1017    public Object irbWaitCycleN(Object n) {
1018        try {
1019            Long nTick = (Long)n ;
1020            SimTime cycleUntil =
1021                currentTime.newSimTimeWithAdvanceTick(nTick.intValue()) ;
1022            while(currentTime.isBefore(cycleUntil)) {
1023                irbYield() ;
1024            }
1025        } catch(InterruptedException ex) {
1026            return irbKillReason ;
1027        }
1028        return null ;
1029    }

1088    /**
1089     * 汎用 updateEveryTick() ;
1090     */
1091    public boolean updateEveryTick() {
1092        synchronized (stop_simulation) {
1093            deferSimulation() ;
1094
1095            // irb とのやり取り。
1096            try {
1097                irbYield() ;
1098            } catch(Exception ex) {} ;
1099            
1100            // ruby wrapper の preUpdate()
1101            if(useRubyWrapper())
1102                callRubyWrapper("preUpdate", currentTime) ;
1103
1104            // pollution 計算。
1105            if (usePollution())
1106                pollutionHandler.updateAll(currentTime, networkMap,
1107                                           getWalkingAgentCollection());
1108
1109            scenario.update(currentTime, networkMap) ;
1110            // 実行本体
1111            agentHandler.update(currentTime);
1112
1113            // ruby wrapper の postUpdate()
1114            if(useRubyWrapper())
1115                callRubyWrapper("postUpdate", currentTime) ;
1116
1117            // 描画のための呼び出し
1118            launcher.updateEveryTick(currentTime);
1119
1120            // カウンタを進める。
1121            clock.advance() ;
1122            currentTime = clock.newSimTime() ;
1123        }
1124        return isFinished() ;
1125    }
