002package nodagumi.ananPJ.Simulator.Obstructer;
003
004import nodagumi.ananPJ.Agents.AgentBase;
005import nodagumi.ananPJ.misc.SetupFileInfo;

007import nodagumi.Itk.*;

009/**
010 * ガスによるエージェントへの影響.
011 * (WalkAgent.java に書かれていた情報。[2015.07.03 I.Noda]
012 * <pre>
013 * effect of damage, for Chloropicrin (minutes):
014 *  20000, BLACK,  STOP #100% Dead
015 *  2000,  BLACK,  STOP #50% Dead
016 *  1000,  RED,    STOP #Cannot breathe
017 *  200,   YELLOW, 25%  #Cannot walk, can breath
018 *  1,     GREEN,  50%  #Can walk, cannot open eyes
019 *  0,     GREEN,  100% #Normal
020 * </pre>
021 */
022public class Pollution extends ObstructerBase {
023    // 参考データ: 三井化学アグロ株式会社 / クロルピクリンの物性・分解性 / 気中濃度と人体への影響
024    //   http://www.mitsui-agro.com/tabid/147/Default.aspx#04a4
025
026    /**
027     * トリアージレベル:Black の下限累積曝露量
028     *
029     * 致死
030     */
031    private double deadlyLevel = 297.6 * 10 * 60;          // 178,560
032
033    /**
034     * トリアージレベル:Red の下限累積曝露量
035     *
036     * 不耐
037     */
038    private double unbearableLevel = 15.0 * 1 * 60;        // 900
039
040    /**
041     * トリアージレベル:Yellow の下限累積曝露量
042     *
043     * 催涙により眼を開けていられない
044     */
045    private double incapacitatedLevel = 0.3 * 0.5 * 60;    // 9
046
047    /**
048     * 歩行速度に影響を与える最低累積曝露量
049     *
050     * 最低刺激
051     */
052    private double irritantLevel = 1.3;
053
054    /**
055     * 歩行不可能になる最低累積曝露量
056     */
057    private double nonAmbulatoryLevel = 9.0;    // = incapacitatedLevel
058
059    /**
060     * 現状の暴露量
061     */
062    private double currentExposureAmount = 0.0 ;
063
064    /**
065     * 累積暴露量
066     */
067    private double accumulatedExposureAmount = 0.0;
068
069    /**
070     * Pollution 用の fallback パラメータ
071     */
072    private static Term config = null;
073
074    /**
075     * fallback パラメータからの値の取得(double)
076     */
077    public static double getConfigParameter(String slot, double fallback) {
078        return SetupFileInfo.fetchFallbackDouble(config, slot, fallback);
079    }
080
081    //--------------------------------------------------
082    /**
083     * コンストラクタ
084     */
085    public Pollution() {}
086
087    /**
088     * 初期化.
089     */
090    public void init(AgentBase agent) {
091        this.agent = agent;
092        if (config == null) {
093            config =
094                SetupFileInfo.filterFallbackTerm(fallbackParameters,
095                                                 getClass().getSimpleName()) ;
096        }
097        deadlyLevel = getConfigParameter("deadlyLevel", deadlyLevel);
098        unbearableLevel = getConfigParameter("unbearableLevel", unbearableLevel);
099        incapacitatedLevel = getConfigParameter("incapacitatedLevel", incapacitatedLevel);
100        irritantLevel = getConfigParameter("irritantLevel", irritantLevel);
101        nonAmbulatoryLevel = getConfigParameter("nonAmbulatoryLevel", nonAmbulatoryLevel);
102    }
103
104    /**
105     * エージェントを暴露させる
106     */
107    public void expose(double exposureAmount) {
108        if (dead || agent.isEvacuated()) {
109            return;
110        }
111
112        currentExposureAmount = exposureAmount;
113        accumulatedExposureAmount += exposureAmount;
114        currentTriageLevel = calcTriage();
115        dead = calcDead();
116
117        /* [2015.07.02 I.Noda]
118         * Emergency は obsolete なので、使わない。
119         * ここの部分は、なにか別の方策が必要。
120         * 一番適切なのは、RationalAgent か RubyAgent で処理する。
121         */
122        /*
123        if (currentTriageLevel != TriageLevel.GREEN) {
124            agent.setGoal(SpecialTerm.Emergency) ;
125        }
126        */
127
128        if (currentTriageLevel != lastTriageLevel) {
129            agent.getMap().getNotifier().agentTriageChanged(agent);
130            lastTriageLevel = currentTriageLevel;
131        }
132    }
133
134    /**
135     * 歩行速度に影響を与える
136     *
137     * ※呼び出し元で speed を更新した後に呼ばれる
138     */
139    public double calcAffectedSpeed(double originalSpeed) {
140        if (accumulatedExposureAmount >= nonAmbulatoryLevel) {
141            return 0.0 ;
142        } else if (accumulatedExposureAmount >= irritantLevel) {
143            double speed =(agent.getSpeed()
144                           * (nonAmbulatoryLevel - accumulatedExposureAmount)
145                           / nonAmbulatoryLevel);
146            return speed ;
147        } else {
148            return originalSpeed ;
149        }
150    }
151
152    /**
153     * 現状の累積曝露量によるエージェントのトリアージレベルを求める
154     */
155    protected TriageLevel calcTriage() {
156        if (accumulatedExposureAmount >= deadlyLevel)
157            return TriageLevel.BLACK;
158        else if (accumulatedExposureAmount >= unbearableLevel)
159            return TriageLevel.RED;
160        else if (accumulatedExposureAmount >= incapacitatedLevel)
161            return TriageLevel.YELLOW;
162        else
163            return TriageLevel.GREEN;
164    }
165
166    /**
167     * dumpResult 用の値。
168     */
169    public double currentValueForLog() {
170        return currentExposureAmount ;
171    }
172
173    /**
174     * dumpResult 用の値。
175     */
176    public double accumulatedValueForLog() {
177        return accumulatedExposureAmount ;
178    }
179}
