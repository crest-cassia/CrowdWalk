001// -*- mode: java; indent-tabs-mode: nil -*-
002package nodagumi.ananPJ.Simulator.Obstructer;
003
004import nodagumi.ananPJ.Agents.AgentBase;
005import nodagumi.ananPJ.misc.SetupFileInfo;
006
007import nodagumi.Itk.*;
008
009/**
010 * 洪水によるエージェントへの影響.
011 */
012public class Flood extends ObstructerBase {
013    // 引用元: 千葉県ホームページ / 浸水深のランク分け
014    //   http://www.pref.chiba.lg.jp/bousaik/tsunamityosa/documents/gaiyo3.pdf
015
016    /**
017     * トリアージレベル:Black の下限浸水深(m)
018     *
019     * 道路歩行中の人は、ほぼ絶望である。(木など高いものに登るしかない)
020     */
021    private double waterDepthThreshold3 = 1.0;
022
023    /**
024     * トリアージレベル:Red の下限浸水深(m)
025     *
026     * 流速が遅く、路面の状況が良い場合には歩行が可能ではあるが、人的被害発生の可能性は非常に高い。
027     */
028    private double waterDepthThreshold2 = 0.8;
029
030    /**
031     * トリアージレベル:Yellow の下限浸水深(m)
032     *
033     * 歩くにはかなり困難であるが、大人であれば生命の危険性は少ない。
034     * ただし、流速が速い場合や子供や路面に障害物がある場合には人的被害発生の可能性が高い。
035     */
036    private double waterDepthThreshold1 = 0.3;
037
038    /**
039     * 歩行不可能になる浸水深(m)
040     */
041    private double nonAmbulatoryDepth = 1.0;
042
043    /**
044     * 現状の浸水深
045     */
046    private double currentDepth = 0.0;
047
048    /**
049     * Flood 用の fallback パラメータ
050     */
051    private static Term config = null;
052
053    /**
054     * fallback パラメータからの値の取得(double)
055     */
056    public static double getConfigParameter(String slot, double fallback) {
057        return SetupFileInfo.fetchFallbackDouble(config, slot, fallback);
058    }
059
060    //--------------------------------------------------
061    /**
062     * コンストラクタ
063     */
064    public Flood() {}
065
066    /**
067     * 初期化.
068     */
069    public void init(AgentBase agent) {
070        this.agent = agent;
071        if (config == null) {
072            config =
073                SetupFileInfo.filterFallbackTerm(fallbackParameters,
074                                                 getClass().getSimpleName()) ;
075        }
076        waterDepthThreshold3 =
077            getConfigParameter("waterDepthThreshold3", waterDepthThreshold3);
078        waterDepthThreshold2 =
079            getConfigParameter("waterDepthThreshold2", waterDepthThreshold2);
080        waterDepthThreshold1 =
081            getConfigParameter("waterDepthThreshold1", waterDepthThreshold1);
082        nonAmbulatoryDepth =
083            getConfigParameter("nonAmbulatoryDepth", nonAmbulatoryDepth);
084    }
085
086    /**
087     * エージェントを冠水させる
088     */
089    public void expose(double depth) {
090        if (dead || agent.isEvacuated()) {
091            return;
092        }
093
094        currentDepth = depth;
095        currentTriageLevel = calcTriage();
096        dead = calcDead();
097
098        if (currentTriageLevel != lastTriageLevel) {
099            agent.getMap().getNotifier().agentTriageChanged(agent);
100            lastTriageLevel = currentTriageLevel;
101        }
102    }
103
104    /* effect of flood, this damage does not increase */
105    // 呼び出し元でもとめた speed に影響を加える。
106    public double calcAffectedSpeed(double originalSpeed) {
107        if (dead) {
108            return 0.0 ;
109        }
110
111        // 歩行速度への影響
112        // ※スピードがマイナスになるのを避ける
113        double speed =
114            (originalSpeed *
115             (nonAmbulatoryDepth - Math.min(currentDepth, nonAmbulatoryDepth))
116             / nonAmbulatoryDepth);
117
118        return speed ;
119    }
120
121    /**
122     * 現状の浸水深によるエージェントのトリアージレベルを求める
123     */
124    protected TriageLevel calcTriage() {
125        if (currentDepth >= waterDepthThreshold3)
126            return TriageLevel.BLACK;
127        else if (currentDepth >= waterDepthThreshold2)
128            return TriageLevel.RED;
129        else if (currentDepth >= waterDepthThreshold1)
130            return TriageLevel.YELLOW;
131        else
132            return TriageLevel.GREEN;
133    }
134
135    /**
136     * dumpResult 用の値。
137     */
138    public double currentValueForLog() {
139        return currentDepth ;
140    }
141
142    /**
143     * dumpResult 用の値。
144     */
145    public double accumulatedValueForLog() {
146        return currentDepth ;
147    }
148}
