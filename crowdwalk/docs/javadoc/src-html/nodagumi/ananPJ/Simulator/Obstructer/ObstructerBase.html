001// -*- mode: java; indent-tabs-mode: nil -*-
002package nodagumi.ananPJ.Simulator.Obstructer;
003
004import java.io.InputStream;
005import java.io.InputStreamReader;
006
007import nodagumi.ananPJ.Agents.AgentBase;
008import nodagumi.ananPJ.misc.SetupFileInfo;
009
010import nodagumi.Itk.*;
011
012//======================================================================
013/**
014 * エージェントの歩行及び生命維持を妨害する要素.
015 */
016public abstract class ObstructerBase {
017    //============================================================
018    //============================================================
019    /**
020     * トリアージレベルを表す enum。
021     */
022    public static enum TriageLevel { 
023        //========================================
024        //::::::::::::::::::::::::::::::::::::::::
025        GREEN, YELLOW, RED, BLACK ;
026
027        //========================================
028        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
029        /** 全Level のリスト */
030        static public TriageLevel[] allList ;
031
032        /** 全Level の数 */
033        static public int allListSize ;
034
035        // 初期設定
036        static {
037            allList = TriageLevel.values() ;
038            allListSize = allList.length ;
039        }
040
041    }
042
043    /**
044     * Obstructer type とクラス名の対応表
045     */
046    public static final String OBSTRUCTER_CLASSES = "/obstructer_classes.json";
047
048    /**
049     * Obstructer用の fallback パラメータ
050     */
051    public static Term fallbackParameters = null;
052
053    /**
054     * fallback パラメータの準備
055     */
056    public static void setupCommonParameters(Term wholeFallbacks) {
057        fallbackParameters =
058            SetupFileInfo.filterFallbackTerm(wholeFallbacks, "obstructer") ;
059    }
060
061    /**
062     * エージェントへのリンク
063     */
064    protected AgentBase agent = null ;
065
066    /**
067     * 現状のトリアージレベル
068     */
069    protected TriageLevel currentTriageLevel = TriageLevel.GREEN;
070
071    /**
072     * 前回のトリアージレベル(更新チェック用)
073     */
074    protected TriageLevel lastTriageLevel = TriageLevel.GREEN;
075
076    /**
077     * エージェントが死亡しているかどうかのフラグ
078     */
079    protected boolean dead = false;
080
081    public abstract void init(AgentBase agent);
082    public abstract void expose(double exposureAmount);
083    public abstract double calcAffectedSpeed(double speed) ;
084    protected abstract TriageLevel calcTriage();
085
086    /* the state of the agent */
087    public TriageLevel getTriage() {
088        return currentTriageLevel ;
089    }
090
091    /* the state of the agent */
092    public int getTriageInt() {
093        return getTriage().ordinal() ;
094    }
095
096    /* the state of the agent */
097    public String getTriageName() {
098        return getTriage().name() ;
099    }
100
101    /**
102     * 現状の暴露量によるエージェントの死亡判定
103     */
104    protected boolean calcDead() {
105        return getTriage().ordinal() >= TriageLevel.BLACK.ordinal();
106    }
107
108    /**
109     * エージェントが死亡しているかどうか
110     */
111    public boolean isDead() {
112        return dead;
113    }
114
115    //--------------------------------------------------
116    /**
117     * dumpResult 用の値。(現在値)
118     */
119    abstract public double currentValueForLog() ;
120
121    //--------------------------------------------------
122    /**
123     * dumpResult 用の値。(累積値)
124     */
125    abstract public double accumulatedValueForLog() ;
126
127    //============================================================
128    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
129    private static ClassFinder classFinder = new ClassFinder() ;
130
131    //============================================================
132    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
133    /**
134     * デフォルトのPollutionクラス登録
135     */
136    static {
137        classFinder.aliasByJson(resourceToString(OBSTRUCTER_CLASSES));
138    }
139
140    /**
141     * サブクラスのインスタンスを生成する
142     */
143    public static ObstructerBase createInstance(String className) {
144        try {
145            return (ObstructerBase)classFinder.newByName(className) ;
146        } catch (Exception ex) {
147            Itk.logError("pollution_type の設定が間違っています。",
148                         className) ;
149            Itk.quitWithStackTrace(ex) ;
150        }
151        return null;
152    }
153
154    /**
155     * サブクラスのインスタンスを生成して初期化する
156     */
157    public static ObstructerBase createAndInitialize(String className, AgentBase agent) {
158        ObstructerBase obstructer = createInstance(className);
159        obstructer.init(agent);
160        return obstructer;
161    }
162
163    /**
164     * リソースファイルを文字列に変換する
165     */
166    public static String resourceToString(String resourceName) {
167        StringBuilder builder = new StringBuilder();
168        try {
169            InputStream is = ObstructerBase.class.getResourceAsStream(resourceName);
170            InputStreamReader reader = new InputStreamReader(is);
171            char[] buff = new char[1024];
172            int length = reader.read(buff);
173            while (length >= 0) {
174                builder.append(buff, 0, length);
175                length = reader.read(buff);
176            }
177            reader.close();
178        } catch (Exception ex) {
179            Itk.logError("リソースファイルの読み込みに失敗しました", resourceName);
180            Itk.quitByError() ;
181        }
182        return builder.toString();
183    }
184}
