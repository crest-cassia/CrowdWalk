093    public PollutionHandler(String scheduleFileName,
094                            ArrayList<MapArea> _areaList,
095                            SimTime _baseTime, double interpolationInterval) {
096        baseTime = _baseTime ;
097        if (scheduleFileName == null || scheduleFileName.isEmpty()) {
098            Itk.logInfo("Load Pollution File", "(none)") ;
099            nextInstantTime = null ;
100        } else {
101            readData(scheduleFileName);
102            Itk.logInfo("Load Pollution File", scheduleFileName);
103            Itk.logInfo("MAX Pollution Level", maxPollutionLevel) ;
104            linearInterpolation(interpolationInterval);
105            pollutionInstantIterator = pollutionInstantList.iterator();
106            if (pollutionInstantIterator.hasNext()) {
107                currentInstant = pollutionInstantIterator.next();
108                nextInstantTime = currentInstant.atTime ;
109            } else {
110                nextInstantTime = null ;
111            }
112        }
113        
114        setupPollutedAreas(_areaList);
115    }

118    /**
119     * pollution data を読み込む。
120     * <ul>
121     *   <li> ファイルの形式は、CSV。</li>
122     *   <li> "#" で始まる行はコメント行 </li>
123     *   <li> 各行の先頭は、開始からの時刻。続いて、各エリアの density が並ぶ。</li>
124     *   <li> 各エリアとの対応は、各 Area (MapAreaRectangle) の tag と、density の序数。</li>
125     *   <li> 各エリアは、序数と同じ整数値のタグを持つ。</li>
126     * </ul>
127     * 読み込まれたデータは、pollutionInstantList に入れられる。
128     * pollutionInstantList の各要素は、[時刻、density1, density2, ...] という
129     * PollutionInsatnt の配列。
130     * <p>
131     * [2015.06.12 I.Noda]
132     * 新たに、先頭行が "#" で始まる場合、その開始行を、以下のように解釈
133     * するようにする。
134     *
135     *      #<開始時刻>, タグ1, タグ2, タグ3, ...
136     *
137     * この場合、タグ1,2,3は、上記の整数値タグの代わりに用いられる。
138     * つまり、整数値タグでないタグをもつ Area を指定できる。
139     * なお、タグ1,2,3で指定した以上のデータが並んでいる場合は、
140     * 自動的に、その序数の整数値を持つタグが使われる。
141     */
142    private void readData(String fileName) {
143        pollutionInstantList.clear();
144        try {
145            CSVParser csvParser = new CSVParser() ;
146            int lineCount = 0 ;
147
148            BufferedReader reader = new BufferedReader(new FileReader(fileName));
149            String line = reader.readLine();
150            while (line != null) {
151                if (! line.trim().startsWith("#")) {
152                    String[] strItems = csvParser.parseLine(line) ;
153                    PollutionInstant instant = new PollutionInstant() ;
154                    instant.atTime =
155                        baseTime.newSimTime()
156                        .setRelativeTime(Double.parseDouble(strItems[0])) ;
157
158                    for (int index = 1; index < strItems.length; index++) {
159                        double value = Double.parseDouble(strItems[index]) ;
160                        // 先頭は time なので、１つずらす。
161                        instant.setValue(index-1, value) ;
162                        // 新しいタグ生成が必要ならば、それをチェック。
163                        if((index-1) >= tagList.size()) {
164                            String tag = (new Integer(index)).toString() ;
165                            addTag(Itk.intern(tag)) ;
166                        }
167                        // pollution の最大値を求める
168                        if (value > maxPollutionLevel) {
169                            maxPollutionLevel = value;
170                        }
171                    }
172                    pollutionInstantList.add(instant);
173                } else if(lineCount == 0) { // # で始まる先頭行の場合。
174                    // 形式を、
175                    // # <開始時刻>, <tag1>, <tag2>, <tag3>, <tag4>,... とする。
176                    // <開始時刻>は無くても良い。（その場合でもカンマは必要）
177                    String[] strItems = csvParser.parseLine(line) ;
178                    try {
179                        String timeStr = strItems[0].replaceAll("[\\# ]*","") ;
180                        if(timeStr.length() > 0) {
181                            int beginTime = Itk.scanTimeStringToInt(timeStr) ;
182                        } 
183                    } catch(Exception ex) {
184                        Itk.logWarn("Illegal Time Format", strItems[0]) ;
185                    }
186                    // [2015.06.12 I.Noda] 現状で、開始時刻は未対応。
187                    for(int index = 1; index < strItems.length ; index++) {
188                        addTag(Itk.intern(strItems[index])) ;
189                    }
190                }
191                lineCount++ ;
192                line = reader.readLine();
193            }
194        } catch (IOException e) {
195            Itk.quitWithStackTrace(e);
196        }
197    }

200    /**
201     * pollution データを interval 秒区分で線形補間する
202     */
203    private void linearInterpolation(double interval) {
204        if (interval <= 0.0 || pollutionInstantList.isEmpty()) {
205            return;
206        }
207        ArrayList<PollutionInstant> interpolatedPollutionInstantList =
208            new ArrayList<PollutionInstant>();
209        PollutionInstant lastInstant = null;
210        for (PollutionInstant instant : pollutionInstantList) {
211            if (lastInstant != null) {
212                double lastEventTime = lastInstant.atTime.getRelativeTime() ;
213                double eventTime = instant.atTime.getRelativeTime() ;
214                if ((eventTime - lastEventTime) > interval) {
215                    // 線形補間
216                    for (double time = lastEventTime + interval;
217                         time < eventTime;
218                         time += interval) {
219                        PollutionInstant interpolatedInstant
220                            = new PollutionInstant() ;
221                        interpolatedInstant.atTime =
222                            baseTime.newSimTime().setRelativeTime(time);
223
224                        for (int index = 0; index < instant.valueSize(); index++) {
225                            double a = (time - lastEventTime) / (eventTime - lastEventTime);    // 補間係数
226                            double v = (lastInstant.getValue(index) +
227                                        a * (instant.getValue(index) -
228                                             lastInstant.getValue(index)));
229                            interpolatedInstant.setValue(index, v) ;
230                        }
231                        interpolatedPollutionInstantList.add(interpolatedInstant);
232                    }
233                }
234            }
235            interpolatedPollutionInstantList.add(instant);
236            lastInstant = instant;
237        }
238        pollutionInstantList.clear();
239        pollutionInstantList.addAll(interpolatedPollutionInstantList);
240    }

243    /**
244     * Pollution Area のタグがついた MapArea を探す。
245     * [2015.06.12 I.Noda]
246     * 現状の設定では、数字のみのタグを、Pollution 用のタグと解釈する。
247     * これは、あまりにひどい設計なので、修正が必要。
248     * また、ある Index を示すタグを持つ Area はただひとつであることを
249     * 仮定している。
250     * この設定も、まずいか、あるいは、ただひとつであることをチェックする
251     * 機能が必要。
252     */
253    private void setupPollutedAreas(ArrayList<MapArea> areas) {
254        for (MapArea area : areas) {
255            String matchedTag = null ;
256            for(String tag : tagList) {
257                if(area.hasTag(tag)) {
258                    if(areaIndexTable.containsKey(area)) {
259                        Itk.logWarn("The area matchs with multiple pollution tag",
260                                    area, tag) ;
261                    } else {
262                        Integer index = tagIndexTable.get(tag) ;
263                        areaIndexTable.put(area, index) ;
264                        area.pollutionLevel = newPollutionLevelInfo() ;
265                        area.pollutionIsUpdated() ;
266                    }
267                }
268            }
269        }
270    }

274     * 毎回呼ばれるメインのサイクル。
275     */
276    public void updateAll(SimTime currentTime,
277                          NetworkMap map,
278                          Collection<AgentBase> agents) {
279
280        if (nextInstantTime != null && currentTime.isAfter(nextInstantTime)) {
281            updatePollution();
282            updateLinks(map) ;
283        }
284        updateAgents(agents) ;
285    }

288    /**
289     * pollution を更新する際に呼ばれる。
290     */
291    private void updatePollution() {
292        Itk.logDebug("PC: updating pollution ",nextInstantTime);
293
294        for (HashMap.Entry<MapArea, Integer> entry : areaIndexTable.entrySet()) {
295            MapArea area = entry.getKey() ;
296            Integer index = entry.getValue() ;
297
298            double pollutionLevel = currentInstant.getValue(index) ;
299            area.getPollutionLevel().setCurrentLevel(pollutionLevel) ;
300            area.pollutionIsUpdated() ;
301        }
302
303        if (pollutionInstantIterator.hasNext()) {
304            currentInstant = pollutionInstantIterator.next();
305            nextInstantTime = currentInstant.atTime ;
306        } else {
307            nextInstantTime = null ;
308        }
309    }

312    /**
313     * Link の update。
314     * pollution対象リンクの汚染フラグを更新する(汚染度が0に戻ることも考慮する)
315     */
316    private void updateLinks(NetworkMap map) {
317        for (MapLink link : map.getLinks()) {
318            if (link.getIntersectedMapAreas().isEmpty()) {
319                continue;
320            }
321            link.setPolluted(false);
322            for (MapArea area : link.getIntersectedMapAreas()) {
323                if (area.isPolluted()) {
324                    link.setPolluted(true);
325                    break;
326                }
327            }
328        }
329    }

331    //------------------------------------------------------------
332    /**
333     * Agent の update。
334     */
335    private void updateAgents(Collection<AgentBase> agents) {
336        for (AgentBase agent : agents) {
337            if (agent.isEvacuated())
338                continue;
339            if (! agent.getCurrentLink().isPolluted()) {
340                agent.exposed(0.0);
341                continue;
342            }
343
344            Double pollutionLevel = null;
345            Point3D point = new Point3D(agent.getPosition().getX(),
346                    agent.getPosition().getY(),
347                    agent.getHeight() + AGENT_HEIGHT);
348
349            for (MapArea area : agent.getCurrentLink().getIntersectedMapAreas()) {
350                if (area.contains(point)) {
351                    pollutionLevel = area.getPollutionLevel().getCurrentLevel() ;
352                    agent.exposed(pollutionLevel * baseTime.getTickUnit());
353                    Itk.logNone("polluted", agent.ID, pollutionLevel) ;
354                    break;
355                }
356            }
357            if (pollutionLevel == null) {
358                agent.exposed(0.0);
359            }
360        }
361    }

378    /**
379     * Tag の登録。
380     */
381    private boolean addTag(String tag) {
382        if(tagIndexTable.containsKey(tag)) {
383            Itk.logWarn("duplicated tag",tag) ;
384            return false ;
385        } else {
386            int index = tagList.size() ;
387            tagList.add(tag) ;
388            tagIndexTable.put(tag, index) ;
389            return true ;
390        }
391    }

402    /**
403     * ある特定時点での Pollution 状況。
404     */
405    public static class PollutionInstant {
406        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
407        /**
408         * タイムスライスの時刻。
409         * 実際には、このタイムスライスが終わる時刻。
410         */
411        public SimTime atTime = null ;
412
413        /**
414         * Pollution の値
415         */
416        private ArrayList<Double> value = new ArrayList<Double>() ;
417
418        //--------------------------------------------------
419        /**
420         * Pollution の値を取得。
421         */
422        public Double getValue(int index) {
423            return value.get(index) ;
424        }
425
426        //--------------------------------------------------
427        /**
428         * Pollution の値をセット。
429         */
430        public void setValue(int index, double val) {
431            while(value.size() <= index) { value.add(0.0) ;}
432            value.set(index, val) ;
433        }
434
435        //--------------------------------------------------
436        /**
437         * Pollution の値の数。
438         */
439        public int valueSize() {
440            return value.size() ;
441        }

446    /**
447     * エリアにおける Pollution の効果の情報
448     */
449    public static class PollutionLevelInfo {
450        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
451        /**
452         * PollutionHandler へのリンク。
453         */
454        private PollutionHandler handler = null ;
455
456        /**
457         * 現在の効果。
458         */
459        private double currentLevel = 0.0 ;
460
461        /**
462         * 一つ前の効果。
463         */
464        private double lastLevel = 0.0 ;
465
466        //--------------------------------------------------
467        /**
468         * コンストラクタ。
469         */
470        public PollutionLevelInfo(PollutionHandler _handler) {
471            handler = _handler ;
472            currentLevel = 0.0 ;
473            lastLevel = 0.0 ;
474        }
475
476        //--------------------------------------------------
477        /**
478         * 実レベルを設定。
479         */
480        public void setCurrentLevel(double level) {
481            currentLevel = level ;
482        } ;
483
484        //--------------------------------------------------
485        /**
486         * 実レベルを取得。
487         */
488        public double getCurrentLevel() {
489            return currentLevel ;
490        }
491
492        //--------------------------------------------------
493        /**
494         * 正規化レベルを取得。
495         */
496        public double getNormalizedLevel() {
497            return currentLevel / handler.getMaxPollutionLevel() ;
498        }
499
500        //--------------------------------------------------
501        /**
502         * 汚染されているかどうかのチェック。
503         */
504        public boolean isPolluted() {
505            return currentLevel > 0.0 ;
506        }
507
508        //--------------------------------------------------
509        /**
510         * 変化を検出。
511         * @param update : 検出した後、lastLevelを更新するかどうか？
512         * @return もし変化していたら true。変化無ければ false。
513         */
514        public boolean isChanged(boolean update) {
515            boolean changed = (currentLevel != lastLevel) ;
516            if(update) {
517                lastLevel = currentLevel ;
518            }
519            return changed ;
520        } ;
521    }
