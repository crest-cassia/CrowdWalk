024/**
025 * CrowdWalk の起動を司る(CUI シミュレータ専用)
026 */
027public class CuiCrowdWalkLauncher {
028    public static String optionsFormat = "[-c] [-h] [-l <LEVEL>] [-N] [-t <FILE>] [-f <FALLBACK>]* [-v] [-V]"; // これはメソッドによる取得も可能
029    public static String commandLineSyntax = String.format("crowdwalk %s <properties-file>", optionsFormat);
030
031    /**
032     * バージョン情報の取得(最終コミットログの出力で代用)
033     */
034    public static String getVersion() {
035        return ObstructerBase.resourceToString("/commit_version.txt");
036    }
037
038    /**
039     * コマンドラインオプションの定義
040     */
041    public static void defineOptions(Options options) {
042        options.addOption("c", "cui", false, "互換性維持のためのオプション(動作には影響しない)");
043        options.addOption("h", "help", false, "この使い方を表示して終了する");
044        options.addOption(OptionBuilder.withLongOpt("log-level")
045            .withDescription("ログレベルを指定する\nLEVEL = Trace | Debug | Info | Warn | Error | Fatal")
046            .hasArg().withArgName("LEVEL").create("l"));
047        options.addOption("N", "disable-no-hint-for-goal-log", false, "\"No hint for goal\" ログを出力しない");
048        options.addOption(OptionBuilder.withLongOpt("tick")
049            .withDescription("tick 情報を FILE に出力する")
050            .hasArg().withArgName("FILE").create("t"));
051        options.addOption(OptionBuilder.withLongOpt("fallback")
052                          .withDescription("fallback の先頭に追加する")
053                          .hasArg().withArgName("JSON").create("f"));
054        options.addOption("v", "version", false, "バージョン情報を表示して終了する");
055        options.addOption("V", "validate", false, "マップデータを検証する\n問題があればシミュレーションを実行せずに終了する");
056    }
057
058    /**
059     * コマンドラインオプションを解析して指定された処理を実行する
060     * @param args : main メソッドの args 引数
061     */
062    public static void parseCommandLine(String[] args, Options options) {
063        String propertiesFilePath = null;
064        ArrayList<String> fallbackStringList = new ArrayList<String>() ;
065
066        try {
067            CommandLine commandLine = new PosixParser().parse(options, args);
068            // ヘルプ表示オプションもしくはコマンドライン引数エラー
069            if (commandLine.hasOption("help") || commandLine.getArgs().length > 1) {
070                printHelp(options);
071                Itk.quitSafely() ;
072            }
073            // バージョン表示
074            if (commandLine.hasOption("version")) {
075                System.err.println(
076                    "CrowdWalk Version\n" +
077                    "----------------------------------------------------------------\n" +
078                    getVersion() +
079                    "----------------------------------------------------------------");
080                Itk.quitSafely() ;
081            }
082
083            // fallback への追加
084            if (commandLine.hasOption("fallback")) {
085                for(String fallback : commandLine.getOptionValues("fallback")) {
086                    fallbackStringList.add(fallback) ;
087                }
088            }
089
090            // プロパティファイルの指定あり
091            if (commandLine.getArgs().length == 1) {
092                propertiesFilePath = commandLine.getArgs()[0];
093            }
094
095            // ログレベルの指定
096            if (commandLine.hasOption("log-level")) {
097                setLogLevel(commandLine.getOptionValue("log-level"));
098            }
099
100            // "No hint for goal" ログを出力しない
101            CrowdWalkPropertiesHandler.setDisableNoHintForGoalLog(commandLine.hasOption("disable-no-hint-for-goal-log"));
102
103            // マップデータを検証する
104            CrowdWalkPropertiesHandler.setValidation(commandLine.hasOption("validate"));
105
106            if (propertiesFilePath == null) {
107                printHelp(options);
108                Itk.quitByError() ;
109            }
110            CuiSimulationLauncher launcher
111                = launchCuiSimulator(propertiesFilePath, fallbackStringList);
112            // tick 情報の出力
113            if (launcher != null && commandLine.hasOption("tick")) {
114                String tickFilePath = commandLine.getOptionValue("tick");
115                saveTick(tickFilePath, launcher.simulator.currentTime);
116            }
117        } catch (ParseException e) {
118            Itk.logError(e.getMessage()) ;
119            printHelp(options);
120            Itk.quitByError() ;
121        } catch (Exception e) {
122            Itk.quitWithStackTrace(e) ;
123        }
124    }
125
126    /**
127     * ログレベルを設定する
128     */
129    public static void setLogLevel(String logLevelName) throws Exception {
130        for (Map.Entry<Itk.LogLevel, String> entry : Itk.LogTag.entrySet()) {
131            if (entry.getValue().equals(Itk.LogTagPrefix + logLevelName)) {
132                Itk.logLevel = entry.getKey();
133                return;
134            }
135        }
136        throw new Exception("Option error - ログレベル名が間違っています: "
137                            + logLevelName);
138    }
139
140    /**
141     * CUI シミュレータを実行する
142     */
143    public static CuiSimulationLauncher
144        launchCuiSimulator(String propertiesFilePath,
145                           ArrayList<String> commandLineFallbacks)
146    {
147        CuiSimulationLauncher launcher =
148            new CuiSimulationLauncher(propertiesFilePath,
149                                      commandLineFallbacks);
150        if (CrowdWalkPropertiesHandler.validation() && ! launcher.getMap().validate()) {
151            Itk.logError_("Failed to validate the map data.");
152            return null;
153        }
154        launcher.initialize();
155        launcher.start();
156        return launcher;
157    }
158
159    /**
160     * tick 情報をファイルに出力する
161     */
162    public static void saveTick(String filePath, SimTime currentTime) throws IOException {
163        PrintWriter writer = new PrintWriter(filePath);
164        if (filePath.toLowerCase().endsWith(".json")) {
165            writer.write("{\"tick\" : " + currentTime.getRelativeTime() + "}");
166        } else {
167            writer.println(currentTime.getRelativeTime());
168        }
169        writer.close();
170    }
171
172    /**
173     * コマンドラインヘルプを標準出力に表示する
174     */
175    public static void printHelp(Options options) {
176        System.out.println("CrowdWalk: CUI シミュレータ専用版");
177        HelpFormatter formatter = new HelpFormatter();
178        int usageWidth = 7 + commandLineSyntax.length();    // "usege: " + commandLineSyntax
179        formatter.setWidth(Math.max(usageWidth, 80));       // 行の折りたたみ最小幅は80桁
180        formatter.printHelp(commandLineSyntax, options);
181    }
182
183    public static void main(String[] args) {
184        Options options = new Options();
185        defineOptions(options);
186        parseCommandLine(args, options);
187    }
