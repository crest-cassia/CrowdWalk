003
004import java.io.IOException;
005import java.io.ObjectInputStream;

009import nodagumi.ananPJ.NetworkMap.Link.MapLink;
010import nodagumi.ananPJ.NetworkMap.Node.MapNode;
011import nodagumi.Itk.Term;
012
013//======================================================================
014/**
015 * 最適ルーティングの情報。
016 */
017public class NavigationHint implements Comparable<NavigationHint> {
018    //::::::::::::::::::::::::::::::::::::::::::::::::::
019    /** デフォルトの主観モード */
020    static public final Term DefaultMentalMode = null ;
021    
022    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
023    /** 主観モード */
024    public Term mentalMode ;
025    /** 最終目標ゴールタグ */
026    public String goalTag ;
027    /** 最終目標ゴールのノード */
028    public MapNode goalNode;
029    /** 起点ノード */
030    public MapNode fromNode ;
031    /** 向かう方向のノード */
032    public MapNode toNode;
033    /** 辿るリンク */
034    public MapLink viaLink;
035    /** ゴールまでの距離 */
036    public double distance;
037
038    //--------------------------------------------------
039    /**
040     * constructor.
041     */
042    public NavigationHint(Term _mentalMode, String _goalTag, MapNode _goalNode,
043                          MapNode _fromNode, MapLink _viaLink, MapNode _toNode,
044                          double _distance) {
045        set(_mentalMode, _goalTag, _goalNode,
046            _fromNode, _viaLink, _toNode, _distance) ;
047    }
048    
049    //--------------------------------------------------
050    /**
051     * 値設定
052     */
053    public NavigationHint set(Term _mentalMode, String _goalTag,
054                              MapNode _goalNode,
055                              MapNode _fromNode, MapLink _viaLink,
056                              MapNode _toNode, double _distance) {
057        mentalMode = _mentalMode ;
058        goalTag = _goalTag ;
059        goalNode = _goalNode ;
060        fromNode = _fromNode ;
061        toNode = _toNode ;
062        viaLink = _viaLink ;
063        distance = _distance;
064
065        return this ;
066    }
067
068    //--------------------------------------------------
069    /**
070     * compare.
071     */
072    public int compareTo(NavigationHint e) {
073        if (e.distance < distance) return -1;
074        else if (e.distance > distance) return 1;
075        else return 0;
076    }
077}
