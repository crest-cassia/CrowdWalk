029//======================================================================
030/**
031 * 思考を表す式の処理系 (misc. functions)
032 */
033public class NaviFormulaMap extends NaviFormula {
034    //============================================================
035    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
036    /**
037     * 代表インスタンス
038     * (複数あっても無駄なので、単一である方が良い)
039     * registerFormulas で登録。
040     */
041    static ThinkFormula singleton ;
042
043    //============================================================
044    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
045    /**
046     * Lexicon に登録するもの
047     */
048    static private final String Lex_length = Itk.intern(":length") ;
049    static private final String Lex_width = Itk.intern(":width") ;
050    static private final String Lex_hasTag = Itk.intern("hasTag") ;
051    static private final String Lex_tag = Itk.intern("tag") ;
052
053
054    //============================================================
055    //------------------------------------------------------------
056    /**
057     * 登録
058     */
059    static public boolean registerFormulas(Lexicon lexicon)
060    {
061        try {
062            Class currentClass =
063                new Object(){}.getClass().getEnclosingClass() ;
064            singleton =
065                (ThinkFormula)(currentClass.newInstance()) ;
066        } catch(Exception ex) {
067            Itk.logError("wrong class definition") ;
068            Itk.dumpStackTraceOf(ex) ;
069        }
070
071        ThinkFormula.register(Lex_length, singleton, false, lexicon) ;
072        ThinkFormula.register(Lex_width,  singleton, false, lexicon) ;
073        ThinkFormula.register(Lex_hasTag, singleton, false, lexicon) ;
074
075        return true ;
076    }
077
078    //------------------------------------------------------------
079    /**
080     * 呼び出し.
081     * <ul>
082     *   <li>{@link #call_length "length"}</li>
083     *   <li>{@link #call_width "width"}</li>
084     *   <li>{@link #call_hasTag "hasTag"}</li>
085     * </ul>
086     */
087    @Override
088    public Term call(String head, Term expr,
089                     ThinkEngine engine, Object env) {
090        if(head == Lex_length) {
091            return call_length(head, expr, engine, env) ;
092        } else if(head == Lex_width) {
093            return call_width(head, expr, engine, env) ;
094        } else if(head == Lex_hasTag) {
095            return call_hasTag(head, expr, engine, env) ;
096        } else {
097            Itk.logWarn("unknown expression", "expr=", expr) ;
098            return Term_Null ;
099        }
100    }
101
102    //------------------------------------------------------------
103    /**
104     * length。
105     * <pre>
106     *   {"":":length"} | ":length"
107     * </pre>
108     * 対象リンクの長さを返す。
109     */
110    public Term call_length(String head, Term expr,
111                            ThinkEngine engine, Object _env) {
112        NaviFormulaEnv env = (NaviFormulaEnv)_env ;
113        return new Term(getLink(env).getLength()) ;
114    }
115
116    //------------------------------------------------------------
117    /**
118     * width。
119     * <pre>
120     *   {"":":width"} | ":width"
121     * </pre>
122     */
123    public Term call_width(String head, Term expr,
124                           ThinkEngine engine, Object env) {
125        return new Term(getLink(env).getWidth()) ;
126    }
127
128    //------------------------------------------------------------
129    /**
130     * タグチェック。
131     * <pre>
132     *   {"":"hasTag",
133     *    "tag": __TagString__}
134     * </pre>
135     */
136    public Term call_hasTag(String head, Term expr,
137                            ThinkEngine engine, Object env) {
138        boolean result = false ;
139        if(expr.hasArg(Lex_tag)) {
140            Term tag = expr.getArgTerm(Lex_tag) ;
141            result = getLink(env).hasTag(tag) ;
142        } else {
143            Itk.logError("hasTag formula should have 'tag' slot.") ;
144            Itk.logError_("expr=", expr) ;
145            Itk.quitByError() ;
146        }
147        return ThinkFormula.booleanTerm(result) ;
148    }
149
150    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
151    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
152    //------------------------------------------------------------
153    //============================================================
154} // class NaviFormulaMap
