025//======================================================================
026/**
027 * Navigation 用 主観距離計算エンジン.
028 *
029 * ルールは、Agent の ThinkEngine の記法に準拠。
030 * <ul>
031 *   <li>{@link NaviFormula NaviFormula: 式要素 トップ}</li>
032 * </ul>
033 * また、上記にマッチしないアトム（配列でもオブジェクトでもないデータ）は、
034 * その値を直接返す。（リテラル扱い）
035 */
036public class NaviEngine extends ThinkEngine {
037    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
038    /**
039     * log tag
040     */
041    static final public String LogTagPrefix = "Navi:" ;
042    public String logTag() {
043        return LogTagPrefix ;
044    }
045
046    //------------------------------------------------------------
047    /**
048     * コンストラクタ
049     */
050    public NaviEngine(){
051        setLexicon(NaviFormula.lexicon) ;
052    }
053
054    //------------------------------------------------------------
055    /**
056     * 推論(top)
057     */
058    public double calc(MapLink _link, MapNode _fromNode, Term _rule) {
059        NaviFormula.NaviFormulaEnv env =
060            new NaviFormula.NaviFormulaEnv(_link, _fromNode) ;
061        Term result = think(_rule, env) ;
062        return result.getDouble() ;
063    }
064
065    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
066    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
067    //------------------------------------------------------------
068    //============================================================
069} // class naviEngine
