025//======================================================================
026/**
027 * 主観によるMapの距離計算を表す式の処理系。
028 * 組み込まれている Formula リスト
029 * <ul>
030 *  <li>{@link nodagumi.ananPJ.Agents.Think.ThinkFormulaLogical#call Logical: 論理形式および実行制御}</li>
031 *  <li>{@link nodagumi.ananPJ.Agents.Think.ThinkFormulaArithmetic#call Arithmetic: 数値処理など}</li>
032 *  <li>{@link NaviFormulaMap#call Map: マップパラメータ関係}</li>
033 *  <li>{@link nodagumi.ananPJ.Agents.Think.ThinkFormulaMisc#call Misc: その他}</li>
034 * </ul>
035 *
036 * <h3>新しい Formula の作成方法 </h3>
037 * <a name="addToExistingThinkFormula"></a>
038 * <a href="../../../ananPJ/Agents/Think/ThinkFormula.html#addToExistingThinkFormula">go to ThinkFormula#addToExistingThinkFormula</a>
039 */
040abstract public class NaviFormula extends ThinkFormula {
041    //============================================================
042    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
043    /**
044     * Formula Lexicon
045     */
046    static public Lexicon lexicon = new Lexicon() ;
047    static {
048        ThinkFormulaMisc.registerFormulas(NaviFormula.lexicon) ;
049        ThinkFormulaLogical.registerFormulas(NaviFormula.lexicon) ;
050        ThinkFormulaArithmetic.registerFormulas(NaviFormula.lexicon) ;
051        NaviFormulaMap.registerFormulas(NaviFormula.lexicon) ;
052    }
053
054    //============================================================
055    //------------------------------------------------------------
056    /**
057     * Formula を検索
058     */
059    static public ThinkFormula findFormula(String head) {
060        return (ThinkFormula)(lexicon.lookUp(head)) ;
061    }
062
063    //============================================================
064    //------------------------------------------------------------
065    /**
066     * Formula を検索
067     */
068    static public ThinkFormula findFormula(Term head) {
069        return (ThinkFormula)(lexicon.lookUp(head.getString())) ;
070    }
071
072    //============================================================
073    /** 
074     * NaviFormula の計算のための環境情報. 
075     */
076    public static class NaviFormulaEnv {
077        public MapLink link ;
078        public MapNode fromNode ;
079        public NaviFormulaEnv(MapLink _link, MapNode _fromNode) {
080            link = _link ;
081            fromNode = _fromNode ;
082        }
083    }
084    
085    //------------------------------------------------------------
086    /**
087     * コンストラクタ
088     */
089    public NaviFormula(){}
090
091    //------------------------------------------------------------
092    /**
093     * engine から link の取り出し。
094     */
095    public MapLink getLink(Object _env) {
096        NaviFormulaEnv env = (NaviFormulaEnv)_env ;
097        return env.link ;
098    }
099
100    //------------------------------------------------------------
101    /**
102     * engine から fromNode の取り出し。
103     */
104    public MapNode getFromNode(Object _env) {
105        NaviFormulaEnv env = (NaviFormulaEnv)_env ;
106        return env.fromNode ;
107    }
108
109    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
110    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
111    //------------------------------------------------------------
112    //============================================================
113} // class ThinkFormula
