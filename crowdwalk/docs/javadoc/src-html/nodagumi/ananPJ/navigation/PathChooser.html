020/**
021 * 探索において、各リンクおよびゴールノードのコストを計算する。
022 * 変更ルールの記述は、property.json の中で以下のように行う。
023 * <pre>
024 *   { ...
025 *     "mental_map_rules" : { "modeName1" : __Rule__,
026 *                            "modeName2" : __Rule__,
027 *                            ... },
028 *   ...}
029 * </pre>
030 *   __Rule__ は、Agent の ThinkEngine の {@link nodagumi.ananPJ.Agents.Think.ThinkFormula ThinkFormula}に準拠。
031 */
032public class PathChooser {
033    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
034    /** 主観モードのタグ。デフォルトは null */
035    Term mentalMode ;
036    /** ルールオブジェクト (これから拡張するため)*/
037    Term modifyRule ;
038
039    /** ルールオブジェクト (これから拡張するため)*/
040    NaviEngine engine ; 
041    
042    //----------------------------------------------------------------------
043    /**
044     * constractor.
045     */
046    public PathChooser() {
047        mentalMode = null ;
048        modifyRule = null ;
049        engine = null ;
050    }
051
052    /**
053     * constractor.
054     */
055    public PathChooser(String _mentalMode, Term _modifyRule) {
056        this(new Term(_mentalMode, false), _modifyRule) ;
057    }
058    
059    /**
060     * constractor.
061     */
062    public PathChooser(Term _mentalMode, Term _modifyRule) {
063        mentalMode = _mentalMode ;
064        modifyRule = _modifyRule ;
065        engine = new NaviEngine() ;
066    }
067    
068    //----------------------------------------------------------------------
069    /**
070     * リンクコストの計算。
071     */
072    public double calcLinkCost(MapLink link, MapNode fromNode) {
073        if(!link.isAvailableFrom(fromNode)) {
074            return Double.POSITIVE_INFINITY ;
075        } else if(mentalMode == null) {
076            return 1.0 * link.getLength() ;
077        } else {
078            synchronized(link){
079                if(link.hasMentalLength(mentalMode, fromNode)) {
080                    return link.getMentalLength(mentalMode, fromNode) ;
081                } else {
082                    double cost = engine.calc(link, fromNode, modifyRule) ;
083                    link.setMentalLength(mentalMode, fromNode, cost) ;
084                    return cost ;
085                }
086            }
087        }
088    }
089
090    //----------------------------------------------------------------------
091    /**
092     * リンクコストの計算。
093     */
094    public double calcGoalNodeCost(MapNode node) {
095        return 0.0 ;
096    }
097}
