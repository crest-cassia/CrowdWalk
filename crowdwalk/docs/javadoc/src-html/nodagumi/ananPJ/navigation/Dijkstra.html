022//======================================================================
023/**
024 * Dijkstra法で最短経路探索。
025 */
026public class Dijkstra {
027    //============================================================
028    /**
029     * 探索結果の格納用クラス。
030     */
031    static public class Result
032        extends LinkedHashMap<MapNode, NavigationHint> {}
033
034    //============================================================
035    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
036    /**
037     * 標準の PathChooser。
038     * これ以外の Chooser を使う機会があるのか、不明。
039     */
040    public static PathChooser DefaultPathChooser =
041        new PathChooser();
042
043    //============================================================
044    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
045    /**
046     * Chooser テーブル。
047     */
048    static public HashMap<String, PathChooser> pathChooserTable =
049        new HashMap<String, PathChooser>() ;
050
051    //============================================================
052    //------------------------------------------------------------
053    /**
054     * 主観的距離計算機を取得。
055     */
056    static public PathChooser getPathChooser(Term mentalMode) {
057        return getPathChooser(Term.convertValueToString(mentalMode)) ;
058    }
059    
060    /** */
061    static public PathChooser getPathChooser(String mentalMode) {
062        if(mentalMode == null) {
063            return DefaultPathChooser ;
064        } else {
065            return pathChooserTable.get(mentalMode) ;
066        }
067    }
068
069    /**
070     * 主観的距離計算機を、安全に取得。thread safe. (Term 引数)
071     */
072    static public PathChooser getOrNewPathChooserSafe(Term mentalMode,
073                                                      NetworkMap networkMap) {
074        return getOrNewPathChooserSafe(Term.convertValueToString(mentalMode),
075                                       networkMap) ;
076    }
077    /**
078     * 主観的距離計算機を、安全に取得。thread safe. (String 引数)
079     */
080    static public PathChooser getOrNewPathChooserSafe(String mentalMode,
081                                                      NetworkMap networkMap) {
082        synchronized(pathChooserTable){
083            PathChooser chooser = Dijkstra.DefaultPathChooser ;
084            if(mentalMode != null) {
085                chooser = getPathChooser(mentalMode) ;
086                if(chooser == null) {
087                    chooser = newPathChooser(mentalMode, networkMap) ;
088                }
089            }
090            return chooser ;
091        }
092    }
093        
094    /**
095     * 主観的距離計算機を新規登録。
096     */
097    static public PathChooser newPathChooser(Term mentalMode,
098                                             NetworkMap networkMap) {
099        PathChooser chooser =
100            new PathChooser(mentalMode,
101                            networkMap.getMentalMapRule(mentalMode)) ;
102        pathChooserTable.put(mentalMode.getString(), chooser) ;
103
104        return chooser ;
105    }
106    /** */
107    static public PathChooser newPathChooser(String mentalMode,
108                                             NetworkMap networkMap) {
109        return newPathChooser((mentalMode == null ?
110                               (Term)null :
111                               new Term(mentalMode, false)),
112                              networkMap) ;
113    }
114
115    //============================================================
116    //------------------------------------------------------------
117    /**
118     * 探索メインルーチン。
119     * 上記の calc は、無駄が多いと思われる。何度も同じノードを展開している。
120     * 下記はそれを避ける。
121     * @param mentalMode 主観的モード
122     * @param goalTag 探索するゴールのタグ。
123     * @param subgoals 目標とするゴール集合。
124     * @param networkMap 地図。
125     * @return 探索結果。Result class のインスタンス。
126     */
127    static public Result calc(Term mentalMode,
128                              String goalTag,
129                              MapNodeTable subgoals,
130                              NetworkMap networkMap) {
131        //String timerStr =
132        //    ("calc(" + (Term.convertValueToString(mentalMode)) +
133        //     "," +  goalTag + ")") ;
134        //Itk.timerStart(timerStr) ;
135
136        PathChooser chooser = getOrNewPathChooserSafe(mentalMode,
137                                                      networkMap) ;
138            
139        Result frontier = new Result();
140        ArrayList<MapNode> closedList = new ArrayList<MapNode>() ;
141        Result result = new Result() ;
142
143        // ゴールノードは、initial cost で。
144        for (MapNode subgoal : subgoals) {
145            double cost = chooser.calcGoalNodeCost(subgoal);
146            NavigationHint hint =
147                new NavigationHint(mentalMode,
148                                   goalTag, subgoal,
149                                   subgoal, null, null, cost) ;
150            frontier.put(subgoal, hint) ;
151            result.put(subgoal, hint) ;
152        }
153
154        // 探索ループ。
155        while (true) {
156            NavigationHint bestHint =
157                new NavigationHint(null, null, null,
158                                   null, null, null, Double.POSITIVE_INFINITY) ;
159            closedList.clear() ;
160            for (MapNode frontierNode : frontier.keySet()) {
161                int countPerNode = 0 ;
162                for (MapLink preLink :
163                         frontierNode.getValidReverseLinkTable()) {
164                    MapNode preNode = preLink.getOther(frontierNode);
165                    if (result.containsKey(preNode)) continue;
166                    countPerNode ++ ;
167                    NavigationHint frontierHint = result.get(frontierNode) ;
168                    double dist = (frontierHint.distance
169                                   + chooser.calcLinkCost(preLink, preNode)) ;
170
171                    if (dist < bestHint.distance) {
172                        bestHint.set(mentalMode, goalTag,
173                                     frontierHint.goalNode,
174                                     preNode, preLink, frontierNode, dist) ;
175                    }
176                }
177                // すべてのリンク先がすでに探査済みであれば、そのノードは close
178                if(countPerNode == 0) {
179                    closedList.add(frontierNode) ;
180                }
181            }
182            // close されたノードを frontiner から除去。
183            for(MapNode node : closedList) {
184                frontier.remove(node) ;
185            }
186            // 新たに付け加えるものがなければ、探査終わり。
187            if (bestHint.fromNode == null) {
188                break;
189            }
190            // 新しノードを frontier に追加。
191            frontier.put(bestHint.fromNode, bestHint) ;
192            result.put(bestHint.fromNode, bestHint) ;
193        }
194
195        //Itk.logWarn("Dijkstra:", timerStr) ;
196        //Itk.timerShowLap(timerStr) ;
197
198        return result ;
199    }
200
201}
