014package nodagumi.ananPJ.NetworkMap.Gate;
015
016import nodagumi.ananPJ.NetworkMap.OBNode;
017import nodagumi.ananPJ.Agents.AgentBase;
018import nodagumi.ananPJ.Scenario.GateEvent;
019import nodagumi.ananPJ.misc.SimTime;
020
021import nodagumi.Itk.*;
022

024//======================================================================
025/**
026 * Ruby によりエージェント毎の通過制御できる Gate。
027 * {@link nodagumi.ananPJ.Scenario.Scenario} の
028 * {@link nodagumi.ananPJ.Scenario.CloseGateEvent} の記述において、
029 * 以下のように指定する。
030 * <pre>
031 *  { "type":"CloseGate",
032 *    "gateClass":"RubyGate",
033 *    "rubyClass":__GateClassInRuby__,
034 *    "atTime":__Time__,
035 *    "placeTag":__Tag__,
036 *    "gateTag":__Tag__,
037 *    __otherItem__:__Value__,
038 *    ...}
039 *  
040 *  __Time__ ::= "hh:mm:ss"
041 *  __GateClassInRuby__ : Ruby側の対応するクラス。
042 * </pre>
043 * <p>
044 * __GateClassInRuby__ は、RubyGateBase クラスの子クラスとして
045 * 読み込まれる Ruby プログラムの中で定義されている必要がある。
046 * エージェントがこのゲートを通過する際には毎サイクル、
047 * {@code isClosed(agent, currentTime)} メソッドが呼び出される。
048 * このメソッドの返り値が {@code true} の場合は、エージェントは通過できず、
049 * 足止めとなる。
050 * 返り値が{@code false} の場合は通過できると判断される。
051 * このメソッドの {@code super} は、
052 * 通常の GateBase の {@code isClosed()}、つまり、
053 * Gate が closed 状態なら {@code true}、open 状態なら {@code false} を返す。
054 * __GateClassInRuby__ で {@code isClosed()} が定義されない場合、この
055 * {@code super} がそのまま呼び出される。
056 * </p><p>
057 * {@code __otherItem__} を含め、上記のシナリオで指定したイベント定義情報は、
058 * __GateClassInRuby__ から {@code getEventDef()} で参照できる。
059 * {@code getEventDef()} は、イベント定義情報を {@link nodagumi.Itk.Term} の形式で
060 * 返す。
061 * </p><p>
062 * 一度作られた gate は、__PlaceTag__ 毎に __GateTag__ で管理されるため、
063 * 同じ __GateTag__ に対する {@link nodagumi.ananPJ.Scenario.GateEvent} は、
064 * 新たにインスタンスを生成せず、もとのインスタンスが利用される。
065 * よって、{@code getEventDef()} で取得できる情報は、初出の 
066 * {@link nodagumi.ananPJ.Scenario.CloseGateEvent} に指定された情報となる。
067 */
068public class RubyGate extends GateBase {
069    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
070    /**
071     * Ruby Engine
072     */
073    public ItkRuby rubyEngine = null ;
074    
075    /**
076     * Ruby Object
077     */
078    public Object rubyGate = null ;
079    
080    /**
081     * ルビークラス名
082     */
083    public String rubyClass ;
084
085    //------------------------------------------------------------
086    /**
087     * コンストラクタ。
088     * @param _tag:: gate tag.
089     * @param _eventDef:: event definition in Term.
090     * @param _closed:: initial state of closed value.
091     */
092    public RubyGate(String _tag, GateEvent _event, boolean _closed,
093                    OBNode _place) {
094        super(_tag, _event, _closed, _place) ;
095        
096        rubyClass = event.eventDef.getArgString("rubyClass") ;
097        if(rubyClass == null) {
098            Itk.logError("ruby class is not specified in RubyGate.",
099                         _tag, _event, _closed) ;
100            Itk.quitByError() ;
101        }
102        Itk.logDebug("RubyGate:rubyClass", rubyClass) ;
103
104        setupRubyEngine(event.getSimulator().getRubyEngine()) ;
105    }
106
107    //------------------------------------------------------------
108    /**
109     * Ruby Engine の設定
110     */
111    public void setupRubyEngine(ItkRuby _rubyEngine) {
112        if(_rubyEngine == null) {
113            Itk.logError("RubyGate require Ruby Engine") ;
114            Itk.quitByError() ;
115        }
116        
117        rubyEngine = _rubyEngine ;
118
119        rubyGate =
120            rubyEngine.newInstanceOfClass(rubyClass, this) ;
121    }
122
123    //------------------------------------------------------------
124    /**
125     * 閉じているかどうか？
126     * @param currnetTime : シミュレーション時刻
127     * @param agent: 対象となるエージェント
128     * @return このゲートが閉じているかどうか
129     */
130    public boolean isClosed(AgentBase agent, SimTime currentTime) {
131        Object ret =
132            rubyEngine.callMethod(rubyGate, "isClosed",
133                                  agent, currentTime) ;
134        return (boolean)ret ;
135    }
136
137    //------------------------------------------------------------
138    /**
139     * ゲートの開閉
140     */
141    public GateBase switchGate(GateEvent _event, boolean _closed) {
142        rubyEngine.callMethod(rubyGate, "switchGate",
143                              _event, _closed) ;
144        return this ;
145    }
146
147    //----------------------------------------
148    /**
149     * ゲートの開閉の元定義。（Ruby CallBack 用）
150     */
151    public GateBase super_switchGate(GateEvent _event, boolean _closed) {
152        super.switchGate(_event, _closed) ;
153        return this ;
154    }
155    
156    //============================================================
157    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
158    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
159    //------------------------------------------------------------
160
161} // class RubyGate
