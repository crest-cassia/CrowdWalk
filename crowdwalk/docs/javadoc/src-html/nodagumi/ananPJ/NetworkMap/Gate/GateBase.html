013package nodagumi.ananPJ.NetworkMap.Gate;
014
015import nodagumi.ananPJ.NetworkMap.OBNode;
016import nodagumi.ananPJ.Agents.AgentBase;
017import nodagumi.ananPJ.Scenario.GateEvent;
018import nodagumi.ananPJ.misc.SimTime;
019
020
021import nodagumi.Itk.*;
022
023
024//======================================================================
025/**
026 * 通行規制制御用クラス
027 */
028public class GateBase {
029    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
030    /**
031     * OBNode内でこのゲートを参照するためのタグ
032     */
033    public String tag;
034
035    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
036    /**
037     * もととなる初出Event 
038     */
039    public GateEvent event ;
040
041    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
042    /**
043     * Gate が設置された場所
044     */
045    public OBNode place ;
046
047    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
048    /**
049     * 現在閉じている（通行止め）かどうか？
050     */
051    public boolean closed ;
052
053    //----------------------------------------
054    /**
055     * コンストラクタ
056     */
057    public GateBase(String _tag, GateEvent _event, boolean _closed,
058                    OBNode _place) {
059        tag = _tag ;
060        event = _event ;
061        closed = _closed ;
062        place = _place ;
063    }
064
065    //----------------------------------------
066    /**
067     * 閉じているかどうか？
068     * 拡張のために、時刻とエージェントを受け取る。
069     * @param currnetTime : シミュレーション時刻
070     * @param agent: 対象となるエージェント
071     * @return デフォルトでは、単にこのゲートが閉じているかどうか
072     */
073    public boolean isClosed(AgentBase agent, SimTime currentTime) {
074        return isClosed() ;
075    }
076
077    //----------------------------------------
078    /**
079     * 閉じているかどうか？
080     * 拡張のために、時刻とエージェントを受け取る。
081     * @param currentTime : シミュレーション時刻
082     * @param agent: 対象となるエージェント
083     * @return デフォルトでは、単にこのゲートが閉じているかどうか
084     */
085    public boolean isOpened(AgentBase agent, SimTime currentTime) {
086        return !isClosed(agent, currentTime) ;
087    }
088
089    //----------------------------------------
090    /**
091     * 閉じているかどうか？
092     */
093    public boolean isClosed() {
094        return closed ;
095    }
096
097    //----------------------------------------
098    /**
099     * 開いているかどうか？
100     */
101    public boolean isOpened() {
102        return !isClosed() ;
103    }
104
105    //----------------------------------------
106    /**
107     * ゲートの開閉
108     */
109    public GateBase switchGate(GateEvent _event, boolean _closed) {
110        closed = _closed ;
111        return this ;
112    }
113
114    //----------------------------------------
115    /**
116     * ゲートを閉じる
117     */
118    public GateBase close() {
119        return switchGate(null, true) ;
120    }
121
122    //----------------------------------------
123    /**
124     * ゲートを開ける
125     */
126    public GateBase open() {
127        return switchGate(null, false) ;
128    }
129
130    //------------------------------------------------------------
131    // アクセスメソッド。（主としてrubyより)
132    //----------------------------------------
133    /**
134     * closed の値取得。
135     */
136    public boolean getClosed() {
137        return closed ;
138    }
139
140    //----------------------------------------
141    /**
142     * tag の値取得。
143     */
144    public String getTag() {
145        return tag ;
146    }
147
148    //----------------------------------------
149    /**
150     * イベント定義の取得。
151     */
152    public GateEvent getEvent() {
153        return event ;
154    } ;
155    
156    //----------------------------------------
157    /**
158     * 場所の取得。
159     */
160    public OBNode getPlace() {
161        return place ;
162    } ;
163    //============================================================
164    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
165    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
166    //------------------------------------------------------------
167
168} // class GateBase
