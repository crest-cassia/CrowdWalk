013/* Parent of agent/node/link (leaf of OB) */
014public abstract class OBMapPart extends OBNode {
015
016        /**
017         * 引数なしconstractor。 ClassFinder.newByName で必要。
018         */
019        public OBMapPart() {} ;
020
021    public OBMapPart(String _ID){
022                init(_ID) ;
023        }
024
025        /**
026         * 初期化。constractorから分離。
027         */
028    @Override
029    public void init(String _ID){
030        super.init(_ID);
031        this.setAllowsChildren(false);
032    }
033
034    @Override
035    public boolean isLeaf() {
036        return true;
037    }
038
039    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
040    /** 
041     * ルールで用いるタグ類
042     */
043    static private final String Slot_tag            = Itk.intern("tag") ;
044    static private final String Slot_type           = Itk.intern("type") ;
045    static private final String Slot_directed   = Itk.intern("directed") ;
046    static private final String Slot_factor     = Itk.intern("factor") ;
047    static private final String Slot_set                = Itk.intern("set") ;
048    static private final String Slot_multiply   = Itk.intern("multiply") ;
049    static private final String Slot_add                = Itk.intern("add") ;
050    
051    final public int DefaultOne = 1 ;
052    //------------------------------------------------------------
053    /**
054     * ルールに従い速度等のvalueに制限を加える。
055         * ルールの書き方は、以下の通り。
056         * <pre>
057         *   _rule_ ::= { "type" : _Type_,
058         *                "tag" : _String_,
059         *                "factor" : _Double_ } 
060     *            | { "type" : _Type_,
061     *                "directed" : true,
062         *                "tag" : _String_,
063         *                "factor" : [_Double_, _Double_] } 
064         *   _Type_ ::= "set" | "multiply" | "add"
065         * </pre>
066         * "tag" には、link, node が所有すべきタグを指定する。
067         * このタグを持つlink,node 上のエージェントが受ける影響を求める。
068         * "factor" は、"type" により意味が異なる。
069         * <ul>
070         *   <li> "set" : value を "factor" で置き換え。 </li>
071         *   <li> "multiply" : value に "factor" を乗ずる。 </li>
072         *   <li> "add" : value に "factor" で加える。 </li>
073         * </ul>
074     * 2つ目の形式では、リンクの方向性で扱いが異なる。
075     * agent の現在地(currentPlace)が、forwardDirection なら、
076     * "factor" の第0要素を、
077     * backwardDirection なら、"factor" の第1要素を、計算に用いる。
078     * @param value: 元になる値
079     * @param rule: ルール
080     * @param agent: 対象となるエージェント。リンク上にいる。
081     * @param currnetTime: 現在時刻
082     * @return 変更された値
083     */
084    public double applyRestrictionRule(double value, Term rule,
085                                       AgentBase agent, SimTime currnetTime) {
086        String tag = rule.getArgString(Slot_tag) ;
087        if(hasTag(tag)) {
088            // factor の値を、directed などから決める。
089            boolean directed = rule.getArgBoolean(Slot_directed) ;
090            double factor ;
091            if(directed) {
092                boolean isForward = agent.isForwardDirection() ;
093                Term factorList = rule.getArgTerm(Slot_factor) ;
094                factor = factorList.getNthDouble((isForward ? 0 : 1)) ;
095            } else {
096                factor = rule.getArgDouble(Slot_factor) ;
097            }
098            // type ごとの処理
099            String type = rule.getArgString(Slot_type) ;
100            if(type == Slot_set) {
101                value = factor ;
102            } else if(type == Slot_multiply) {
103                value *= factor ;
104            } else if(type == Slot_add) {
105                value += factor ;
106            } else {
107                Itk.logError("unknown restriction rule type",
108                             "rule=", rule) ;
109            }
110        }
111        return value ;
112    }
113

116
117}
