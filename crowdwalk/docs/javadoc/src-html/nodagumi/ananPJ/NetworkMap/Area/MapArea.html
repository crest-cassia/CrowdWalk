002package nodagumi.ananPJ.NetworkMap.Area;
003
004import java.awt.Shape;
005import java.awt.geom.Line2D;
006import java.awt.geom.Point2D;
007import java.util.ArrayList;
008
009import math.geom3d.Point3D;
010
011import nodagumi.ananPJ.NetworkMap.OBNode;
012import nodagumi.ananPJ.Simulator.PollutionHandler.*;
013
014//======================================================================
015/**
016 * 地図上のエリア。
017 */
018public abstract class MapArea extends OBNode {
019
020    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
021    /**
022     * 現在の Pollution 情報
023     */
024    public PollutionLevelInfo pollutionLevel = null ;
025
026    //------------------------------------------------------------
027    /**
028     * コンストラクタ
029     */
030    public MapArea(String _id) {
031        super(_id);
032    }
033
034    //------------------------------------------------------------
035    /**
036     * Pollution Level を格納。
037     */
038    public void pollutionIsUpdated() {
039        if(pollutionLevel.isChanged(true)) {
040            map.getNotifier().pollutionLevelChanged(this);
041        }
042    }
043
044    //------------------------------------------------------------
045    /**
046     * Pollution Level を取得。
047     */
048    public PollutionLevelInfo getPollutionLevel() {
049        return pollutionLevel ;
050    }
051
052    //------------------------------------------------------------
053    /**
054     * Pollution されているかどうか。
055     */
056    public boolean isPolluted() {
057        return pollutionLevel.isPolluted() ;
058    }
059
060    //------------------------------------------------------------
061    /**
062     * 包含判定。
063     */
064    public abstract boolean contains(Point2D point);
065
066    //------------------------------------------------------------
067    /**
068     * 包含判定。
069     */
070    public abstract boolean contains(Point3D point);
071
072    //------------------------------------------------------------
073    /**
074     * 交差判定。
075     */
076    public abstract boolean intersectsLine(Line2D line);
077
078    //------------------------------------------------------------
079    /**
080     * 形状を取得。
081     */
082    public abstract Shape getShape();
083
084    //------------------------------------------------------------
085    /**
086     * 高さの範囲の最小値を取得。
087     */
088    public abstract double getMinHeight();
089
090    //------------------------------------------------------------
091    /**
092     * 高さの範囲の最大値を取得。
093     */
094    public abstract double getMaxHeight();
095
096    //------------------------------------------------------------
097    /**
098     * 頂点リスト。
099     */
100    public abstract ArrayList<Point2D> getAllVertices();        // tkokada
101
102    //------------------------------------------------------------
103    /**
104     * 方向。
105     */
106    public abstract double getAngle();  // tkokada
107
108    //------------------------------------------------------------
109    /**
110     * to/from DOM codes
111     */
112    static public String getNodeTypeString() {
113        return "Area";
114    }
115
116    //------------------------------------------------------------
117    /**
118     * タイプ。
119     */
120    @Override
121    public NType getNodeType() {
122        return NType.AREA;
123    }
124
125    //------------------------------------------------------------
126    /**
127     * 末端判定。
128     */
129    @Override
130    public boolean isLeaf() {
131        return true;
132    }
133
134}
