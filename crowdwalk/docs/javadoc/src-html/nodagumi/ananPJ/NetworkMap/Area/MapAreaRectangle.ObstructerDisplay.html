080     * コンストラクタ
081     */
082    public MapAreaRectangle(String _id,
083            Rectangle2D _bounds,
084            double _min_height,
085            double _max_height,
086            double _angle) {
087        this(_id);
088        bounds = _bounds;
089        minHeight = _min_height;
090        maxHeight = _max_height;
091        angle = _angle * Math.PI / 180; // tkokada
092    }

094    //------------------------------------------------------------
095    /**
096     * 包含判定
097     */
098    @Override
099    public boolean contains(Point3D point) {
100        if (point.getZ() < minHeight || point.getZ() > maxHeight) {
101            //Itk.logWarn("height not match");
102            return false;
103        }
104        if (!bounds.contains(rotatePoint(
105                new Point2D.Double(bounds.getCenterX(), bounds.getCenterY()),
106                new Point2D.Double(point.getX(), point.getY()),
107                -angle))) {
108            return false;
109        }
110        return true;
111    }
112
113    //------------------------------------------------------------
114    /**
115     * 包含判定
116     */
117    @Override
118    public boolean contains(Point2D point) {
119        // tkokada
120        return bounds.contains(rotatePoint(new Point2D.Double(bounds.getCenterX(), bounds.getCenterY()), point, -angle));
121    }

160    /**
161     * 頂点リスト。
162     */
163    @Override // tkokada
164    public ArrayList<Point2D> getAllVertices() {
165        ArrayList<Point2D> allVertices = new ArrayList<Point2D>();
166        Point2D center = new Point2D.Double(bounds.getCenterX(), bounds.getCenterY());
167        allVertices.add(rotatePoint(center, new Point2D.Double(bounds.getMinX(), bounds.getMinY()), angle));
168        allVertices.add(rotatePoint(center, new Point2D.Double(bounds.getMaxX(), bounds.getMinY()), angle));
169        allVertices.add(rotatePoint(center, new Point2D.Double(bounds.getMinX(), bounds.getMaxY()), angle));
170        allVertices.add(rotatePoint(center, new Point2D.Double(bounds.getMaxX(), bounds.getMaxY()), angle));
171        
172        return allVertices;
173    }

184    //------------------------------------------------------------
185    /**
186     * DOM からの取得。
187     */
188    @Override
189    protected void getAttributesFromDom(Element element) {
190        super.getAttributesFromDom(element);
191
192        ID = element.getAttribute("id");
193
194        double x1 = Double.parseDouble(element.getAttribute("pWestX"));
195        double x2 = Double.parseDouble(element.getAttribute("pEastX"));
196        double y1 = Double.parseDouble(element.getAttribute("pNorthY"));
197        double y2 = Double.parseDouble(element.getAttribute("pSouthY"));
198        if (y1 > y2) {
199            if (! nsWarned) {
200                Itk.logWarn("MapArea coordinate error",
201                            String.format("pNorthY = %s > pSouthY = %s", y1, y2));
202                nsWarned = true;
203                // Itk.quitByError() ;
204            }
205            double y = y1;
206            y1 = y2;
207            y2 = y;
208        }
209        if (x1 > x2) {
210            if (! weWarned) {
211                Itk.logWarn("MapArea coordinate error",
212                            String.format("pWestX = %s > pEastX = %s", x1, x2));
213                weWarned = true;
214                // Itk.quitByError() ;
215            }
216            double x = x1;
217            x1 = x2;
218            x2 = x;
219        }
220        bounds = new Rectangle2D.Double(x1, y1, x2 - x1, y2 - y1);
221
222        minHeight = Double.parseDouble(element.getAttribute("minHeight"));
223        maxHeight = Double.parseDouble(element.getAttribute("maxHeight"));
224        try {
225            angle = Double.parseDouble(element.getAttribute("angle"));
226        } catch (Exception e) {
227            angle = 0;
228        }
229    }
230

235    @Override
236    public Element toDom(Document dom, String tagname) {
237        Element element = super.toDom(dom, getNodeTypeString());
238
239        element.setAttribute("id", ID);
240        element.setAttribute("pWestX", "" + bounds.getMinX());
241        element.setAttribute("pEastX", "" + bounds.getMaxX());
242        element.setAttribute("pNorthY", "" + bounds.getMinY());
243        element.setAttribute("pSouthY", "" + bounds.getMaxY());
244        element.setAttribute("minHeight", "" + minHeight);
245        element.setAttribute("maxHeight", "" + maxHeight);
246        element.setAttribute("angle", "" + angle);
247
248        return element;
249    }

251    //------------------------------------------------------------
252    /**
253     * 回転。
254     */
255    // tkokada added
256    private Point2D rotatePoint(Point2D origin, Point2D point, double angle) {
257        return new Point2D.Double(
258                origin.getX() + (point.getX() - origin.getX()) * Math.cos(angle) - (point.getY() - origin.getY()) * Math.sin(angle),
259                origin.getY() + (point.getX() - origin.getX()) * Math.sin(angle) + (point.getY() - origin.getY()) * Math.cos(angle));
260    }

267        MapAreaRectangle area = new MapAreaRectangle(null);
268        area.getAttributesFromDom(element);
269        return area;
270    }

281    /**
282     * このオブジェクトの状態をテキストで取得する
283     */
284    public String getStatusText() {
285        Rectangle2D bounds = (Rectangle2D)this.getShape();
286        PollutionLevelInfo pollutionLevel = this.getPollutionLevel();
287        StringBuilder buff = new StringBuilder();
288        buff.append("Area ID: ").append(this.ID).append("\n");
289        buff.append("pWestX: ").append(bounds.getMinX()).append("\n");
290        buff.append("pEastX: ").append(bounds.getMaxX()).append("\n");
291        buff.append("pNorthY: ").append(bounds.getMinY()).append("\n");
292        buff.append("pSouthY: ").append(bounds.getMaxY()).append("\n");
293        buff.append("current level: ").append(pollutionLevel.getCurrentLevel()).append("\n");
294        buff.append("normalized level: ").append(pollutionLevel.getNormalizedLevel()).append("\n");
295        buff.append("tags: ").append(this.getTagString()).append("\n");
296        return buff.toString();
297    }
