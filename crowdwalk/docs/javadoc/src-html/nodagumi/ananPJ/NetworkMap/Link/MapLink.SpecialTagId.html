331    /**
332     * 主観的距離の取得。
333     * 指定された tag の距離が定義されていない時は、
334     * UndefinedMentalLength を返す。
335     * @param mentalMode 主観距離の指定タグ。
336     *        mentalMode が null (NavigationHint.DefaultMentalMode)
337     *        なら、もとの length。
338     * @param fromNode リンクに侵入するノード。
339     *        距離が非対称の時に使う。（[2016-01-30 I.Noda]未実装）
340     * @return 距離が定義されていればその値。定義されていなければ、
341     */
342    public double getMentalLength(Term mentalMode, MapNode fromNode) {
343        if(mentalMode == NavigationHint.DefaultMentalMode) {
344            return length ;
345        } else if(hasMentalLength(mentalMode, fromNode)) {
346            return mentalLengthTable.get(mentalMode.getString()) ;
347        } else {
348            Itk.logWarn("undefined mental length",
349                        "[link ID=", this.ID, "] ",
350                        "calc on-the-fly.") ;
351            PathChooser chooser = Dijkstra.getPathChooser(mentalMode) ;
352            if(chooser == null) {
353                Itk.logError("unknown mentalMode. ",
354                             "mode=", mentalMode) ;
355                Itk.quitByError() ;
356            } 
357            return chooser.calcLinkCost(this, fromNode) ;
358        }
359    }

468    /**
469     * リンク上の agents のソートと各レーンへの振り分け。
470     */
471    private void setup_lanes() {
472        if (! agents.isEmpty()) {
473            Collections.sort(agents) ;
474        }
475        forwardLane.clear() ;
476        backwardLane.clear() ;
477        for (AgentBase agent : agents) {
478            if (agent.isForwardDirection()) {
479                forwardLane.add(agent);
480            } else {
481                backwardLane.add(agent);
482            }
483        }
484        /* [2015.01.03 I.Noda]
485         * 以下の sort は、agents がすでに sort されているので無駄なはず。
486         * なので排除。
487         */
488        //Collections.sort(forwardLane);
489        //Collections.sort(backwardLane);
490    }

492    private void add_agent_to_lane(AgentBase agent) {
493        if (agent.isForwardDirection()) {
494            forwardLane.add(agent);
495            Collections.sort(forwardLane);    
496        } else {
497            backwardLane.add(agent);
498            Collections.sort(backwardLane);
499        }
500    }

512    static public double dWidth = 1.0 ;
513
514    //------------------------------------------------------------
515    /**
516     * リンクのレーンの幅。
517     * レーンの幅は、forward/backward のレーンに存在するエージェント数に
518     * 比例して、元の width から割り振られる。
519     * 1 以下にはしない。
520     */
521    public int getLaneWidth(Direction dir) {
522        int d = getLane(dir).size() ;
523        int lane_width = (int)(d * (width / dWidth) / (forwardLane.size() +
524                    backwardLane.size()));
525        if (lane_width == 0) {
526            lane_width = 1;
527        }
528        return lane_width ;
529    }

541    /**
542     * リンク上のある点の位置の計算。リンク上の視点からの距離を指定。
543     * ただし、リンクが真っ直ぐであると仮定。
544     */
545    public Point2D calcPosition(double position) {
546        double x = fromNode.getX() + (toNode.getX() - fromNode.getX()) * 
547            position / length;
548        double y = fromNode.getY() + (toNode.getY() - fromNode.getY()) * 
549            position / length;
550
551        return new Point2D.Double(x,y);
552    }

568    public void agentEnters(AgentBase agent) {
569        agents.add(agent);
570        /* [2015.01.09 I.Noda]
571         * 以下の処理はおそらく無駄。
572         * しかもいちいち sort しているので、計算時間の無駄のはず。
573        if (agent.isForwardDirection() || agent.isBackwardDirection()) {
574            add_agent_to_lane(agent);
575        }
576        */
577
578        /* alert message を新しいエージェントに伝える */
579        for(HashMap.Entry<Term, SimTime> entry : alertMessageTable.entrySet()) {
580            agent.alertMessage(entry.getKey(), entry.getValue()) ;
581        }
582    }
583
584    public void agentEntersOnLane(AgentBase agent) {
585        agentEnters(agent) ;
586        if (agent.isForwardDirection() || agent.isBackwardDirection()) {
587            add_agent_to_lane(agent);
588        }
589    }

591    public void agentExits (AgentBase agent) {
592        assert(agents.contains(agent));
593        agents.remove(agent);
594        if (! forwardLane.remove(agent)) {
595            backwardLane.remove(agent);
596        }
597    }

744        if(id == null || id.isEmpty()) {
745            Itk.logError("Link element in DOM has no ID" + element) ;
746            Itk.quitByError() ;

749        double length = Double.parseDouble(element.getAttribute("length"));

752        nodes[0] = element.getAttribute("from");

1109    public double calcRestrictedSpeed(double speed, AgentBase agent,
1110                                      SimTime currentTime) {
1111        /* 分断制御 
1112         */
1113        if(isGateClosed(agent, currentTime)) {
1114            /*
1115             * [2018.10.18 I.Noda]
1116             * リンク上での分断制御をやめてみる。
1117             * ノード側のバグが判明したので、復活。
1118             * ゲートのplaceタグを持つノードでもちゃんと止まるようになる。
1119             */
1120            speed = 0 ;
1121        }
1122
1123        /* 制約ルール適用 */
1124        for(int i = 0 ; i < speedRestrictRule.getArraySize() ; i++) {
1125            Term rule = speedRestrictRule.getNthTerm(i) ;
1126            speed = applyRestrictionRule(speed, rule, agent, currentTime) ;
1127        }
1128
1129        return speed ;
1130    }
