025/**
026 * MapLink のテーブル。 
027 * {@literal ArrayList<MapLink>} を置き換え、各種機能を提供する。
028 */
029public class MapLinkTable extends OBNodeTable<MapLink> {
030    //------------------------------------------------------------
031    /**
032     * 指定された tag を持つリンクを取り出す。
033     * @param tag tag名
034     * @return tag 名を持つリンクを集めた MapLinkTable
035     */
036    public MapLinkTable findTaggedLinks(String tag){
037        return findTaggedLinks(tag, new MapLinkTable()) ;
038    }
039
040    //------------------------------------------------------------
041    /**
042     * 指定された tag を持つリンクを取り出す。
043     * @param tag tag名
044     * @param table 格納用テーブル
045     * @return tag 名を持つリンクを集めた MapLinkTable。table がそのまま返る。
046     */
047    public MapLinkTable findTaggedLinks(String tag, MapLinkTable table){
048        return (MapLinkTable)super.findTaggedOBNodes(tag, table) ;
049    }
050
051} // class MapLinkTable
