001package nodagumi.ananPJ.NetworkMap.Polygon;
002
003import java.util.ArrayList;
004import org.w3c.dom.Document;
005import org.w3c.dom.Element;
006import org.w3c.dom.NodeList;
007
008import math.geom3d.Point3D;
009
010/**
011 * ポリゴンを構成する三角形メッシュの頂点座標
012 */
013public class TriangleMeshes {
014    /**
015     * DOM 要素のタグ名
016     */
017    public static final String TAG_NAME = "TriangleMeshes";
018
019    /**
020     * coordinates が法線ベクトルを含んでいるか?
021     */
022    private final boolean normal = false;
023
024    /**
025     * 頂点座標
026     */
027    private Coordinates coordinates = null;
028
029    /**
030     * コンストラクタ
031     */
032    public TriangleMeshes(Coordinates coordinates) {
033        this.coordinates = coordinates;
034    }
035
036    /**
037     * コンストラクタ
038     */
039    public TriangleMeshes(ArrayList<Point3D> coordinates) {
040        this.coordinates = new Coordinates(coordinates, 3);
041    }
042
043    /**
044     * コンストラクタ.
045     *
046     * @param element DOM の TriangleMeshes 要素
047     */
048    public TriangleMeshes(Element element) throws Exception {
049        String tagName = element.getTagName();
050        if (! tagName.equals(TAG_NAME)) {
051            throw new Exception("Tag name is not \"" + TAG_NAME + "\" : " + tagName);
052        }
053
054        NodeList elements = element.getElementsByTagName("coordinates");
055        if (elements.getLength() > 1) {
056            throw new Exception("Too many \"coordinates\" element");
057        } else if (elements.getLength() == 1) {
058            coordinates = new Coordinates((Element)elements.item(0), 3);
059        } else {
060            coordinates = new Coordinates(3);
061        }
062    }
063
064    /**
065     * coordinates が法線ベクトルを含んでいるか?
066     */
067    public boolean isNormal() {
068        return normal;
069    }
070
071    /**
072     * 頂点座標を取得する
073     */
074    public Coordinates getCoordinates() {
075        return coordinates;
076    }
077
078    /**
079     * 頂点座標配列を取得する
080     */
081    public float[] getPoints() {
082        float[] points = new float[coordinates.getValue().size() * 3];
083        for (int index = 0; index < coordinates.getValue().size(); index++) {
084            Point3D point = coordinates.getValue().get(index);
085            points[index * 3] = (float)point.getX();
086            points[index * 3 + 1] = (float)point.getY();
087            points[index * 3 + 2] = (float)point.getZ();
088        }
089        return points;
090    }
091
092    /**
093     * DOM 要素を生成する
094     */
095    public Element createElement(Document dom) {
096        Element element = dom.createElement(TAG_NAME);
097        element.appendChild(coordinates.createElement(dom));
098        return element;
099    }
100}
