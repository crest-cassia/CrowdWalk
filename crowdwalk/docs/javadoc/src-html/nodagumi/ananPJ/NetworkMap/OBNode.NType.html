079    public Element toDom(Document dom, String tagname) {
080        Element element = dom.createElement(tagname);
081        for (String tag_str : tags) {
082            Element tnode = dom.createElement("tag");
083            Text tag_text = dom.createTextNode(tag_str);
084            tnode.appendChild(tag_text);
085            element.appendChild(tnode);
086        }
087        return element;
088    }
089
090    protected void getAttributesFromDom(Element element) {
091        NodeList children = element.getChildNodes();
092        for (int i = 0; i < children.getLength(); ++i) {
093            if (children.item(i)  instanceof Element) {
094                Element child = (Element)children.item(i);
095                if (!child.getTagName().equals("tag")) continue;
096                addTag(Itk.intern(child.getTextContent())) ;
097            }
098        }
099    }
100
101    public static String getNodeTypeString() {return null;}
102    public static OBNode fromDom(Element element) {
103        final String tagName = element.getTagName();
104        if (tagName.equals(MapPartGroup.getNodeTypeString())) {
105            return MapPartGroup.fromDom(element);
106        } else if (tagName.equals(MapNode.getNodeTypeString())) {
107            return MapNode.fromDom(element);
108        } else if (tagName.equals(MapLink.getNodeTypeString())) {
109            return MapLink.fromDom(element);
110        } else if (tagName.equals(MapAreaRectangle.getNodeTypeString())) {
111            return MapAreaRectangle.fromDom(element);
112        } else if (tagName.equals(OBNodeSymbolicLink.getNodeTypeString())) {
113            return OBNodeSymbolicLink.fromDom(element);
114        } else if (tagName.equals(MapPolygon.getNodeTypeString())) {
115            return MapPolygon.fromDom(element);
116        } else if (tagName.equals("tag")){
117            return null;
118        } else {
119            Itk.logWarn("", tagName + " not known to OBnode.fromDom");
120            return null;
121        }
122    }

136     * ID の番号部分(0詰めを除く)を取得
137     */
138    public String getIdNumber() {
139        if (idNumber == null) {
140            idNumber = ID.replaceFirst("[^0-9]*", "");
141            // 番号部分なしならば "" を返す
142            if (idNumber.isEmpty()) {
143                return idNumber;
144            }
145            idNumber = idNumber.replaceFirst("0*", "");
146            // 0のみ
147            if (idNumber.isEmpty()) {
148                idNumber = "0";
149            }
150        }
151        return idNumber;
152    }

162    public boolean hasTag (String _tag) {
163        if (_tag == null) return false;
164
165        //return tags.contains(_tag) ;
166        return Itk.containsItself(tags, _tag) ;
167    }
168
169    /**
170     * 部分文字列 _tag を含んでいるタグがあるか?
171     */
172    public boolean hasSubTag(String subtag) {
173        if (subtag == null || subtag.isEmpty())
174            return false;
175
176        for (String tag : tags) {
177            if (tag.contains(subtag))
178                return true;
179        }
180        return false;
181    }
182
183    public Matcher matchTag (String pattern_str) {
184        Pattern pattern = Pattern.compile(pattern_str);
185        for (final String tag : tags) {
186            Matcher matcher = pattern.matcher(tag);
187            if (matcher.matches()) {
188                return matcher;
189            }
190        }
191        return null;
192    }

198    public boolean addTag(String _tag) {
199        if (_tag == null || _tag.equals("")) return false;
200
201        if (hasTag(_tag)) {
202            return false;
203        }
204        tags.add(_tag);
205        return true;
206    }

211    public boolean replaceTag(String srcTag, String dstTag) {
212        int index = tags.indexOf(srcTag);
213        if (index == -1) {
214            return false;
215        }
216        tags.set(index, dstTag);
217        return true;
218    }

241    public String getTagString() {
242        StringBuffer str = new StringBuffer();
243        boolean first = true;
244        for (final String tag : tags) {
245            if (!first) {
246                str.append(",");
247            }
248            str.append(tag);
249            first = false;
250        }

304     * ゲート（分断制御用交通規制）のチェック
305     */
306    public boolean isGateClosed(AgentBase agent, SimTime currentTime) {
307        if(gateTable == null) {
308            return false ;
309        } else {
310            for(String gateTag: gateTable.keySet()) {
311                GateBase gate = gateTable.get(gateTag) ;
312                if(gate.isClosed(agent, currentTime)) return true ;
313            }
314            return false ;
315        }
316    }

329    /**
330     * ゲート操作
331     */
332    public GateBase switchGate(String gateTag,
333                               GateEvent event,
334                               boolean closed) {
335        if(gateTable == null) {
336            gateTable = new HashMap<String, GateBase>() ;
337        }
338        
339        GateBase gate = gateTable.get(gateTag) ;
340        if(gate == null) {
341            gate = newGate(gateTag, event, closed) ;
342            gateTable.put(gateTag, gate) ;
343        } else {
344            gate.switchGate(event, closed) ;
345        }
346        Itk.logInfo("OBNode::switchGate", gateTag,
347                    (closed ? "close" : "open"), this) ;
348        return gate ;
349    }

351    //------------------------------------------------------------
352    /**
353     * ゲート操作
354     */
355    public GateBase newGate(String gateTag,
356                            GateEvent event,
357                            boolean closed) {
358        String gateClass =
359            (event.eventDef == null ? null :
360             event.eventDef.getArgString("gateClass")) ;
361        if(gateClass == null) {
362            Itk.logDebug("newGate", "no gateClass, use GateBase.") ;
363            return new GateBase(gateTag, event, closed, this) ;
364        } else if(gateClass.equals("RubyGate")) {
365            Itk.logDebug("newGate", "use RubyGate.") ;
366            return new RubyGate(gateTag, event, closed, this) ;
367        } else {
368            Itk.logError("Unknown Gate Class", gateTag, gateClass, event) ;
369            Itk.quitByError() ;
370            return null ; // never reach
371        }
372    }
