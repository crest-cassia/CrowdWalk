221    /**
222     * OBNode の挿入
223     */
224    public void insertOBNode (OBNode parent, OBNode node) {
225        addObject(node.ID, node);
226        insertNodeInto(node, parent, parent.getChildCount());
227
228        OBNode.NType type = node.getNodeType();
229        if (type == OBNode.NType.NODE) {
230            nodesCache.add((MapNode)node);
231        } else if (type == OBNode.NType.LINK) {
232            linksCache.add((MapLink)node);
233        } else if (type == OBNode.NType.AGENT) {
234            Itk.logWarn("insert Agent OBNode is obsolete.", node) ;
235        } else if (type == OBNode.NType.GROUP) {
236            /* no operation */
237        } else if (type == OBNode.NType.AREA) {
238            /* no operation */
239        } else if (type == OBNode.NType.SYMLINK) {
240            /* no operation */
241        } else if (type == OBNode.NType.POLYGON) {
242            /* no operation */
243        } else {
244            Itk.logError("unkown type added");
245        }
246
247    }
248
249    /**
250     * OBノード削除
251     */
252    public void removeOBNode (OBNode parent, OBNode node) {
253        removeObject(node.ID);
254
255        OBNode.NType type = node.getNodeType();
256        if (type != OBNode.NType.SYMLINK) {
257            clearSymlinks(node);
258        }
259
260        boolean linkRemoved = false;
261        switch (type) {
262        case NODE:
263            nodesCache.remove((MapNode)node);
264            break;
265        case LINK:
266            linksCache.remove((MapLink)node);
267            linkRemoved = true;
268            break;
269        case AGENT:
270            Itk.logWarn("remove Agent OBNode is obsolete") ;
271            break;
272        case GROUP:
273            while (node.getChildCount() > 0) {
274                removeOBNode(node, (OBNode)node.getChildAt(0));
275            }
276            break;
277        case AREA:
278            break;
279        case SYMLINK:
280            break;
281        case POLYGON:
282            break;
283        default:
284            Itk.logError("unkown type removed", type) ;
285        }
286        removeNodeFromParent(node);
287
288        if (linkRemoved) {
289            notifier.linkRemoved((MapLink)node);
290        }
291    }

375                              null,
376                              new OBTreeCrawlFunctor() {
377                                  @Override
378                                  public void apply(OBNode node, OBNode parent) {
379                                      if (node.getNodeType() ==
380                                          OBNode.NType.SYMLINK) {
381                                          OBNodeSymbolicLink symlink =
382                                              (OBNodeSymbolicLink)node;
383                                          if (symlink.getOriginal() == orig) {
384                                              Itk.logInfo("deleted!");
385                                              removeOBNode(parent, node);
386                                          }
387                                      }
388                                  }
389                              });
390    }

394     * tree のクローラ操作クラス
395     */
396    protected static abstract class OBTreeCrawlFunctor {
397        public abstract void apply(OBNode node,
398                                   OBNode parent);
399    }

401    //------------------------------------------------------------
402    /**
403     * tree のクローラ操作
404     */
405    private void applyToAllChildrenRec(OBNode node,
406                                       OBNode parent,
407                                       OBTreeCrawlFunctor func) {
408        func.apply(node, parent);
409        for (int i = 0; i < node.getChildCount(); ++i) {
410            OBNode child = (OBNode)node.getChildAt(i);
411            if (child instanceof OBNode) {
412                applyToAllChildrenRec(child, node, func);
413            }
414        }
415    }

498    /**
499     * グループ
500     */
501    public ArrayList<MapPartGroup> getGroups() {
502        ArrayList<MapPartGroup> groups = new ArrayList<MapPartGroup>();
503        for (OBNode node : getOBCollection()) {
504            if (node.getNodeType() == OBNode.NType.GROUP &&
505                !node.getTags().isEmpty()) {
506                groups.add((MapPartGroup)node);
507            }
508        }
509        Collections.sort(groups, new Comparator<MapPartGroup>() {
510                public int compare(MapPartGroup lhs, MapPartGroup rhs) {
511                    Matcher f1 = lhs.matchTag("((B?)\\d+)F");
512                    Matcher f2 = rhs.matchTag("((B?)\\d+)F");
513
514                    if (f1 != null && f2 != null) {
515                        int i1;
516                        if (f1.group(1).startsWith("B")) {
517                            i1 = -Integer.parseInt(f1.group(1).substring(1));
518                        } else {
519                            i1 = Integer.parseInt(f1.group(1));
520                        }
521                        int i2;
522                        if (f2.group(1).startsWith("B")) {
523                            i2 = -Integer.parseInt(f2.group(1).substring(1));
524                        } else {
525                            i2 = Integer.parseInt(f2.group(1));
526                        }
527
528                        return i1 - i2;
529                    } else if (f1 != null) {
530                        return -1;
531                    } else if (f2 != null) {
532                        return 1;
533                    }
534
535                    String tc1Name = lhs.getTagString();
536                    String tc2Name = rhs.getTagString();
537                    if (tc1Name.compareTo(tc2Name) < 0) {
538                        return -1;
539                    } else if (tc1Name.compareTo(tc2Name) > 0) {
540                        return 1;
541                    }
542                    return 0;
543                }
544            });
545        return groups;
546    }

692    /**
693     * 経路探索
694     * @return 探索成功した結果。すでにノードには情報は格納されている。
695     */
696    public Dijkstra.Result calcGoalPath(Term mentalMode,
697                                        String goalTag) {
698        MapNodeTable goals = new MapNodeTable();
699        for (MapNode node : getNodes()) {
700            if (node.hasTag(goalTag)) goals.add(node);
701        }
702        for (MapLink link : getLinks()) {
703            if (link.hasTag(goalTag)) {
704                goals.add(link.getFrom());
705                goals.add(link.getTo());
706            }
707        }
708        if (goals.size() == 0) {
709            Itk.logWarn("No Goal", goalTag) ;
710            synchronized(validRouteKeys) {
711                validRouteKeys.put(goalTag, false) ;
712            }
713            return null ;
714        }
715        Itk.logInfo("Found Goal", goalTag, ":", goals.size()) ;
716
717        Dijkstra.Result result =
718            Dijkstra.calc(mentalMode,
719                          goalTag,
720                          goals,
721                          this) ;
722
723        synchronized(getNodes()) {
724            synchronized(validRouteKeys) {
725                validRouteKeys.put(goalTag, true);
726            }
727            for (MapNode node : result.keySet()) {
728                NavigationHint hint = result.get(node);
729                node.addNavigationHint(mentalMode, goalTag, hint) ;
730            }
731        }
732        return result ;
733    }

789    /**
790     * マップに整合性があるかチェック。
791     * @return 整合性がなければ false。
792     */
793    public boolean checkConsistency() {
794        int counter = 0;
795        for (MapLink link : getLinks()) {
796            if (!getNodes().contains(link.getFrom())) {
797                Itk.logWarn("link's fromNode does not exist.",
798                            link, link.getFrom()) ;
799                counter += 1;
800            } else if (!getNodes().contains(link.getTo())) {
801                Itk.logWarn("link's toNode does not exist.",
802                            link, link.getTo()) ;
803                counter += 1;
804            }
805        }
806        if (counter > 0) {
807            Itk.logWarn("EvacuationSimulator invalid links: ", counter);
808            return false ;
809        } else {
810            return true ;
811        }
812    }

814    /**
815     * マップデータを検証する.
816     *
817     * @return 問題があれば false
818     */
819    public boolean validate() {
820        // マップに整合性があるかチェック
821        boolean succeeded = checkConsistency();
822
823        ArrayList<MapLink> loopedLinks = new ArrayList();
824        HashMap<String, ArrayList<MapLink>> duplicatedLinks = new HashMap();
825        for (MapLink link : getLinks()) {
826            // ループしたリンクがないかチェック
827            if (link.getFrom() == link.getTo()) {
828                Itk.logWarn_("Looped link found", link.toShortInfo());
829                loopedLinks.add(link);
830                succeeded = false;
831            }
832            // 長さ 0m 以下のリンクがないかチェック
833            if (link.getLength() <= 0.0 && ! loopedLinks.contains(link)) {
834                Itk.logWarn_("Zero length link found", link.toShortInfo());
835                succeeded = false;
836            }
837            // 重複リンクがないかチェック
838            MapNode fromNode = link.getFrom();
839            MapNode toNode = link.getTo();
840            for (MapLink _link : fromNode.getLinks()) {
841                if (_link != link && _link.getOther(fromNode) == toNode) {
842                    String key = null;
843                    if (fromNode.getID().compareTo(toNode.getID()) < 0) {
844                        key = fromNode.getID() + " " + toNode.getID();
845                    } else {
846                        key = toNode.getID() + " " + fromNode.getID();
847                    }
848                    ArrayList<MapLink> links = duplicatedLinks.get(key);
849                    if (links == null) {
850                        links = new ArrayList();
851                        links.add(link);
852                        links.add(_link);
853                        duplicatedLinks.put(key, links);
854                    } else {
855                        if (! links.contains(link)) {
856                            links.add(link);
857                        }
858                        if (! links.contains(_link)) {
859                            links.add(_link);
860                        }
861                    }
862                    succeeded = false;
863                }
864            }
865        }
866
867        int n = 1;
868        for (String key : duplicatedLinks.keySet()) {
869            for (MapLink link : duplicatedLinks.get(key)) {
870                Itk.logWarn_("Duplicated link found", "place #" + n + " " + link.toShortInfo());
871            }
872            n++;
873        }
874
875        return succeeded;
876    }

965     * リンク設定。
966     */
967    @SuppressWarnings("unchecked")
968    private void setupLinks(OBNode ob_node) {
969        if (OBNode.NType.LINK == ob_node.getNodeType()) {
970            addObject(ob_node.ID, ob_node);
971            MapLink link = (MapLink) ob_node;
972            linksCache.add(link);
973
974            String[] nodes = (String[])ob_node.getUserObject();
975            MapNode from_node = (MapNode)getObject(nodes[0]);
976            MapNode to_node = (MapNode)getObject(nodes[1]);
977
978            if (from_node == null) {
979                Itk.logWarn("from_node is null:", nodes[0]) ;
980            }
981            if (to_node == null) {
982                Itk.logWarn("to_node is null:", nodes[1]) ;
983            }
984            from_node.addLink(link);
985            to_node.addLink(link);
986            try {
987                link.setFromTo(from_node, to_node);
988            } catch (Exception e) {
989                Itk.dumpStackTraceOf(e) ;
990                Itk.logWarn("Warning setting up network", "Try to set from/to of a link, which alread has it setted link ID: " + link.ID);
991            }
992        } else if (OBNode.NType.GROUP == ob_node.getNodeType()) {
993            Enumeration e = ob_node.children();
994            while (e.hasMoreElements()) {
995                OBNode child = (OBNode)e.nextElement();
996                setupLinks(child);
997            }
998        }

1035    /**
1036     * 元となる OBNode が存在しないシンボリックリンクをすべて削除する
1037     */
1038    private void checkDanglingSymlinks(OBNode node) {
1039        class CheckSymlink extends OBTreeCrawlFunctor {
1040            public int dangling_symlink_count = 0;
1041            boolean found = false;
1042            @Override
1043            public void apply(OBNode node,
1044                              OBNode parent) {
1045                if (node.getNodeType() == OBNode.NType.SYMLINK) {
1046                    OBNode original = ((OBNodeSymbolicLink)node).getOriginal();
1047                    if (original == null) {
1048                        dangling_symlink_count++;
1049                        found = true;
1050                        removeOBNode(parent, node);
1051                    }
1052                }
1053            }
1054
1055            public boolean mustCheckMore() {
1056                boolean ret = found;
1057                found = false;
1058                return ret;
1059            }
1060        }
1061        CheckSymlink checker = new CheckSymlink();
1062        do {
1063            applyToAllChildrenRec(node, null, checker);
1064        } while(checker.mustCheckMore());
1065        if (checker.dangling_symlink_count != 0) {
1066            Itk.logWarn("Corrupt file", "Removed " + checker.dangling_symlink_count + " dangling symlink(s) on loading");
1067        }
1068    }

1076    /**
1077     * マップに外接する矩形を算出する
1078     */
1079    public Rectangle2D calcRectangle() {
1080        double north = 0.0;
1081        double south = 0.0;
1082        double west = 0.0;
1083        double east = 0.0;
1084        for (MapNode node : getNodes()) {
1085            if (north == 0.0 && south == 0.0) {
1086                north = node.getY();
1087                south = node.getY();
1088                west = node.getX();
1089                east = node.getX();
1090            }
1091            if (node.getY() < north) {
1092                north = node.getY();
1093            }
1094            if (node.getY() > south) {
1095                south = node.getY();
1096            }
1097            if (node.getX() < west) {
1098                west = node.getX();
1099            }
1100            if (node.getX() > east) {
1101                east = node.getX();
1102            }
1103        }
1104        return new Rectangle2D.Double(west, north, east - west, south - north);
1105    }

1107    /**
1108     * このオブジェクトのコピーを作成して返す
1109     */
1110    public NetworkMap clone() {
1111        final NetworkMap networkMap = new NetworkMap();
1112        // コンストラクタが作ったグループは削除する
1113        networkMap.removeObject(((MapPartGroup)networkMap.root).ID);
1114
1115        applyToAllChildrenRec((OBNode)root, null, new OBTreeCrawlFunctor() {
1116            public void apply(OBNode node, OBNode parent) {
1117                MapPartGroup parentGroup = null;
1118                if (parent != null) {
1119                    parentGroup = (MapPartGroup)networkMap.getObject(parent.ID);
1120                }
1121                String id = node.ID;
1122                OBNode.NType type = node.getNodeType();
1123                if (type == OBNode.NType.GROUP) {
1124                    MapPartGroup group = new MapPartGroup(id);
1125                    ((MapPartGroup)node).copyAttributes(group);
1126                    if (parent == null) {
1127                        networkMap.addObject(id, group);
1128                        networkMap.setRoot(group);
1129                    } else {
1130                        networkMap.insertOBNode(parentGroup, group);
1131                    }
1132                } else if (type == OBNode.NType.NODE) {
1133                    MapNode srcNode = (MapNode)node;
1134                    MapNode newNode = new MapNode(id, (Point2D)srcNode.getPosition().clone(), srcNode.getHeight());
1135                    for (String tag : srcNode.getTags()) {
1136                        newNode.getTags().add(tag);
1137                    }
1138                    networkMap.insertOBNode(parentGroup, newNode);
1139                } else if (type == OBNode.NType.LINK) {
1140                    MapLink srcLink = (MapLink)node;
1141                    MapLink newLink = new MapLink(id, srcLink.getLength(), srcLink.getWidth());
1142                    for (String tag : srcLink.getTags()) {
1143                        newLink.getTags().add(tag);
1144                    }
1145                    networkMap.insertOBNode(parentGroup, newLink);
1146                } else if (type == OBNode.NType.AREA) {
1147                    MapArea srcArea = (MapArea)node;
1148                    Rectangle2D bounds = (Rectangle2D)srcArea.getShape();
1149                    MapAreaRectangle newArea = new MapAreaRectangle(id, bounds, srcArea.getMinHeight(), srcArea.getMaxHeight(), srcArea.getAngle());
1150                    for (String tag : srcArea.getTags()) {
1151                        newArea.getTags().add(tag);
1152                    }
1153                    networkMap.insertOBNode(parentGroup, newArea);
1154                } else if (type == OBNode.NType.SYMLINK) {
1155                    OBNodeSymbolicLink srcSymlink = (OBNodeSymbolicLink)node;
1156                    if (srcSymlink.getOriginal() != null) {
1157                        // 元のオリジナルを仮代入しておく
1158                        OBNodeSymbolicLink symlink = new OBNodeSymbolicLink(id, srcSymlink.getOriginal());
1159                        networkMap.insertOBNode(parentGroup, symlink);
1160                    }
1161                } else if (type == OBNode.NType.POLYGON) {
1162                    networkMap.insertOBNode(parentGroup, ((MapPolygon)node).clone());
1163                }
1164            }
1165        });
1166        for (MapLink link : networkMap.getLinks()) {
1167            MapLink srcLink = (MapLink)this.getObject(link.ID);
1168            MapNode fromNode = (MapNode)networkMap.getObject(srcLink.getFrom().ID);
1169            MapNode toNode = (MapNode)networkMap.getObject(srcLink.getTo().ID);
1170            try {
1171                link.setFromTo(fromNode, toNode);
1172            } catch (Exception e) {
1173                Itk.logError("NetworkMap replication error", e.getMessage());
1174                return null;
1175            }
1176            link.prepareAdd();
1177        }
1178        networkMap.applyToAllChildrenRec((OBNode)networkMap.root, null, new OBTreeCrawlFunctor() {
1179            public void apply(OBNode node, OBNode parent) {
1180                if (node.getNodeType() == OBNode.NType.SYMLINK) {
1181                    OBNodeSymbolicLink symlink = (OBNodeSymbolicLink)node;
1182                    OBNode original = networkMap.getObject(symlink.getOriginal().ID);
1183                    symlink.setOriginal(original);
1184                }
1185            }
1186        });
1187
1188        return networkMap;
1189    }
