075    private void addAttributesToDom(Element element) {
076        element.setAttribute("id", ID);
077        element.setAttribute("pNorthWestX", "" + pNorthWest.getX());
078        element.setAttribute("pNorthWestY", "" + pNorthWest.getY());
079        element.setAttribute("pSouthEastX", "" + pSouthEast.getX());
080        element.setAttribute("pSouthEastY", "" + pSouthEast.getY());
081        element.setAttribute("pTheta", "" + pTheta);
082
083        element.setAttribute("tx", "" + tx);
084        element.setAttribute("ty", "" + ty);
085        element.setAttribute("sx", "" + sx);
086        element.setAttribute("sy", "" + sy);
087        element.setAttribute("r", "" + r);
088        
089        element.setAttribute("scale", "" + getScale());
090        element.setAttribute("minHeight", "" + getMinHeight());
091        element.setAttribute("maxHeight", "" + getMaxHeight());
092        element.setAttribute("defaultHeight", "" + getDefaultHeight());
093        element.setAttribute("imageFileName", getImageFileName());
094        element.setAttribute("zone", "" + getZone());
095    }

097    @Override
098    protected void getAttributesFromDom(Element element) {
099        super.getAttributesFromDom(element);
100        ID = element.getAttribute("id");
101        double x = Double.parseDouble(element.getAttribute("pNorthWestX"));
102        double y = Double.parseDouble(element.getAttribute("pNorthWestY"));
103        pNorthWest = new Point2D.Double(x, y);
104        
105        x = Double.parseDouble(element.getAttribute("pSouthEastX"));
106        y = Double.parseDouble(element.getAttribute("pSouthEastY"));
107        pSouthEast = new Point2D.Double(x, y);
108        pTheta = Double.parseDouble(element.getAttribute("pTheta"));
109
110        tx = Double.parseDouble(element.getAttribute("tx"));
111        ty = Double.parseDouble(element.getAttribute("ty"));
112        sx = Double.parseDouble(element.getAttribute("sx"));
113        sy = Double.parseDouble(element.getAttribute("sy"));
114        r = Double.parseDouble(element.getAttribute("r"));
115        if (sx != sy) {
116                        Itk.logWarn("MapPartGroup", element.getAttribute("tag"),"group.",
117                                                "Aspect Ratio mismatch.",
118                                                "sx=", sx, "sy=", sy, ".") ;
119        }
120
121        setScale(Double.parseDouble(element.getAttribute("scale")));
122        setMinHeight(Double.parseDouble(element.getAttribute("minHeight")));
123        setMaxHeight(Double.parseDouble(element.getAttribute("maxHeight")));
124        setDefaultHeight(Double.parseDouble(element.getAttribute("defaultHeight")));
125        setImageFileName(element.getAttribute("imageFileName"));
126
127        String _zone = element.getAttribute("zone");
128        if (_zone != null && ! _zone.isEmpty()) {
129            setZone(Integer.parseInt(_zone));
130        }
131    }

147    @SuppressWarnings("unchecked")
148    @Override
149    public Element toDom(Document dom, String tagname) {
150        Element element = super.toDom(dom, getNodeTypeString());
151
152        addAttributesToDom(element);
153        Enumeration e = children();
154        while (e.hasMoreElements()) {
155            OBNode child = (OBNode)e.nextElement();
156            if (child != null) {
157                element.appendChild(child.toDom(dom, null));
158            }
159        }
160        return element;
161    }

163    public static OBNode fromDom(Element element) {
164        MapPartGroup group = new MapPartGroup(null);
165        group.getAttributesFromDom(element);
166        NodeList elm_children = element.getChildNodes();
167        for (int i = 0; i < elm_children.getLength(); ++i) {
168            if (elm_children.item(i) instanceof Element) {
169                Element node = (Element)elm_children.item(i);
170                OBNode child = OBNode.fromDom(node);
171                if (child != null) {
172                    group.add(child);
173                }
174            }
175        }
176        return group;
177    }

307    @SuppressWarnings("unchecked")
308    public MapNodeTable getChildNodesAndSymlinks() {
309        MapNodeTable children = new MapNodeTable();
310        Enumeration e = this.children();
311        while (e.hasMoreElements()) {
312            OBNode child = (OBNode)e.nextElement();
313            if (child.getNodeType() == OBNode.NType.NODE) {
314                children.add((MapNode)child);
315            } else if (child.getNodeType() == OBNode.NType.SYMLINK) {
316                OBNode orig = ((OBNodeSymbolicLink)child).getOriginal();
317                if (orig.getNodeType() == OBNode.NType.NODE) {
318                    children.add((MapNode)orig);
319                }
320            }
321        }
322        return children;
323    }

325    // 基準座標から象限を絞って抽出する
326    public MapNodeTable getChildNodesAndSymlinks(MapNode from, int quadrant) {
327        MapNodeTable children = new MapNodeTable();
328        Enumeration e = this.children();
329        while (e.hasMoreElements()) {
330            OBNode child = (OBNode)e.nextElement();
331            if (child.getNodeType() == OBNode.NType.NODE) {
332                if (from.include((MapNode)child, quadrant)) {
333                    children.add((MapNode)child);
334                }
335            } else if (child.getNodeType() == OBNode.NType.SYMLINK) {
336                OBNode orig = ((OBNodeSymbolicLink)child).getOriginal();
337                if (orig.getNodeType() == OBNode.NType.NODE) {
338                    if (from.include((MapNode)orig, quadrant)) {
339                        children.add((MapNode)orig);
340                    }
341                }
342            }
343        }
344        return children;
345    }
