013package nodagumi.ananPJ.NetworkMap.Node;
014
015import java.util.Collection;
016import java.util.ArrayList;
017import java.util.HashMap;
018
019import nodagumi.ananPJ.NetworkMap.OBNodeTable ;
020import nodagumi.ananPJ.NetworkMap.Node.MapNode ;
021
022import nodagumi.Itk.*;
023
024//======================================================================
025/**
026 * MapNode のテーブル。 
027 * {@code ArrayList<MapNode>} を置き換え、各種機能を提供する。
028 */
029public class MapNodeTable extends OBNodeTable<MapNode> {
030    //------------------------------------------------------------
031    /**
032     * Constructor with no args
033     */
034    public MapNodeTable() { 
035        super() ; 
036    }
037
038    //------------------------------------------------------------
039    /**
040     * Constructor from Collection
041     * これがなぜ必要なのかわからない。
042     * Generic 型ではコンストラクタは継承されない？
043     */
044    public MapNodeTable(Collection<MapNode> origin) {
045        super(origin) ;
046    }
047
048    //------------------------------------------------------------
049    /**
050     * 指定された tag を持つノードを取り出す。
051     * @param tag tag名
052     * @return tag 名を持つノードを集めた MapNodeTable
053     */
054    public MapNodeTable findTaggedNodes(String tag){
055        return findTaggedNodes(tag, new MapNodeTable()) ;
056    }
057
058    //------------------------------------------------------------
059    /**
060     * 指定された tag を持つノードを取り出す。
061     * @param tag tag名
062     * @param table 格納用テーブル
063     * @return tag 名を持つノードを集めた MapNodeTable。table がそのまま返る。
064     */
065    public MapNodeTable findTaggedNodes(String tag, MapNodeTable table){
066        return (MapNodeTable)super.findTaggedOBNodes(tag, table) ;
067    }
068
069    //------------------------------------------------------------
070    /**
071     * 指定された tag をもつノードの、指定したprefixを持つタグを集める。
072     * @param tag tag名
073     * @param tag prefix
074     * @return tag 名を持つノードのタグを集めた MapNodeTable。
075     */
076    public ArrayList<String> findPrefixedTagsOfTaggedNodes(String tag,
077                                                           String prefix) {
078        ArrayList<String> table = new ArrayList<String>() ;
079
080        for(MapNode node : this) {
081            if(node.hasTag(tag)) {
082                for(String nodeTag : node.getTags()) {
083                    if(nodeTag.contains(prefix)) {
084                        table.add(nodeTag) ;
085                    }
086                }
087            }
088        }
089
090        return table ;
091    }
092
093} // class MapNodeTable
