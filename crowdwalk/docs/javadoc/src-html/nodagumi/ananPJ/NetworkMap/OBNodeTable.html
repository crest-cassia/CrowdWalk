013package nodagumi.ananPJ.NetworkMap;
014
015import java.util.Collection;
016import java.util.ArrayList;
017import java.util.HashMap;
018import java.util.Random;
019
020import nodagumi.ananPJ.NetworkMap.OBNode;
021
022import nodagumi.Itk.*;
023
024//======================================================================
025/**
026 * OBNode のテーブル。 
027 * {@literal ArrayList<OBNode>} を置き換え、各種機能を提供する。
028 */
029public class OBNodeTable<T extends OBNode> extends ArrayList<T> {
030    //------------------------------------------------------------
031    /**
032     * Constructor with no args
033     */
034    public OBNodeTable() { 
035        super() ; 
036    }
037
038    //------------------------------------------------------------
039    /**
040     * Constructor from Collection
041     */
042    public OBNodeTable(Collection<T> origin) {
043        this() ;
044        for(T obNode  : origin) {
045            add(obNode) ;
046        }
047    }
048
049    //------------------------------------------------------------
050    /**
051     * 指定された tag を持つリンクの存在チェック
052     * @param tag tag名
053     * @return 存在したら true。
054     */
055    public boolean tagExistP(String tag) {
056        for(OBNode obNode : this) {
057            if(obNode.hasTag(tag)) {
058                return true ;
059            }
060        }
061        return false ;
062    }
063
064    //------------------------------------------------------------
065    /**
066     * 指定された id を持つOBNodeの取得
067     * @param id tag名
068     * @return 存在したらそのOBNode。
069     */
070    public T findById(String id) {
071        for(T obNode : this) {
072            if(obNode.getID().equals(id)) {
073                return obNode ;
074            }
075        }
076        return null ;
077    }
078
079    //------------------------------------------------------------
080    /**
081     * 指定された tag を持つOBNodeを取り出す。
082     * @param tag tag名
083     * @return tag 名を持つOBNodeを集めた OBNodeTable
084     */
085    public OBNodeTable<T> findTaggedOBNodes(String tag){
086        return findTaggedOBNodes(tag, new OBNodeTable<T>()) ;
087    }
088
089    //------------------------------------------------------------
090    /**
091     * 指定された tag を持つリンクを取り出す。
092     * @param tag tag名
093     * @param table 格納用テーブル
094     * @return tag 名を持つOBNodeを集めた OBNodeTable。table がそのまま返る。
095     */
096    public OBNodeTable findTaggedOBNodes(String tag, OBNodeTable table){
097        for(OBNode obNode : this) {
098            if(obNode.hasTag(tag)) {
099                table.add(obNode) ;
100            }
101        }
102        return table ;
103    }
104
105    //------------------------------------------------------------
106    /**
107     * テーブル内の要素をランダムに選択
108     * @param random 乱数生成器
109     * @return 選択された要素
110     */
111    public T chooseRandom(Random random) {
112        return get(random.nextInt(size())) ;
113    }
114
115} // class OBNodeTable<T>
