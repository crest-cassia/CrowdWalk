284    /**
285     * 地図の読み込み
286     */
287    protected NetworkMap readMapWithName(String file_name)
288            throws IOException {
289        FileInputStream fis = new FileInputStream(file_name);
290        Document doc = ItkXmlUtility.singleton.streamToDoc(fis);
291        if (doc == null) {
292            Itk.logError("Could not read map.");
293            return null;
294        }
295        NodeList toplevel = doc.getChildNodes();
296        if (toplevel == null) {
297            Itk.logError("", "BasiciSimulationLauncher.readMapWithName " +
298                         "invalid inputted DOM object.");
299            return null;
300        }
301        // NetMAS based map
302        NetworkMap tmpMap = new NetworkMap() ;
303        if (false == tmpMap.fromDOM(doc))
304            return null;
305        tmpMap.sortNodesById();
306        tmpMap.sortLinksById();
307        Itk.logInfo("Load Map File", file_name) ;
308        setupFileInfo.setNetworkMapFile(file_name);
309        return tmpMap;
310    }

318                                      ArrayList<String> _commandLineFallbacks) {
319        properties = new CrowdWalkPropertiesHandler(_propertiesFile);
320        propertiesFile = _propertiesFile;
321        commandLineFallbacks = _commandLineFallbacks;
322
323        // random
324        random = new Random(properties.getRandseed()) ;
325        // files
326        setNetworkMapFile(properties.getNetworkMapFile());
327        setPollutionFile(properties.getPollutionFile());
328        setGenerationFile(properties.getGenerationFile());
329        setScenarioFile(properties.getScenarioFile());
330        setFallbackFile(properties.getFallbackFile(),
331                        commandLineFallbacks) ;
332        // ending condition
333        setExitCount(properties.getExitCount()) ;
334        setIsAllAgentSpeedZeroBreak(properties.getIsAllAgentSpeedZeroBreak());
335
336    }

345    /**
346     * シミュレーションの初期化。
347     */
348    protected void initializeSimulatorEntity() {
349
350        simulator = new EvacuationSimulator(map, this, random) ;
351
352        simulator.setProperties(properties);
353        simulator.setup();
354      
355        // model.begin set files (pol, gen, sce) to map
356        simulator.begin() ;
357
358        simulator.setIsAllAgentSpeedZeroBreak(isAllAgentSpeedZeroBreak);
359
360        // rand seed setup
361        random.setSeed(properties.getRandseed());
362
363        counter = 0;
364        finished = false;
365    }
366
367    //------------------------------------------------------------
368    /**
369     * シミュレーションのステップ（synchronize していない）
370     */
371    protected void simulateOneStepBare() {
372        finished = simulator.updateEveryTick();
373        int relativeTime = (int)simulator.clock.getRelativeTime() ;
374        counter++ ;
375        if ((counter % 100) == 0)
376            Itk.logDebug("Cycle",
377                         "at:", simulator.currentTime.getAbsoluteTimeString(),
378                         "("+counter+")",
379                         "time:", Itk.getCurrentTimeStrShort(),
380                         "walking:",
381                         simulator.getAgentHandler().numOfWalkingAgents()) ;
382    }
383
384    /**
385     * EvacuationSimulator のステップ処理が終了してカウントアップされる直前に呼び出される。
386     */
387    public abstract void updateEveryTick(SimTime currentTime);
388
389    //------------------------------------------------------------
390    /**
391     * シミュレーションのメインループ。
392     * 初期化などは済んでいるものとする。
393     * また、pause で止まった後の再開もこれで行う。
394     */
395    protected void simulateMainLoop() {
396        if (finished || paused) {
397            return;
398        }
399        while (!finished && !paused) {
400            simulateOneStepBare() ;
401            if (exitCount > 0 && counter > exitCount) {
402                finished = true;
403                break;
404            }
405        }
406        if (finished) {
407            finish() ;
408        }
409    }
