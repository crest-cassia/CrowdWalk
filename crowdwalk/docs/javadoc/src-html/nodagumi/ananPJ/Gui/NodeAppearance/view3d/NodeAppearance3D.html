001package nodagumi.ananPJ.Gui.NodeAppearance.view3d;
002
003import java.util.HashMap;
004
005import nodagumi.ananPJ.Gui.NodeAppearance.NodeAppearanceBase;
006import nodagumi.ananPJ.Gui.SimulationPanel3D;
007
008import nodagumi.Itk.Itk;
009
010/**
011 * ノード 3D 表示の定義情報を扱う
012 */
013public class NodeAppearance3D extends NodeAppearanceBase {
014    /**
015     * ノードを 3D 描画するオブジェクト
016     */
017    private NodeViewBase3D view;
018
019    /**
020     * コンストラクタ
021     */
022    public NodeAppearance3D(SimulationPanel3D panel, HashMap parameters) {
023        super(parameters);
024
025        if (isValidFor3D()) {
026            try {
027                view = (NodeViewBase3D)classFinder.newByName(viewClassName3D);
028                view.init(panel, viewParameters3D, this);
029            } catch (Exception e) {
030                Itk.quitWithStackTrace(e);
031            }
032        }
033    }
034
035    /**
036     * 描画クラスのインスタンスを取得する
037     */
038    public NodeViewBase3D getView() {
039        return view;
040    }
041}
