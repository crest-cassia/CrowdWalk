001package nodagumi.ananPJ.Gui.NodeAppearance.view2d;
002
003import java.awt.Graphics2D;
004import java.awt.geom.AffineTransform;
005import java.awt.geom.Point2D;
006import java.awt.image.BufferedImage;
007import java.io.File;
008import java.io.IOException;
009import java.util.HashMap;
010import javax.imageio.ImageIO;
011
012import nodagumi.ananPJ.Gui.SimulationPanel2D;
013import nodagumi.ananPJ.NetworkMap.Node.MapNode;
014
015/**
016 * 固定スケール画像ノードの 2D 表示
017 */
018public class FixedScaleImageNode2D extends NodeViewBase2D {
019    /**
020     * 画像
021     */
022    protected BufferedImage image;
023
024    /**
025     * 画像の幅(Pixel)
026     */
027    protected int width;
028
029    /**
030     * 画像の高さ(Pixel)
031     */
032    protected int height;
033
034    /**
035     * コンストラクタ
036     */
037    public FixedScaleImageNode2D() {}
038
039    /**
040     * 初期設定
041     */
042    public void init(SimulationPanel2D panel, HashMap parameters, NodeAppearance2D nodeAppearance) throws Exception {
043        super.init(panel, parameters, nodeAppearance);
044
045        String fileName = getStringParameter("fileName", null);
046        if (fileName == null) {
047            throw new Exception("Node appearance error: Image file name is not specified with " + getClass().getSimpleName() + ".");
048        }
049        String filePath = panel.getProperties().furnishPropertiesDirPath(fileName, true, false);
050        try {
051            image = ImageIO.read(new File(filePath));
052        } catch (IOException e) {
053            throw new Exception("Node appearance error: \"" + filePath + "\" " + e.getMessage() + " with " + getClass().getSimpleName() + ".");
054        }
055        width = image.getWidth();
056        height = image.getHeight();
057    }
058
059    /**
060     * ノードの半径を取得する
061     */
062    public double getRadius() {
063        double scale = panel.getDrawingScale();
064        return Math.max(width, height) / 2.0 / scale;
065    }
066
067    /**
068     * ノードを描画する
069     */
070    public void draw(MapNode node, Graphics2D g2) {
071        double scale = panel.getDrawingScale();
072        Point2D point = panel.getRotatedPoint(node);
073
074        AffineTransform at = g2.getTransform();
075        g2.translate(point.getX(), point.getY());
076        g2.rotate(Math.toRadians(panel.getAngle()));
077        g2.scale(1.0 / scale, 1.0 / scale);
078        g2.drawImage(image, -width / 2, -height / 2, width, height, panel);
079        g2.setTransform(at);
080    }
081}
