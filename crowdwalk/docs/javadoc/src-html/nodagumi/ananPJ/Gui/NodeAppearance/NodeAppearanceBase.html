001package nodagumi.ananPJ.Gui.NodeAppearance;
002
003import java.util.HashMap;
004import java.util.regex.Matcher;
005import java.util.regex.Pattern;
006
007import nodagumi.ananPJ.CrowdWalkLauncher;
008import nodagumi.ananPJ.misc.JsonicHashMapGetter;
009import nodagumi.ananPJ.NetworkMap.Node.MapNode;
010import nodagumi.ananPJ.Simulator.Obstructer.ObstructerBase;
011import nodagumi.Itk.*;
012
013/**
014 * ノード表示の定義情報を扱うベースクラス。
015 * <ul>
016 * <li>設定方法: 
017 * propaties_file の "node_appearance_file" に、その定義ファイルを指定する。
018 * </li>
019 * <li> 定義ファイルの形式:
020 * <a target="_blank" href="https://github.com/crest-cassia/CrowdWalk/wiki/Node-appearance-%E3%81%AE%E8%A8%AD%E5%AE%9A%E6%96%B9%E6%B3%95">こちら</a>を参照。
021 * </li>
022 * </ul>
023 */
024public class NodeAppearanceBase extends JsonicHashMapGetter {
025    /**
026     * ノードの表示スタイルを決めるクラスの一覧
027     */
028    public static final String CLASSES = "/node_appearance_classes.json";
029
030    /**
031     * クラスローダー
032     */
033    protected ClassFinder classFinder = new ClassFinder();
034
035    /**
036     * このスタイルを適用するノードのタグ
037     */
038    protected String tag;
039
040    /**
041     * タグが正規表現だった場合の正規表現パターン
042     */
043    protected Pattern tagPattern = null;
044
045    /**
046     * ノードを 2D 描画するクラスの名前
047     */
048    protected String viewClassName2D;
049
050    /**
051     * ノードを 2D 描画するクラスの初期化パラメータ
052     */
053    protected HashMap viewParameters2D;
054
055    /**
056     * ノードを 3D 描画するクラスの名前
057     */
058    protected String viewClassName3D;
059
060    /**
061     * ノードを 3D 描画するクラスの初期化パラメータ
062     */
063    protected HashMap viewParameters3D;
064
065    /**
066     * コンストラクタ
067     */
068    public NodeAppearanceBase(HashMap parameters) {
069        classFinder.aliasByJson(ObstructerBase.resourceToString(CLASSES));
070
071        setParameters(parameters);
072        try {
073            // 適用対象となるノードタグ
074            tag = getStringParameter("tag", "*");
075            Matcher matcher = Pattern.compile("^\\/(.*)\\/$").matcher(tag);
076            // タグが正規表現か?
077            if (matcher.find()) {
078                tagPattern = Pattern.compile(matcher.group(1));
079            }
080
081            // NodeViewBase2D インスタンス生成用の情報を取得
082            HashMap<String, Object> view2DPart = getHashMapParameter("2D_View", null);
083            if (view2DPart != null) {
084                viewClassName2D = (String)view2DPart.get("className");
085                if (viewClassName2D == null) {
086                    throw new Exception("Node appearance file format error.");
087                }
088                if (! isUsableClass(viewClassName2D)) {
089                    throw new Exception("Node appearance file error: invalid className - " + viewClassName2D);
090                }
091                viewParameters2D = (HashMap<String, Object>)view2DPart.get("parameters");
092            }
093
094            // NodeViewBase3D インスタンス生成用の情報を取得
095            HashMap<String, Object> view3DPart = getHashMapParameter("3D_View", null);
096            if (view3DPart != null && CrowdWalkLauncher.isUsable3dSimulator()) {
097                viewClassName3D = (String)view3DPart.get("className");
098                if (viewClassName3D == null) {
099                    throw new Exception("Node appearance file format error.");
100                }
101                if (! isUsableClass(viewClassName3D)) {
102                    throw new Exception("Node appearance file error: invalid className - " + viewClassName3D);
103                }
104                viewParameters3D = (HashMap<String, Object>)view3DPart.get("parameters");
105            }
106        } catch (Exception e) {
107            Itk.quitWithStackTrace(e);
108        }
109    }
110
111    /**
112     * 使用可能なクラスか?
113     */
114    protected boolean isUsableClass(String className) {
115        return classFinder.isClassName(className);
116    }
117
118    /**
119     * このスタイルを適用するタグを取得する
120     */
121    public String getTag() {
122        return tag;
123    }
124
125    /**
126     * タグに該当する文字列か?
127     */
128    public boolean isTagApplied(String text) {
129        if (tagPattern == null) {
130            if (tag.equals("*") || text.indexOf(tag) != -1) {
131                return true;
132            }
133        } else {
134            return tagPattern.matcher(text).find();
135        }
136        return false;
137    }
138
139    /**
140     * 2D View に使用可能
141     */
142    public boolean isValidFor2D() {
143        return viewClassName2D != null;
144    }
145
146    /**
147     * 3D View に使用可能
148     */
149    public boolean isValidFor3D() {
150        return viewClassName3D != null;
151    }
152}
