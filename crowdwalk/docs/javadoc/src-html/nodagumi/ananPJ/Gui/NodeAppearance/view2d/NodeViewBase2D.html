001package nodagumi.ananPJ.Gui.NodeAppearance.view2d;
002
003import java.awt.BasicStroke;
004import java.awt.Color;
005import java.awt.Graphics2D;
006import java.awt.geom.Ellipse2D;
007import java.awt.geom.Point2D;
008import java.util.HashMap;
009
010import nodagumi.ananPJ.Gui.Color2D;
011import nodagumi.ananPJ.Gui.SimulationPanel2D;
012import nodagumi.ananPJ.NetworkMap.Node.MapNode;
013import nodagumi.ananPJ.misc.JsonicHashMapGetter;
014
015/**
016 * ノード 2D 描画のベースクラス
017 */
018public abstract class NodeViewBase2D extends JsonicHashMapGetter {
019    /**
020     * 2D シミュレーションパネル
021     */
022    protected SimulationPanel2D panel;
023
024    /**
025     * ノードの 2D 表示の定義情報
026     */
027    protected NodeAppearance2D nodeAppearance;
028
029    /**
030     * ノードの色
031     */
032    protected Color color = Color2D.BLACK2;
033
034    /**
035     * 初期設定
036     */
037    public void init(SimulationPanel2D panel, HashMap parameters, NodeAppearance2D nodeAppearance) throws Exception {
038        this.panel = panel;
039        this.nodeAppearance = nodeAppearance;
040        setParameters(parameters);
041
042        String colorName = getStringParameter("color", "BLACK2");
043        color = Color2D.getColor(colorName);
044        if (color == null) {
045            color = Color2D.BLACK2;
046        }
047        float transparency = getFloatParameter("transparency", 0.0f);
048        float[] compArray = color.getComponents(null);
049        color = new Color(compArray[0], compArray[1], compArray[2], 1.0f - transparency);
050    }
051
052    /**
053     * ノードの半径を取得する
054     */
055    public abstract double getRadius();
056
057    /**
058     * ノードとラベルを描画する
059     */
060    public void draw(MapNode node, Graphics2D g2, boolean showLabel) {
061        draw(node, g2);
062        if (showLabel) {
063            drawLabel(node, g2, Color.darkGray, panel.getBackground());
064        }
065    }
066
067    /**
068     * ノードを描画する
069     */
070    public abstract void draw(MapNode node, Graphics2D g2);
071
072    /**
073     * ノードのホバーを描画する
074     */
075    public void drawHover(MapNode node, Graphics2D g2) {
076        double scale = panel.getDrawingScale();
077        double diameter = 8.0 / scale;
078        Point2D point = panel.getRotatedPoint(node);
079        double x = point.getX() - diameter / 2.0;
080        double y = point.getY() - diameter / 2.0;
081
082        g2.setColor(panel.HOVER_COLOR);
083        g2.setStroke(new BasicStroke((float)(3.0 / scale)));
084        g2.draw(new Ellipse2D.Double(x, y, diameter, diameter));
085        drawLabel(node, g2, Color.darkGray, panel.HOVER_BG_COLOR);
086    }
087
088    /**
089     * ラベルを描画する
090     */
091    public void drawLabel(MapNode node, Graphics2D g2, Color color, Color bgColor) {
092        String text = node.getTagString();
093        if (! text.isEmpty()) {
094            double cx = node.getX();
095            double cy = node.getY() - getRadius();
096            g2.setColor(color);
097            panel.drawText(g2, cx, cy, SimulationPanel2D.TextPosition.UPPER_CENTER, text, bgColor);
098        }
099    }
100}
