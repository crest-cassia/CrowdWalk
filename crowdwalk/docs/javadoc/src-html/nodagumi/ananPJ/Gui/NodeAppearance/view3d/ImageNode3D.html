001package nodagumi.ananPJ.Gui.NodeAppearance.view3d;
002
003import java.io.File;
004import java.util.HashMap;
005
006import javafx.geometry.Point2D;
007import javafx.scene.image.Image;
008import javafx.scene.paint.Color;
009import javafx.scene.paint.PhongMaterial;
010import javafx.scene.shape.MeshView;
011import javafx.scene.shape.Shape3D;
012import javafx.scene.shape.TriangleMesh;
013
014import nodagumi.ananPJ.Gui.SimulationPanel3D;
015import nodagumi.ananPJ.NetworkMap.Node.MapNode;
016
017/**
018 * 画像ノードの 3D 表示.
019 *
020 * width * height サイズの長方形を形作る三角形メッシュに画像テクスチャを貼り付ける
021 */
022public class ImageNode3D extends NodeViewBase3D {
023    private final static float[] TEX_COORDS = {
024        0, 0,
025        0, 1,
026        1, 1,
027        1, 0
028    };
029
030    private final static int[] FACES = {
031        0, 0, 1, 1, 2, 2,
032        2, 2, 3, 3, 0, 0
033    };
034
035    /**
036     * 画像の頂点コンポーネントが定義された三角形メッシュ
037     */
038    protected TriangleMesh mesh;
039
040    /**
041     * コンストラクタ
042     */
043    public ImageNode3D() {}
044
045    /**
046     * 初期設定
047     */
048    public void init(SimulationPanel3D panel, HashMap parameters, NodeAppearance3D nodeAppearance) throws Exception {
049        super.init(panel, parameters, nodeAppearance);
050
051        String fileName = getStringParameter("fileName", null);
052        if (fileName == null) {
053            throw new Exception("Node appearance error: Image file name is not specified with ImageNode3D.");
054        }
055        String filePath = panel.getPropertiesHandler().furnishPropertiesDirPath(fileName, true, false);
056        Image image = new Image(new File(filePath).toURI().toString());
057        material.setDiffuseColor(Color.WHITE);
058        material.setDiffuseMap(image);
059
060        double width = getDoubleParameter("width", 0.0);
061        double height = getDoubleParameter("height", 0.0);
062        if (width <= 0.0 || height <= 0.0) {
063            throw new Exception("Node appearance error: width or height is not specified correctly with ImageNode3D.");
064        }
065
066        Point2D[] vertices = new Point2D[4];
067        vertices[0] = new Point2D(0.0, 0.0);
068        vertices[1] = new Point2D(0.0, height);
069        vertices[2] = new Point2D(width, height);
070        vertices[3] = new Point2D(width, 0.0);
071
072        float[] points = new float[4 * 3];
073        for (int index = 0; index < 4; index++) {
074            points[index * 3] = (float)(vertices[index].getX() - width / 2.0);
075            points[index * 3 + 1] = (float)(vertices[index].getY() - height / 2.0);
076            points[index * 3 + 2] = 0.0f;
077        }
078
079        mesh = new TriangleMesh();
080        mesh.getPoints().setAll(points);
081        mesh.getTexCoords().setAll(TEX_COORDS);
082        mesh.getFaces().setAll(FACES);
083    }
084
085    /**
086     * ノードのシェイプを生成する
087     */
088    public Shape3D[] createShapes(MapNode node) {
089        Shape3D shape = new MeshView(mesh);
090        shape.setMaterial(material);
091        java.awt.geom.Point2D pos = node.getPosition();
092        shape.setTranslateX(pos.getX());
093        shape.setTranslateY(pos.getY());
094        shape.setTranslateZ(-node.getHeight() * panel.getVerticalScale());
095
096        Shape3D[] shapes = { shape };
097        return shapes;
098    }
099}
