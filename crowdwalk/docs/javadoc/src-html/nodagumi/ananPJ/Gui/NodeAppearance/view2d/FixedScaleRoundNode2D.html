001package nodagumi.ananPJ.Gui.NodeAppearance.view2d;
002
003import java.awt.Graphics2D;
004import java.awt.geom.Ellipse2D;
005import java.awt.geom.Point2D;
006import java.util.HashMap;
007
008import nodagumi.ananPJ.Gui.SimulationPanel2D;
009import nodagumi.ananPJ.NetworkMap.Node.MapNode;
010
011/**
012 * 固定スケールの丸いノードの 2D 表示
013 */
014public class FixedScaleRoundNode2D extends RoundNode2D {
015    /**
016     * コンストラクタ
017     */
018    public FixedScaleRoundNode2D() {}
019
020    /**
021     * ノードの半径を取得する
022     */
023    public double getRadius() {
024        double scale = panel.getDrawingScale();
025        return diameter / 2.0 / scale;
026    }
027
028    /**
029     * ノードを描画する
030     */
031    public void draw(MapNode node, Graphics2D g2) {
032        double radius = getRadius();
033        g2.setColor(color);
034        Point2D point = panel.getRotatedPoint(node);
035        double x = point.getX() - radius;
036        double y = point.getY() - radius;
037        g2.fill(new Ellipse2D.Double(x, y, radius * 2.0, radius * 2.0));
038    }
039}
