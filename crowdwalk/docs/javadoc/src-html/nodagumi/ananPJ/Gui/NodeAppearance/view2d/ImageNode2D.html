001package nodagumi.ananPJ.Gui.NodeAppearance.view2d;
002
003import java.awt.Graphics2D;
004import java.awt.geom.AffineTransform;
005import java.awt.geom.Point2D;
006import java.util.HashMap;
007
008import nodagumi.ananPJ.Gui.SimulationPanel2D;
009import nodagumi.ananPJ.NetworkMap.Node.MapNode;
010
011/**
012 * 画像ノードの 2D 表示
013 */
014public class ImageNode2D extends FixedScaleImageNode2D {
015    /**
016     * マップ上の画像の幅(m)
017     */
018    protected double widthOnMap;
019
020    /**
021     * マップ上の画像の高さ(m)
022     */
023    protected double heightOnMap;
024
025    /**
026     * コンストラクタ
027     */
028    public ImageNode2D() {}
029
030    /**
031     * 初期設定
032     */
033    public void init(SimulationPanel2D panel, HashMap parameters, NodeAppearance2D nodeAppearance) throws Exception {
034        super.init(panel, parameters, nodeAppearance);
035
036        widthOnMap = getDoubleParameter("width", 0.0);
037        heightOnMap = getDoubleParameter("height", 0.0);
038        if (widthOnMap <= 0.0 || heightOnMap <= 0.0) {
039            throw new Exception("Node appearance error: width or height is not specified correctly with ImageNode2D.");
040        }
041    }
042
043    /**
044     * ノードの半径を取得する
045     */
046    public double getRadius() {
047        return Math.max(widthOnMap, heightOnMap) / 2.0;
048    }
049
050    /**
051     * ノードを描画する
052     */
053    public void draw(MapNode node, Graphics2D g2) {
054        AffineTransform at = g2.getTransform();
055        Point2D point = panel.getRotatedPoint(node);
056        g2.translate(point.getX(), point.getY());
057        g2.rotate(Math.toRadians(panel.getAngle()));
058        g2.scale(widthOnMap / width, heightOnMap / height);
059        g2.drawImage(image, -width / 2, -height / 2, width, height, panel);
060        g2.setTransform(at);
061    }
062}
