489    public class TrianglePolygon extends TriangleMesh {
490        private final float[] texCoords = {
491            0, 0
492        };
493        private final int[] faces = {
494            0, 0, 1, 0, 2, 0
495        };
496
497        public TrianglePolygon(Point3D[] vertices) {
498            float[] points = new float[3 * 3];
499            for (int index = 0; index < 3; index++) {
500                points[index * 3] = (float)vertices[index].getX();
501                points[index * 3 + 1] = (float)vertices[index].getY();
502                points[index * 3 + 2] = (float)(vertices[index].getZ() * verticalScale);
503            }
504            this.getPoints().setAll(points);
505            this.getTexCoords().setAll(texCoords);
506            this.getFaces().setAll(faces);
507        }
508    }

541    public SimulationPanel3D(int panelWidth, int panelHeight, NetworkMap networkMap, double verticalScale, CrowdWalkPropertiesHandler properties, ArrayList<GsiTile> mapTiles, Coastline coastline, ArrayList<HashMap> linkAppearanceConfig, ArrayList<HashMap> nodeAppearanceConfig) {
542        this.networkMap = networkMap;
543        this.verticalScale = verticalScale;
544        this.properties = properties;
545        legacy = CrowdWalkLauncher.legacy || properties.isLegacy();
546
547        init(linkAppearanceConfig, nodeAppearanceConfig);
548
549        // シーングラフの構築
550        SubScene networkMapScene = createNetworkMapScene(panelWidth, panelHeight, mapTiles, coastline);
551        getChildren().add(networkMapScene);
552        getChildren().add(createLogoPane());
553        pickingPane = createPickingPane();
554        getChildren().add(pickingPane);
555        getChildren().add(createStatusPane());
556
557        // StackPane のリサイズに SubScene を追従させる
558        // ※高さのリサイズ時にサブシーンのスケールが追従してしまうのは JavaFX の仕様(回避方法は見つからない)
559        networkMapScene.heightProperty().bind(this.heightProperty());
560        networkMapScene.widthProperty().bind(this.widthProperty());
561        networkMapScene.setManaged(false);
562
563        // シミュレーションビューのマウス操作定義
564        final SimulationPanel3D pane = this;
565        networkMapScene.setOnMousePressed(event -> {
566            lastMouseX = event.getX();
567            lastMouseY = event.getY();
568        });
569        networkMapScene.setOnMouseDragged(event -> {
570            if (isViewPointOperationEnabled()) {
571                double dx = event.getX() - lastMouseX;
572                double dy = event.getY() - lastMouseY;
573                double sightHeight = Math.tan((VIEW_ANGLE / 2.0) * Math.PI / 180.0) * elevationTranslate.getZ() * 2.0 / zoomScale;
574                switch (event.getButton()) {
575                case PRIMARY:
576                    if (event.isControlDown()) {
577                        // 縦横スクロール
578                        scroll(dx * sightHeight / pane.getHeight(), dy * sightHeight / pane.getHeight());
579                    } else {
580                        // マウスの縦移動で俯瞰角度を変える
581                        double angle = mapRotateX.getAngle() + dy / DRAG_SENSITIVITY;
582                        // 自然な角度で見下ろせる範囲に限定する(Shift キー押下で解除)
583                        if (! event.isShiftDown()) {
584                            if (angle > 0.0) {
585                                angle = 0.0;
586                            } else if (angle < -90.0) {
587                                angle = -90.0;
588                            }
589                        }
590                        mapRotateX.setAngle(angle);
591
592                        // マウスの横移動で地図を回転させる
593                        dx /= DRAG_SENSITIVITY;
594                        if (event.getY() > pane.getHeight() / 2) {
595                            dx = -dx;
596                        }
597                        angle = mapRotateZ.getAngle() + dx;
598                        if (angle > 180.0) {
599                            angle -= 360.0;
600                        } else if (angle < -180.0) {
601                            angle += 360.0;
602                        }
603                        mapRotateZ.setAngle(angle);
604                    }
605                    break;
606                case SECONDARY:
607                    // 縦横スクロール
608                    scroll(dx * sightHeight / pane.getHeight(), dy * sightHeight / pane.getHeight());
609                    break;
610                }
611            }
612            lastMouseX = event.getX();
613            lastMouseY = event.getY();
614        });
615        networkMapScene.setOnScroll(event -> {
616            // 拡大・縮小
617            if (! isViewPointOperationEnabled() || event.getDeltaY() == 0.0) {
618                return;
619            } else if (event.getDeltaY() > 0.0) {
620                zoomScale *= 1.1;
621            } else {
622                zoomScale *= 0.9;
623            }
624            updateZoomScale();
625        });
626    }

628    /**
629     * 初期設定
630     */
631    public void init(ArrayList<HashMap> linkAppearanceConfig, ArrayList<HashMap> nodeAppearanceConfig) {
632        areas = networkMap.getAreas();
633
634        // リンクの分別
635        for (MapLink link : networkMap.getLinks()) {
636            if (link.getFrom() == link.getTo()) {
637                // 不正なリンクは無視する
638                Itk.logWarn_("Looped link found", "ID=" + link.ID);
639                continue;
640            }
641            if (legacy) {
642                if (link.hasSubTag("POLYGON")) {
643                    polygonLinks.add(link);
644                } else if (link.hasSubTag("STRUCTURE")) {
645                    structureLinks.add(link);
646                } else {
647                    regularLinks.add(link);
648                }
649            } else {
650                regularLinks.add(link);
651            }
652        }
653
654        // 実在の地形ベースのマップでなければセンタリングマージンを付加する
655        setMarginAdded(regularLinks.size() < MINIMUM_REAL_MAP_LINKS);
656
657        // ホバー用のマテリアル
658        hoverMaterial = new PhongMaterial();
659        hoverMaterial.setDiffuseColor(HOVER_COLOR);
660
661        try {
662            // Link appearance の準備
663            String[] exclusionTags = new String[0];
664            if (legacy) {
665                exclusionTags = new String[]{"POLYGON"};
666            }
667            EdgePoints edgePoints = new EdgePoints(networkMap, exclusionTags);
668            for (HashMap parameters : linkAppearanceConfig) {
669                LinkAppearance3D appearance = new LinkAppearance3D(this, parameters, edgePoints);
670                if (! appearance.isValidFor3D()) {
671                    Itk.logFatal("Link appearance error", "3D view not defined");
672                    Itk.quitByError() ;
673                }
674                linkAppearances.add(appearance);
675            }
676
677            // Node appearance の準備
678            for (HashMap parameters : nodeAppearanceConfig) {
679                NodeAppearance3D appearance = new NodeAppearance3D(this, parameters);
680                if (! appearance.isValidFor3D()) {
681                    Itk.logFatal("Node appearance error", "3D view not defined");
682                    Itk.quitByError() ;
683                }
684                nodeAppearances.add(appearance);
685            }

830    /**
831     * ポリゴン表示用のリンクを元にタグ別のポリゴン座標リストを作成する.
832     *
833     * ※ legacy モード時のみ有効
834     */
835    public HashMap<String, ArrayList<Point3D>> createPolygonPoints(MapLinkTable links) {
836        HashMap<String, ArrayList<Point3D>> polygonPoints = new HashMap();
837        if (! legacy) {
838            return polygonPoints;
839        }
840
841        // タグごとにリンクを選別する
842        HashMap<String, MapLinkTable> polygonLinksByTag = new HashMap();
843        for (MapLink link : links) {
844            for (String tag : link.getTags()) {
845                if (tag.contains("POLYGON")) {
846                    MapLinkTable _links = polygonLinksByTag.get(tag);
847                    if (_links == null) {
848                        _links = new MapLinkTable();
849                        polygonLinksByTag.put(tag, _links);
850                    }
851                    _links.add(link);
852                }
853            }
854        }
855        // 選別したリンクを元に座標リストを作成する
856        for (String tag : polygonLinksByTag.keySet()) {
857            // 通過点順に MapNode を収集する
858            MapLinkTable _links = polygonLinksByTag.get(tag);
859            MapLink currentLink = _links.get(0);
860            MapNode start = currentLink.getFrom();
861            MapNode next = currentLink.getTo();
862            MapNodeTable polygonNodes = new MapNodeTable();
863            polygonNodes.add(start);
864            while (next != start) {
865                polygonNodes.add(next);
866                for (MapLink link : _links) {
867                    if (link == currentLink) {
868                        continue;
869                    }
870                    if (next == link.getFrom() || next == link.getTo()) {
871                        currentLink = link;
872                        if (currentLink.getFrom() != next)
873                            next = currentLink.getFrom();
874                        else
875                            next = currentLink.getTo();
876                        break;
877                    }
878                }
879            }
880            // 収集した MapNode の座標リストを作成する
881            ArrayList<Point3D> points = new ArrayList();
882            for (MapNode node : polygonNodes) {
883                points.add(new Point3D(node.getX(), node.getY(), -node.getHeight()));
884            }
885            polygonPoints.put(tag, points);
886        }
887        return polygonPoints;
888    }

938     * マップをセンタリングする
939     */
940    public void centering(boolean withScaling, double paneWidth, double paneHeight) {
941        Rectangle2D bounds = networkMap.calcRectangle();
942        Point2D centerOfMap = calcCenter(centeringByNodeAverage);
943
944        double length = bounds.getHeight();
945        if (bounds.getWidth() / bounds.getHeight() >= paneWidth / paneHeight) {
946            length = bounds.getWidth() * paneHeight / paneWidth;
947            if (isMarginAdded() && paneWidth > CENTERING_MARGIN) {
948                length *= (paneWidth + CENTERING_MARGIN) / paneWidth;
949            }
950        } else {
951            if (isMarginAdded() && paneHeight > CENTERING_MARGIN) {
952                length *= (paneHeight + CENTERING_MARGIN) / paneHeight;
953            }
954        }
955        if (isMarginAdded() && paneWidth > CENTERING_MARGIN && paneHeight > CENTERING_MARGIN) {
956            double width = Math.min(paneWidth, paneHeight);
957            length *= (width + CENTERING_MARGIN) / width;
958        }
959        double z = (length / 2.0) / Math.tan((VIEW_ANGLE / 2.0) * Math.PI / 180.0);
960
961        elevationTranslate.setZ(z);
962        mapRotateX.setAngle(0.0);
963        mapRotateZ.setAngle(0.0);
964        mapTranslate.setX(-centerOfMap.getX());
965        mapTranslate.setY(-centerOfMap.getY());
966        mapScale.setPivotX(centerOfMap.getX());
967        mapScale.setPivotY(centerOfMap.getY());
968        if (withScaling) {
969            setZoomScale(1.0);
970        }
971    }

975     *
976     * TODO: NetworkMap クラスに移動する
977     */
978    public Point2D calcCenter(boolean nodeAverage) {
979        Rectangle2D bounds = networkMap.calcRectangle();
980        double x = 0.0;
981        double y = 0.0;
982        if (nodeAverage) {
983            // 元々の CrowdWalk で使われていた方法
984            int count = 0;
985            for (MapNode node : networkMap.getNodes()) {
986                java.awt.geom.Point2D pos = node.getPosition();
987                x += pos.getX();
988                y += pos.getY();
989                count++;
990            }
991            x /= count;
992            y /= count;
993        } else {
994            x = bounds.getMinX() + bounds.getWidth() / 2.0;
995            y = bounds.getMinY() + bounds.getHeight() / 2.0;
996        }
997        return new Point2D(x, y);
998    }

1142     * ホバーを表示する
1143     */
1144    private synchronized void hoverOn(String id, double x, double y) {
1145        currentHoverObject = null;
1146
1147        String[] words = id.split(" ");
1148        if (words[0].equals("Node")) {
1149            hoverType = NType.NODE;
1150            currentHoverObject = networkMap.getObject(words[1]);
1151        } else if (words[0].equals("Link")) {
1152            hoverType = NType.LINK;
1153            currentHoverObject = networkMap.getObject(words[1]);
1154        } else if (words[0].equals("Area")) {
1155            hoverType = NType.AREA;
1156            currentHoverObject = areaMap.get(words[1]);
1157        } else if (words[0].equals("Agent")) {
1158            hoverType = NType.AGENT;
1159            currentHoverObject = agentMap.get(words[1]);
1160        }
1161        if (currentHoverObject == null) {
1162            hoverType = null;
1163            return;
1164        }
1165
1166        Shape3D shape = null;
1167        switch (hoverType) {
1168        case NODE:
1169            Shape3D[] shapes = nodeShapes.get(currentHoverObject);
1170            if (shapes == null) {
1171                currentHoverObject = null;
1172                hoverType = null;
1173                return;
1174            }
1175            currentMaterials.clear();
1176            for (Shape3D _shape : shapes) {
1177                currentMaterials.add(_shape.getMaterial());
1178                _shape.setMaterial(hoverMaterial);
1179            }
1180            break;
1181        case LINK:
1182            shape = pickingLinkShapes.get(currentHoverObject);
1183            break;
1184        case AREA:
1185            shape = areaShapes.get(currentHoverObject);
1186            break;
1187        case AGENT:
1188            shape = agentShapes.get(currentHoverObject);
1189            break;
1190        }
1191        if (shape == null) {
1192            currentHoverObject = null;
1193            hoverType = null;
1194            return;
1195        }
1196
1197        switch (hoverType) {
1198        case LINK:
1199            currentMaterial = shape.getMaterial();
1200            shape.setMaterial(hoverMaterial);
1201            break;
1202        case AREA:
1203            PhongMaterial material = (PhongMaterial)shape.getMaterial();
1204            material.setDiffuseColor(HOVER_COLOR);
1205            break;
1206        case AGENT:
1207            AgentBase agent = (AgentBase)currentHoverObject;
1208            getAgentView(agent).switchToHover(agent, shape);
1209            break;
1210        }
1211
1212        // ID + タグを表示
1213        Point2D panePosition = pickingPane.localToScreen(0.0, 0.0);
1214        pickedObjectLabel.relocate(x - panePosition.getX() + 16, y - panePosition.getY());
1215        String tag = currentHoverObject.getTagString();
1216        pickedObjectLabel.setText(currentHoverObject.getID() + (tag.isEmpty() ? "" : " : " + tag));
1217        pickedObjectLabel.setVisible(true);
1218    }

1921     * エリアの表示色を返す
1922     */
1923    public Color getAreaColor(double currentLevel, double normalizedLevel) {
1924        double density = 0.0;
1925        if (pollutionColorSaturation == 0.0) {
1926            density = Math.min(normalizedLevel, 1.0);
1927        } else {
1928            density = Math.min(currentLevel / pollutionColorSaturation, 1.0);
1929        }
1930        density = Math.max(density, 0.0);   // マイナスの水位はサポートしない
1931        double opacity = Math.min(density / 1.5, 1.0);
1932
1933        Color color = null;
1934        switch (obstructerDisplay) {
1935        case RED:
1936            color = new Color(density, 0.0, 0.0, opacity);
1937            break;
1938        case BLUE:
1939            color = new Color(0.0, 0.0, density, opacity);
1940            break;
1941        case HSV:
1942            double hue = Math.min((1.0 - density) * 0.65, 1.0);
1943            color = Color.hsb(hue, 1.0, 1.0);
1944            break;
1945        case ORANGE:
1946            color = new Color(1.0, Math.min(1.0 - density / 2.0, 1.0), 0.0, opacity);
1947            break;
1948        }
1949        return color;
1950    }
