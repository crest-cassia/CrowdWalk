113    private class TimeSpinnerValueFactory extends SpinnerValueFactory<LocalTime> {
114        private LocalTime maxValue;
115        private LocalTime minValue;
116
117        public TimeSpinnerValueFactory(LocalTime min, LocalTime max, LocalTime initialValue) {
118            minValue = min;
119            maxValue = max;
120            setValue(initialValue);
121            setConverter(new LocalTimeStringConverter(DateTimeFormatter.ofPattern("HH:mm:ss"), null));
122        }
123
124        public void decrement(int steps) {
125            LocalTime time = (LocalTime)getValue();
126            if (time.isAfter(minValue)) {
127                setValue(time.minusSeconds(steps));
128            }
129        }
130
131        public void increment(int steps) {
132            LocalTime time = (LocalTime)getValue();
133            if (time.isBefore(maxValue)) {
134                setValue(time.plusSeconds(steps));
135            }
136        }
137    }

358    private void init(CrowdWalkPropertiesHandler properties) {
359        deferFactor = launcher.getDeferFactor();
360        areas = launcher.getMap().getAreas();
361
362        if (properties.getPropertiesFile() != null) {
363            try {
364                String filePath = properties.getFilePath("camera_file", null);
365                if (filePath != null) {
366                    if (filePath.toLowerCase().endsWith(".json")) {
367                        loadCamerawork(filePath);
368                    } else {
369                        Itk.logError("camera_file should be a json file with .json suffix." ) ;
370                        Itk.quitByError() ;
371                    }
372                }
373            } catch (Exception e) {
374                Itk.quitWithStackTrace(e) ;
375            }
376        }
377
378        // ヘルプ画面の準備
379
380        MutableDataSet options = new MutableDataSet();
381        options.setFrom(ParserEmulationProfile.MARKDOWN);
382        options.set(Parser.EXTENSIONS, Arrays.asList(
383            TablesExtension.create()
384        ));
385        parser = Parser.builder(options).build();
386        renderer = HtmlRenderer.builder(options).build();
387
388        webView.setOnKeyPressed(event -> {
389            if (event.isControlDown()) {
390                if (event.getCode() == KeyCode.W) {
391                    helpStage.close();
392                } else if (event.getCode() == KeyCode.PLUS || event.getCode() == KeyCode.UP) {
393                    helpZoom += 0.1;
394                    webView.setZoom(helpZoom);
395                } else if (event.getCode() == KeyCode.MINUS || event.getCode() == KeyCode.DOWN) {
396                    helpZoom -= 0.1;
397                    webView.setZoom(helpZoom);
398                } else if (event.getCode() == KeyCode.DIGIT0) {
399                    helpZoom = 1.0;
400                    webView.setZoom(helpZoom);
401                }
402            }
403        });
404
405        Button okButton = new Button("  OK  ");
406        okButton.setOnAction(event -> helpStage.close());
407        BorderPane buttonPane = new BorderPane();
408        buttonPane.setPadding(new Insets(4, 8, 4, 8));
409        buttonPane.setRight(okButton);
410
411        BorderPane borderPane = new BorderPane();
412        borderPane.setCenter(webView);
413        borderPane.setBottom(buttonPane);
414
415        Scene scene = new Scene(borderPane);
416        scene.getStylesheets().add("stylesheet.css");
417        helpStage.setScene(scene);
418    }

500    private Pane createControlButtonPane() {
501        Image start_icon = new Image(getClass().getResourceAsStream("/img/start.png"));
502        startIcon = new ImageView(start_icon);
503        startIcon.setViewport(new Rectangle2D(-10, 0, 32, 12));
504        Image pause_icon = new Image(getClass().getResourceAsStream("/img/pause.png"));
505        pauseIcon = new ImageView(pause_icon);
506        pauseIcon.setViewport(new Rectangle2D(-10, 0, 32, 12));
507        Image step_icon = new Image(getClass().getResourceAsStream("/img/step.png"));
508        ImageView stepIcon = new ImageView(step_icon);
509        stepIcon.setViewport(new Rectangle2D(-7, 0, 32, 12));
510
511        start_button = new ToggleButton();
512        start_button.setGraphic(startIcon);
513        start_button.setOnAction(e -> {
514            if (start_button.isSelected()) {
515                start_button.setGraphic(pauseIcon);
516                verticalScaleControl.setDisable(true);
517                resetButton.setDisable(true);
518                launcher.start();
519            } else {
520                start_button.setGraphic(startIcon);
521                launcher.pause();
522                verticalScaleControl.setDisable(false);
523                resetButton.setDisable(false);
524            }
525            update_buttons();
526        });
527
528        step_button = new Button();
529        step_button.setGraphic(stepIcon);
530        step_button.setOnAction(e -> {
531            start_button.setDisable(true);
532            step_button.setDisable(true);
533            // ボタンクリックでいきなりプログラムが終了することがないようにするため
534            if (exit_with_simulation_finished_cb.isSelected()) {
535                exit_with_simulation_finished_cb.fire();
536            }
537
538            // 別スレッドでステップ実行
539            Task<Boolean> task = new Task<Boolean>() {
540                @Override
541                protected Boolean call() throws Exception {
542                    launcher.step();
543                    Platform.runLater(() -> start_button.setDisable(false));
544                    return true; 
545                }
546            };
547            Thread thread = new Thread(task);
548            thread.setDaemon(true);
549            thread.start();
550        });
551
552        Label label = new Label("wait");
553
554        ScrollBar deferFactorControl = new ScrollBar();
555        deferFactorControl.setPrefSize(200, 20);
556        deferFactorControl.setMax(300);
557        deferFactorControl.setValue(deferFactor);
558        deferFactorControl.valueProperty().addListener((ov, oldValue, newValue) -> {
559            setSimulationDeferFactor(newValue.intValue());
560            simulationDeferFactorValue.setText("" + deferFactor);
561        });
562
563        simulationDeferFactorValue = new Label("" + deferFactor);
564        simulationDeferFactorValue.setAlignment(Pos.CENTER_RIGHT);
565
566        FlowPane innerPane = new FlowPane(12, 0);
567        BorderStroke borderStroke = new BorderStroke(Color.LIGHTGREY, BorderStrokeStyle.SOLID, new CornerRadii(2), BorderWidths.DEFAULT);
568        innerPane.setBorder(new Border(borderStroke));
569        innerPane.setPadding(new Insets(8, 0, 8, 0));
570        innerPane.setAlignment(Pos.CENTER);
571        innerPane.getChildren().addAll(start_button, step_button, label, deferFactorControl, simulationDeferFactorValue);
572
573        StackPane pane = new StackPane();
574        pane.setPadding(new Insets(0, 8, 0, 8));
575        pane.getChildren().add(innerPane);
576
577        return pane;
578    }

1480    /**
1481     * カメラワークを更新する
1482     */
1483    public void updateCamerawork(SimTime updateTime) {
1484        if (replayCheckBox.isSelected() && camerawork.size() > 0) {
1485            double time = updateTime.getRelativeTime();
1486            CameraShot last_camera = null;
1487            CameraShot next_camera = null;
1488            for (CameraShot camera : camerawork) {
1489                if (camera.time > time) {
1490                    next_camera = camera;
1491                    break;
1492                }
1493                last_camera = camera;
1494            }
1495            // 最初の CameraShot の time が 0.0 ではなく、まだその time に達していない場合
1496            if (last_camera == null) {
1497                if (next_camera.agentSize != getAgentSize()) {
1498                    agent_size_control.setValue(next_camera.agentSize);
1499                }
1500                panel.updateTransform(next_camera.translate, next_camera.pivot, next_camera.rotateX, next_camera.rotateZ, next_camera.zoom);
1501            }
1502            // 最後の CameraShot とそれ以降
1503            else if (next_camera == null) {
1504                if (last_camera.time < time && ! forceUpdateCamerawork) {
1505                    return;
1506                }
1507                if (last_camera.agentSize != getAgentSize()) {
1508                    agent_size_control.setValue(last_camera.agentSize);
1509                }
1510                panel.updateTransform(last_camera.translate, last_camera.pivot, last_camera.rotateX, last_camera.rotateZ, last_camera.zoom);
1511            }
1512            // 上の条件以外
1513            else {
1514                if (time > last_camera.time && ! forceUpdateCamerawork) {
1515                    double nextCameraTime = next_camera.time;   // 一時退避
1516                    next_camera.time = last_camera.time;
1517                    if (last_camera.equals(next_camera)) {
1518                        next_camera.time = nextCameraTime;
1519                        return;
1520                    }
1521                    next_camera.time = nextCameraTime;
1522                }
1523                double ratio = (time - last_camera.time) / (next_camera.time - last_camera.time);
1524
1525                double agent_size = (next_camera.agentSize - last_camera.agentSize) * ratio + last_camera.agentSize;
1526                if (agent_size != getAgentSize()) {
1527                    agent_size_control.setValue(agent_size);
1528                }
1529
1530                Point3D translate = next_camera.translate.subtract(last_camera.translate).multiply(ratio).add(last_camera.translate);
1531                Point3D pivot = next_camera.pivot.subtract(last_camera.pivot).multiply(ratio).add(last_camera.pivot);
1532                double rotateX = (next_camera.rotateX - last_camera.rotateX) * ratio + last_camera.rotateX;
1533                double drot = next_camera.rotateZ - last_camera.rotateZ;
1534                // 補完する角度が180度を超えている場合は、(360度- 角度)の遷移と判断する
1535                if (drot > 180.0 || drot <= -180.0) {
1536                    if (Math.signum(drot) == 1.0) {
1537                        drot -= 360.0;
1538                    } else {
1539                        drot += 360.0;
1540                    }
1541                }
1542                double rotateZ = drot * ratio + last_camera.rotateZ;
1543                double zoom_scale = (next_camera.zoom - last_camera.zoom) * ratio + last_camera.zoom;
1544                panel.updateTransform(translate, pivot, rotateX, rotateZ, zoom_scale);
1545            }
1546            forceUpdateCamerawork = false;
1547        }
1548    }
