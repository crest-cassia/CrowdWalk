003import java.awt.Color;
004import java.io.InputStream;
005import java.math.BigDecimal;
006import java.util.LinkedHashMap;
007import java.util.Map;
008
009import net.arnx.jsonic.JSON;
010
011import nodagumi.ananPJ.NetworkMap.Link.*;
012import nodagumi.Itk.Itk ;
013
014/**
015 * リンクの表示スタイル(幅, 色, 透明度)をタグ別に指定するために使用するクラス
016 */
017public class LinkAppearance2D {
018    public static final String NONE_TAG = "__NONE__";
019
020    public boolean widthFixed = false;
021    public double widthRatio = 1.0;
022    public Color color = Color2D.DEFAULT_LINK_COLOR;
023    public float transparency = 0.75f;
024
025    public LinkAppearance2D(Boolean _widthFixed, BigDecimal _widthRatio,
026            String colorName, BigDecimal _transparency, LinkAppearance2D defaultValue) {
027        if (defaultValue != null) {
028            widthFixed = defaultValue.widthFixed;
029            widthRatio = defaultValue.widthRatio;
030            color = defaultValue.color;
031            transparency = defaultValue.transparency;
032        }
033        if (_widthFixed != null) {
034            widthFixed = _widthFixed;
035        }
036        if (_widthRatio != null) {
037            widthRatio = _widthRatio.doubleValue();
038        }
039        if (colorName != null) {
040            color = Color2D.getColor(colorName);
041        }
042        if (_transparency != null) {
043            transparency = _transparency.floatValue();
044        }
045        float[] compArray = color.getComponents(null);
046        color = new Color(compArray[0], compArray[1], compArray[2], 1.0f - transparency);
047    }
048
049    public static void load(InputStream is,
050            LinkedHashMap<String, LinkAppearance2D> linkAppearances) throws Exception {
051        /* [2014.12.27] I.Noda. to adapt new version os JSONIC. */
052        //Map<String, Object> map = (Map<String, Object>)JSON.decode(is);
053        JSON json = new JSON(JSON.Mode.TRADITIONAL);
054        Map<String, Object> map = (Map<String, Object>)json.parse(is);
055        LinkAppearance2D noneTagAppearance = null;
056        for (Map.Entry<String, Object> entry : map.entrySet()) {
057            String tag = Itk.intern(entry.getKey());
058            Map<String, Object> items = (Map<String, Object>)entry.getValue();
059            BigDecimal widthRatio = (BigDecimal)items.get("width_ratio");
060            if (widthRatio == null) {
061                widthRatio = (BigDecimal)items.get("width");
062            }
063            LinkAppearance2D appearance = new LinkAppearance2D(
064                (Boolean)items.get("width_fixed"),
065                widthRatio,
066                (String)items.get("color"),
067                (BigDecimal)items.get("transparency"),
068                linkAppearances.get(tag)
069            );
070            if (tag.equals(NONE_TAG)) {
071                noneTagAppearance = appearance;
072            } else {
073                linkAppearances.put(tag, appearance);
074            }
075        }
076        // LinkedHashMap の最後に追加
077        if (noneTagAppearance != null) {
078            linkAppearances.put(NONE_TAG, noneTagAppearance);
079        }
080    }
081
082    public static LinkAppearance2D getAppearance(
083            LinkedHashMap<String, LinkAppearance2D> linkAppearances, MapLink link) {
084        for (Map.Entry<String, LinkAppearance2D> entry : linkAppearances.entrySet()) {
085            String tag = entry.getKey();
086            if (tag.equals(NONE_TAG) || link.hasTag(tag)) {
087                return entry.getValue();
088            }
089        }
090        return null;
091    }
092}
