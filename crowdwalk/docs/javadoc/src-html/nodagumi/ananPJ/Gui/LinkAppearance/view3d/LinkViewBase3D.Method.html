001package nodagumi.ananPJ.Gui.LinkAppearance.view3d;
002
003import java.util.HashMap;
004
005import javafx.geometry.Point2D;
006import javafx.geometry.Point3D;
007import javafx.scene.paint.Color;
008import javafx.scene.paint.PhongMaterial;
009import javafx.scene.shape.CullFace;
010import javafx.scene.shape.DrawMode;
011import javafx.scene.shape.MeshView;
012import javafx.scene.shape.Shape3D;
013
014import nodagumi.ananPJ.Gui.FxColor;
015import nodagumi.ananPJ.Gui.LinkAppearance.EdgePoints;
016import nodagumi.ananPJ.Gui.SimulationPanel3D;
017import nodagumi.ananPJ.NetworkMap.Link.MapLink;
018import nodagumi.ananPJ.NetworkMap.Node.MapNode;
019import nodagumi.ananPJ.misc.JsonicHashMapGetter;
020
021/**
022 * リンク 3D 描画のベースクラス
023 */
024public abstract class LinkViewBase3D extends JsonicHashMapGetter {
025    /**
026     * リンクの描画方法
027     */
028    public enum Method {
029        FILLING,
030        BORDERING
031    };
032
033    /**
034     * リンクを擬似的な1ドット幅のラインで表示する時の幅(1mm)
035     */
036    public static double THIN_LINK_WIDTH = 0.001;
037
038    /**
039     * ピッキング用マテリアル
040     */
041    private static PhongMaterial pickingMaterial;
042
043    /**
044     * 3D シミュレーションパネル
045     */
046    protected SimulationPanel3D panel;
047
048    /**
049     * リンクの 3D 表示の定義情報
050     */
051    protected LinkAppearance3D linkAppearance;
052
053    /**
054     * リンクのマテリアル
055     */
056    protected PhongMaterial material;
057
058    /**
059     * リンクの描画方法
060     */
061    protected Method method = Method.FILLING;
062
063    static {
064        pickingMaterial = new PhongMaterial();
065        pickingMaterial.setDiffuseColor(Color.TRANSPARENT);
066    }
067
068    /**
069     * 初期設定
070     */
071    public void init(SimulationPanel3D panel, HashMap parameters, LinkAppearance3D linkAppearance) throws Exception {
072        this.panel = panel;
073        this.linkAppearance = linkAppearance;
074        setParameters(parameters);
075
076        String colorName = getStringParameter("color", "DEFAULT_LINK_COLOR");
077        Color color = FxColor.getColor(colorName);
078        if (color == null) {
079            color = FxColor.DEFAULT_LINK_COLOR;
080        }
081        double transparency = getDoubleParameter("transparency", 0.0);
082        color = new Color(color.getRed(), color.getGreen(), color.getBlue(), 1.0 - transparency);
083
084        material = new PhongMaterial();
085        material.setDiffuseColor(color);
086
087        String methodName = getStringParameter("method", method.name());
088        method = Method.valueOf(methodName.toUpperCase());
089    }
090
091    /**
092     * 最終的なリンクの幅を取得する
093     */
094    public abstract double getWidth(MapLink link);
095
096    /**
097     * リンクをポリゴン表示するための頂点座標を求める
098     */
099    public Point3D[] calcVertices(MapNode from, MapNode to, double width, double dz) {
100        Point3D p1 = new Point3D(to.getX() - from.getX(), to.getY() - from.getY(), 0);
101        Point3D p2 = p1.normalize().crossProduct(0, 0, width / 2.0);
102        double dx = p2.getX();
103        double dy = p2.getY();
104
105        Point3D[] vertices = new Point3D[4];
106        vertices[0] = new Point3D(from.getX() + dx, from.getY() + dy, -from.getHeight() + dz);
107        vertices[1] = new Point3D(from.getX() - dx, from.getY() - dy, -from.getHeight() + dz);
108        vertices[3] = new Point3D(to.getX() - dx, to.getY() - dy, -to.getHeight() + dz);
109        vertices[2] = new Point3D(to.getX() + dx, to.getY() + dy, -to.getHeight() + dz);
110        return vertices;
111    }
112
113    /**
114     * リンクをポリゴン表示するための頂点座標を求める
115     */
116    public Point3D[] calcVertices(Point3D point1, Point3D point2, double width, double dz) {
117        Point3D p1 = new Point3D(point2.getX() - point1.getX(), point2.getY() - point1.getY(), 0);
118        Point3D p2 = p1.normalize().crossProduct(0, 0, width / 2.0);
119        double dx = p2.getX();
120        double dy = p2.getY();
121
122        Point3D[] vertices = new Point3D[4];
123        vertices[0] = new Point3D(point1.getX() + dx, point1.getY() + dy, -point1.getZ() + dz);
124        vertices[1] = new Point3D(point1.getX() - dx, point1.getY() - dy, -point1.getZ() + dz);
125        vertices[3] = new Point3D(point2.getX() - dx, point2.getY() - dy, -point2.getZ() + dz);
126        vertices[2] = new Point3D(point2.getX() + dx, point2.getY() + dy, -point2.getZ() + dz);
127        return vertices;
128    }
129
130    /**
131     * 幅のある実線(中を塗りつぶした長方形シェイプ)を生成する
132     */
133    public Shape3D createSingleLine(MapLink link) {
134        Point3D[] vertices = calcVertices(link.getFrom(), link.getTo(), getWidth(link), 0.0);
135        Shape3D shape = new MeshView(panel.new QuadPolygon(vertices));
136        shape.setDrawMode(DrawMode.FILL);
137        shape.setMaterial(material);
138        return shape;
139    }
140
141    /**
142     * 実線(線分と見分けが付かない超鋭角の三角形シェイプ)を生成する
143     */
144    public Shape3D createSolidLine(Point3D point1, Point3D point2) {
145        Point3D[] vertices = calcVertices(point1, point2, THIN_LINK_WIDTH, 0.0);
146        Shape3D shape = new MeshView(panel.new TrianglePolygon(vertices));
147        shape.setDrawMode(DrawMode.LINE);
148        shape.setCullFace(CullFace.NONE);    // これがないと時々表示が消える
149        shape.setMaterial(material);
150        return shape;
151    }
152
153    /**
154     * リンクの縁取りラインを生成する
155     */
156    public Shape3D[] createBorderLine(MapLink link) {
157        EdgePoints.Points points = linkAppearance.getEdgePoints(link);
158        if (points == null) {
159            return null;
160        }
161        Point2D a1 = points.fromNodeLeftPoint;
162        Point2D a2 = points.toNodeRightPoint;
163        Point2D b1 = points.fromNodeRightPoint;
164        Point2D b2 = points.toNodeLeftPoint;
165        if (a1 == null || a2 == null || b1 == null || b2 == null) {
166            return null;
167        }
168
169        Point3D point1 = new Point3D(a1.getX(), a1.getY(), link.getFrom().getHeight());
170        Point3D point2 = new Point3D(a2.getX(), a2.getY(), link.getTo().getHeight());
171        Shape3D borderLine1 = createSolidLine(point1, point2);
172
173        Point3D point3 = new Point3D(b1.getX(), b1.getY(), link.getFrom().getHeight());
174        Point3D point4 = new Point3D(b2.getX(), b2.getY(), link.getTo().getHeight());
175        Shape3D borderLine2 = createSolidLine(point3, point4);
176
177        Shape3D[] shapes = { borderLine1, borderLine2 };
178        return shapes;
179    }
180
181    /**
182     * リンクのシェイプを生成する
183     */
184    public Shape3D[] createShapes(MapLink link) {
185        switch (method) {
186        case FILLING:
187            Shape3D[] shapes = { createSingleLine(link) };
188            return shapes;
189        case BORDERING:
190            return createBorderLine(link);
191        }
192        return null;
193    }
194
195    /**
196     * ピッキング用の透明シェイプを生成する
197     */
198    public Shape3D createPickingShape(MapLink link) {
199        Point3D[] vertices = calcVertices(link.getFrom(), link.getTo(), link.getWidth(), -0.1); // 10cm 上方に配置する
200        Shape3D shape = new MeshView(panel.new QuadPolygon(vertices));
201        shape.setDrawMode(DrawMode.FILL);
202        shape.setMaterial(pickingMaterial);
203        return shape;
204    }
205}
