001package nodagumi.ananPJ.Gui.LinkAppearance.view2d;
002
003import java.util.HashMap;
004
005import nodagumi.ananPJ.Gui.SimulationPanel2D;
006import nodagumi.ananPJ.NetworkMap.Link.MapLink;
007
008/**
009 * 固定幅リンクの 2D 表示
010 */
011public class FixedWidth2D extends LinkViewBase2D {
012    /**
013     * リンクの幅
014     */
015    protected double width = 1.0;
016
017    /**
018     * コンストラクタ
019     */
020    public FixedWidth2D() {}
021
022    /**
023     * 初期設定
024     */
025    public void init(SimulationPanel2D panel, HashMap parameters, LinkAppearance2D linkAppearance) throws Exception {
026        super.init(panel, parameters, linkAppearance);
027        width = getDoubleParameter("width", width);
028    }
029
030    /**
031     * 最終的なリンクの幅を取得する
032     */
033    public double getWidth(MapLink link) {
034        return width;
035    }
036}
