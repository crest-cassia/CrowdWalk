001package nodagumi.ananPJ.Gui.LinkAppearance.view2d;
002
003import java.awt.BasicStroke;
004import java.awt.Color;
005import java.awt.Graphics2D;
006import java.awt.geom.GeneralPath;
007import java.awt.geom.Line2D;
008import java.awt.geom.Point2D;
009import java.util.HashMap;
010
011import nodagumi.ananPJ.Gui.Color2D;
012import nodagumi.ananPJ.Gui.LinkAppearance.EdgePoints;
013import nodagumi.ananPJ.Gui.SimulationFrame2D;
014import nodagumi.ananPJ.Gui.SimulationPanel2D;
015import nodagumi.ananPJ.NetworkMap.Link.MapLink;
016import nodagumi.ananPJ.NetworkMap.Node.MapNode;
017import nodagumi.ananPJ.misc.JsonicHashMapGetter;
018
019/**
020 * リンク 2D 描画のベースクラス
021 */
022public abstract class LinkViewBase2D extends JsonicHashMapGetter {
023    /**
024     * リンクの描画方法
025     */
026    public enum Method {
027        FILLING,
028        BORDERING
029    };
030
031    /**
032     * 2D シミュレーションパネル
033     */
034    protected SimulationPanel2D panel;
035
036    /**
037     * リンクの 2D 表示の定義情報
038     */
039    protected LinkAppearance2D linkAppearance;
040
041    /**
042     * リンクの色
043     */
044    protected Color color = Color2D.DEFAULT_LINK_COLOR;
045
046    /**
047     * リンクの描画方法
048     */
049    protected Method method = Method.FILLING;
050
051    /**
052     * 初期設定
053     */
054    public void init(SimulationPanel2D panel, HashMap parameters, LinkAppearance2D linkAppearance) throws Exception {
055        this.panel = panel;
056        this.linkAppearance = linkAppearance;
057        setParameters(parameters);
058
059        String colorName = getStringParameter("color", "DEFAULT_LINK_COLOR");
060        color = Color2D.getColor(colorName);
061        if (color == null) {
062            color = Color2D.DEFAULT_LINK_COLOR;
063        }
064        float transparency = getFloatParameter("transparency", 0.0f);
065        float[] compArray = color.getComponents(null);
066        color = new Color(compArray[0], compArray[1], compArray[2], 1.0f - transparency);
067
068        String methodName = getStringParameter("method", method.name());
069        method = Method.valueOf(methodName.toUpperCase());
070    }
071
072    /**
073     * 最終的なリンクの幅を取得する
074     */
075    public abstract double getWidth(MapLink link);
076
077    /**
078     * リンクの Line2D を生成する
079     */
080    public Line2D[] createLinkLines(MapLink link) {
081        Line2D[] lines = null;
082        switch (method) {
083        case FILLING:
084            lines = new Line2D[1];
085            lines[0] = new Line2D.Double(panel.getRotatedPoint(link.getFrom()), panel.getRotatedPoint(link.getTo()));
086            break;
087        case BORDERING:
088            EdgePoints.Points points = linkAppearance.getEdgePoints(link);
089            if (points == null) {
090                break;
091            }
092            javafx.geometry.Point2D a1 = points.fromNodeLeftPoint;
093            javafx.geometry.Point2D a2 = points.toNodeRightPoint;
094            javafx.geometry.Point2D b1 = points.fromNodeRightPoint;
095            javafx.geometry.Point2D b2 = points.toNodeLeftPoint;
096            if (a1 == null || a2 == null || b1 == null || b2 == null) {
097                break;
098            }
099            lines = new Line2D[2];
100            lines[0] = new Line2D.Double(panel.calcRotatedPoint(a1.getX(), a1.getY()), panel.calcRotatedPoint(a2.getX(), a2.getY()));
101            lines[1] = new Line2D.Double(panel.calcRotatedPoint(b1.getX(), b1.getY()), panel.calcRotatedPoint(b2.getX(), b2.getY()));
102            break;
103        }
104        return lines;
105    }
106
107    /**
108     * リンクとラベルを描画する
109     */
110    public void draw(MapLink link, Graphics2D g2, boolean showLabel) {
111        draw(link, g2);
112        if (showLabel) {
113            drawLabel(link, g2, Color.darkGray, panel.getBackground());
114        }
115    }
116
117    /**
118     * リンクを描画する
119     */
120    public void draw(MapLink link, Graphics2D g2) {
121        g2.setColor(color);
122        Line2D[] lines = null;
123        switch (method) {
124        case FILLING:
125            lines = linkAppearance.getLinkLines(link);
126            if (lines == null) {
127                break;
128            }
129            g2.setStroke(new BasicStroke((float)getWidth(link), BasicStroke.CAP_BUTT, BasicStroke.JOIN_MITER));
130            g2.draw(lines[0]);
131            break;
132        case BORDERING:
133            lines = linkAppearance.getLinkLines(link);
134            if (lines == null) {
135                break;
136            }
137            double scale = panel.getDrawingScale();
138            g2.setStroke(new BasicStroke((float)(1.0 / scale), BasicStroke.CAP_BUTT, BasicStroke.JOIN_MITER));
139            g2.draw(lines[0]);
140            g2.draw(lines[1]);
141            break;
142        }
143    }
144
145    /**
146     * リンクのホバーを描画する
147     */
148    public void drawHover(MapLink link, Graphics2D g2) {
149        double scale = panel.getDrawingScale();
150
151        g2.setColor(panel.HOVER_COLOR);
152        g2.fill(getLinkRect(link, scale));
153        drawLabel(link, g2, Color.darkGray, panel.HOVER_BG_COLOR);
154    }
155
156    /**
157     * ラベルを描画する
158     */
159    public void drawLabel(MapLink link, Graphics2D g2, Color color, Color bgColor) {
160        String text = link.getTagString();
161        if (! text.isEmpty()) {
162            Point2D middlePoint = link.getMiddlePoint();
163            double scale = panel.getDrawingScale();
164            double cx = middlePoint.getX();
165            double cy = middlePoint.getY() - getWidth(link) / scale / 2.0;
166            g2.setColor(color);
167            panel.drawText(g2, cx, cy, SimulationPanel2D.TextPosition.UPPER_CENTER, text, bgColor);
168        }
169    }
170
171    /**
172     * ホバー描画用の Path2D を生成する
173     */
174    public GeneralPath getLinkRect(MapLink link, double scale) {
175        MapNode fromNode = link.getFrom();
176        MapNode toNode = link.getTo();
177        double x1 = panel.getRotatedX(fromNode);
178        double y1 = panel.getRotatedY(fromNode);
179        double x2 = panel.getRotatedX(toNode);
180        double y2 = panel.getRotatedY(toNode);
181        double fwidth = Math.max(getWidth(link), 4.0) / scale;
182
183        double dx = (x2 - x1);
184        double dy = (y2 - y1);
185        double a = Math.sqrt(dx*dx + dy*dy);
186
187        double edx = fwidth * dx / a / 2;
188        double edy = fwidth * dy / a / 2;
189
190        GeneralPath p = new GeneralPath();
191        p.moveTo(x1 - edy, y1 + edx);
192        p.lineTo(x1 + edy, y1 - edx);
193        p.lineTo(x2 + edy, y2 - edx);
194        p.lineTo(x2 - edy, y2 + edx);
195        p.lineTo(x1 - edy, y1 + edx);
196        p.closePath();
197
198        return p;
199    }
200}
