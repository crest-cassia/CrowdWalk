001package nodagumi.ananPJ.Gui.LinkAppearance;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005
006import javafx.geometry.Point2D;
007import javafx.geometry.Point3D;
008
009import nodagumi.ananPJ.NetworkMap.Link.MapLink;
010import nodagumi.ananPJ.NetworkMap.Node.MapNode;
011import nodagumi.ananPJ.NetworkMap.NetworkMap;
012
013/**
014 * リンクの縁取りライン(交差点できれいに繋がる)を生成するための座標を計算して保持する
015 */
016public class EdgePoints {
017    /**
018     * リンクの縁が交差する座標
019     */
020    public class Points {
021        public Point2D fromNodeLeftPoint = null;
022        public Point2D fromNodeRightPoint = null;
023        public Point2D toNodeLeftPoint = null;
024        public Point2D toNodeRightPoint = null;
025
026        public Points() {}
027
028        public void setLeftPoint(MapLink link, MapNode node, Point2D point) {
029            if (link.getFrom() == node) {
030                fromNodeLeftPoint = point;
031            } else {
032                toNodeLeftPoint = point;
033            }
034        }
035
036        public void setRightPoint(MapLink link, MapNode node, Point2D point) {
037            if (link.getFrom() == node) {
038                fromNodeRightPoint = point;
039            } else {
040                toNodeRightPoint = point;
041            }
042        }
043    }
044
045    /**
046     * 地図データ。
047     */
048    private NetworkMap networkMap;
049
050    /**
051     * 除外するリンクを示すタグ
052     */
053    private String[] exclusionTags;
054
055    /**
056     * マップ全体のリンクの縁が交差する座標
057     */
058    private HashMap<MapLink, Points> edgePoints = null;
059
060    /**
061     * コンストラクタ
062     */
063    public EdgePoints(NetworkMap networkMap, String[] exclusionTags) {
064        this.networkMap = networkMap;
065        this.exclusionTags = exclusionTags;
066    }
067
068    /**
069     * リンクの縁取りラインを生成するための座標を取得する
070     */
071    public Points getEdgePoints(MapLink link) {
072        synchronized (this) {
073            if (edgePoints == null) {
074                edgePoints = calcEdgePoints();
075            }
076        }
077        return edgePoints.get(link);
078    }
079
080    /**
081     * リンクの縁が交差する座標を求める
082     */
083    public HashMap<MapLink, Points> calcEdgePoints() {
084        // ノードに接続されたリンク(正規のリンクのみ)
085        HashMap<MapNode, ArrayList<MapLink>> regularLinksAtNode = new HashMap();
086        for (MapNode node : networkMap.getNodes()) {
087            ArrayList<MapLink> links = new ArrayList();
088            for (MapLink link : node.getLinks()) {
089                if (link.getOther(node) == node) {
090                    continue;
091                }
092                int index = 0;
093                while (index < exclusionTags.length && ! link.hasSubTag(exclusionTags[index])) {
094                    index++;
095                }
096                if (index < exclusionTags.length) {
097                    continue;
098                }
099                links.add(link);
100            }
101            regularLinksAtNode.put(node, links);
102        }
103
104        HashMap<MapLink, Points> linkPoints = new HashMap();
105        for (MapNode node : networkMap.getNodes()) {
106            ArrayList<MapLink> links = regularLinksAtNode.get(node);
107            if (links.isEmpty()) {
108                continue;
109            }
110            if (links.size() == 1) {
111                MapLink link = links.get(0);
112                MapNode oppositeNode = link.getOther(node);
113                // 単独リンクの処理
114                if (regularLinksAtNode.get(oppositeNode).size() == 1) {
115                    Point3D p1 = new Point3D(oppositeNode.getX() - node.getX(), oppositeNode.getY() - node.getY(), 0);
116                    Point3D p2 = p1.normalize().crossProduct(0, 0, link.getWidth() / 2.0);
117                    double dx = p2.getX();
118                    double dy = p2.getY();
119                    Point2D a1 = new Point2D(node.getX() - dx, node.getY() - dy);
120                    Point2D b1 = new Point2D(node.getX() + dx, node.getY() + dy);
121                    if (! linkPoints.containsKey(link)) {
122                        linkPoints.put(link, new Points());
123                    }
124                    linkPoints.get(link).setLeftPoint(link, node, a1);
125                    linkPoints.get(link).setRightPoint(link, node, b1);
126                }
127                continue;
128            }
129
130            Point2D nodePoint = new Point2D(node.getX(), node.getY());
131            for (int index = 0; index < links.size(); index++) {
132                MapLink link = links.get(index);
133                MapLink prevLink = links.get((index + links.size() - 1) % links.size());
134                MapLink nextLink = links.get((index + 1) % links.size());
135                MapNode prevOppositeNode = prevLink.getOther(node);
136                MapNode oppositeNode = link.getOther(node);
137                MapNode nextOppositeNode = nextLink.getOther(node);
138                Point2D oppositeNodePoint = new Point2D(oppositeNode.getX(), oppositeNode.getY());
139
140                if (! linkPoints.containsKey(link)) {
141                    linkPoints.put(link, new Points());
142                }
143                if (! linkPoints.containsKey(nextLink)) {
144                    linkPoints.put(nextLink, new Points());
145                }
146                Point3D p1 = new Point3D(oppositeNode.getX() - node.getX(), oppositeNode.getY() - node.getY(), 0);
147                Point3D p2 = p1.normalize().crossProduct(0, 0, link.getWidth() / 2.0);
148                double dx = p2.getX();
149                double dy = p2.getY();
150                // links は node を中心として -π を起点とした時計回りの順にソートされている
151                // link 直線に dx, dy をプラスしたものが道幅の(links 順に見て)起点側の縁を表す
152                // よって link 直線に dx, dy をマイナスしたものと、次の link 直線に dx, dy をプラスしたものの交点座標が、かど座標となる
153                Point2D a1 = new Point2D(node.getX() - dx, node.getY() - dy);
154                Point2D a2 = new Point2D(oppositeNode.getX() - dx, oppositeNode.getY() - dy);
155
156                p1 = new Point3D(nextOppositeNode.getX() - node.getX(), nextOppositeNode.getY() - node.getY(), 0);
157                p2 = p1.normalize().crossProduct(0, 0, nextLink.getWidth() / 2.0);
158                dx = p2.getX();
159                dy = p2.getY();
160                Point2D b1 = new Point2D(node.getX() + dx, node.getY() + dy);
161                Point2D b2 = new Point2D(nextOppositeNode.getX() + dx, nextOppositeNode.getY() + dy);
162
163                double prevAngle = angle(prevOppositeNode, node, oppositeNode);
164                double angle = angle(oppositeNode, node, nextOppositeNode);
165                if (
166                    // 曲がり角が突き出てしまう
167                    angle <= 30.0 && links.size() == 2
168                    // 交点座標が遙か彼方になってしまうかもしれない
169                    || angle >= 178.0
170                    // 道幅の差が大きいためきれいに繋がらない
171                    || angle >= 165.0 && (Math.max(link.getWidth(), nextLink.getWidth()) / Math.min(link.getWidth(), nextLink.getWidth())) > 1.5
172                ) {
173                    linkPoints.get(link).setLeftPoint(link, node, a1);
174                    linkPoints.get(nextLink).setRightPoint(nextLink, node, b1);
175                } else {
176                    Point2D intersectionPoint = intersection(a1, a2, b1, b2);
177                    if (
178                        // 交点座標が道幅の1.5倍以上ノードから離れている
179                        nodePoint.distance(intersectionPoint) > Math.max(link.getWidth(), nextLink.getWidth()) * 1.5
180                        // 交点座標がノードよりも先にはみ出ている
181                        && nodePoint.distance(intersectionPoint) > nodePoint.distance(oppositeNodePoint)
182                    ) {
183                        linkPoints.get(link).setLeftPoint(link, node, a1);
184                        linkPoints.get(nextLink).setRightPoint(nextLink, node, b1);
185                    } else {
186                        linkPoints.get(link).setLeftPoint(link, node, intersectionPoint);
187                        linkPoints.get(nextLink).setRightPoint(nextLink, node, intersectionPoint);
188                    }
189                }
190                if (regularLinksAtNode.get(oppositeNode).size() == 1) {
191                    linkPoints.get(link).setRightPoint(link, oppositeNode, a2);
192                }
193                if (regularLinksAtNode.get(nextOppositeNode).size() == 1) {
194                    linkPoints.get(nextLink).setLeftPoint(nextLink, nextOppositeNode, b2);
195                }
196            }
197        }
198        return linkPoints;
199    }
200
201    /**
202     * 3点のノードがなす角度を求める.
203     *
204     * 0.0～180.0 を返す
205     */
206    public static double angle(MapNode nodeA, MapNode nodeB, MapNode nodeC) {
207        Point2D pointA = new Point2D(nodeA.getX(), nodeA.getY());
208        Point2D pointB = new Point2D(nodeB.getX(), nodeB.getY());
209        Point2D pointC = new Point2D(nodeC.getX(), nodeC.getY());
210        return pointB.angle(pointA, pointC);
211    }
212
213    /**
214     * 2直線の交点座標を求める.
215     *
216     * ※直線 a, b は平行ではないこと。
217     */
218    public static Point2D intersection(Point2D a1, Point2D a2, Point2D b1, Point2D b2) {
219        double f1 = a2.getX() - a1.getX();
220        double g1 = a2.getY() - a1.getY();
221        double f2 = b2.getX() - b1.getX();
222        double g2 = b2.getY() - b1.getY();
223        double dx = b1.getX() - a1.getX();
224        double dy = b1.getY() - a1.getY();
225        double t1 = (f2 * dy - g2 * dx) / (f2 * g1 - f1 * g2);
226        return new Point2D(a1.getX() + f1 * t1, a1.getY() + g1 * t1);
227    }
