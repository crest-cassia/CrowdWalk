001package nodagumi.ananPJ.Gui.LinkAppearance.view2d;
002
003import nodagumi.ananPJ.NetworkMap.Link.MapLink;
004
005/**
006 * 実線リンクの 2D 表示
007 */
008public class SolidLine2D extends LinkViewBase2D {
009    /**
010     * コンストラクタ
011     */
012    public SolidLine2D() {}
013
014    /**
015     * 最終的なリンクの幅を取得する
016     */
017    public double getWidth(MapLink link) {
018        double scale = panel.getDrawingScale();
019        return 1.0 / scale;
020    }
021}
