001package nodagumi.ananPJ.Gui.LinkAppearance.view3d;
002
003import javafx.geometry.Point3D;
004import javafx.scene.shape.CullFace;
005import javafx.scene.shape.DrawMode;
006import javafx.scene.shape.MeshView;
007import javafx.scene.shape.Shape3D;
008
009import nodagumi.ananPJ.NetworkMap.Link.MapLink;
010
011/**
012 * 実線リンクの 3D 表示
013 */
014public class SolidLine3D extends LinkViewBase3D {
015    /**
016     * コンストラクタ
017     */
018    public SolidLine3D() {}
019
020    /**
021     * 最終的なリンクの幅を取得する
022     */
023    public double getWidth(MapLink link) {
024        return THIN_LINK_WIDTH;
025    }
026
027    /**
028     * 線分と見分けが付かない超鋭角の三角形シェイプを生成する
029     */
030    public Shape3D[] createShapes(MapLink link) {
031        Point3D[] vertices = calcVertices(link.getFrom(), link.getTo(), THIN_LINK_WIDTH, 0.0);
032        Shape3D shape = new MeshView(panel.new TrianglePolygon(vertices));
033        shape.setDrawMode(DrawMode.LINE);
034        shape.setCullFace(CullFace.NONE);    // これがないと時々表示が消える
035        shape.setMaterial(material);
036
037        Shape3D[] shapes = { shape };
038        return shapes;
039    }
040}
