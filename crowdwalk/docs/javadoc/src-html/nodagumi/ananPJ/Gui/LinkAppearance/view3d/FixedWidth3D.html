001package nodagumi.ananPJ.Gui.LinkAppearance.view3d;
002
003import java.util.HashMap;
004
005import nodagumi.ananPJ.Gui.SimulationPanel3D;
006import nodagumi.ananPJ.NetworkMap.Link.MapLink;
007
008/**
009 * 固定幅リンクの 3D 表示
010 */
011public class FixedWidth3D extends LinkViewBase3D {
012    /**
013     * リンクの幅
014     */
015    protected double width = 1.0;
016
017    /**
018     * コンストラクタ
019     */
020    public FixedWidth3D() {}
021
022    /**
023     * 初期設定
024     */
025    public void init(SimulationPanel3D panel, HashMap parameters, LinkAppearance3D linkAppearance) throws Exception {
026        super.init(panel, parameters, linkAppearance);
027
028        if (method != Method.FILLING) {
029            throw new Exception("Link appearance error: only \"filling\" method can be specified with FixedWidth3D.");
030        }
031        width = getDoubleParameter("width", width);
032    }
033
034    /**
035     * 最終的なリンクの幅を取得する
036     */
037    public double getWidth(MapLink link) {
038        return width;
039    }
040}
