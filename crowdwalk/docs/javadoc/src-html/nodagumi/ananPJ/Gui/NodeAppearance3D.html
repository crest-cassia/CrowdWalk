003import java.io.InputStream;
004import java.math.BigDecimal;
005import java.util.ArrayList;
006import java.util.LinkedHashMap;
007import java.util.Map;
008
009import javafx.scene.image.Image;

011import javafx.scene.paint.PhongMaterial;
012
013import net.arnx.jsonic.JSON;
014
015import nodagumi.ananPJ.NetworkMap.Node.*;
016import nodagumi.Itk.Itk ;
017
018/**
019 * ノードの表示スタイル(直径, 色, 透明度)をタグ別に指定するために使用するクラス
020 */
021public class NodeAppearance3D {
022    public static final String NONE_TAG = "__NONE__";
023
024    public double diameter = 1.5;
025    public Color color = FxColor.BLACK2;
026    public double opacity = 0.25;
027    public PhongMaterial material;
028
029    public NodeAppearance3D(BigDecimal _diameter, String colorName,
030            BigDecimal _transparency, NodeAppearance3D defaultValue) {
031        if (defaultValue != null) {
032            diameter = defaultValue.diameter;
033            color = defaultValue.color;
034            opacity = defaultValue.opacity;
035        }
036        if (_diameter != null) {
037            diameter = _diameter.doubleValue();
038        }
039        if (colorName != null) {
040            color = FxColor.getColor(colorName);
041        }
042        if (_transparency != null) {
043            opacity = 1.0 - _transparency.doubleValue();
044        }
045        color = new Color(color.getRed(), color.getGreen(), color.getBlue(), opacity);
046
047        material = new PhongMaterial();
048        material.setDiffuseColor(color);
049    }
050
051    public static void load(InputStream is,
052            LinkedHashMap<String, NodeAppearance3D> nodeAppearances) throws Exception {
053        JSON json = new JSON(JSON.Mode.TRADITIONAL);
054        Map<String, Object> map = (Map<String, Object>)json.parse(is);
055        NodeAppearance3D noneTagAppearance = null;
056        for (Map.Entry<String, Object> entry : map.entrySet()) {
057            String tag = Itk.intern(entry.getKey());
058            Map<String, Object> items = (Map<String, Object>)entry.getValue();
059            NodeAppearance3D appearance = new NodeAppearance3D(
060                (BigDecimal)items.get("diameter"),
061                (String)items.get("color"),
062                (BigDecimal)items.get("transparency"),
063                nodeAppearances.get(tag)
064            );
065            if (tag.equals(NONE_TAG)) {
066                noneTagAppearance = appearance;
067            } else {
068                nodeAppearances.put(tag, appearance);
069            }
070        }
071        // LinkedHashMap の最後に追加
072        if (noneTagAppearance != null) {
073            nodeAppearances.put(NONE_TAG, noneTagAppearance);
074        }
075    }
076
077    public static NodeAppearance3D getAppearance(
078            LinkedHashMap<String, NodeAppearance3D> nodeAppearances, MapNode node) {
079        for (Map.Entry<String, NodeAppearance3D> entry : nodeAppearances.entrySet()) {
080            String tag = entry.getKey();
081            if (tag.equals(NONE_TAG) || node.hasTag(tag)) {
082                return entry.getValue();
083            }
084        }
085        return null;
086    }
087
088    public static NodeAppearance3D getAppearance(
089            LinkedHashMap<String, NodeAppearance3D> nodeAppearances, ArrayList<String> tags) {
090        for (Map.Entry<String, NodeAppearance3D> entry : nodeAppearances.entrySet()) {
091            String tag = entry.getKey();
092            if (tag.equals(NONE_TAG) || tags.contains(tag)) {
093                return entry.getValue();
094            }
095        }
096        return null;
097    }
098}
