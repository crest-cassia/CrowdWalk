165    /**
166     * 地理院タイル画像を必要ならダウンロードして読み込む
167     */
168    public boolean loadImage(String cachePath, String tileName, int x, int y, int zoom, int zone) {
169        String ext = getExt(tileName);
170
171        if (! GsiAccessor.downloadGsiTileImage(cachePath, tileName, x, y, zoom, ext)) {
172            return false;
173        }
174
175        filePath = GsiAccessor.getFileNameOfGsiTileImage(cachePath, tileName, x, y, zoom, ext);
176        try {
177            BufferedImage tileImage = ImageIO.read(new FileInputStream(filePath));
178            // 隣のタイルとの間に隙間が出来るのを防ぐため1ピクセル分コピーして拡張する
179            image = new BufferedImage(GSI_TILE_SIZE + 1, GSI_TILE_SIZE + 1, BufferedImage.TYPE_INT_ARGB);
180            Graphics2D g2 = image.createGraphics();
181            g2.drawImage(tileImage, 1, 1, null);
182            g2.drawImage(tileImage, 0, 1, null);
183            g2.drawImage(tileImage, 1, 0, null);
184            g2.drawImage(tileImage, 0, 0, null);
185        } catch (IOException e) {
186            Itk.dumpStackTraceOf(e) ;
187            return false;
188        }
189
190        // 経緯度(WGS84 地理座標系)を平面直角座標に換算する
191        CoordinateTransform transform = createCoordinateTransform("EPSG:4326", JGD2000_JPR_EPSG_NAMES[zone]);
192
193        // 地理院タイルの左上点に当たる CrowdWalk 座標を求める
194        Rectangle2D tileRect = GsiAccessor.tile2boundingBox(x, y, zoom);
195        Point2D jpr = transformCoordinate(transform, tileRect.getX(), tileRect.getY());
196        point = convertJPR2CW(jpr.getY(), jpr.getX());
197
198        // 地理院タイルの右上点に当たる CrowdWalk 座標を求める
199        tileRect = GsiAccessor.tile2boundingBox(x + 1, y, zoom);
200        jpr = transformCoordinate(transform, tileRect.getX(), tileRect.getY());
201        upperRightPoint = convertJPR2CW(jpr.getY(), jpr.getX());
202
203        // 地理院タイルの左下点に当たる CrowdWalk 座標を求める
204        tileRect = GsiAccessor.tile2boundingBox(x, y + 1, zoom);
205        jpr = transformCoordinate(transform, tileRect.getX(), tileRect.getY());
206        lowerLeftPoint = convertJPR2CW(jpr.getY(), jpr.getX());
207
208        // 地理院タイルの右下点に当たる CrowdWalk 座標を求める
209        tileRect = GsiAccessor.tile2boundingBox(x + 1, y + 1, zoom);
210        jpr = transformCoordinate(transform, tileRect.getX(), tileRect.getY());
211        lowerRightPoint = convertJPR2CW(jpr.getY(), jpr.getX());
212
213        scaleX = (lowerRightPoint.getX() - point.getX()) / GSI_TILE_SIZE;
214        scaleY = (lowerRightPoint.getY() - point.getY()) / GSI_TILE_SIZE;
215        tileNumberX = x;
216        tileNumberY = y;
217
218        return true;
219    }

294    public static Point2D convertJPR2CW(double x, double y) {
295        return new Point2D.Double(y, -x);

298    /**

301    public static Point2D convertCW2JPR(double x, double y) {
302        return new Point2D.Double(-y, x);
303    }

305    /**
306     * Proj4J の座標変換オブジェクトを生成する
307     */
308    public static CoordinateTransform createCoordinateTransform(String srcEpsgName, String targetEpsgName) {
309        CRSFactory crsFactory = new CRSFactory();
310        CoordinateReferenceSystem sourceCRS = crsFactory.createFromName(srcEpsgName);
311        CoordinateReferenceSystem targetCRS = crsFactory.createFromName(targetEpsgName);
312        CoordinateTransformFactory transformFactory = new CoordinateTransformFactory();
313        return transformFactory.createTransform(sourceCRS, targetCRS);
314    }

316    /**
317     * 座標変換.
318     *
319     * ※Proj4J は平面直角座標系のX軸/Y軸ルールなど関知しないため、座標を指定する時は注意
320     */
321    public static Point2D transformCoordinate(CoordinateTransform transform, double x, double y) {
322        ProjCoordinate gcsPoint = new ProjCoordinate(x, y);
323        ProjCoordinate pcsPoint = new ProjCoordinate();
324        pcsPoint = transform.transform(gcsPoint, pcsPoint);
325        return new Point2D.Double(pcsPoint.x, pcsPoint.y);
326    }

328    /**
329     * 座標変換.
330     *
331     * ※Proj4J は平面直角座標系のX軸/Y軸ルールなど関知しないため、座標を指定する時は注意
332     */
333    public static Point2D transformCoordinate(CoordinateTransform transform, Point2D point) {
334        return transformCoordinate(transform, point.getX(), point.getY());
335    }

347    /**
348     * マップを覆う範囲の地理院タイルをロードする
349     */
350    public static ArrayList<GsiTile> loadGsiTiles(NetworkMap networkMap, String tileName, int zoom, int zone) {
351        String cachePath = makeCachePath();
352        ArrayList<GsiTile> gsiTiles = new ArrayList();
353
354        // マップの外接矩形座標を求める
355        Rectangle2D region = networkMap.calcRectangle();
356        double north = region.getY();
357        double south = north + region.getHeight();
358        double west = region.getX();
359        double east = west + region.getWidth();
360        Itk.logInfo("region(CrowdWalk)", String.format("(%f, %f)-(%f, %f)", west, north, east, south));
361
362        // CrowdWalk 座標を経緯度(WGS84 地理座標系)に換算する
363        CoordinateTransform transform = createCoordinateTransform(JGD2000_JPR_EPSG_NAMES[zone], "EPSG:4326");
364        Point2D jpr1 = convertCW2JPR(west, north);
365        Point2D point1 = transformCoordinate(transform, jpr1.getY(), jpr1.getX());
366        Point2D jpr2 = convertCW2JPR(east, south);
367        Point2D point2 = transformCoordinate(transform, jpr2.getY(), jpr2.getX());
368        north = point1.getY();
369        west = point1.getX();
370        south = point2.getY();
371        east = point2.getX();
372        Itk.logInfo("region(longitude,latitude)", String.format("(%f, %f)-(%f, %f)", west, north, east, south));
373
374        Point northWest = GsiAccessor.getTileNumber(north, west, zoom);
375        Point southEast = GsiAccessor.getTileNumber(south, east, zoom);
376        // 一回り大きな範囲を取得する
377        northWest.setLocation(northWest.x - 1, northWest.y - 1);
378        southEast.setLocation(southEast.x + 1, southEast.y + 1);
379        Itk.logInfo("Load GSI tiles", String.format("(%d, %d)-(%d, %d)", northWest.x, northWest.y, southEast.x, southEast.y));
380
381        // 地理院タイルをロードする
382        for (int y = northWest.y; y <= southEast.y; y++) {
383            for (int x = northWest.x; x <= southEast.x; x++) {
384                GsiTile gsiTile = new GsiTile();
385                if (! gsiTile.loadImage(cachePath, tileName, x, y, zoom, zone)) {
386                    Itk.logError("Load GSI tile error", String.format("(%d, %d)", x, y));
387                    return gsiTiles;
388                }
389                gsiTiles.add(gsiTile);
390            }
391        }
392
393        return gsiTiles;
394    }
