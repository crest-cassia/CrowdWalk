001package nodagumi.ananPJ.Gui;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005
006import javafx.scene.Node;
007import javafx.scene.Scene;
008import javafx.scene.control.Menu;
009import javafx.scene.control.MenuBar;
010import javafx.scene.control.MenuItem;
011import javafx.scene.input.KeyCombination;
012import javafx.scene.layout.BorderPane;
013import javafx.stage.Stage;
014
015import nodagumi.ananPJ.NetworkMap.NetworkMap;
016import nodagumi.ananPJ.NetworkMap.Node.MapNode;
017import nodagumi.ananPJ.misc.CrowdWalkPropertiesHandler;
018
019/**
020 * 3D マップ表示専用のウィンドウ
021 */
022public class MapViewFrame {
023    /**
024     * ウィンドウフレーム
025     */
026    private Stage stage;
027
028    /**
029     * マップパネル
030     */
031    public SimulationPanel3D panel;
032
033    private String linkAppearanceFile = null;
034
035    public MapViewFrame() {}
036
037    public MapViewFrame(String title, int width, int height, NetworkMap networkMap, CrowdWalkPropertiesHandler properties, ArrayList<HashMap> linkAppearanceConfig, ArrayList<HashMap> nodeAppearanceConfig) {
038        stage = new Stage();
039        stage.setTitle(title);
040
041        // メニュー
042        Node menuBar = createMenu();
043
044        // シミュレーションパネル
045        for (MapNode node : networkMap.getNodes()) {
046            node.sortLinkTableByAngle();
047        }
048        panel = new SimulationPanel3D(width, height, networkMap, 1.0, properties, null, null, linkAppearanceConfig, nodeAppearanceConfig);
049        panel.setPrefSize(width, height);
050
051        BorderPane borderPane = new BorderPane();
052        borderPane.setTop(menuBar);
053        borderPane.setCenter(panel);
054
055        Scene scene = new Scene(borderPane);
056        stage.setScene(scene);
057    }
058
059    /**
060     * メニューの生成
061     */
062    private Node createMenu() {
063        MenuBar menuBar = new MenuBar();
064
065        //// File menu ////
066
067        Menu fileMenu = new Menu("File");
068
069        MenuItem miClose = new MenuItem("Close");
070        miClose.setOnAction(e -> stage.close());
071        miClose.setAccelerator(KeyCombination.valueOf("Ctrl+W"));
072
073        fileMenu.getItems().addAll(miClose);
074
075        //// View menu ////
076
077        Menu viewMenu = new Menu("View");
078
079        MenuItem miCentering = new MenuItem("Centering");
080        miCentering.setOnAction(e -> {
081            panel.centering(false, panel.getWidth(), panel.getHeight());
082        });
083
084        MenuItem miCenteringWithScaling = new MenuItem("Centering with scaling");
085        miCenteringWithScaling.setOnAction(e -> {
086            panel.centering(true, panel.getWidth(), panel.getHeight());
087        });
088
089        viewMenu.getItems().addAll(miCentering, miCenteringWithScaling);
090
091        menuBar.getMenus().addAll(fileMenu, viewMenu);
092
093        return menuBar;
094    }
095
096    public void show() {
097        stage.show();
098    }
099}
