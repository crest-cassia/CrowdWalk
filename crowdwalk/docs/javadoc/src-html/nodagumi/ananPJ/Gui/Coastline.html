021 * 国土地理院の海岸線データを使って海面領域を表すポリゴンを作成する
022 */
023public class Coastline extends JsonicHashMapGetter {
024    /**
025     * 閉じた海岸線(島)の座標リスト
026     */
027    private ArrayList<ArrayList<Point2D>> islands = new ArrayList();
028
029    /**
030     * 閉じていない海岸線の先頭座標をキーとするハッシュテーブル
031     */
032    private HashMap<String, ArrayList<Point2D>> linesFirstReference = new HashMap();
033
034    /**
035     * 閉じていない海岸線の末尾座標をキーとするハッシュテーブル
036     */
037    private HashMap<String, ArrayList<Point2D>> linesLastReference = new HashMap();
038
039    /**
040     * 海岸線の座標リスト
041     */
042    private ArrayList<ArrayList<Point2D>> outerBoundaries = new ArrayList();

044    /**
045     * 交差する辺の情報を持つ座標リスト
046     */
047    private class DividingLine<T> extends ArrayList {
048        public int entranceSide;
049        public int exitSide;
050
051        /**
052         * get(int index) のオーバーライド
053         */
054        public Point2D get(int index) {
055            return (Point2D)super.get(index);
056        }
057
058        /**
059         * 開始座標を返す
060         */
061        public Point2D getEntrancePoint() {
062            if (isEmpty()) {
063                return null;
064            }
065            return get(0);
066        }
067
068        /**
069         * 終端座標を返す
070         */
071        public Point2D getExitPoint() {
072            if (isEmpty()) {
073                return null;
074            }
075            return get(size() - 1);
076        }
077
078        /**
079         * 開始座標の X 値または Y 値を返す
080         */
081        public Double getEntranceValue() {
082            if (isEmpty()) {
083                return 0.0;
084            }
085            Point2D point = get(0);
086            return new Double(entranceSide % 2 == 0 ? point.getX() : point.getY());
087        }
088
089        /**
090         * 終端座標の X 値または Y 値を返す
091         */
092        public Double getExitValue() {
093            if (isEmpty()) {
094                return 0.0;
095            }
096            Point2D point = get(size() - 1);
097            return new Double(exitSide % 2 == 0 ? point.getX() : point.getY());
098        }
099    }

106    /**
107     * 海岸線データの GeoJSON ファイル(平面直角座標系)を読み込んで座標データを取り出す
108     */
109    public void read(String filePath, Rectangle2D boundary) throws Exception {
110        Itk.logInfo("Read coastline file", filePath);
111        setParameters(JSON.decode(new FileInputStream(filePath)));
112        ArrayList<HashMap> features = (ArrayList<HashMap>)getArrayListParameter("features", null);
113        if (features == null) {
114            throw new Exception("File parsing error - \"" + filePath + "\" : \"features\" elements are missing.");
115        }
116
117        for (HashMap feature : features) {
118            setParameters(feature);
119            HashMap geometry = getHashMapParameter("geometry", null);
120            if (geometry == null) {
121                throw new Exception("File parsing error - \"" + filePath + "\" : \"geometry\" elements are missing.");
122            }
123            setParameters(geometry);
124            ArrayList<ArrayList<BigDecimal>> coordinates = (ArrayList<ArrayList<BigDecimal>>)getArrayListParameter("coordinates", null);
125            if (coordinates == null) {
126                throw new Exception("File parsing error - \"" + filePath + "\" : \"coordinates\" elements are missing.");
127            }
128            ArrayList<Point2D> coords = new ArrayList();
129            for (ArrayList<BigDecimal> coordinate : coordinates) {
130                if (coordinate.size() != 2) {
131                    throw new Exception("File parsing error - \"" + filePath + "\" : invalid coordinates: " + coordinate);
132                }
133                double east = coordinate.get(0).doubleValue();
134                double north = coordinate.get(1).doubleValue();
135                // CrowdWalk 座標系に変換して格納
136                coords.add(new Point2D.Double(east, -north));
137            }
138            if (! coords.isEmpty()) {
139                if (coords.size() == 1) {
140                    throw new Exception("File parsing error - \"" + filePath + "\" : invalid size of coordinates: " + coords);
141                }
142                appendCoordinates(coords);
143            }
144        }
145    }

147    /**
148     * 海岸線データを閉じたものと閉じていないものに分類して保存する.
149     *
150     * @param _coordinates 繋がった海岸線の座標リスト
151     */
152    private void appendCoordinates(ArrayList<Point2D> _coordinates) throws Exception {
153        ArrayList<Point2D> coordinates = (ArrayList<Point2D>)_coordinates.clone();
154        int lastIndex = coordinates.size() - 1;
155        Point2D firstPoint = coordinates.get(0);
156        Point2D lastPoint = coordinates.get(lastIndex);
157
158        if (firstPoint.equals(lastPoint)) {
159            // 閉じた海岸線(島)
160            islands.add(eliminateRedundancy(coordinates));
161        } else {
162            ArrayList<Point2D> lineFirstReference = linesFirstReference.get(lastPoint.toString());
163            ArrayList<Point2D> lineLastReference = linesLastReference.get(firstPoint.toString());
164
165            if (lineFirstReference != null && lineLastReference != null) {
166                // coordinates を間に挿入して line と line を接続し、閉じた場合には islands に移動する
167                if (lineFirstReference == lineLastReference) {
168                    // 閉じた海岸線(島)だった
169                    lineLastReference.addAll(coordinates);
170                    islands.add(eliminateRedundancy(lineLastReference));
171                    linesFirstReference.remove(lastPoint.toString());
172                    linesLastReference.remove(firstPoint.toString());
173                } else {
174                    coordinates.remove(lastIndex);
175                    coordinates.remove(0);
176                    // lineLastReference = lineLastReference + coordinates + lineFirstReference
177                    lineLastReference.addAll(coordinates);
178                    lineLastReference.addAll(lineFirstReference);
179
180                    // lineFirstReference を削除
181                    linesFirstReference.remove(lineFirstReference.get(0).toString());
182                    linesLastReference.remove(lineFirstReference.get(lineFirstReference.size() - 1).toString());
183
184                    // lineLastReference の linesLastReference 側を新しい末尾で更新
185                    linesLastReference.remove(firstPoint.toString());
186                    linesLastReference.put(lineLastReference.get(lineLastReference.size() - 1).toString(), lineLastReference);
187                }
188            } else if (lineFirstReference != null) {
189                // coordinates を line の前部に挿入する
190                coordinates.remove(lastIndex);
191                lineFirstReference.addAll(0, coordinates);  // 前方に伸ばす。終端点は変わらない
192                linesFirstReference.remove(lastPoint.toString());
193                linesFirstReference.put(firstPoint.toString(), lineFirstReference);
194            } else if (lineLastReference != null) {
195                // coordinates を line の後部に追加する
196                coordinates.remove(0);
197                lineLastReference.addAll(coordinates);      // 後方に伸ばす。開始点は変わらない
198                linesLastReference.remove(firstPoint.toString());
199                linesLastReference.put(lastPoint.toString(), lineLastReference);
200            } else {
201                if (linesFirstReference.containsKey(firstPoint.toString())) {
202                    Itk.logWarn("Duplicated coastline coordinate was found(firstPoint)", "(" + firstPoint.getX() + ", " + firstPoint.getY() + ")");
203                    if (linesLastReference.containsKey(lastPoint.toString())) {
204                        Itk.logWarn("Duplicated coastline coordinate was found(lastPoint)", "(" + lastPoint.getX() + ", " + lastPoint.getY() + ")");
205                    }
206                    return;
207                }
208                if (linesLastReference.containsKey(lastPoint.toString())) {
209                    Itk.logWarn("Duplicated coastline coordinate was found(lastPoint)", "(" + lastPoint.getX() + ", " + lastPoint.getY() + ")");
210                    return;
211                }
212                linesFirstReference.put(firstPoint.toString(), coordinates);
213                linesLastReference.put(lastPoint.toString(), coordinates);
214            }
215        }
216    }

218    /**
219     * 海岸線の必要な範囲を切り取ってポリゴン化する.
220     *
221     * @param boundary ポリゴン化する矩形範囲
222     * @param upperLeftIsOnTheSea boundary の左上点は海上か?
223     */
224    public void polygonization(Rectangle2D boundary, boolean upperLeftIsOnTheSea) {
225        Line2D[] boundaryLines = new Line2D[4];     // { 上, 右, 下, 左 } 座標は右回りを描く
226        boundaryLines[0] = new Line2D.Double(boundary.getX(), boundary.getY(), boundary.getMaxX(), boundary.getY());
227        boundaryLines[1] = new Line2D.Double(boundary.getMaxX(), boundary.getY(), boundary.getMaxX(), boundary.getMaxY());
228        boundaryLines[2] = new Line2D.Double(boundary.getMaxX(), boundary.getMaxY(), boundary.getX(), boundary.getMaxY());
229        boundaryLines[3] = new Line2D.Double(boundary.getX(), boundary.getMaxY(), boundary.getX(), boundary.getY());
230        Area boundaryArea = new Area(boundary);
231        ArrayList<ArrayList<Point2D>> coastlines = new ArrayList(linesFirstReference.values());
232
233        // boundary 内部に完全に収まっていない島の扱い
234        for (int index = islands.size() - 1; index >= 0; index--) {
235            ArrayList<Point2D> coordinates = islands.get(index);
236            Area original = new Area(pointListToPath2D(coordinates));
237            Area clippedIsland = (Area)original.clone();
238            clippedIsland.intersect(boundaryArea);
239            // boundary の外側にある島は削除する
240            if (clippedIsland.isEmpty()) {
241                islands.remove(index);
242            }
243            // boundary からはみ出す島は、始点を boundary の外に移動して閉じていない海岸線として扱う
244            else if (! clippedIsland.equals(original)) {
245                ArrayList<Point2D> wCoordinates = new ArrayList(coordinates);
246                wCoordinates.addAll(coordinates);
247                int startIndex = -1;
248                for (int idx = 0; idx < wCoordinates.size(); idx++) {
249                    if (! boundary.contains(wCoordinates.get(idx)) && boundary.contains(wCoordinates.get(idx + 1))) {
250                        startIndex = idx;
251                        break;
252                    }
253                }
254                ArrayList<Point2D> _coordinates = new ArrayList();
255                for (int idx = 0; idx < coordinates.size() - 1; idx++) {
256                    _coordinates.add(wCoordinates.get(startIndex + idx));
257                }
258                islands.remove(index);
259                coastlines.add(_coordinates);
260            }
261        }
262
263        if (coastlines.isEmpty()) {
264            // 四方を海に囲まれて、boundary の四辺上に島が存在していない
265            if (upperLeftIsOnTheSea) {
266                ArrayList<Point2D> outerBoundary = new ArrayList();
267                outerBoundary.add(new Point2D.Double(boundary.getX(), boundary.getY()));
268                outerBoundary.add(new Point2D.Double(boundary.getMaxX(), boundary.getY()));
269                outerBoundary.add(new Point2D.Double(boundary.getMaxX(), boundary.getMaxY()));
270                outerBoundaries.add(outerBoundary);
271            }
272            // 完全に陸上なので海面は存在しない
273            else {
274                // 島が存在していたら削除する(念のため)
275                if (! islands.isEmpty()) {
276                    islands.clear();
277                }
278            }
279            return;
280        }
281
282        // 閉じていない海岸線を、boundary を分割する線に分解する
283        // TreeMap を使っているのでラインの始点が boundary の辺の端点に近い順にソートされる
284        TreeMap<Double, DividingLine<Point2D>>[] dividingLines = new TreeMap[4];
285        for (int side = 0; side < 4; side++) {
286            dividingLines[side] = new TreeMap();
287        }
288        for (ArrayList<Point2D> coastline : coastlines) {
289            int lastIndex = coastline.size() - 1;
290            boolean inside = false;
291            DividingLine<Point2D> dividingLine = new DividingLine();
292            // ラインセグメントごとに処理する
293            Point2D firstPoint = null;
294            for (Point2D lastPoint : coastline) {
295                if (firstPoint == null) {
296                    firstPoint = lastPoint;
297                    continue;
298                }
299
300                // ラインセグメントが完全に boundary の内部にある
301                if (inside && boundary.contains(lastPoint)) {
302                    dividingLine.add(lastPoint);
303                    firstPoint = lastPoint;
304                    continue;
305                }
306
307                // ラインセグメントが boundary と交差する
308                Line2D lineSegment = new Line2D.Double(firstPoint, lastPoint);
309                for (int side = 0; side < 4; side++) {
310                    if (lineSegment.intersectsLine(boundaryLines[side])) {
311                        // 交点
312                        Point2D iPoint = intersection(lineSegment, boundaryLines, side);
313                        // 内部 -> 外部
314                        if (inside) {
315                            dividingLine.add(iPoint);
316                            dividingLine.exitSide = side;
317                            dividingLines[side].put(dividingLine.getExitValue(), dividingLine);
318                            dividingLine = new DividingLine<Point2D>();
319                            inside = false;
320                        }
321                        // 外部 -> 内部
322                        else if (boundary.contains(lastPoint)) {
323                            dividingLine.add(iPoint);
324                            dividingLine.add(lastPoint);
325                            dividingLine.entranceSide = side;
326                            dividingLines[side].put(dividingLine.getEntranceValue(), dividingLine);
327                            inside = true;
328                        }
329                        // 外部 -> 内部 -> 外部
330                        else {
331                            Point2D otherPoint = null;
332                            int otherSide = side + 1;
333                            while (otherSide < 4) {
334                                if (lineSegment.intersectsLine(boundaryLines[otherSide])) {
335                                    otherPoint = intersection(lineSegment, boundaryLines, otherSide);
336                                    break;
337                                }
338                                otherSide++;
339                            }
340                            if (firstPoint.distance(iPoint) < firstPoint.distance(otherPoint)) {
341                                dividingLine.add(iPoint);
342                                dividingLine.add(otherPoint);
343                                dividingLine.entranceSide = side;
344                                dividingLine.exitSide = otherSide;
345                            } else {
346                                dividingLine.add(otherPoint);
347                                dividingLine.add(iPoint);
348                                dividingLine.entranceSide = otherSide;
349                                dividingLine.exitSide = side;
350                            }
351                            dividingLines[dividingLine.entranceSide].put(dividingLine.getEntranceValue(), dividingLine);
352                            dividingLines[dividingLine.exitSide].put(dividingLine.getExitValue(), dividingLine);
353                            dividingLine = new DividingLine<Point2D>();
354                        }
355                        break;
356                    }
357                }
358                firstPoint = lastPoint;
359            }
360        }
361
362        ArrayList<Point2D> outerBoundary = makeOuterBoundary(boundaryLines, upperLeftIsOnTheSea, dividingLines);
363        while (! outerBoundary.isEmpty()) {
364            int length = outerBoundary.size();
365            if (length > 1 && outerBoundary.get(0).equals(outerBoundary.get(length - 1))) {
366                outerBoundary.remove(length - 1);
367            }
368            outerBoundaries.add(outerBoundary);
369            // 二回目以降の呼び出しでは必ず陸地からスタートする
370            outerBoundary = makeOuterBoundary(boundaryLines, false, dividingLines);
371        }
372    }

374    /**
375     * boundary を分割する線に分解された海岸線を元に outerBoundary を作る.
376     *
377     * boundary の辺を右回りに辿りながら、見つかった海面領域を右回りに囲んでいく
378     * outerBoundary に使用された海岸線は dividingLines から削除される
379     */
380    private ArrayList<Point2D> makeOuterBoundary(Line2D[] boundaryLines, boolean startFromTheSea, TreeMap<Double, DividingLine<Point2D>>[] dividingLines) {
381        ArrayList<Point2D> outerBoundary = new ArrayList<Point2D>();
382        int side = 0;
383
384        int totalDividingLines = 0;
385        for (int _side = 0; _side < 4; _side++) {
386            totalDividingLines += dividingLines[_side].size();
387        }
388        if (totalDividingLines <= 1) {
389            return outerBoundary;
390        }
391
392        // 左上点が海上なので outerBoundary の始点とする
393        if (startFromTheSea) {
394            outerBoundary.add(new Point2D.Double(boundaryLines[0].getX1(), boundaryLines[0].getY1()));
395        }
396        // 陸->海の境界点を見つけて outerBoundary の始点とする
397        else {
398            Double value = null;
399            for (; side < 4; side++) {
400                double cornerPoint = (side % 2 == 0 ? boundaryLines[side].getX1() : boundaryLines[side].getY1());
401                if (side < 2) {
402                    value = dividingLines[side].higherKey(cornerPoint);
403                } else {
404                    value = dividingLines[side].lowerKey(cornerPoint);
405                }
406                if (value != null) {
407                    break;
408                }
409            }
410            // 海岸線はもう残っていない
411            if (value == null) {
412                return outerBoundary;
413            }
414
415            DividingLine<Point2D> dividingLine = dividingLines[side].get(value);
416            Point2D point = dividingLine.get(0);
417            if (value.equals(dividingLine.getEntranceValue())) {
418                outerBoundary.add(dividingLine.getEntrancePoint());
419            } else {
420                outerBoundary.add(dividingLine.getExitPoint());
421            }
422        }
423        while (true) {
424            Point2D point = outerBoundary.get(outerBoundary.size() - 1);
425            // outerBoundary が閉じたら終了
426            if (outerBoundary.size() > 1 && point.equals(outerBoundary.get(0))) {
427                break;
428            }
429            // この辺に接する海岸線は残っていない
430            if (dividingLines[side].isEmpty()) {
431                outerBoundary.add(boundaryLines[side].getP2());
432                side = (side + 1) % 4;
433                continue;
434            }
435            // 辺上で最も近くにある開始点(または終端点)を見つける
436            Double lastValue = (side % 2 == 0 ? point.getX() : point.getY());
437            Double value = null;
438            if (side < 2) {
439                value = dividingLines[side].higherKey(lastValue);
440            } else {
441                value = dividingLines[side].lowerKey(lastValue);
442            }
443            // 見つからなければ次の辺に移る
444            if (value == null) {
445                outerBoundary.add(boundaryLines[side].getP2());
446                side = (side + 1) % 4;
447                continue;
448            }
449
450            // 開始点(または終端点)が boundary の辺上に存在して、辺の端点に最も近い海岸線
451            DividingLine<Point2D> dividingLine = dividingLines[side].get(value);
452            Double entranceValue = dividingLine.getEntranceValue();
453            Double exitValue = dividingLine.getExitValue();
454            // 終端点で見つけた海岸線だったら座標を逆順に並べ替える
455            if (value.equals(exitValue)) {
456                Collections.reverse(dividingLine);
457            }
458            point = dividingLine.get(0);
459            // その海岸線を outerBoundary につなげる
460            outerBoundary.addAll(dividingLine);
461            // つなげた海岸線を元の map から削除して、その海岸線の反対側の辺に移る
462            dividingLines[side].remove(value);
463            if (value.equals(entranceValue)) {
464                side = dividingLine.exitSide;
465                dividingLines[side].remove(exitValue);
466            } else {
467                side = dividingLine.entranceSide;
468                dividingLines[side].remove(entranceValue);
469            }
470        }
471
472        return outerBoundary;
473    }

475    /**
476     * 座標リストから Path2D を作成する.
477     *
478     * 先頭と末尾の座標が同じならば末尾の座標は除外する
479     */
480    public static Path2D pointListToPath2D(ArrayList<Point2D> coordinates) {
481        int length = coordinates.size();
482        if (length > 1 && coordinates.get(0).equals(coordinates.get(length - 1))) {
483            length--;
484        }
485        Path2D path = new Path2D.Double();
486        for (int index = 0; index < length; index++) {
487            Point2D point = coordinates.get(index);
488            if (path.getCurrentPoint() == null) {
489                path.moveTo(point.getX(), point.getY());
490            } else {
491                path.lineTo(point.getX(), point.getY());
492            }
493        }
494        path.closePath();
495        return path;
496    }

512    /**
513     * outerBoundary の内側に存在する閉じた海岸線(島)のリストを取得する
514     */
515    public ArrayList<ArrayList<Point2D>> getInnerBoundaries(ArrayList<Point2D> outerBoundary) {
516        ArrayList<ArrayList<Point2D>> innerBoundaries = new ArrayList();
517        Area outerArea = new Area(pointListToPath2D(outerBoundary));
518
519        for (ArrayList<Point2D> island : islands) {
520            Area islandArea = new Area(pointListToPath2D(island));
521            Area _islandArea = (Area)islandArea.clone();
522            _islandArea.intersect(outerArea);
523            if (_islandArea.equals(islandArea)) {
524                innerBoundaries.add(island);
525            }
526        }
527        return innerBoundaries;
528    }

530    /**
531     * 先頭と末尾の座標が同じならば末尾の座標を削除する
532     */
533    public static ArrayList<Point2D> eliminateRedundancy(ArrayList<Point2D> coordinates) {
534        int length = coordinates.size();
535        if (length > 1 && coordinates.get(0).equals(coordinates.get(length - 1))) {
536            coordinates.remove(length - 1);
537        }
538        return coordinates;
539    }

541    /**
542     * line と boundary の交点座標を求める
543     */
544    public static Point2D intersection(Line2D line, Line2D[] boundaryLines, int side) {
545        Point2D point = intersection(line, boundaryLines[side]);
546        if (side == 0 || side == 2) {
547            return new Point2D.Double(point.getX(), boundaryLines[side].getY1());
548        } else {
549            return new Point2D.Double(boundaryLines[side].getX1(), point.getY());
550        }
551    }

553    /**
554     * 交点座標を求める
555     */
556    public static Point2D intersection(Line2D l1, Line2D l2) {
557        double f1 = l1.getX2() - l1.getX1();
558        double g1 = l1.getY2() - l1.getY1();
559        double f2 = l2.getX2() - l2.getX1();
560        double g2 = l2.getY2() - l2.getY1();
561        double dx = l2.getX1() - l1.getX1();
562        double dy = l2.getY1() - l1.getY1();
563        double t1 = (f2 * dy - g2 * dx) / (f2 * g1 - f1 * g2);
564        return new Point2D.Double(l1.getX1() + f1 * t1, l1.getY1() + g1 * t1);
565    }
