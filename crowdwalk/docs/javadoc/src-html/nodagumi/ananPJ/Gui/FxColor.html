053    public static Color speedToColor(double speed) {
054        //float f = ((float) speed) * 0.3333f;
055        /* [2015.01.29 I.Noda]
056         * 歩行速度の性質から、原則はじめると急速に減速する。
057         * なので、speed = 1 付近(自由速度に近い部分)を拡大する。
058         */
059        double hue = Math.min(Math.pow(speed, 5) * 0.35, 1.0) * 360.0;
060        Color c_rgb = Color.hsb(hue, 0.8588, 0.698);
061        return c_rgb;
062    }
063
064    public static Color getColor(String name) {
065        for (int index = 0; index < NAMES.length; index++) {
066            if (NAMES[index].equals(name)) {
067                return COLORS[index];
068            }
069        }
070        return null;
071    }
