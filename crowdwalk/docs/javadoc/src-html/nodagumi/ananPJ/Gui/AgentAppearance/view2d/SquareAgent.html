001package nodagumi.ananPJ.Gui.AgentAppearance.view2d;
002
003import java.awt.Graphics2D;
004import java.awt.geom.AffineTransform;
005import java.awt.geom.Rectangle2D;
006
007/**
008 * 四角いエージェントの 2D 表示
009 */
010public class SquareAgent extends RoundAgent {
011    /**
012     * コンストラクタ
013     */
014    public SquareAgent() {}
015
016    /**
017     * エージェント形状を描画する
018     */
019    protected void drawShape(Graphics2D g2, double x, double y, double size) {
020        AffineTransform at = g2.getTransform();
021        g2.rotate(Math.toRadians(panel.getAngle()), x, y);
022        g2.fill(new Rectangle2D.Double(x - size / 2.0, y - size / 2.0, size, size));
023        g2.setTransform(at);
024    }
025
026    /**
027     * ホバー形状を描画する
028     */
029    protected void drawHoverShape(Graphics2D g2, double x, double y, double size) {
030        AffineTransform at = g2.getTransform();
031        g2.rotate(Math.toRadians(panel.getAngle()), x, y);
032        g2.draw(new Rectangle2D.Double(x - size / 2.0, y - size / 2.0, size, size));
033        g2.setTransform(at);
034    }
035}
