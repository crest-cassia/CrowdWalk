001package nodagumi.ananPJ.Gui.AgentAppearance.view3d;
002
003import java.util.HashMap;
004
005import javafx.geometry.Point3D;
006import javafx.scene.paint.Color;
007import javafx.scene.paint.PhongMaterial;
008import javafx.scene.shape.CullFace;
009import javafx.scene.shape.DrawMode;
010import javafx.scene.shape.MeshView;
011import javafx.scene.shape.Shape3D;
012import javafx.scene.shape.TriangleMesh;
013import javafx.scene.transform.Rotate;
014
015import nodagumi.ananPJ.Agents.AgentBase;
016import nodagumi.ananPJ.Gui.SimulationPanel3D.TrianglePolygon;
017
018/**
019 * 平面三角形エージェントの 3D 表示
020 */
021public class PlaneTriangleAgent extends SphereAgent {
022    /**
023     * エージェントと Rotate オブジェクトの対応表
024     */
025    protected HashMap<AgentBase, Rotate> rotations = new HashMap();
026
027    /**
028     * エージェントのシェープを生成する
029     */
030    public Shape3D createShape(AgentBase agent, double size) {
031        Shape3D shape = new MeshView(makeTrianglePolygon(size));
032        shape.setDrawMode(DrawMode.FILL);
033        shape.setCullFace(CullFace.NONE);
034
035        PhongMaterial material = new PhongMaterial();
036        material.setDiffuseColor(getAgentColor(agent));
037        shape.setMaterial(material);
038
039        double angle = Math.atan2(agent.getNextNode().getY() - agent.getPrevNode().getY(), agent.getNextNode().getX() - agent.getPrevNode().getX()) * 180.0 / Math.PI;
040        Rotate rotate = new Rotate(angle + 90.0, Rotate.Z_AXIS);
041        shape.getTransforms().add(rotate);
042        rotations.put(agent, rotate);
043
044        return shape;
045    }
046
047    /**
048     * エージェントが移動した
049     */
050    public void agentMoved(AgentBase agent, Shape3D shape) {
051        Rotate rotate = rotations.get(agent);
052        double angle = Math.atan2(agent.getNextNode().getY() - agent.getPrevNode().getY(), agent.getNextNode().getX() - agent.getPrevNode().getX()) * 180.0 / Math.PI;
053        rotate.setAngle(angle + 90.0);
054    }
055
056    /**
057     * エージェントのサイズを変更する
058     */
059    public void changeAgentSize(AgentBase agent, Shape3D shape, double size) {
060        ((MeshView)shape).setMesh(makeTrianglePolygon(size));
061    }
062
063    /**
064     * 平面三角形ポリゴンを生成する
065     */
066    protected TrianglePolygon makeTrianglePolygon(double size) {
067        double height = Math.sqrt(size * size - Math.pow(size / 2.0, 2.0));
068        double x1 = 0.0;
069        double y1 = -2.0 * height / 3.0;
070        double x2 = -(size / 2.0);
071        double y2 = height / 3.0;
072        double x3 = size / 2.0;
073        double y3 = y2;
074
075        Point3D[] vertices = new Point3D[3];
076        vertices[0] = new Point3D(x1, y1, 0.0);
077        vertices[1] = new Point3D(x2, y2, 0.0);
078        vertices[2] = new Point3D(x3, y3, 0.0);
079        return panel.new TrianglePolygon(vertices);
080    }
081}
