001package nodagumi.ananPJ.Gui.AgentAppearance.view2d;
002
003import java.awt.BasicStroke;
004import java.awt.Graphics2D;
005import java.awt.Polygon;
006import java.awt.geom.AffineTransform;
007import java.awt.geom.Point2D;
008import java.awt.geom.Rectangle2D;
009
010import math.geom3d.Vector3D;
011import org.apache.batik.ext.awt.geom.Polygon2D;
012
013import nodagumi.ananPJ.Agents.AgentBase;
014
015/**
016 * 三角形エージェントの 2D 表示
017 */
018public class TriangleAgent extends RoundAgent {
019    /**
020     * 三角形ポリゴンのキャッシュ
021     */
022    private Polygon2D triangleCache = null;
023    private double lastSize = 0.0;
024
025    /**
026     * コンストラクタ
027     */
028    public TriangleAgent() {}
029
030    /**
031     * エージェントを描画する
032     */
033    public void draw(AgentBase agent, Graphics2D g2) {
034        double scale = g2.getTransform().getScaleX();
035        Point2D pos = panel.calcRotatedPoint(agent.getPosition());
036        Vector3D swing = agent.getSwing();
037        double size = frame.getAgentSize() / scale;
038        double x = pos.getX() + swing.getX();
039        double y = pos.getY() + swing.getY();
040        double angle = Math.atan2(agent.getNextNode().getY() - agent.getPrevNode().getY(), agent.getNextNode().getX() - agent.getPrevNode().getX());
041
042        AffineTransform at = g2.getTransform();
043        g2.translate(x, y);
044        g2.rotate(Math.toRadians(Math.toDegrees(angle + Math.PI / 2.0) + panel.getAngle()));
045        g2.setColor(getAgentColor(agent));
046        g2.fill(makeTriangle(size));
047        g2.setTransform(at);
048    }
049
050    /**
051     * エージェントのホバーを描画する
052     */
053    public void drawHover(AgentBase agent, Graphics2D g2) {
054        drawLabel(agent, g2, panel.HOVER_COLOR, panel.HOVER_BG_COLOR);
055
056        double scale = g2.getTransform().getScaleX();
057        Point2D pos = panel.calcRotatedPoint(agent.getPosition());
058        Vector3D swing = agent.getSwing();
059        double size = hoverSize / scale;
060        double x = pos.getX() + swing.getX();
061        double y = pos.getY() + swing.getY();
062        double angle = Math.atan2(agent.getNextNode().getY() - agent.getPrevNode().getY(), agent.getNextNode().getX() - agent.getPrevNode().getX());
063
064        AffineTransform at = g2.getTransform();
065        g2.translate(x, y);
066        g2.rotate(Math.toRadians(Math.toDegrees(angle + Math.PI / 2.0) + panel.getAngle()));
067        g2.setStroke(new BasicStroke((float)(3.0 / scale)));
068        g2.setColor(panel.HOVER_COLOR);
069        g2.draw(makeTriangle(size));
070        g2.setTransform(at);
071    }
072
073    /**
074     * 三角形ポリゴンを生成する
075     */
076    protected Polygon2D makeTriangle(double size) {
077        if (size == lastSize) {
078            return triangleCache;
079        }
080
081        double height = Math.sqrt(size * size - Math.pow(size / 2.0, 2.0));
082        double x1 = 0.0;
083        double y1 = -2.0 * height / 3.0;
084        double x2 = -(size / 2.0);
085        double y2 = height / 3.0;
086        double x3 = size / 2.0;
087        double y3 = y2;
088
089        Polygon2D polygon = new Polygon2D();
090        polygon.addPoint((float)x1, (float)y1);
091        polygon.addPoint((float)x2, (float)y2);
092        polygon.addPoint((float)x3, (float)y3);
093
094        lastSize = size;
095        triangleCache = polygon;
096
097        return polygon;
098    }
099}
