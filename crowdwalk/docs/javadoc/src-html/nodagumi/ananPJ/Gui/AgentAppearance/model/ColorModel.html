001package nodagumi.ananPJ.Gui.AgentAppearance.model;
002
003import java.awt.Color;
004
005import nodagumi.ananPJ.Gui.Color2D;
006
007/**
008 * エージェントの状態変化を色で示す
009 */
010public abstract class ColorModel extends AgentAppearanceModel {
011    /**
012     * 表示色のRGB値
013     */
014    protected int rgb = Color2D.DEFAULT_AGENT_COLOR.getRGB();
015
016    /**
017     * appearance value を返す
018     */
019    public Object getValue() {
020        return new Color(rgb);
021    }
022
023    /**
024     * 表示色の青成分(0～255)を返す
025     */
026    public int getBlue() {
027        return rgb & 0xFF;
028    }
029
030    /**
031     * 表示色の緑成分(0～255)を返す
032     */
033    public int getGreen() {
034        return (rgb >> 8) & 0xFF;
035    }
036
037    /**
038     * 表示色の赤成分(0～255)を返す
039     */
040    public int getRed() {
041        return (rgb >> 16) & 0xFF;
042    }
043}
