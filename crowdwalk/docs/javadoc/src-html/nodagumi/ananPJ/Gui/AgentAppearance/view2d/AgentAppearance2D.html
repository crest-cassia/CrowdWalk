001package nodagumi.ananPJ.Gui.AgentAppearance.view2d;
002
003import java.util.HashMap;
004
005import nodagumi.ananPJ.Gui.AgentAppearance.AgentAppearanceBase;
006import nodagumi.ananPJ.Gui.SimulationFrame2D;
007import nodagumi.ananPJ.GuiSimulationLauncher2D;
008
009import nodagumi.Itk.Itk;
010
011/**
012 * エージェントの 2D 表示の定義情報を扱う
013 */
014public class AgentAppearance2D extends AgentAppearanceBase {
015    /**
016     * エージェントを 2D 描画するオブジェクト
017     */
018    private AgentViewBase2D view;
019
020    /**
021     * コンストラクタ
022     */
023    public AgentAppearance2D(GuiSimulationLauncher2D launcher, SimulationFrame2D frame, HashMap parameters) {
024        super(launcher, parameters);
025
026        if (isValidFor2D()) {
027            try {
028                view = (AgentViewBase2D)classFinder.newByName(viewClassName2D);
029                view._init(launcher, frame, model, viewParameters2D);
030                view.init();
031            } catch (Exception e) {
032                Itk.quitWithStackTrace(e) ;
033            }
034        }
035    }
036
037    public AgentViewBase2D getView() {
038        return view;
039    }
040}
