001package nodagumi.ananPJ.Gui.AgentAppearance.view3d;
002
003import java.math.BigDecimal;
004import java.util.ArrayList;
005import java.util.HashMap;
006
007import javafx.geometry.Point3D;
008import javafx.scene.paint.Color;
009import javafx.scene.paint.PhongMaterial;
010import javafx.scene.shape.CullFace;
011import javafx.scene.shape.DrawMode;
012import javafx.scene.shape.MeshView;
013import javafx.scene.shape.Shape3D;
014import javafx.scene.shape.TriangleMesh;
015import javafx.scene.transform.Rotate;
016
017import nodagumi.ananPJ.Agents.AgentBase;
018
019/**
020 * ユーザー定義のポリゴンエージェントの 3D 表示
021 */
022public class TriangleMeshAgent extends SphereAgent {
023    /**
024     * 頂点座標
025     */
026    protected float[] points;
027
028    /**
029     * faces
030     */
031    protected int[] faces;
032
033    /**
034     * 描画モード
035     */
036    protected DrawMode drawMode = DrawMode.FILL;
037
038    /**
039     * ポリゴンに加える基本回転の角度
040     */
041    protected double rotationAngle = 0.0;
042
043    /**
044     * ポリゴンに加える基本回転の回転軸
045     */
046    protected Point3D rotationAxis = Rotate.Z_AXIS;
047
048    /**
049     * エージェントと Rotate オブジェクトの対応表
050     */
051    protected HashMap<AgentBase, Rotate> rotations = new HashMap();
052
053    /**
054     * 初期設定 2
055     */
056    public void init() throws Exception {
057        ArrayList<BigDecimal> points = getBigDecimalArrayList("points", null);
058        if (points == null) {
059            throw new Exception("Agent appearance error: points is not specified with TriangleMeshAgent.");
060        }
061        this.points = new float[points.size()];
062        for (int index = 0; index < points.size(); index++) {
063            this.points[index] = points.get(index).floatValue();
064        }
065
066        ArrayList<BigDecimal> faces = getBigDecimalArrayList("faces", null);
067        if (faces == null) {
068            throw new Exception("Agent appearance error: faces is not specified with TriangleMeshAgent.");
069        }
070        this.faces = new int[faces.size()];
071        for (int index = 0; index < faces.size(); index++) {
072            this.faces[index] = faces.get(index).intValue();
073        }
074
075        switch (getStringParameter("drawMode", "FILL").toUpperCase()) {
076        case "FILL":
077            drawMode = DrawMode.FILL;
078            break;
079        case "LINE":
080            drawMode = DrawMode.LINE;
081            break;
082        }
083
084        rotationAngle = getDoubleParameter("rotationAngle", rotationAngle);
085
086        switch (getStringParameter("rotationAxis", "Z")) {
087        case "X":
088            rotationAxis = Rotate.X_AXIS;
089            break;
090        case "Y":
091            rotationAxis = Rotate.Y_AXIS;
092            break;
093        case "Z":
094            rotationAxis = Rotate.Z_AXIS;
095            break;
096        }
097    }
098
099    /**
100     * エージェントのシェープを生成する
101     */
102    public Shape3D createShape(AgentBase agent, double size) {
103        float points[] = new float[this.points.length];
104        for (int index = 0; index < points.length; index++) {
105            points[index] = this.points[index] * (float)size;
106        }
107        TriangleMesh mesh = new TriangleMesh();
108        mesh.getTexCoords().addAll(0, 0);
109        mesh.getPoints().setAll(points);
110        mesh.getFaces().setAll(faces);
111
112        Shape3D shape = new MeshView(mesh);
113        shape.setDrawMode(drawMode);
114        shape.setCullFace(CullFace.NONE);
115
116        PhongMaterial material = new PhongMaterial();
117        material.setDiffuseColor(getAgentColor(agent));
118        shape.setMaterial(material);
119
120        double angle = Math.atan2(agent.getNextNode().getY() - agent.getPrevNode().getY(), agent.getNextNode().getX() - agent.getPrevNode().getX()) * 180.0 / Math.PI;
121        Rotate rotate = new Rotate(angle, Rotate.Z_AXIS);
122        if (rotationAngle == 0.0) {
123            shape.getTransforms().add(rotate);  // この transform はリアクティブなので rotate の値が変わると反応する
124        } else {
125            Rotate baseRotate = new Rotate(rotationAngle, rotationAxis);
126            shape.getTransforms().addAll(rotate, baseRotate);
127        }
128        rotations.put(agent, rotate);
129
130        return shape;
131    }
132
133    /**
134     * エージェントが移動した
135     */
136    public void agentMoved(AgentBase agent, Shape3D shape) {
137        // 進行方向に向きを変える
138        Rotate rotate = rotations.get(agent);
139        double angle = Math.atan2(agent.getNextNode().getY() - agent.getPrevNode().getY(), agent.getNextNode().getX() - agent.getPrevNode().getX()) * 180.0 / Math.PI;
140        rotate.setAngle(angle);
141    }
142
143    /**
144     * エージェントのサイズを変更する
145     */
146    public void changeAgentSize(AgentBase agent, Shape3D shape, double size) {
147        float _size = (float)size;
148        float[] points = new float[this.points.length];
149        for (int index = 0; index < points.length; index++) {
150            points[index] = this.points[index] * _size;
151        }
152        TriangleMesh mesh = (TriangleMesh)((MeshView)shape).getMesh();
153        mesh.getPoints().setAll(points);
154    }
155}
