001package nodagumi.ananPJ.Gui;
002
003import java.io.FileInputStream;
004import java.io.InputStream;
005import java.math.BigDecimal;
006import java.util.ArrayList;
007import java.util.HashMap;
008import java.util.regex.Matcher;
009import java.util.regex.Pattern;
010
011import net.arnx.jsonic.JSON;
012
013import nodagumi.ananPJ.misc.JsonicHashMapGetter;
014import nodagumi.Itk.Itk;
015
016/**
017 * ポリゴン表示スタイルの定義情報
018 */
019public class PolygonAppearance extends JsonicHashMapGetter {
020    /**
021     * このスタイルを適用するタグ
022     */
023    protected String tag;
024
025    /**
026     * タグの正規表現パターン
027     */
028    protected Pattern tagPattern = null;
029
030    protected String drawMode = "FILL";
031    protected String colorValue = null;
032    protected double transparency = 0.0;
033    protected String fileName = null;
034    protected String faceCulling = "NONE";
035    protected float[] texCoords = null;
036    protected int[] faces = null;
037
038    /**
039     * コンストラクタ
040     */
041    public PolygonAppearance(HashMap parameters) throws Exception {
042        setParameters(parameters);
043
044        // 適用対象となるタグ
045        tag = getStringParameter("tag", "*");
046        Matcher matcher = Pattern.compile("^\\/(.*)\\/$").matcher(tag);
047        // タグが正規表現か?
048        if (matcher.find()) {
049            tagPattern = Pattern.compile(matcher.group(1));
050        }
051
052        drawMode = getStringParameter("drawMode", drawMode).toUpperCase();
053        faceCulling = getStringParameter("faceCulling", faceCulling).toUpperCase();
054        HashMap<String, Object> color = getHashMapParameter("color", null);
055        HashMap<String, Object> texture = getHashMapParameter("texture", null);
056
057        if (color != null) {
058            setParameters(color);
059            colorValue = getStringParameter("value", null);
060            if (colorValue == null) {
061                throw new Exception("Polygon appearance file format error.");
062            }
063            transparency = getDoubleParameter("transparency", transparency);
064            if (transparency < 0.0) {
065                transparency = 0.0;
066            } else if (transparency > 1.0) {
067                transparency = 1.0;
068            }
069            ArrayList<BigDecimal> list = getBigDecimalArrayList("faces", null);
070            if (list != null) {
071                faces = new int[list.size()];
072                for (int index = 0; index < list.size(); index++) {
073                    faces[index] = list.get(index).intValue();
074                }
075            }
076        }
077
078        if (texture != null) {
079            setParameters(texture);
080            fileName = getStringParameter("fileName", null);
081            if (fileName == null) {
082                throw new Exception("Polygon appearance file format error.");
083            }
084
085            ArrayList<BigDecimal> list = getBigDecimalArrayList("texCoords", null);
086            if (list == null) {
087                throw new Exception("Polygon appearance file format error.");
088            }
089            texCoords = new float[list.size()];
090            for (int index = 0; index < list.size(); index++) {
091                texCoords[index] = list.get(index).floatValue();
092            }
093
094            list = getBigDecimalArrayList("faces", null);
095            if (list == null) {
096                throw new Exception("Polygon appearance file format error.");
097            }
098            faces = new int[list.size()];
099            for (int index = 0; index < list.size(); index++) {
100                faces[index] = list.get(index).intValue();
101            }
102        }
103    }
104
105    public String getTag() {
106        return tag;
107    }
108
109    /**
110     * タグに該当する文字列か?
111     */
112    public boolean isTagApplied(String text) {
113        if (tagPattern == null) {
114            if (tag.equals("*") || text.indexOf(tag) != -1) {
115                return true;
116            }
117        } else {
118            return tagPattern.matcher(text).find();
119        }
120        return false;
121    }
122
123    public HashMap getHashMapSection(String name) {
124        HashMap section = null;
125        try {
126            section = getHashMapParameter(name, null);
127        } catch (Exception e) {
128            Itk.quitWithStackTrace(e) ;
129        }
130        return section;
131    }
132
133    public HashMap getColor() {
134        return getHashMapSection("color");
135    }
136
137    public HashMap getTexture() {
138        return getHashMapSection("texture");
139    }
140
141    public String getDrawMode() {
142        return drawMode;
143    }
144
145    public String getColorValue() {
146        return colorValue;
147    }
148
149    public double getTransparency() {
150        return transparency;
151    }
152    
153    public double getOpacity() {
154        return 1.0 - transparency;
155    }
156
157    public String getTextureFileName() {
158        return fileName;
159    }
160
161    public String getFaceCulling() {
162        return faceCulling;
163    }
164
165    public float[] getTexCoords() {
166        return texCoords;
167    }
168
169    public int[] getFaces() {
170        return faces;
171    }
172
173    /**
174     * polygon appearance file を読み込む
175     */
176    public static ArrayList<PolygonAppearance> load(String fileName) throws Exception {
177        JSON json = new JSON(JSON.Mode.TRADITIONAL);
178        ArrayList<PolygonAppearance> appearances = new ArrayList();
179        if (fileName != null && ! fileName.isEmpty()) {
180            InputStream is = new FileInputStream(fileName);
181            for (HashMap parameters : (ArrayList<HashMap>)json.parse(is)) {
182                appearances.add(new PolygonAppearance(parameters));
183            }
184        }
185        InputStream is = PolygonAppearance.class.getResourceAsStream("/polygon_appearance.json");
186        for (HashMap parameters : (ArrayList<HashMap>)json.parse(is)) {
187            appearances.add(new PolygonAppearance(parameters));
188        }
189        return appearances;
190    }
191}
