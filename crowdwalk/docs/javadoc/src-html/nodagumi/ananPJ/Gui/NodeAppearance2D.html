003import java.awt.Color;
004import java.io.InputStream;
005import java.math.BigDecimal;
006import java.util.LinkedHashMap;
007import java.util.Map;
008
009import net.arnx.jsonic.JSON;
010
011import nodagumi.ananPJ.NetworkMap.Node.*;
012import nodagumi.Itk.Itk ;
013
014/**
015 * ノードの表示スタイル(直径, 色, 透明度)をタグ別に指定するために使用するクラス
016 */
017public class NodeAppearance2D {
018    public static final String NONE_TAG = "__NONE__";
019
020    public double diameter = 1.5;
021    public Color color = Color2D.BLACK2;
022    public float transparency = 0.75f;
023
024    public NodeAppearance2D(BigDecimal _diameter, String colorName,
025            BigDecimal _transparency, NodeAppearance2D defaultValue) {
026        if (defaultValue != null) {
027            diameter = defaultValue.diameter;
028            color = defaultValue.color;
029            transparency = defaultValue.transparency;
030        }
031        if (_diameter != null) {
032            diameter = _diameter.doubleValue();
033        }
034        if (colorName != null) {
035            color = Color2D.getColor(colorName);
036        }
037        if (_transparency != null) {
038            transparency = _transparency.floatValue();
039        }
040        float[] compArray = color.getComponents(null);
041        color = new Color(compArray[0], compArray[1], compArray[2], 1.0f - transparency);
042    }
043
044    public static void load(InputStream is,
045            LinkedHashMap<String, NodeAppearance2D> nodeAppearances) throws Exception {
046        /* [2014.12.27] I.Noda. to adapt new version os JSONIC. */
047        //Map<String, Object> map = (Map<String, Object>)JSON.decode(is);
048        JSON json = new JSON(JSON.Mode.TRADITIONAL);
049        Map<String, Object> map = (Map<String, Object>)json.parse(is);
050        NodeAppearance2D noneTagAppearance = null;
051        for (Map.Entry<String, Object> entry : map.entrySet()) {
052            String tag = Itk.intern(entry.getKey());
053            Map<String, Object> items = (Map<String, Object>)entry.getValue();
054            NodeAppearance2D appearance = new NodeAppearance2D(
055                (BigDecimal)items.get("diameter"),
056                (String)items.get("color"),
057                (BigDecimal)items.get("transparency"),
058                nodeAppearances.get(tag)
059            );
060            if (tag.equals(NONE_TAG)) {
061                noneTagAppearance = appearance;
062            } else {
063                nodeAppearances.put(tag, appearance);
064            }
065        }
066        // LinkedHashMap の最後に追加
067        if (noneTagAppearance != null) {
068            nodeAppearances.put(NONE_TAG, noneTagAppearance);
069        }
070    }
071
072    public static NodeAppearance2D getAppearance(
073            LinkedHashMap<String, NodeAppearance2D> nodeAppearances, MapNode node) {
074        for (Map.Entry<String, NodeAppearance2D> entry : nodeAppearances.entrySet()) {
075            String tag = entry.getKey();
076            if (tag.equals(NONE_TAG) || node.hasTag(tag)) {
077                return entry.getValue();
078            }
079        }
080        return null;
081    }
082}
