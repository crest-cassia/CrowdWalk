013package nodagumi.Itk ;
014
015import java.lang.ClassNotFoundException ; 
016import java.lang.InstantiationException ;
017import java.lang.IllegalAccessException ;
018import java.lang.reflect.InvocationTargetException ;
019
020import java.lang.reflect.Method;
021
022import java.util.HashMap ;
023import java.util.Map ;
024
025import net.arnx.jsonic.JSON ;
026import net.arnx.jsonic.JSONException ;
027
028//======================================================================
029/**
030 * クラスの名前からクラスオブジェクトを探すためのツール
031 */
032public class ClassFinder {
033    //============================================================
034    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
035    static public ClassFinder primal = new ClassFinder() ;
036
037    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
038    /**
039     * alias を格納しておくテーブル
040     */
041    public HashMap<String, String> aliasTable =
042        new HashMap<String, String>() ;
043
044    //------------------------------------------------------------
045    /**
046     * クラスオブジェクトを持ってくる。
047     * @param className クラスの名前。alias名もしくは fullpath。
048     */
049    public Class<?> get(String className)
050        throws ClassNotFoundException
051    {
052        return Class.forName(fullname(className)) ;
053    }
054
055    //------------------------------------------------------------
056    /**
057     * クラスオブジェクトを持ってくる。
058     * @param className クラスの名前。alias名もしくは fullpath。
059     */
060    public boolean isClassName(String className) 
061    {
062        try {
063            get(className) ;
064            return true ;
065        } catch (ClassNotFoundException ex) {
066            return false ;
067        }
068    }
069
070    //------------------------------------------------------------
071    /**
072     * alias table を参照しつつ、fullname を探す。
073     * もし alias されていなければ、そのまま返す。
074     * @param name 探す名前
075     */
076    public String fullname(String name)
077    {
078        String fname = aliasTable.get(name) ;
079        if(fname == null) {
080            return name ;
081        } else {
082            return fullname(fname) ;
083        }
084    }
085
086    //------------------------------------------------------------
087    /**
088     * クラスを見つけて、インスタンスを生成する。
089     * @param name クラスの名前
090     */
091    public Object newByName(String name)
092        throws ClassNotFoundException, 
093               InstantiationException, 
094               IllegalAccessException
095    {
096        Class<?> klass = get(name) ;
097        return klass.newInstance() ;
098    }
099
100    //------------------------------------------------------------
101    /**
102     * alias を登録する。(別名)
103     * @param shortName alias 名。
104     * @param fullName alias される名前。
105     */
106    public String alias(String shortName,
107                        String fullName) {
108        return registerAlias(shortName, fullName) ;
109    }
110
111    //------------------------------------------------------------
112    /**
113     * alias を登録する。
114     * @param shortName alias 名。
115     * @param fullName alias される名前。
116     */
117    public String registerAlias(String shortName,
118                                String fullName) {
119        aliasTable.put(shortName, fullName) ;
120        return shortName ;
121    }
122
123    //------------------------------------------------------------
124    /**
125     * alias を登録する。(別名)
126     * @param shortName alias 名。
127     * @param klass クラスオブジェクト
128     */
129    public String alias(String shortName,
130                        Class<?> klass) {
131        return registerAlias(shortName, klass) ;
132    }
133
134    //------------------------------------------------------------
135    /**
136     * alias を登録する。
137     * @param shortName alias 名。
138     * @param klass クラスオブジェクト
139     */
140    public String registerAlias(String shortName,
141                                Class<?> klass) {
142        return registerAlias(shortName, klass.getName()) ;
143    }
144
145    //------------------------------------------------------------
146    /**
147     * JSON で alias をまとめて定義する。
148     * @param json JSON 文字列
149     */
150    public void aliasByJson(String json) {
151        Map<String, Object> map = (Map<String, Object>)JSON.decode(json);
152        for(Map.Entry<String, Object> entry : map.entrySet()) {
153            alias(entry.getKey(), (String)entry.getValue()) ;
154        }
155    }
156
157    //------------------------------------------------------------
158    /**
159     * alias table を JSON に治す。
160     */
161    public String aliasToJson()
162        throws JSONException 
163    {
164        return aliasToJson(false) ;
165    }
166
167    //------------------------------------------------------------
168    /**
169     * alias table を JSON に治す。
170     * @param pprint prity print で出力する。
171     */
172    public String aliasToJson(boolean pprint) 
173        throws JSONException 
174    {
175        return JSON.encode(aliasTable, pprint) ;
176    }
177
178    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
179    /**
180     * dummy instance を格納しておくテーブル
181     * instance method を呼ぶためのもの。
182     * ゼロ引数で Constructor を呼べるものに限る。
183     */
184    static public HashMap<Class<?>, Object> DummyTable =
185        new HashMap<Class<?>, Object>() ;
186
187    //------------------------------------------------------------
188    /**
189     * class と dummy instance の登録
190     */
191    public void registerClassDummy(Class<?> klass) {
192        _registerClassDummy(klass) ;
193    }
194
195    //------------------------------------------------------------
196    /**
197     * class と dummy instance の登録 (static 版)
198     */
199    static public void _registerClassDummy(Class<?> klass) {
200        try {
201            Object object = klass.newInstance() ;
202            DummyTable.put(klass, object) ;
203        } catch (Exception ex) {
204            Itk.dumpStackTraceOf(ex) ;
205            Itk.logError("can not register the class:" + klass) ;
206        }
207    }
208
209    //------------------------------------------------------------
210    /**
211     * class の dummy instance の取得
212     */
213    static public Object getClassDummy(Class<?> klass) {
214        return _getClassDummy(klass) ;
215    }
216
217    //------------------------------------------------------------
218    /**
219     * class の dummy instance の取得 (static 版)
220     */
221    static public Object _getClassDummy(Class<?> klass) {
222        return DummyTable.get(klass) ;
223    }
224
225    //------------------------------------------------------------
226    /**
227     * クラス名、メソッド名を指定して、dummy object で instance method
228     * を呼び出す。
229     * (注意)
230     * 現状では、引数が primitive type だとうまく行かない。
231     * (クラス名が boxing でわからなくなる)
232     * @param className クラスの名前。alias 名でも良い。
233     * @param methodName メソッド名。
234     * @param staticP static メソッドかどうか。
235     * @param args 引数。
236     */
237    public Object callMethodForClass(String className,
238                                     String methodName,
239                                     boolean staticP,
240                                     Object... args)
241        throws ClassNotFoundException,
242               NoSuchMethodException,
243               IllegalAccessException,
244               InvocationTargetException
245    {
246        Class<?> klass = get(className) ;
247        return callMethodForClass(klass, methodName, staticP, args) ;
248    }
249    //------------------------------------------------------------
250    /**
251     * クラス名、メソッド名を指定して、dummy object で instance method
252     * を呼び出す。
253     * (注意)
254     * 現状では、引数が primitive type だとうまく行かない。
255     * (クラス名が boxing でわからなくなる)
256     * @param klass クラス。
257     * @param methodName メソッド名。
258     * @param staticP static メソッドかどうか。
259     * @param args 引数。
260     */
261    public Object callMethodForClass(Class<?> klass,
262                                            String methodName,
263                                            boolean staticP,
264                                            Object... args)
265        throws NoSuchMethodException,
266               IllegalAccessException,
267               InvocationTargetException
268    {
269        return _callMethodForClass(klass, methodName, staticP, args) ;
270    }
271
272    //------------------------------------------------------------
273    /**
274     * クラス名、メソッド名を指定して、dummy object で instance method
275     * を呼び出す。（static 版）
276     * (注意)
277     * 現状では、引数が primitive type だとうまく行かない。
278     * (クラス名が boxing でわからなくなる)
279     * @param klass クラス。
280     * @param methodName メソッド名。
281     * @param staticP static メソッドかどうか。
282     * @param args 引数。
283     */
284    static public Object _callMethodForClass(Class<?> klass,
285                                             String methodName,
286                                             boolean staticP,
287                                             Object... args)
288        throws NoSuchMethodException,
289               IllegalAccessException,
290               InvocationTargetException
291    {
292        Class<?>[] argClassList = new Class<?>[args.length] ;
293
294        Object instance = (staticP ? null : getClassDummy(klass)) ;
295
296        for(int i = 0 ; i < args.length ; i++) {
297            argClassList[i] = args[i].getClass() ;
298        }
299
300        Method method = klass.getMethod(methodName,argClassList) ;
301
302        return method.invoke(instance, args) ;
303    }
304
305} // class ClassFinder
