166    public void testRecycle() {
167        RingBuffer<TestRecycleFoo> rb0 =
168            new RingBuffer<TestRecycleFoo>(10, ExpandType.Recycle, true) ;
169        rb0.fillElements(TestRecycleFoo.class) ;
170
171        Itk.dbgVal("rb0",rb0) ;
172        Itk.dbgVal("buffer", rb0.getBuffer()) ;
173        Itk.dbgVal("array", rb0.toArrayList()) ;
174        for(int i = 0 ; i < 5 ; i++) {
175            rb0.shiftHead().set(i+1,(double)(i * i)) ;
176        }
177        Itk.dbgVal("rb0",rb0) ;
178
179        for(int i = 0 ; i < 15 ; i++) {
180            rb0.shiftHead().set(i + 100, (double)(3 * i)) ;
181            Itk.dbgVal("rb0", rb0) ;
182            Itk.dbgVal("buffer", rb0.getBuffer()) ;
183            Itk.dbgVal("array", rb0.toArrayList()) ;
184        }
185    }
186    static class TestRecycleFoo {
187        public int id = 0;
188        public double v = 0.0;
189        public void set(int _id, double _v) {
190            id = _id ; v = _v ;
191        }
192        public String toString(){ return ("#Foo[" + id + "]") ; }
193    } ;
