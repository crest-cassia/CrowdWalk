013package nodagumi.Itk;
014
015import java.io.IOException;
016import com.opencsv.CSVParser ;
017
018import nodagumi.Itk.Itk;
019
020//======================================================================
021/**
022 * String 配列を、先頭から順番に使っていく時のためのクラス。
023 */
024public class ShiftingStringList {
025    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
026    /**
027     * 現在位置
028     */
029    private int index = 0 ;
030
031    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
032    /**
033     * カラムの列
034     */
035    private String[] stringList = null ;
036
037    //------------------------------------------------------------
038    /**
039     * コンストラクタ
040     */
041    public ShiftingStringList(String[] _stringList) {
042        index = 0 ;
043        stringList = _stringList ;
044    }
045
046    //------------------------------------------------------------
047    /**
048     * shift: index をひとつずらす
049     */
050    public int shift() {
051        return shift(1) ;
052    }
053
054    //------------------------------------------------------------
055    /**
056     * shift: index を n ずらす
057     */
058    public int shift(int n) {
059        index += n ;
060        return index ;
061    }
062
063    //------------------------------------------------------------
064    /**
065     * 残りの長さ
066     */
067    public int length() {
068        return totalLength() - index ;
069    }
070
071    //------------------------------------------------------------
072    /**
073     * もとの長さ
074     */
075    public int totalLength() {
076        return stringList.length ;
077    }
078
079    //------------------------------------------------------------
080    /**
081     * 空かどうかのチェック
082     */
083    public boolean isEmpty() {
084        return length() <= 0 ;
085    }
086
087    //------------------------------------------------------------
088    /**
089     * 現在の先頭を見る。（取り除かない）
090     */
091    public String top() {
092        return nth(0) ;
093    }
094
095    //------------------------------------------------------------
096    /**
097     * 現在のindexよりn番目を取り出す。
098     */
099    public String nth(int n) {
100        return stringList[index + n] ;
101    }
102
103    //------------------------------------------------------------
104    /**
105     * 現在の先頭を見る。（取り除く）
106     */
107    public String get() {
108        String string = top() ;
109        shift() ;
110        return string ;
111    }
112
113    //============================================================
114    // static definitions
115
116    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
117    /**
118     * shared CSV parser
119     */
120    static private CSVParser csvParser = new CSVParser() ;
121
122    //------------------------------------------------------------
123    /**
124     * CSV parser の特殊文字の変更
125     */
126    static public void setCsvSpecialChars(char separatorChar,
127                                          char quoteChar,
128                                          char escapeChar) {
129        csvParser = new CSVParser(separatorChar, quoteChar, escapeChar) ;
130    }
131
132    //------------------------------------------------------------
133    /**
134     * CSV の一行 (row)  を読み込んで、ShiftingStringList をひとつ作る。
135     */
136    static public ShiftingStringList newFromCsvRow(String csvRow) 
137        throws IOException
138    {
139        return new ShiftingStringList(csvParser.parseLine(csvRow)) ;
140    }
141
142} // class ShitingStringList.java
