020//======================================================================
021/**
022 * リングバッファ。
023 */
024public class RingBuffer<E> implements Iterable<E> {
025    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
026    //============================================================
027    static public enum ExpandType {
028        /** 自動拡張 */
029        Auto,
030        /** 固定長(満杯チェックあり) */
031        Fixed,
032        /** 固定長(満杯チェックなし) */
033        Overwrite,
034        /** 固定長(オブジェクト再利用) */
035        Recycle ;
036    }
037        
038    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
039    /**
040     * 拡張タイプ。
041     */
042    private ExpandType _type ;
043
044    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
045    /**
046     * 繰り返しタイプ
047     */
048    private boolean _iterateBackward ;
049
050    //::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
051    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
052    static public double defaultExpandRatio = 2.0 ;
053    
054    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
055    /**
056     * 拡張係数。
057     */
058    private double _expandRatio = defaultExpandRatio ;
059
060    /**
061     * バッファサイズ。
062     */
063    private int _size ;
064
065    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
066    /**
067     * 新たにデータを入れる先の index。
068     */
069    private int _head ;
070
071    /**
072     * 次にデータを取り出す index。
073     */
074    private int _tail ;
075
076    //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
077    /**
078     * データの格納先。
079     */
080    private ArrayList<E> _buffer ;

082    //------------------------------------------------------------
083    /**
084     * constructor。
085     * @param initSize: リングバッファの初期サイズ。
086     */
087    public RingBuffer(int initSize){
088        this(initSize, ExpandType.Auto) ;
089    }
090
091    /**
092     * constructor。
093     * @param initSize: リングバッファの初期サイズ。
094     * @param type: 拡張タイプ。
095     */
096    public RingBuffer(int initSize, ExpandType type){
097        this(initSize, type, false) ;
098    }
099    
100    /**
101     * constructor。
102     * @param initSize: リングバッファの初期サイズ。
103     * @param type: 拡張タイプ。
104     * @param iterateBackward: 繰り返し方向を逆にするかどうか。
105     */
106    public RingBuffer(int initSize, ExpandType type, boolean iterateBackward){
107        _type = type ;
108        _iterateBackward = iterateBackward ;
109        _size = initSize ;
110        _buffer = allocate(_size) ;
111        _head = 0 ;
112        _tail = 0 ;
113    }
114
115    //------------------------------------------------------------
116    // アクセス関係。
117    //--------------------
118    /** 拡張タイプ取得。*/
119    public ExpandType getType() { return _type ; }
120    /** 拡張タイプ設定。*/
121    public void setType(ExpandType type) { _type = type ; }
122
123    //--------------------
124    /** 繰り返しタイプ取得。*/
125    public boolean getIterateBackward() { return _iterateBackward ; }
126    /** 繰り返しタイプ設定。*/
127    public void setIterateBackward() { setIterateBackward(true) ; } ;
128    /** 繰り返しタイプ設定。*/
129    public void setIterateForward() { setIterateBackward(false) ; } ;
130    /** 繰り返しタイプ設定。*/
131    public void setIterateBackward(boolean backward) {
132        _iterateBackward = backward ;
133    }
134
135    //--------------------
136    /** 拡張係数取得。 */
137    public double getExpandRatio() { return _expandRatio ; }
138    /** 拡張係数設定。*/
139    public void setExpandRatio(double ratio) { _expandRatio = ratio ; }
140    
141    //--------------------
142    /** サイズ取得。*/
143    public int getSize() { return _size ; }
144
145    //--------------------
146    /** head index取得。*/
147    public int getHeadIndex() { return _head ; }
148    /** tail index取得。*/
149    public int getTailIndex() { return _tail ; }
150
151    //--------------------
152    /** バッファー取得。*/
153    public ArrayList<E> getBuffer() { return _buffer ; }
154
155    //------------------------------------------------------------
156    /**
157     * メモリ確保。
158     * _buffer にはセットしない。
159     */
160    private ArrayList<E> allocate(int size) {
161        ArrayList<E> newBuf = new ArrayList<E>(size) ;
162        for(int i = 0 ; i < size ; i++) {
163            newBuf.add(null) ;
164        }
165        return newBuf ;
166    }
167
168    /**
169     * _buffer の null 要素にインスタンスを埋める。
170     * Recycle の場合に、インスタンス生成後、呼ばれるべきである。
171     * その他のタイプの場合、あまり意味はない。 
172     */
173    public void fillElements(Class<?> klass) {
174        try {
175            for(int i = 0 ; i < _size ; i++) {
176                if(_buffer.get(i) == null) {
177                    E element = (E)klass.newInstance() ;
178                    _buffer.set(i, element) ;
179                }
180            }
181        } catch(Exception e) {
182            throw new RuntimeException(e) ;
183        }
184    }
185
186    //------------------------------------------------------------
187    /**
188     * 空チェック。
189     * @return true if empty buffer
190     */
191    public boolean isEmpty() {
192        return ((_head - _tail + _size) % _size == 0) ;
193    }
194
195    //------------------------------------------------------------
196    /**
197     * 満杯 チェック。
198     * @return true if full buffer
199     */
200    public boolean isFull() {
201        return (((_tail - _head + _size) % _size) == 1) ;
202    }
203
204    //------------------------------------------------------------
205    /**
206     * index が、head と tail の間かチェック。
207     * @return true if index is between head and tail
208     */
209    public boolean isBody(int index) {
210        return (index >= _tail) && (index < _head) ;
211    }
212
213    //------------------------------------------------------------
214    /**
215     * index の縮退表現。
216     * @param index
217     * @return 正規化されたindex。
218     */
219    public int reducedIndex(int index) {
220        return (index % _size) ;
221    }
222
223    //------------------------------------------------------------
224    /**
225     * インデックスの縮約。
226     */
227    public void reduceIndexes() {
228        _head = _head % _size ;
229        _tail = _tail % _size ;
230        if(_head < _tail) {
231            _head = _head + _size ;
232        }
233    }
234    
235    //------------------------------------------------------------
236    /**
237     * head の前進。
238     */
239    public void advanceHead() {
240        _head += 1 ;
241    }
242
243    //------------------------------------------------------------
244    /**
245     * tail の前進。
246     */
247    public void advanceTail() {
248        _tail += 1 ;
249        if(_tail >= _size) {
250            reduceIndexes() ;
251        }
252    }
253
254    //------------------------------------------------------------
255    /**
256     * index の部分の取り出し。
257     * @param index
258     * @return index を取り出し。
259     */
260    public E get(int index) {
261        return _buffer.get(reducedIndex(index)) ;
262    }
263
264    //------------------------------------------------------------
265    /**
266     * index への代入。
267     * @param index
268     * @param value
269     * @return index を取り出し。
270     */
271    public E set(int index, E value) {
272        return _buffer.set(reducedIndex(index), value) ;
273    }
274
275    //------------------------------------------------------------
276    /**
277     * buffer の head 側に入れる。
278     * @param value.
279     * @return true if succeed to enqueue the value.
280     */
281    public boolean enqueue(E value) {
282        if(isFull()) {
283            if(_type == ExpandType.Auto) {
284                expand(_expandRatio) ;
285                return enqueue(value) ;
286            } else if(_type == ExpandType.Overwrite) {
287                dequeue() ;
288                return enqueue(value) ;
289            } else {
290                return false ;
291            }
292        } else {
293            set(_head, value) ;
294            advanceHead() ;
295            return true ;
296        }
297    }
298    
299    //------------------------------------------------------------
300    /**
301     * buffer の tail 側から取り出す。
302     * @return null if empty。
303     */
304    public E dequeue() {
305        if(isEmpty()) {
306            return null ;
307        } else {
308            E value = get(_tail) ;
309            if(_type != ExpandType.Recycle) {
310                set(_tail, null) ;
311            }
312            advanceTail() ;
313            return value ;
314        }
315    }
316
317    //------------------------------------------------------------
318    /**
319     * buffer の tail を覗く。
320     * @return null if empty。
321     */
322    public E peekTail() {
323        if(isEmpty()) {
324            return null ;
325        } else {
326            return get(_tail) ;
327        }
328    }
329    
330    //------------------------------------------------------------
331    /**
332     * buffer の head を shift を移動する。
333     * もし、head がすでに tail に接しているなら、まず tail を shift してから
334     * head を shift する。
335     * Recycle の場合に使われる。（他では意味がない。）
336     * @return 現状の head 位置にある element を返す。
337     */
338    public E shiftHead() {
339        if(isFull()) {
340            advanceTail() ;
341        } 
342        E value = get(_head) ;
343        advanceHead() ;
344        return value ;
345    }
346    
347    //------------------------------------------------------------
348    /**
349     * 拡張
350     */
351    public boolean expand(double ratio) {
352        if(ratio <= 1.0) {
353            return false ;
354        } else {
355            int newSize = (int)Math.ceil(((double)_size) * ratio) ;
356            ArrayList<E> newBuffer = allocate(newSize) ;
357            for(int i = 0 ; i < _size ; i++) {
358                newBuffer.set(i, get(_tail + i)) ;
359            }
360            int newHead = _head - _tail ;
361            while(newHead < 0) { newHead += _size ;}
362            _tail = 0 ;
363            _head = newHead ;
364            _size = newSize ;
365            _buffer = newBuffer ;
366            return true ;
367        }
368    }

370    //------------------------------------------------------------
371    /**
372     * 繰り返し演算子。
373     */
374    public Iterator<E> iterator() {
375        if(_iterateBackward) {
376            return backwardIterator() ;
377        } else {
378            return forwardIterator() ;
379        }
380    }
381    
382    //------------------------------------------------------------
383    /**
384     * 反対方向繰り返し演算子。
385     */
386    public Iterator<E> reverseIterator() {
387        if(_iterateBackward) {
388            return forwardIterator() ;
389        } else {
390            return backwardIterator() ;
391        }
392    }
393
394    //------------------------------------------------------------
395    /**
396     * 繰り返し演算子。（順方向）
397     */
398    public Iterator<E> forwardIterator() {
399        final RingBuffer<E> ring = this ;
400        return new Iterator<E>(){
401            private RingBuffer<E> _ringBuffer = ring ;
402            private int _index = ring.getTailIndex() ;
403            public boolean hasNext() {
404                return _index < _ringBuffer.getHeadIndex() ; }
405            public E next() {
406                E value = _ringBuffer.get(_index) ;
407                _index += 1 ;
408                return value ;
409            }
410            public void remove() {} ;
411        } ;
412    }
413    
414    //------------------------------------------------------------
415    /**
416     * 繰り返し演算子。（逆方向）
417     */
418    public Iterator<E> backwardIterator() {
419        final RingBuffer<E> ring = this ;
420        return new Iterator<E>(){
421            private RingBuffer<E> _ringBuffer = ring ;
422            private int _index = ring.getHeadIndex() ;
423            public boolean hasNext() {
424                return _index > _ringBuffer.getTailIndex() ; }
425            public E next() {
426                _index -= 1 ;
427                return _ringBuffer.get(_index) ;
428            }
429            public void remove() {} ;
430        } ;
431    }

444                "]") ;
445    }

464     * new T() とか、T.class() とかが呼べない。

466     * 現状の RingBuffer の実装ではうまく行かない。

469    static Class<?> ___guessElementClass(Object object) {
470        try {
471            Class<?> clazz = object.getClass() ;
472            Itk.dbgVal("clazz", clazz) ;
473            Type type = clazz.getGenericSuperclass() ;
474            Itk.dbgVal("type", type) ;
475            ParameterizedType ptype = (ParameterizedType)type ;
476            Itk.dbgVal("ptype", ptype) ;
477            Type[] actualTypeArguments = ptype.getActualTypeArguments() ;
478            Itk.dbgVal("atype", actualTypeArguments[0]) ;
479            // ここのキャストでエラーが起きる。
480            Class<?> elementClass = (Class<?>)actualTypeArguments[0] ;
481            Itk.dbgVal("cClass", elementClass) ;
482            return elementClass ;
483        } catch(Exception e) {
484            throw new RuntimeException(e) ;
485        }
486    }
