127    /**
128     * Member情報の定義の登録。
129     * 以下のように記述することを想定。
130     *    jsonFormatter.registerMember(jsonFormatter.new Member("foo"){
131     *         public String value(T object) { return object.foo() ; }
132     *         }) ;
133     * この定義のあと、jsonFormatter.addMember("foo") や、
134     * jsonFormatter.setMember(["foo", "bar", ...]) などで実際のフォーマット
135     * を指定。
136     * @param member:: member 情報
137     */
138    public JsonFormatter registerMember(Member member) {
139        memberTable.put(member.key, member) ;
140        return this ;
141    }

144    /**
145     * Member情報追加。
146     * 以下のように記述することを想定。
147     * <pre>{@code
148     *    jsonFormatter.addMember(jsonFormatter.new Member("foo"){
149     *         public String value(T object) { return object.foo() ; }
150     *         }) ;
151     * }</pre>
152     * 同時に、registerMemberも行う。
153     * @param member:: member 情報
154     */
155    public JsonFormatter addMember(Member member) {
156        registerMember(member) ;
157        memberList.add(member) ;
158        return this ;
159    }

162    /**
163     * key名によるMember情報追加。
164     * 以下のように記述することを想定。
165     * <pre>{@code
166     *    jsonFormatter.addMember("foo") ;
167     * }</pre>
168     * 指定された memberKey に対応する Member の定義は、すでに registerMember
169     * で行われているものとする。
170     * @param memberKey:: member のキー
171     */
172    public JsonFormatter addMember(String memberKey) {
173        Member member = memberTable.get(memberKey) ;
174        if(member != null) {
175            memberList.add(member) ;
176            return this ;
177        } else {
178            Itk.logError("unknown key string for member:", memberKey) ;
179            return null ;
180        }
181    }

445    /**
446     * データ行のTerm出力。
447     * @param object : Column#value に引き渡すデータ。
448     * @return 内容を追加した buffer。
449     */
450    public Term outputRecordToNewTerm(T object, Object... auxObjects) {
451        Term term = new Term() ;
452        for(Member member : memberList) {
453            Object value = null ;
454            switch(auxObjects.length) {
455            case 0:
456                value = member.value(object) ;
457                break ;
458            case 1:
459                value = member.value(object, auxObjects[0]) ;
460                break ;
461            case 2:
462                value = member.value(object, auxObjects[0], auxObjects[1]) ;
463                break ;
464            case 3:
465                value = member.value(object, auxObjects[0], auxObjects[1],
466                                     auxObjects[2]) ;
467                break ;
468            case 4:
469                value = member.value(object, auxObjects[0], auxObjects[1],
470                                     auxObjects[2], auxObjects[3]) ;
471                break ;
472            case 5:
473                value = member.value(object, auxObjects[0], auxObjects[1],
474                                     auxObjects[2], auxObjects[3],
475                                     auxObjects[4]) ;
476                break ;
477            case 6:
478                value = member.value(object, auxObjects[0], auxObjects[1],
479                                     auxObjects[2], auxObjects[3],
480                                     auxObjects[4], auxObjects[4]) ;
481                break ;
482            default:
483                Itk.logError("too many auxObjects[].") ;
484                Itk.quitByError() ;
485            }
486            addMemberInTerm(term, member, value) ;
487        }
488
489        return term ;
490    }

493    /**
494     * output a term to a buffer with format.
495     */
496    public StringBuilder outputTermToBuffer(StringBuilder buffer,
497                                            Term term) {
498        // output separater and newline if need.
499        boolean prettyPrintP = (overallStyle == OverallStyle.PrettyPrint) ;
500        if(overallStyle != OverallStyle.RecordPerLine) {
501            if(recordCounter > 0) {
502                buffer.append(memberSeparator) ;
503                if(prettyPrintP) { buffer.append("\n") ; } ;
504            } else {
505                if(!prettyPrintP) { buffer.append(" ") ; } ;
506            }
507        }
508        
509        // output body
510        buffer.append(term.toJson(prettyPrintP)) ;
511
512        recordCounter++ ;
513
514        return buffer ;
515    }

518    /**
519     * member を Term にセット。
520     * @param term : 出力する先の Term。
521     * @param member : 出力する member
522     * @return 内容を追加した term
523     */
524    public Term addMemberInTerm(Term term, Member member, Object value) {
525        Term targetTerm = term ;
526        for(int i = 0 ; i < member.keyNest.size() - 1 ; i++) {
527            String key = member.keyNest.get(i) ;
528            Object nestedValue = targetTerm.getArg(key) ;
529            if(nestedValue == null) {
530                nestedValue = new Term() ;
531                targetTerm.setArg(key, nestedValue, false) ;
532            } else if (!(nestedValue instanceof Term)) {
533                Itk.logError("Inconsistent keys in members of JsonFormatter:",
534                             "whole key=", member.key,
535                             ", current key=", key) ;
536                Itk.quitByError() ;
537            }
538            targetTerm = (Term)nestedValue ;
539        }
540        targetTerm.setArg(member.lastKey(), value, false) ;
541        
542        return term ;
543    }

589    public class Member {
590        //@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@
591        /**
592         * Member の key
593         */
594        public String key ;
595
596        /**
597         * Member の key の入れ子リスト
598         */
599        public ArrayList<String> keyNest = null ;
600
601
602        //----------------------------------------
603        /**
604         * コンストラクタ
605         */
606        public Member(String _key) {
607            extractKeyNest(_key) ;
608        }
609
610        //----------------------------------------
611        /**
612         * key の入れ子構造分解
613         */
614        public void extractKeyNest(String _key) {
615            key = _key ;
616            keyNest = new ArrayList<String>() ;
617            for(String subKey : key.split(keyNestSeparator)) {
618                keyNest.add(subKey) ;
619            }
620        }
621
622        //----------------------------------------
623        /**
624         * key が入れ子かどうかのチェック
625         */
626        public String lastKey() {
627            return keyNest.get(keyNest.size()-1) ;
628        }
629            
630        //----------------------------------------
631        /**
632         * 値の出力。
633         * 使われる場合、各々の Member で再定義されるべき。
634         * 戻り値は、String, Term, Integer, Double のいずれかであるべき。
635         */
636        public Object value(T object0) {
637            Itk.logError("Member#value(object) is not implemented.") ;
638            return null ;
639        }
