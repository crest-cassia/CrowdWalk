013package nodagumi.Itk;
014
015import static org.junit.Assert.*;
016import java.util.ArrayList;
017import org.junit.Test;
018import java.util.logging.Logger;
019
020import nodagumi.Itk.*;
021
022//======================================================================
023/**
024 * CSV Formatter Test
025 */
026public class CsvFormatter_Test {
027    //============================================================
028    class Foo {
029        public int x ;
030        public Foo(int _x) { x = _x ; } ;
031    }
032
033    static public CsvFormatter<Foo> formatter = new CsvFormatter<Foo>() ;
034    static {
035        formatter
036            .addColumn(formatter.new Column("x") {
037                    public String value(Foo f) {
038                        return Integer.toString(f.x) ; }})
039            .addColumn(formatter.new Column("xx") {
040                    public String value(Foo f) {
041                        return Integer.toString(f.x * f.x) ; }})
042            .addColumn(formatter.new Column("xxx") {
043                    public String value(Foo f) {
044                        return Integer.toString(f.x * f.x * f.x) ; }})
045            .addColumn(formatter.new Column("1/x") {
046                    public String value(Foo f) {
047                        return Double.toString(1.0 / f.x) ; }}) ;
048    }
049
050    //------------------------------------------------------------
051    /**
052     */
053    @Test
054    public void test1() {
055        Itk.dbgVal(formatter.outputHeaderToBuffer()) ;
056        Foo f1 = new Foo(1) ;
057        Foo f2 = new Foo(2) ;
058        Foo f3 = new Foo(3) ;
059        Itk.dbgVal(formatter.outputValueToBuffer(f1)) ;
060        Itk.dbgVal(formatter.outputValueToBuffer(f2)) ;
061        Itk.dbgVal(formatter.outputValueToBuffer(f3)) ;
062    }
063
064    //------------------------------------------------------------
065    /**
066     */
067    @Test
068    public void test2() {
069        Logger logger = Logger.getLogger("foo") ;
070        ArrayList<Foo> ff = new ArrayList<Foo>() ;
071        ff.add(new Foo(1)) ;
072        ff.add(new Foo(2)) ;
073        ff.add(new Foo(3)) ;
074
075        formatter.outputHeaderToLoggerInfo(logger) ;
076        formatter.outputAllValueToLoggerInfo(logger, ff) ;
077    }
078
079    //------------------------------------------------------------
080    /**
081     */
082    @Test
083    public void test3() {
084        ArrayList<Foo> ff = new ArrayList<Foo>() ;
085        ff.add(new Foo(1)) ;
086        ff.add(new Foo(2)) ;
087        ff.add(new Foo(3)) ;
088
089        formatter.outputHeaderToStream(System.out) ;
090        formatter.outputAllValueToStream(System.out, ff) ;
091    }
092
093} // class CsvFormatter_Test
