039    @Test
040    public void test_Jsonic() {
041        try {
042            HashMap h = new HashMap<String, Object>() ;
043            //h.put("", "") ;
044            h.put("foo", new ArrayList<Object>()) ;
045            h.put("bar", new ArrayList<Object>()) ;
046            h.put("baz", new ArrayList<Object>()) ;
047            
048            Set<Map.Entry<String, Object>> s = h.entrySet() ;
049            JSONWriter writer = new JSON().getWriter((OutputStream)System.out) ;
050            writer.beginObject() ;
051            for(Map.Entry<String,Object> entry : s) {
052              writer.name(entry.getKey());
053              if(entry.getValue() instanceof List) {
054                  writer.beginArray();
055                  for(Object obj : (List)entry.getValue()) {
056                      writer.value(obj) ;
057                  }
058                  writer.endArray();
059              } else {
060                  writer.value(entry.getValue()) ;
061              }
062            }
063            writer.endObject() ;
064            System.out.close() ;
065        } catch(Exception e) {
066            System.err.println("Error") ;
067        } ;
068    }
