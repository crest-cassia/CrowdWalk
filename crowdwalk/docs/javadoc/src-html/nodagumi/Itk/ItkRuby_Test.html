108        // put, get, callMethod

110        Itk.dbgVal("foo", foo) ;
111        Itk.dbgVal("put_bar", container.put(foo,"@bar", new Integer(100)));
112        Itk.dbgVal("bar", container.get(foo,"@bar")) ;
113        Itk.dbgVal("baz", container.callMethod(foo,"baz")) ;
114
115        // parse, EvalUnit

117        Itk.dbgVal("eunit",eunit) ;
118        Itk.dbgVal("eunit.run",eunit.run()) ;
119        Itk.dbgVal("eunit.run",eunit.run()) ;
120        Itk.dbgVal("eunit.run",eunit.run()) ;
121
122        // get class
123        Object fooClass = container.runScriptlet("Foo") ;
124        Object foo2 = container.callMethod(fooClass, "new") ;
125        container.put(foo2,"@bar",new Integer(10)) ;
126        Itk.dbgVal("foo2.baz", container.callMethod(foo2,"baz")) ;
