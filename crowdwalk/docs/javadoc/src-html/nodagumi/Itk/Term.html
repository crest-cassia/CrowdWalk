033/**
034 * 項 クラス。
035 * 項(Term) は、アトム（シンボルおよび数値）・配列(Array)もしくは
036 * 連想配列(Object)からなる。
037 * Prolog の項に相当。ただし文字列はシンボルと同じとする。
038 * Object のうち、空文字列("")  をキー(slot)とする値(value)があるとき、
039 * それを head とする。head 以外の slot がない場合、それはアトムと同じ
040 * 同じ扱いとする。
041 *
042 * JSON で表すときには、
043 *     {"":_head_, _slot1_:_value1_, _slot2_:_value2_...} 
044 * と表記する。
045 * 引数のない場合は、単なる_head_だけの文字列と等価。
046 * すなわち、
047 *     {"":_head_} == _head_
048 * 配列は、
049 *     [_value_,_value_,...]
050 *
051 * CSV の中での表記では、
052 *     _head_(_slot1_:_value1_,_slot2_:_value2_...)
053 * とする。ただしこの表記方法は obsolete。将来の保証はない。
054 *
055 * また、特例として、slot 名のない表記
056 *     _head_(_value1_,_value2_...)
057 * は以下と同じとみなす。
058 *     _head_("1":_value1_,"2":_value2_...)
059 * CSV形式で配列はない。
060 * 上と同じく、引数なしは引数ゼロと同じ。
061 *     _head_() == _head_
062 *
063 * _head_ の無い (nullである) Objectを許す。
064 *
065 * _head_ も _body_ もない項は null と同じとみなす。
066 */

288        head = null ;
289        body = null ;

450    /**
451     * arg チェック (再帰的)
452     */
453    public boolean hasArgRecursive(String... slots) {
454        int nSlots = slots.length ;
455        if(nSlots < 1) {
456            Itk.logError("hasArgRecursive() should have one or more args.") ;
457            Itk.quitByError() ;
458        }
459
460        Term term = this ;
461        for(int i = 0 ; i < nSlots - 1 ; i++) {
462            if(!term.hasArg(slots[i])) return false ;
463
464            term = term.getArgTerm(slots[i]) ;
465
466            if(term == null) return false ;
467        }
468        
469        return term.hasArg(slots[nSlots - 1]) ;
470    }

556    /**
557     * arg を filter(fallback 付き) して取り出す。
558     * 例えば、
559     * <pre>
560     * { "a" : { "x" : 1, "y" : 2 },
561     *   "b" : { "u" : 3, "v" : 4 },
562     *   "_fallback" : { "a" : { "y" : 5, "z" : 6 }, 
563     *                   "b" : { "u" : 7, "w" : 8 }}}
564     * </pre>
565     * から、"a" で filter すると、以下を構築して返す。
566     * <pre>
567     * { "x" : 1, "y" : 2, "_fallback" : { "y" : 5, "z" : 6 } }
568     * </pre>
569     * @param slot : filter するスロット
570     * @param fallbackSlot : fallback で潜っていくスロット
571     * @return filter された結果。必ずコピーされる。
572     */
573    public Term filterArgTerm(String slot, String fallbackSlot) {
574        Term filteredTerm = Term.newObjectTerm() ;
575        if(hasArg(slot)) {
576            filteredTerm.updateObjectFacile(getArgTerm(slot),false) ;
577        }
578        if(hasArg(fallbackSlot)) {
579            Term fallbackTerm = getArgTerm(fallbackSlot) ;
580            Term filteredFallback =
581                fallbackTerm.filterArgTerm(slot, fallbackSlot) ;
582            filteredTerm.setArg(fallbackSlot, filteredFallback) ;
583        }
584        return filteredTerm ;
585    }

588    /**
589     * fallback 部分について、Term を単一化（フラット化）する。
590     * 例えば、
591     * <pre>
592     * { "a" : { "x" : 1, "y" : 2 },
593     *   "b" : { "u" : 3, "v" : 4 },
594     *   "_fallback" : { "a" : { "y" : 5, "z" : 6 }, 
595     *                   "b" : { "u" : 7, "w" : 8 }}}
596     * </pre>
597     * を、"_fallback" を fallbackSlot として単一化すると、
598     * <pre>
599     * { "a" : { "x" : 1, "y" : 2, "z" : 6 },
600     *   "b" : { "u" : 3, "v" : 4, "w" : 8 }}
601     * </pre>
602     * となる。
603     * @param fallbackSlot : fallback で潜っていくスロット。
604     * @return 作成された Term。
605     */
606    public Term unifyFallbacks(String fallbackSlot) {
607        Term unifiedTerm = Term.newObjectTerm() ;
608        unifyFallbacks_Body(unifiedTerm, this, fallbackSlot) ;
609        return unifiedTerm ;
610    }
611
612    /**
613     * fallback 部分について、Term を単一化（フラット化）する。
614     * 再帰されるメイン処理。
615     * @param unifiedTerm : 単一化の受け皿の Term。
616     * @param originalTerm : 単一化の元になる Term。
617     * @param fallbackSlot : fallback として潜っていくスロット名。
618     */
619    private void unifyFallbacks_Body(Term unifiedTerm,
620                                     Term originalTerm,
621                                     String fallbackSlot) {
622        if(originalTerm.hasArg(fallbackSlot)) {
623            unifyFallbacks_Body(unifiedTerm,
624                                originalTerm.getArgTerm(fallbackSlot),
625                                fallbackSlot) ;
626        }
627        unifiedTerm.updateObjectFacile(originalTerm, true) ;
628        unifiedTerm.clearFallback(fallbackSlot) ;
629    }

687     * boolean への変換
688     */
689    private boolean convertValueToBoolean(Object val) {
690        if(val instanceof Term) {
691            Term term = ((Term)val) ;
692            if(term.isAtom()) {
693                return convertValueToBoolean(term.getHead()) ;
694            }
695        } else if(val instanceof Boolean) {
696            return (Boolean)val ;
697        } else if(val instanceof String) {
698            return Boolean.valueOf((String)val) ;
699        } else if(val == null) {
700            return false ;
701        }
702        Thread.dumpStack() ;
703        Itk.logError("can not convert to boolean:" + this.toString()) ;
704        Itk.logError_("use false") ;
705        return false ;
706    }

1451    /**

1453     * @param patch : update する部分 Object
1454     * @param recursive : 深くまでの update を再帰的に行うか？
1455     */
1456    public Term updateObject(Term patch, boolean recursive)
1457        throws Exception
1458    {
1459        if(!this.isObject() && !this.isZeroArgs())
1460            throw new Exception("Can't update non-Object Term:" + this) ;
1461        if(!patch.isObject() && !patch.isZeroArgs())
1462            throw new Exception("Can't update by non-Object Term:" + patch) ;
1463
1464        if(!patch.isZeroArgs()) {
1465            for(String slot : patch.getBody().keySet()) {
1466                Term value = patch.getArgTerm(slot) ;
1467                Term originalValue = this.getArgTerm(slot) ;
1468                if(recursive &&
1469                   value != null && value.isObject() &&
1470                   originalValue != null && originalValue.isObject()) {
1471                    originalValue.updateObject(value, recursive) ;
1472                } else {
1473                    this.setArg(slot, value) ;
1474                }
1475            }
1476        }
1477        return this ;
1478    }
