143    /**
144     * String の即値の intern の速さ
145     * 結果： n = 100000000 の場合
146     * ITKDBG[Lap:new String]: 0.526 [sec]
147     * ITKDBG[Lap:inline String]: 0.015 [sec]
148     * ITKDBG[Lap:inline intern String]: 57.639 [sec]
149     * ITKDBG[Lap:stored intern String]: 0.016 [sec]
150     * ITKDBG[Lap:stored intern String ==]: 0.015 [sec]
151     */
152    //@Test
153    public void test_String_inline_intern() {
154        String d = Itk.intern("aabbccddeeffggaabbccddeeffggaabbccddeeffgg") ;
155        int n = 100000000 ;
156
157        Itk.timerStart("new String") ;
158        int c = 0 ;
159        for(int i = 0 ; i < n ; i++) {
160            final String x = new String("aabbccddeeffggaabbccddeeffggaabbccddeeffgg") ;
161            if(d.equals(x)) { c++ ; }
162        }
163        Itk.timerShowLap("new String") ;
164        Itk.dbgVal("c=",c) ;
165        
166        Itk.timerStart("inline String") ;
167        c = 0 ;
168        for(int i = 0 ; i < n ; i++) {
169            final String x = "aabbccddeeffggaabbccddeeffggaabbccddeeffgg" ;
170            if(d.equals(x)) { c++ ; }
171        }
172        Itk.timerShowLap("inline String") ;
173        Itk.dbgVal("c=",c) ;
174            
175        Itk.timerStart("inline intern String") ;
176        c = 0 ;
177        for(int i = 0 ; i < n ; i++) {
178            String x = Itk.intern("aabbccddeeffggaabbccddeeffggaabbccddeeffgg");
179            if(d.equals(x)) { c++ ; }
180        }
181        Itk.timerShowLap("inline intern String") ;
182        Itk.dbgVal("c=",c) ;
183
184        Itk.timerStart("stored intern String") ;
185        c = 0 ;
186        {
187            final String x = Itk.intern("aabbccddeeffggaabbccddeeffggaabbccddeeffgg") ;
188            for(int i = 0 ; i < n ; i++) {
189                if(d.equals(x)) { c++ ; }
190            }
191        }
192        Itk.timerShowLap("stored intern String") ;
193        Itk.dbgVal("c=",c) ;
194
195        final String x = Itk.intern("aabbccddeeffggaabbccddeeffggaabbccddeeffgg") ;
196        Itk.timerStart("stored intern String ==") ;
197        c = 0 ;
198        for(int i = 0 ; i < n ; i++) {
199            if(d == x) { c++ ; }
200        }
201        Itk.timerShowLap("stored intern String ==") ;
202        Itk.dbgVal("c=",c) ;
203    }

534        for(int i = 0 ; i < m ; i++) {
535            data0 = data0 + data0 ;

537        String data = data0.intern() ;
538
539        Itk.dbgVal("data",data) ;
540
541        ArrayList<String> listCopy = new ArrayList<String>() ;
542        ArrayList<String> listSame = new ArrayList<String>() ;
543        ArrayList<String> listIntern = new ArrayList<String>() ;
544
545        for(int i = 0 ; i < m ; i++) {
546            String newData = "" + data0 ;
547            listCopy.add(newData) ;
548            listSame.add(data) ;
549            listIntern.add(newData.intern()) ;
550        }
551        int c ;
552        // copy
553        Itk.timerStart("copy") ;
554        c = 0 ;
555        for(int i = 0 ; i < n ; i++) {
556            if(listCopy.get(i % m).equals(data)) {
557                c += 1 ;
558            } ;
559        }
560        Itk.timerShowLap("copy") ;
561        Itk.dbgVal("c",c) ;
562        // intern
563        Itk.timerStart("intern") ;
564        c = 0 ;
565        for(int i = 0 ; i < n ; i++) {
566            if(listIntern.get(i % m).equals(data)) {
567            //if(listIntern.get(i % m) == data) {
568                c += 1 ;
569            } ;
570        }
571        Itk.timerShowLap("intern") ;
572        Itk.dbgVal("c",c) ;
573        // same
574        Itk.timerStart("same") ;
575        c = 0 ;
576        for(int i = 0 ; i < n ; i++) {
577            if(listSame.get(i % m).equals(data)) {
578                c += 1 ;
579            } ;
580        }
581        Itk.timerShowLap("same") ;
582        Itk.dbgVal("c",c) ;
