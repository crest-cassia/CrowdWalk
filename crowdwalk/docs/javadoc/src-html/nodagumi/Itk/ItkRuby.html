057     * script 実行(1行)
058     */
059    public Object eval(String script) {
060        return container.runScriptlet(script) ;
061    }
062
063    /**
064     * script 実行(複数行)
065     */
066    public Object eval(String script0, String script1, String... scriptList) {
067        String script = "" + script0 + "\n" + script1 ;
068        for(String scriptN : scriptList) {
069            script = script + "\n" + scriptN ;
070        }
071        return eval(script) ;
072    }

328    /**
329     * Ruby の LoadPath。
330     * @return path のリスト。
331     * Note: container.getLoadPaths() というのがあるが、なぜか正しく動かない。
332     * なので、ruby の環境での値を直接取るようにする。
333     */
334    public List<String> getLoadPaths() {
335        return (List<String>)this.eval("$LOAD_PATH") ;
336    }
337
338    //------------------------------------------------------------
339    /**
340     * Ruby の LoadPath を設定。
341     * @param pathList path のリスト。
342     * Note: container.setLoadPaths() というのがあるが、なぜか正しく動かない。
343     * なので、ruby の環境で値をセットするようにする。
344     */
345    public void setLoadPaths(List<String> pathList) {
346        this.eval("$LOAD_PATH=[]");
347        for(String path : pathList) {
348            pushLoadPath(path) ;
349        }
350    }
351
352    //------------------------------------------------------------
353    /**
354     * Ruby の LoadPath の末尾に追加。
355     * 追加するかどうかチェックする。
356     * @param path 追加するpath。
357     * @return 追加されれば true。すでに存在して追加されなかった場合は false。
358     * Note: container.setLoadPaths() というのがあるが、なぜか正しく動かない。
359     * なので、ruby の環境で値をセットするようにする。
360     */
361    public boolean pushLoadPath(String path) {
362        List<String> pathList = this.getLoadPaths() ;
363        if(pathList.contains(path)) {
364            return false ;
365        } else {
366            container.callMethod(this.eval("$LOAD_PATH"), "push", path) ;
367            return true ;
368        }
369    }

392     * rubyEngine がインスタンスを持つか確認。
393     * @return 新たにインスタンス生成されたら true。それ以外は false。
394     */
395    public static boolean ensureRubyEngine() {
396        if(rubyEngine == null) {
397            //            Itk.timerStart("Ruby.newInstance()") ;
398            rubyEngine = Ruby.newInstance() ;
399            //            Itk.timerShowLap("Ruby.newInstance()") ;
400            return true ;
401        } else {
402            return false ;
403        }
404    }
