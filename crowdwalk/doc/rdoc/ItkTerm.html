167 def toRuby(object, deeply = true)
168   if(!object.is_a?(Term)) then      # Term のインスタンスではない場合。
169     if(object.is_a?(Array) || object.is_a?(java.util.ArrayList)) then
170       return toRubyFromArray(object, deeply) ;
171     elsif(object.is_a?(Hash) || object.is_a?(java.util.HashMap)) then
172       return toRubyFromHash(object, deeply) ;
173     else
174       return object ;
175     end
176 
177   elsif(object.isNull()) then       # null Termの場合
178     return nil ;
179 
180   elsif(object.isAtom()) then               ## atom の場合
181     if(object.isInt()) then
182       return object.getInt() ;
183     elsif(object.isDouble()) then
184       return object.getDouble() ;
185     else
186       return object.getHead() ;
187     end
188 
189   elsif(object.isArray()) then              ## 配列の場合
190     return toRubyFromArray(object.getArray(), deeply) ;
191 
192   else                                      ## json の Object の場合
193     return toRubyFromHash(object.getBody(), deeply) ;
194   end
195 
196 end
